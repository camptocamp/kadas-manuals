49 bool QgsKMLImport::importFile( const QString& filename, QString& errMsg )
50 {
51  if ( filename.endsWith( ".kmz", Qt::CaseInsensitive ) )
52  {
53  QuaZip quaZip( filename );
54  if ( !quaZip.open( QuaZip::mdUnzip ) )
55  {
56  errMsg = tr( "Unable to open %1." ).arg( QFileInfo( filename ).fileName() );
57  return false;
58  }
59  // Search for kml file to open
60  QStringList kmzFileList = quaZip.getFileNameList();
61  int mainKmlIndex = kmzFileList.indexOf( QRegExp( "[^/]+.kml", Qt::CaseInsensitive ) );
62  if ( mainKmlIndex == -1 || !quaZip.setCurrentFile( kmzFileList[mainKmlIndex] ) )
63  {
64  errMsg = tr( "Corrupt KMZ file." );
65  return false;
66  }
67  QDomDocument doc;
68  QuaZipFile file( &quaZip );
69  if ( !file.open( QIODevice::ReadOnly ) || !doc.setContent( &file ) )
70  {
71  errMsg = tr( "Corrupt KMZ file." );
72  return false;
73  }
74  file.close();
75  return importDocument( doc, errMsg, &quaZip );
76  }
77  else if ( filename.endsWith( ".kml", Qt::CaseInsensitive ) )
78  {
79  QFile file( filename );
80  QDomDocument doc;
81  if ( !file.open( QIODevice::ReadOnly ) || !doc.setContent( &file ) )
82  {
83  errMsg = tr( "Unable to open %1." ).arg( QFileInfo( filename ).fileName() );
84  return false;
85  }
86  return importDocument( doc, errMsg );
87  }
88  return false;

91 bool QgsKMLImport::importDocument( const QDomDocument &doc, QString& errMsg, QuaZip *zip )
92 {
93  QDomElement documentEl = doc.firstChildElement( "kml" ).firstChildElement( "Document" );
94  if ( documentEl.isNull() )
95  {
96  errMsg = tr( "Corrupt KMZ file." );
97  return false;
98  }
99 
100  // Styles / StyleMaps with id
101  QMap<QString, StyleData> styleMap;
102  QDomNodeList styleEls = documentEl.elementsByTagName( "Style" );
103  for ( int iStyle = 0, nStyles = styleEls.size(); iStyle < nStyles; ++iStyle )
104  {
105  QDomElement styleEl = styleEls.at( iStyle ).toElement();
106  if ( !styleEl.attribute( "id" ).isEmpty() )
107  {
108  styleMap.insert( QString( "#%1" ).arg( styleEl.attribute( "id" ) ), parseStyle( styleEl, zip ) );
109  }
110  }
111  QDomNodeList styleMapEls = documentEl.elementsByTagName( "StyleMap" );
112  for ( int iStyleMap = 0, nStyleMaps = styleMapEls.size(); iStyleMap < nStyleMaps; ++iStyleMap )
113  {
114  QDomElement styleMapEl = styleMapEls.at( iStyleMap ).toElement();
115  if ( !styleMapEl.attribute( "id" ).isEmpty() )
116  {
117  QString id = QString( "#%1" ).arg( styleMapEl.attribute( "id" ) );
118  // Just pick the first item of the StyleMap
119  QDomElement pairEl = styleMapEl.firstChildElement( "Pair" );
120  QDomElement styleEl = pairEl.firstChildElement( "Style" );
121  QDomElement styleUrlEl = pairEl.firstChildElement( "styleUrl" );
122  if ( !styleEl.isNull() )
123  {
124  styleMap.insert( id, parseStyle( styleEl, zip ) );
125  }
126  else if ( !styleUrlEl.isNull() )
127  {
128  styleMap.insert( id, styleMap.value( styleUrlEl.text() ) );
129  }
130  }
131  }
132 
133 
134  // Placemarks
135  QDomNodeList placemarkEls = documentEl.elementsByTagName( "Placemark" );
136  for ( int iPlacemark = 0, nPlacemarks = placemarkEls.size(); iPlacemark < nPlacemarks; ++iPlacemark )
137  {
138  QDomElement placemarkEl = placemarkEls.at( iPlacemark ).toElement();
139  QString name = placemarkEl.firstChildElement( "name" ).text();
140 
141  // Geometry
142  QList<QgsAbstractGeometryV2*> geoms = parseGeometries( placemarkEl );
143 
144  if ( geoms.isEmpty() )
145  {
146  // Placemark without geometry
147  QgsDebugMsg( "Could not parse placemark geometry" );
148  continue;
149  }
150 
151  // Style
152  QDomElement styleEl = placemarkEl.firstChildElement( "Style" );
153  QDomElement styleUrlEl = placemarkEl.firstChildElement( "styleUrl" );
154  StyleData style;
155  if ( !styleEl.isNull() )
156  {
157  style = parseStyle( styleEl, zip );
158  }
159  else if ( !styleUrlEl.isNull() )
160  {
161  style = styleMap.value( styleUrlEl.text() );
162  }
163 
164  QMap<QString, QString> attributes = parseExtendedData( placemarkEl );
165 
166  // If there is an icon and the geometry is a point, add as annotation item, otherwise as redlining symbol
167  if ( geoms.size() == 1 && !style.icon.isEmpty() && dynamic_cast<QgsPointV2*>( geoms.front() ) )
168  {
169  QgsPointV2* point = static_cast<QgsPointV2*>( geoms.front() );
170  QgsGeoImageAnnotationItem* item = new QgsGeoImageAnnotationItem( mCanvas );
171  item->setPositionLocked( true );
172  item->setFrameVisible( false );
173  item->setFilePath( style.icon );
174  item->setOffsetFromReferencePoint( style.hotSpot );
175  item->setMapPosition( QgsPoint( point->x(), point->y() ), QgsCRSCache::instance()->crsByAuthId( "EPSG:4326" ) );
176  QgsAnnotationLayer::getLayer( mCanvas, "geoImage", tr( "Pictures" ) )->addItem( item );
177  delete geoms.front();
178  }
179  else
180  {
181  // Attempt to recover redlining attributes
182  QString flags = attributes.value( "flags" );
183  QString tooltip = attributes.value( "tooltip" );
184  QString text = attributes.value( "text" );
185  StyleData redliningStyle;
186  redliningStyle.outlineStyle = QgsSymbolLayerV2Utils::decodePenStyle( attributes.value( "outline_style" ) );
187  redliningStyle.fillStyle = QgsSymbolLayerV2Utils::decodeBrushStyle( attributes.value( "fill_style" ) );
188  redliningStyle.outlineSize = attributes.value( "size" ).toInt();
189  redliningStyle.outlineColor = QgsSymbolLayerV2Utils::decodeColor( attributes.value( "outline" ) );
190  redliningStyle.fillColor = QgsSymbolLayerV2Utils::decodeColor( attributes.value( "fill" ) );
191  // If flags is empty, it is not a redlining object. Clear text, because it messes up rendering without matching flags
192  if ( flags.isEmpty() )
193  {
194  text = "";
195  }
196  else
197  {
198  style = redliningStyle;
199  attributes.remove( "flags" );
200  attributes.remove( "tooltip" );
201  attributes.remove( "text" );
202  attributes.remove( "outline_style" );
203  attributes.remove( "fill_style" );
204  attributes.remove( "size" );
205  attributes.remove( "outline" );
206  attributes.remove( "fill" );
207  }
208 
209  QStringList attributesList;
210  for ( auto it = attributes.begin(), itEnd = attributes.end(); it != itEnd; ++it )
211  {
212  attributesList.append( it.key().toHtmlEscaped() + "=" + it.value().replace( QRegularExpression( "^\\s*(.*[^\\s])\\s*$", QRegularExpression::MultilineOption ), "\\1" ).toHtmlEscaped() );
213  }
214  QString attributesStr = attributesList.join( "&" );
215 
216  for ( QgsAbstractGeometryV2* geom : geoms )
217  {
218  QString geomFlags = flags;
219  if ( geomFlags.isEmpty() )
220  {
221  QgsWKBTypes::Type type = QgsWKBTypes::singleType( QgsWKBTypes::flatType( geom->wkbType() ) );
222  if ( type == QgsWKBTypes::Point )
223  {
224  // KML does not have point styles, so points without associated icon are rendered as text
225  text = name;
226  geomFlags = "bold=1,family=sans,fontSize=14,shape=point,symbol=";
227  }
228  else if ( type == QgsWKBTypes::LineString )
229  {
230  geomFlags = "shape=line";
231  }
232  else
233  {
234  geomFlags = "shape=polygon";
235  }
236  }
237  geom->transform( *QgsCoordinateTransformCache::instance()->transform( "EPSG:4326", "EPSG:3857" ) );
238  mRedliningLayer->addShape( new QgsGeometry( geom ), style.outlineColor, style.fillColor, style.outlineSize, style.outlineStyle, style.fillStyle, geomFlags, tooltip, text, attributesStr );
239  }
240  }
241  }
242 
243  // Ground overlays: group by name of parent folder, if possible
244  if ( zip )
245  {
246  QMap<QString, OverlayData> overlays;
247  QDomNodeList groundOverlayEls = documentEl.elementsByTagName( "GroundOverlay" );
248  for ( int iOverlay = 0, nOverlays = groundOverlayEls.size(); iOverlay < nOverlays; ++iOverlay )
249  {
250  QDomElement groundOverlayEl = groundOverlayEls.at( iOverlay ).toElement();
251  QDomElement bboxEl = groundOverlayEl.firstChildElement( "LatLonBox" ).toElement();
252  QString name = groundOverlayEl.firstChildElement( "name" ).text();
253  // If tile contained in folder, group by folder
254  if ( groundOverlayEl.parentNode().nodeName() == "Folder" )
255  {
256  name = groundOverlayEl.parentNode().firstChildElement( "name" ).text();
257  }
258  TileData tile;
259  tile.iconHref = groundOverlayEl.firstChildElement( "Icon" ).firstChildElement( "href" ).text();
260  tile.bbox.setXMinimum( bboxEl.firstChildElement( "west" ).text().toDouble() );
261  tile.bbox.setXMaximum( bboxEl.firstChildElement( "east" ).text().toDouble() );
262  tile.bbox.setYMinimum( bboxEl.firstChildElement( "south" ).text().toDouble() );
263  tile.bbox.setYMaximum( bboxEl.firstChildElement( "north" ).text().toDouble() );
264  overlays[name].tiles.append( tile );
265  if ( overlays[name].bbox.isEmpty() )
266  {
267  overlays[name].bbox = tile.bbox;
268  }
269  else
270  {
271  overlays[name].bbox.unionRect( tile.bbox );
272  }
273  }
274 
275  // Build VRTs for each overlay group
276  for ( auto it = overlays.begin(), itEnd = overlays.end(); it != itEnd; ++it )
277  {
278  buildVSIVRT( it.key(), it.value(), zip );
279  }
280  }
281 
282  return true;
283 }

285 void QgsKMLImport::buildVSIVRT( const QString& name, OverlayData &overlayData, QuaZip* kmzZip ) const
286 {
287  if ( overlayData.tiles.empty() )
288  {
289  return;
290  }
291 
292  // Prepare vsi output
293  QString vsifilename = QgsTemporaryFile::createNewFile( QString( "%1.zip" ).arg( name ) );
294  QuaZip vsiZip( vsifilename );
295  if ( !vsiZip.open( QuaZip::mdCreate ) )
296  {
297  return;
298  }
299  // Copy tile images to vsi, determine tile sizes
300 for ( TileData& tile : overlayData.tiles )
301  {
302  if ( !kmzZip->setCurrentFile( tile.iconHref ) )
303  {
304  continue;
305  }
306  QuaZipFile kmzTileFile( kmzZip );
307  QuaZipFile vsiTileFile( &vsiZip );
308  if ( !kmzTileFile.open( QIODevice::ReadOnly ) || !vsiTileFile.open( QIODevice::WriteOnly, QuaZipNewInfo( QFileInfo( tile.iconHref ).completeBaseName() + ".png" ) ) )
309  {
310  continue;
311  }
312  QImage tileImage = QImage::fromData( kmzTileFile.readAll() );
313  tile.size = tileImage.size();
314  tileImage.save( &vsiTileFile, "PNG" );
315  vsiTileFile.close();
316  }
317 
318  // Get total size by assuming all tiles have same resolution
319  const TileData& firstTile = overlayData.tiles.front();
320  QSize totSize(
321  qRound( firstTile.size.width() / firstTile.bbox.width() * overlayData.bbox.width() ),
322  qRound( firstTile.size.height() / firstTile.bbox.height() * overlayData.bbox.height() )
323  );
324 
325  // Write vrt
326  QString vrtString;
327  QTextStream vrtStream( &vrtString, QIODevice::WriteOnly );
328  vrtStream << "<VRTDataset rasterXSize=\"" << totSize.width() << "\" rasterYSize=\"" << totSize.height() << "\">" << endl;
329  vrtStream << " <SRS>" << QgsCRSCache::instance()->crsByAuthId( "EPSG:4326" ).toProj4().toHtmlEscaped() << "</SRS>" << endl;
330  vrtStream << " <GeoTransform>" << endl;
331  vrtStream << " " << overlayData.bbox.xMinimum() << "," << ( overlayData.bbox.width() / totSize.width() ) << ", 0," << endl;
332  vrtStream << " " << overlayData.bbox.yMaximum() << ", 0," << ( -overlayData.bbox.height() / totSize.height() ) << endl;
333  vrtStream << " </GeoTransform>" << endl;
334 for ( const QPair<int, QString>& band : QList<QPair<int, QString>> {qMakePair( 1, QString( "Red" ) ), qMakePair( 2, QString( "Green" ) ), qMakePair( 3, QString( "Blue" ) ), qMakePair( 4, QString( "Alpha" ) )} )
335  {
336  vrtStream << " <VRTRasterBand dataType=\"Byte\" band=\"" << band.first << "\">" << endl;
337  vrtStream << " <ColorInterp>" << band.second << "</ColorInterp>" << endl;
338  for ( const TileData& tile : overlayData.tiles )
339  {
340  int i = qRound(( tile.bbox.xMinimum() - overlayData.bbox.xMinimum() ) / overlayData.bbox.width() * totSize.width() );
341  int j = qRound(( overlayData.bbox.yMaximum() - tile.bbox.yMaximum() ) / overlayData.bbox.height() * totSize.height() );
342  vrtStream << " <SimpleSource>" << endl;
343  vrtStream << " <SourceFilename relativeToVRT=\"1\">" << tile.iconHref << "</SourceFilename>" << endl;
344  vrtStream << " <SourceBand>" << band.first << "</SourceBand>" << endl;
345  vrtStream << " <SourceProperties RasterXSize=\"" << tile.size.width() << "\" RasterYSize=\"" << tile.size.height() << "\" DataType=\"Byte\" BlockXSize=\"" << tile.size.width() << "\" BlockYSize=\"1\" />" << endl;
346  vrtStream << " <SrcRect xOff=\"0\" yOff=\"0\" xSize=\"" << tile.size.width() << "\" ySize=\"" << tile.size.height() << "\" />" << endl;
347  vrtStream << " <DstRect xOff=\"" << i << "\" yOff=\"" << j << "\" xSize=\"" << tile.size.width() << "\" ySize=\"" << tile.size.height() << "\" />" << endl;
348  vrtStream << " </SimpleSource>" << endl;
349  }
350  vrtStream << " </VRTRasterBand>" << endl;
351  }
352  vrtStream << "</VRTDataset>" << endl;
353  vrtStream.flush();
354 
355  QuaZipFile vsiVrtFile( &vsiZip );
356  if ( !vsiVrtFile.open( QIODevice::WriteOnly, QuaZipNewInfo( "dataset.vrt" ) ) )
357  {
358  return;
359  }
360  vsiVrtFile.write( vrtString.toLocal8Bit() );
361  vsiVrtFile.close();
362  vsiZip.close();
363 
364  QgsRasterLayer* rasterLayer = new QgsRasterLayer( QString( "/vsizip/%1/dataset.vrt" ).arg( vsifilename ), name, "gdal" );
365  QgsMapLayerRegistry::instance()->addMapLayer( rasterLayer );
366 }

368 QList<QgsPointV2> QgsKMLImport::parseCoordinates( const QDomElement& geomEl ) const
369 {
370  QStringList coordinates = geomEl.firstChildElement( "coordinates" ).text().split( QRegExp( "\\s+" ), QString::SkipEmptyParts );
371  QList<QgsPointV2> points;
372  for ( int i = 0, n = coordinates.size(); i < n; ++i )
373  {
374  QStringList coordinate = coordinates[i].split( "," );
375  if ( coordinate.size() >= 3 )
376  {
377  QgsPointV2 p( QgsWKBTypes::PointZ );
378  p.setX( coordinate[0].toDouble() );
379  p.setY( coordinate[1].toDouble() );
380  p.setZ( coordinate[2].toDouble() );
381  points.append( p );
382  }
383  else if ( coordinate.size() == 2 )
384  {
385  QgsPointV2 p( QgsWKBTypes::Point );
386  p.setX( coordinate[0].toDouble() );
387  p.setY( coordinate[1].toDouble() );
388  points.append( p );
389  }
390  }
391  return points;
392 }

412  }
413  if ( polyStyleEl.firstChildElement( "outline" ).text() == "0" )
414  {
415  style.outlineColor = QColor( Qt::transparent );

417

419  if ( !iconHRefEl.isNull() )
420  {

422  // Only local files in KMZ are supported (also for security reasons)
423  if ( zip && zip->setCurrentFile( filename ) )
424  {
425  QuaZipFile file( zip );
426  if ( file.open( QIODevice::ReadOnly ) )
427  {
428  QImage icon = QImage::fromData( file.readAll() );
429  style.icon = QgsTemporaryFile::createNewFile( "kml_import.png" );
430  icon.save( style.icon );
431 
432  QDomElement hotSpotEl = styleEl.firstChildElement( "IconStyle" ).firstChildElement( "hotSpot" );
433  if ( !hotSpotEl.isNull() )
434  {
435  double x = hotSpotEl.attribute( "x" ).toDouble();
436  double y = hotSpotEl.attribute( "y" ).toDouble();
437  if ( hotSpotEl.attribute( "xunits" ) == "fraction" )
438  {
439  x *= icon.width();
440  }
441  if ( hotSpotEl.attribute( "yunits" ) == "fraction" )
442  {
443  y *= icon.height();
444  }
445  style.hotSpot = QPointF( x, y );
446  }
447  }
448  }
449  }
450  return style;
451 }

453 QList<QgsAbstractGeometryV2*> QgsKMLImport::parseGeometries( const QDomElement& containerEl )
454 {
455  QList<QgsAbstractGeometryV2*> geoms;
456 
457  QDomElement pointEl = containerEl.firstChildElement( "Point" );
458  if ( !pointEl.isNull() )
459  {
460  QList<QgsPointV2> points = parseCoordinates( pointEl );
461  if ( !points.isEmpty() )
462  {
463  geoms.append( points[0].clone() );
464  }
465  }
466 
467  QDomElement lineStringEl = containerEl.firstChildElement( "LineString" );
468  if ( !lineStringEl.isNull() )
469  {
470  QgsLineStringV2* line = new QgsLineStringV2();
471  line->setPoints( parseCoordinates( lineStringEl ) );
472  geoms.append( line );
473  }
474 
475  QDomElement polygonEl = containerEl.firstChildElement( "Polygon" );
476  if ( !polygonEl.isNull() )
477  {
478  QDomElement outerRingEl = polygonEl.firstChildElement( "outerBoundaryIs" ).firstChildElement( "LinearRing" );
479  QgsLineStringV2* exterior = new QgsLineStringV2();
480  exterior->setPoints( parseCoordinates( outerRingEl ) );
481  QgsPolygonV2* poly = new QgsPolygonV2();
482  poly->setExteriorRing( exterior );
483 
484  QDomNodeList innerBoundaryEls = polygonEl.elementsByTagName( "innerBoundaryIs" );
485  for ( int iRing = 0, nRings = innerBoundaryEls.size(); iRing < nRings; ++iRing )
486  {
487  QDomElement innerRingEl = innerBoundaryEls.at( iRing ).toElement().firstChildElement( "LinearRing" );
488  QgsLineStringV2* interior = new QgsLineStringV2();
489  interior->setPoints( parseCoordinates( innerRingEl ) );
490  poly->addInteriorRing( interior );
491  }
492  geoms.append( poly );
493  }
494 
495  QDomElement multiGeometryEl = containerEl.firstChildElement( "MultiGeometry" );
496  if ( !multiGeometryEl.isNull() )
497  {
498  geoms.append( parseGeometries( multiGeometryEl ) );
499  }
500  return geoms;
501 }
