57 void QgsMapToolPan::canvasPressEvent( QMouseEvent * e )
58 {
59  if ( e->button() == Qt::LeftButton )
60  {
61  if ( e->modifiers() == Qt::ShiftModifier || e->modifiers() == Qt::ControlModifier )
62  {
63  mExtentRubberBand = new QgsRubberBand( mCanvas, QGis::Polygon );
64  if ( e->modifiers() == Qt::ShiftModifier )
65  mExtentRubberBand->setColor( QColor( 0, 0, 255, 63 ) );
66  else if ( e->modifiers() == Qt::ControlModifier )
67  {
68  QSettings settings;
69  int red = settings.value( "/Qgis/default_measure_color_red", 255 ).toInt();
70  int green = settings.value( "/Qgis/default_measure_color_green", 0 ).toInt();
71  int blue = settings.value( "/Qgis/default_measure_color_blue", 0 ).toInt();
72  mExtentRubberBand->setFillColor( QColor( red, green, blue, 63 ) );
73  mExtentRubberBand->setBorderColor( QColor( red, green, blue, 255 ) );
74  mExtentRubberBand->setWidth( 3 );
75  }
76  mExtentRect.setTopLeft( e->pos() );
77  mExtentRect.setBottomRight( e->pos() );
78  mExtentRubberBand->setToCanvasRectangle( mExtentRect );
79  mExtentRubberBand->show();
80  }
81  else if ( mAllowItemInteraction )
82  {
83  mPickClick = true;
84  }
85  }
86  else if ( e->button() == Qt::RightButton && mAllowItemInteraction )
87  {
88  emit contextMenuRequested( e->pos(), toMapCoordinates( e->pos() ) );
89  }
90 }

116 void QgsMapToolPan::canvasReleaseEvent( QMouseEvent * e )
117 {
118  if ( e->button() == Qt::LeftButton )
119  {
120  if ( mExtentRubberBand )
121  {
122  if ( e->modifiers() == Qt::ShiftModifier )
123  {
124  // set center and zoom
125  QSize zoomRectSize = mExtentRect.normalized().size();
126  QSize canvasSize = mCanvas->mapSettings().outputSize();
127  double sfx = ( double )zoomRectSize.width() / canvasSize.width();
128  double sfy = ( double )zoomRectSize.height() / canvasSize.height();
129  double scaleFactor = qMax( sfx, sfy );
130 
131  QgsPoint c = mCanvas->getCoordinateTransform()->toMapCoordinates( mExtentRect.center() );
132  QgsRectangle oe = mCanvas->mapSettings().extent();
133  QgsRectangle e(
134  c.x() - oe.width() / 2.0, c.y() - oe.height() / 2.0,
135  c.x() + oe.width() / 2.0, c.y() + oe.height() / 2.0
136  );
137  e.scale( scaleFactor, &c );
138  mCanvas->setExtent( e, true );
139  mCanvas->refresh();
140  }
141  else if ( e->modifiers() == Qt::ControlModifier )
142  {
143  QgsMapToolDeleteItems( canvas() ).deleteItems( mExtentRubberBand->rect(), canvas()->mapSettings().destinationCrs() );
144  }
145 
146  delete mExtentRubberBand;
147  mExtentRubberBand = 0;
148  }
149  else if ( mDragging )
150  {
151  mCanvas->panActionEnd( e->pos() );
152  mDragging = false;
153  mCanvas->setCursor( mCursor );
154  }
155  else if ( mAllowItemInteraction && mPickClick )
156  {
157  QgsFeaturePicker::PickResult result = QgsFeaturePicker::pick( mCanvas, e->pos(), toMapCoordinates( e->pos() ), QGis::AnyGeometry );
158  if ( !result.isEmpty() )
159  {
160  emit itemPicked( result );
161  }
162  }
163  }
164 }

177 void QgsMapToolPan::pinchTriggered( QPinchGesture *gesture )
178 {
179  if ( gesture->state() == Qt::GestureFinished )
180  {
181  //a very small totalScaleFactor indicates a two finger tap (pinch gesture without pinching)
182  if ( 0.98 < gesture->totalScaleFactor() && gesture->totalScaleFactor() < 1.02 )
183  {
184  mCanvas->zoomOut();
185  }
186  else
187  {
188  //Transfor global coordinates to widget coordinates
189  QPoint pos = gesture->centerPoint().toPoint();
190  pos = mCanvas->mapFromGlobal( pos );
191  // transform the mouse pos to map coordinates
192  QgsPoint center = mCanvas->getCoordinateTransform()->toMapPoint( pos.x(), pos.y() );
193  QgsRectangle r = mCanvas->extent();
194  r.scale( 1 / gesture->totalScaleFactor(), &center );
195  mCanvas->setExtent( r );
196  mCanvas->refresh();
197  }
198  mPinching = false;
199  }
