51 QgsRasterBlock * QgsRasterDataProvider::block2( int theBandNo, QgsRectangle const & theExtent, int theWidth, int theHeight, QgsRasterBlockFeedback *feedback )
52 {
53  QgsDebugMsg( QString( "theBandNo = %1 theWidth = %2 theHeight = %3" ).arg( theBandNo ).arg( theWidth ).arg( theHeight ) );
54  QgsDebugMsg( QString( "theExtent = %1" ).arg( theExtent.toString() ) );
55 
56  QgsRasterBlock *block;
57  if ( srcHasNoDataValue( theBandNo ) && useSrcNoDataValue( theBandNo ) )
58  {
59  block = new QgsRasterBlock( dataType( theBandNo ), theWidth, theHeight, srcNoDataValue( theBandNo ) );
60  }
61  else
62  {
63  block = new QgsRasterBlock( dataType( theBandNo ), theWidth, theHeight );
64  }
65 
66  if ( block->isEmpty() )
67  {
68  QgsDebugMsg( "Couldn't create raster block" );
69  return block;
70  }
71 
72  // Read necessary extent only
73  QgsRectangle tmpExtent = extent().intersect( &theExtent );
74 
75  if ( tmpExtent.isEmpty() )
76  {
77  QgsDebugMsg( "Extent outside provider extent" );
78  block->setIsNoData();
79  return block;
80  }
81 
82  double xRes = theExtent.width() / theWidth;
83  double yRes = theExtent.height() / theHeight;
84  double tmpXRes, tmpYRes;
85  double providerXRes = 0;
86  double providerYRes = 0;
87  if ( capabilities() & Size )
88  {
89  providerXRes = extent().width() / xSize();
90  providerYRes = extent().height() / ySize();
91  tmpXRes = qMax( providerXRes, xRes );
92  tmpYRes = qMax( providerYRes, yRes );
93  if ( qgsDoubleNear( tmpXRes, xRes ) ) tmpXRes = xRes;
94  if ( qgsDoubleNear( tmpYRes, yRes ) ) tmpYRes = yRes;
95  }
96  else
97  {
98  tmpXRes = xRes;
99  tmpYRes = yRes;
100  }
101 
102  if ( tmpExtent != theExtent ||
103  tmpXRes > xRes || tmpYRes > yRes )
104  {
105  // Read smaller extent or lower resolution
106 
107  if ( !extent().contains( theExtent ) )
108  {
109  QRect subRect = QgsRasterBlock::subRect( theExtent, theWidth, theHeight, extent() );
110  block->setIsNoDataExcept( subRect );
111  }
112 
113  // Calculate row/col limits (before tmpExtent is aligned)
114  int fromRow = qRound(( theExtent.yMaximum() - tmpExtent.yMaximum() ) / yRes );
115  int toRow = qRound(( theExtent.yMaximum() - tmpExtent.yMinimum() ) / yRes ) - 1;
116  int fromCol = qRound(( tmpExtent.xMinimum() - theExtent.xMinimum() ) / xRes );
117  int toCol = qRound(( tmpExtent.xMaximum() - theExtent.xMinimum() ) / xRes ) - 1;
118 
119  QgsDebugMsg( QString( "fromRow = %1 toRow = %2 fromCol = %3 toCol = %4" ).arg( fromRow ).arg( toRow ).arg( fromCol ).arg( toCol ) );
120 
121  if ( fromRow < 0 || fromRow >= theHeight || toRow < 0 || toRow >= theHeight ||
122  fromCol < 0 || fromCol >= theWidth || toCol < 0 || toCol >= theWidth )
123  {
124  // Should not happen
125  QgsDebugMsg( "Row or column limits out of range" );
126  return block;
127  }
128 
129  // If lower source resolution is used, the extent must beS aligned to original
130  // resolution to avoid possible shift due to resampling
131  if ( tmpXRes > xRes )
132  {
133  int col = floor(( tmpExtent.xMinimum() - extent().xMinimum() ) / providerXRes );
134  tmpExtent.setXMinimum( extent().xMinimum() + col * providerXRes );
135  col = ceil(( tmpExtent.xMaximum() - extent().xMinimum() ) / providerXRes );
136  tmpExtent.setXMaximum( extent().xMinimum() + col * providerXRes );
137  }
138  if ( tmpYRes > yRes )
139  {
140  int row = floor(( extent().yMaximum() - tmpExtent.yMaximum() ) / providerYRes );
141  tmpExtent.setYMaximum( extent().yMaximum() - row * providerYRes );
142  row = ceil(( extent().yMaximum() - tmpExtent.yMinimum() ) / providerYRes );
143  tmpExtent.setYMinimum( extent().yMaximum() - row * providerYRes );
144  }
145  int tmpWidth = qRound( tmpExtent.width() / tmpXRes );
146  int tmpHeight = qRound( tmpExtent.height() / tmpYRes );
147  tmpXRes = tmpExtent.width() / tmpWidth;
148  tmpYRes = tmpExtent.height() / tmpHeight;
149 
150  QgsDebugMsg( QString( "Reading smaller block tmpWidth = %1 theHeight = %2" ).arg( tmpWidth ).arg( tmpHeight ) );
151  QgsDebugMsg( QString( "tmpExtent = %1" ).arg( tmpExtent.toString() ) );
152 
153  QgsRasterBlock *tmpBlock;
154  if ( srcHasNoDataValue( theBandNo ) && useSrcNoDataValue( theBandNo ) )
155  {
156  tmpBlock = new QgsRasterBlock( dataType( theBandNo ), tmpWidth, tmpHeight, srcNoDataValue( theBandNo ) );
157  }
158  else
159  {
160  tmpBlock = new QgsRasterBlock( dataType( theBandNo ), tmpWidth, tmpHeight );
161  }
162 
163  readBlock( theBandNo, tmpExtent, tmpWidth, tmpHeight, tmpBlock->bits() );
164 
165  int pixelSize = dataTypeSize( theBandNo );
166 
167  double xMin = theExtent.xMinimum();
168  double yMax = theExtent.yMaximum();
169  double tmpXMin = tmpExtent.xMinimum();
170  double tmpYMax = tmpExtent.yMaximum();
171 
172  for ( int row = fromRow; row <= toRow; row++ )
173  {
174  double y = yMax - ( row + 0.5 ) * yRes;
175  int tmpRow = floor(( tmpYMax - y ) / tmpYRes );
176 
177  for ( int col = fromCol; col <= toCol; col++ )
178  {
179  double x = xMin + ( col + 0.5 ) * xRes;
180  int tmpCol = floor(( x - tmpXMin ) / tmpXRes );
181 
182  if ( tmpRow < 0 || tmpRow >= tmpHeight || tmpCol < 0 || tmpCol >= tmpWidth )
183  {
184  QgsDebugMsg( "Source row or column limits out of range" );
185  block->setIsNoData(); // so that the problem becomes obvious and fixed
186  delete tmpBlock;
187  return block;
188  }
189 
190  qgssize tmpIndex = ( qgssize )tmpRow * ( qgssize )tmpWidth + tmpCol;
191  qgssize index = row * ( qgssize )theWidth + col;
192 
193  char *tmpBits = tmpBlock->bits( tmpIndex );
194  char *bits = block->bits( index );
195  if ( !tmpBits )
196  {
197  QgsDebugMsg( QString( "Cannot get input block data tmpRow = %1 tmpCol = %2 tmpIndex = %3." ).arg( tmpRow ).arg( tmpCol ).arg( tmpIndex ) );
198  continue;
199  }
200  if ( !bits )
201  {
202  QgsDebugMsg( "Cannot set output block data." );
203  continue;
204  }
205  memcpy( bits, tmpBits, pixelSize );
206  }
207  }
208 
209  delete tmpBlock;
210  }
211  else
212  {
213  readBlock( theBandNo, theExtent, theWidth, theHeight, block->bits(), feedback );
214  }
215 
216  // apply scale and offset
217  block->applyScaleOffset( bandScale( theBandNo ), bandOffset( theBandNo ) );
218  // apply user no data values
219  block->applyNoDataValues( userNoDataValues( theBandNo ) );
220  return block;
221 }

285 QgsRasterIdentifyResult QgsRasterDataProvider::identify( const QgsPoint & thePoint, QgsRaster::IdentifyFormat theFormat, const QgsRectangle &theExtent, int theWidth, int theHeight, int /*theDpi*/ )
286 {
287  QgsDebugMsg( "Entered" );
288  QMap<int, QVariant> results;
289 
290  if ( theFormat != QgsRaster::IdentifyFormatValue || !( capabilities() & IdentifyValue ) )
291  {
292  QgsDebugMsg( "Format not supported" );
293  return QgsRasterIdentifyResult( ERR( tr( "Format not supported" ) ) );
294  }
295 
296  if ( !extent().contains( thePoint ) )
297  {
298  // Outside the raster
299  for ( int bandNo = 1; bandNo <= bandCount(); bandNo++ )
300  {
301  results.insert( bandNo, QVariant() );
302  }
303  return QgsRasterIdentifyResult( QgsRaster::IdentifyFormatValue, results );
304  }
305 
306  QgsRectangle myExtent = theExtent;
307  if ( myExtent.isEmpty() ) myExtent = extent();
308 
309  if ( theWidth == 0 )
310  {
311  theWidth = capabilities() & Size ? xSize() : 1000;
312  }
313  if ( theHeight == 0 )
314  {
315  theHeight = capabilities() & Size ? ySize() : 1000;
316  }
317 
318  // Calculate the row / column where the point falls
319  double xres = ( myExtent.width() ) / theWidth;
320  double yres = ( myExtent.height() ) / theHeight;
321 
322  int col = ( int ) floor(( thePoint.x() - myExtent.xMinimum() ) / xres );
323  int row = ( int ) floor(( myExtent.yMaximum() - thePoint.y() ) / yres );
324 
325  double xMin = myExtent.xMinimum() + col * xres;
326  double xMax = xMin + xres;
327  double yMax = myExtent.yMaximum() - row * yres;
328  double yMin = yMax - yres;
329  QgsRectangle pixelExtent( xMin, yMin, xMax, yMax );
330 
331  for ( int i = 1; i <= bandCount(); i++ )
332  {
333  QgsRasterBlock * myBlock = block( i, pixelExtent, 1, 1 );
334 
335  if ( myBlock )
336  {
337  double value = myBlock->value( 0 );
338 
339  results.insert( i, value );
340  delete myBlock;
341  }
342  else
343  {
344  results.insert( i, QVariant() );
345  }
346  }
347  return QgsRasterIdentifyResult( QgsRaster::IdentifyFormatValue, results );
348 }
