164 void QgsVectorLayerUndoCommandChangeGeometry::undo()
165 {
166  if ( FID_IS_NEW( mFid ) )
167  {
168  // modify added features
169  QgsFeatureMap::iterator it = mBuffer->mAddedFeatures.find( mFid );
170  Q_ASSERT( it != mBuffer->mAddedFeatures.end() );
171  it.value().setGeometry( *mOldGeom );
172 
173  cache()->cacheGeometry( mFid, *mOldGeom );
174  emit mBuffer->geometryChanged( mFid, *mOldGeom );
175  }
176  else
177  {
178  // existing feature
179 
180  if ( !mOldGeom )
181  {
182  mBuffer->mChangedGeometries.remove( mFid );
183 
184  QgsFeature f;
185  if ( layer()->getFeatures( QgsFeatureRequest().setFilterFid( mFid ).setSubsetOfAttributes( QgsAttributeList() ) ).nextFeature( f ) && f.geometry() )
186  {
187  cache()->cacheGeometry( mFid, *f.geometry() );
188  emit mBuffer->geometryChanged( mFid, *f.geometry() );
189  }
190  }
191  else
192  {
193  mBuffer->mChangedGeometries[mFid] = *mOldGeom;
194  cache()->cacheGeometry( mFid, *mOldGeom );
195  emit mBuffer->geometryChanged( mFid, *mOldGeom );
196  }
197  }

246 void QgsVectorLayerUndoCommandChangeAttribute::undo()
247 {
248  QVariant original = mOldValue;
249 
250  if ( FID_IS_NEW( mFid ) )
251  {
252  // added feature
253  QgsFeatureMap::iterator it = mBuffer->mAddedFeatures.find( mFid );
254  Q_ASSERT( it != mBuffer->mAddedFeatures.end() );
255  it.value().setAttribute( mFieldIndex, mOldValue );
256  }
257  else if ( mFirstChange )
258  {
259  // existing feature
260  mBuffer->mChangedAttributeValues[mFid].remove( mFieldIndex );
261  if ( mBuffer->mChangedAttributeValues[mFid].isEmpty() )
262  mBuffer->mChangedAttributeValues.remove( mFid );
263 
264  if ( !mOldValue.isValid() )
265  {
266  // get old value from provider
267  QgsFeature tmp;
268  QgsFeatureRequest request;
269  request.setFilterFid( mFid );
270  request.setFlags( QgsFeatureRequest::NoGeometry );
271  request.setSubsetOfAttributes( QgsAttributeList() << mFieldIndex );
272  QgsFeatureIterator fi = layer()->getFeatures( request );
273  if ( fi.nextFeature( tmp ) )
274  original = tmp.attribute( mFieldIndex );
275  }
276  }
277  else
278  {
279  mBuffer->mChangedAttributeValues[mFid][mFieldIndex] = mOldValue;
280  }
281 
282  emit mBuffer->attributeValueChanged( mFid, mFieldIndex, original );
283 }

341 QgsVectorLayerUndoCommandDeleteAttribute::QgsVectorLayerUndoCommandDeleteAttribute( QgsVectorLayerEditBuffer* buffer, int fieldIndex )
342  : QgsVectorLayerUndoCommand( buffer )
343  , mFieldIndex( fieldIndex )
344 {
345  const QgsFields& fields = layer()->pendingFields();
346  QgsFields::FieldOrigin origin = fields.fieldOrigin( mFieldIndex );
347  mOriginIndex = fields.fieldOriginIndex( mFieldIndex );
348  mProviderField = ( origin == QgsFields::OriginProvider );
349 
350  if ( !mProviderField )
351  {
352  // need to store the field definition
353  mOldField = mBuffer->mAddedAttributes[mOriginIndex];

355 
356  // save values of new features

358  {
359  const QgsFeature& f = it.value();
360  mDeletedValues.insert( f.id(), f.attribute( mFieldIndex ) );
361  }
362 
363  // save changed values
364  for ( QgsChangedAttributesMap::const_iterator it = mBuffer->mChangedAttributeValues.begin(); it != mBuffer->mChangedAttributeValues.end(); ++it )
365  {

372 void QgsVectorLayerUndoCommandDeleteAttribute::undo()
373 {
374  if ( mProviderField )
375  {
376  mBuffer->mDeletedAttributeIds.removeOne( mOriginIndex );
377  }
378  else
379  {
380  // newly added attribute
381  mBuffer->mAddedAttributes.insert( mOriginIndex, mOldField );
382  }
383 
384  mBuffer->updateLayerFields();
385  mBuffer->handleAttributeAdded( mFieldIndex ); // update changed attributes + new features
386 
387  // set previously used attributes of new features
388  for ( QgsFeatureMap::iterator it = mBuffer->mAddedFeatures.begin(); it != mBuffer->mAddedFeatures.end(); ++it )
389  {
390  QgsFeature& f = it.value();
391  f.setAttribute( mFieldIndex, mDeletedValues.value( f.id() ) );
392  }
393  // set previously used changed attributes
394  for ( QMap<QgsFeatureId, QVariant>::const_iterator it = mDeletedValues.begin(); it != mDeletedValues.end(); ++it )
395  {
396  if ( !FID_IS_NEW( it.key() ) )
397  {
398  QgsAttributeMap& attrs = mBuffer->mChangedAttributeValues[it.key()]; // also adds record if nonexistant
399  attrs.insert( mFieldIndex, it.value() );
400  }
401  }
402 
403  emit mBuffer->attributeAdded( mFieldIndex );
404 }
