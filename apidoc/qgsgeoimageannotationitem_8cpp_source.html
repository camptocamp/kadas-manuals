33 QgsGeoImageAnnotationItem* QgsGeoImageAnnotationItem::create( QgsMapCanvas *canvas, const QString &filePath, bool onlyGeoreferenced, QString* errMsg )
34 {
35  QImageReader reader( filePath );
36  if ( reader.format().isEmpty() || reader.size().isEmpty() )
37  {
38  // Image format not supported
39  return 0;
40  }
41  QgsPoint wgs84Pos;
42  bool locked = true;
43  // Fall back to canvas center position if image has no geotags
44  if ( !readGeoPos( filePath, wgs84Pos, errMsg ) )
45  {
46  if ( onlyGeoreferenced )
47  {
48  return 0;
49  }
50  locked = false;
51  const QgsCoordinateTransform* crst = QgsCoordinateTransformCache::instance()->transform( canvas->mapSettings().destinationCrs().authid(), "EPSG:4326" );
52  wgs84Pos = crst->transform( canvas->mapSettings().extent().center() );
53  }
54  QgsGeoImageAnnotationItem* item = new QgsGeoImageAnnotationItem( canvas );
55  if ( locked )
56  {
57  item->setItemFlags( QgsAnnotationItem::ItemMapPositionLocked );
58  }
59  item->setFilePath( filePath );
60  item->setMapPosition( wgs84Pos, QgsCRSCache::instance()->crsByAuthId( "EPSG:4326" ) );
61  QgsAnnotationLayer::getLayer( canvas, "geoImage", tr( "Pictures" ) )->addItem( item );
62  canvas->setMapTool( new QgsMapToolEditAnnotation( canvas, item ) );
63  return item;
64 }

66 bool QgsGeoImageAnnotationItem::readGeoPos( const QString &filePath, QgsPoint &wgs84Pos, QString* errMsg )
67 {
68  // Read EXIF position
69  Exiv2::Image::AutoPtr image;
70  try
71  {
72  image = Exiv2::ImageFactory::open( filePath.toLocal8Bit().data() );
73  }
74  catch ( const Exiv2::Error& )
75  {
76  if ( errMsg ) *errMsg = tr( "Could not read image" );
77  return false;
78  }
79 
80  if ( image.get() == 0 )
81  {
82  if ( errMsg ) *errMsg = tr( "Could not read image" );
83  return false;
84  }
85 
86  image->readMetadata();
87  Exiv2::ExifData &exifData = image->exifData();
88  if ( exifData.empty() )
89  {
90  if ( errMsg ) *errMsg = tr( "Failed to read EXIF tags" );
91  return false;
92  }
93 
94  Exiv2::ExifData::iterator itLatRef = exifData.findKey( Exiv2::ExifKey( "Exif.GPSInfo.GPSLatitudeRef" ) );
95  Exiv2::ExifData::iterator itLatVal = exifData.findKey( Exiv2::ExifKey( "Exif.GPSInfo.GPSLatitude" ) );
96  Exiv2::ExifData::iterator itLonRef = exifData.findKey( Exiv2::ExifKey( "Exif.GPSInfo.GPSLongitudeRef" ) );
97  Exiv2::ExifData::iterator itLonVal = exifData.findKey( Exiv2::ExifKey( "Exif.GPSInfo.GPSLongitude" ) );
98 
99  if ( itLatRef == exifData.end() || itLatVal == exifData.end() ||
100  itLonRef == exifData.end() || itLonVal == exifData.end() )
101  {
102  if ( errMsg ) *errMsg = tr( "Failed to read position EXIF tags" );
103  return false;
104  }
105  QString latRef = QString::fromStdString( itLatRef->value().toString() );
106  QStringList latVals = QString::fromStdString( itLatVal->value().toString() ).split( QRegExp( "\\s+" ) );
107  QString lonRef = QString::fromStdString( itLonRef->value().toString() );
108  QStringList lonVals = QString::fromStdString( itLonVal->value().toString() ).split( QRegExp( "\\s+" ) );
109 
110  double lat = 0, lon = 0;
111  double div = 1;
112  foreach ( const QString& rational, latVals )
113  {
114  QStringList pair = rational.split( "/" );
115  if ( pair.size() != 2 )
116  break;
117  lat += ( pair[0].toDouble() / pair[1].toDouble() ) / div;
118  div *= 60;
119  }
120 
121  div = 1;
122  foreach ( const QString& rational, lonVals )
123  {
124  QStringList pair = rational.split( "/" );
125  if ( pair.size() != 2 )
126  break;
127  lon += ( pair[0].toDouble() / pair[1].toDouble() ) / div;
128  div *= 60;
129  }
130 
131  if ( latRef.compare( "S", Qt::CaseInsensitive ) == 0 )
132  {
133  lat *= -1;
134  }
135  if ( lonRef.compare( "W", Qt::CaseInsensitive ) == 0 )
136  {
137  lon *= -1;
138  }
139  wgs84Pos = QgsPoint( lon, lat );
140  return true;
141 }

162 void QgsGeoImageAnnotationItem::setFilePath( const QString& filePath )
163 {
164  mFilePath = filePath;
165  mName = QFileInfo( mFilePath ).baseName();
166 
167  QImageReader reader( mFilePath );
168 
169  QSize imageSize = reader.size();
170  // Scale such that largest dimension is max 64px
171  if ( imageSize.width() > imageSize.height() )
172  {
173  imageSize.setHeight(( 64. * imageSize.height() ) / imageSize.width() );
174  imageSize.setWidth( 64 );
175  }
176  else
177  {
178  imageSize.setWidth(( 64. * imageSize.width() ) / imageSize.height() );
179  imageSize.setHeight( 64 );
180  }
181 
182  reader.setBackgroundColor( Qt::white );
183  reader.setScaledSize( imageSize );
184  mImage = reader.read().convertToFormat( QImage::Format_ARGB32 );
185  setFrameSize( QSize( imageSize.width() + 4, imageSize.height() + 4 ) );
186 }

188 void QgsGeoImageAnnotationItem::writeXML( QDomDocument& doc ) const
189 {
190  QDomElement documentElem = doc.documentElement();
191  if ( documentElem.isNull() )
192  {
193  return;
194  }
195 
196  QDomElement geoImageAnnotationElem = doc.createElement( "GeoImageAnnotationItem" );
197  mFilePath = QgsProject::instance()->writePath( mFilePath, QString::null, true );
198  geoImageAnnotationElem.setAttribute( "file", mFilePath );
199  QgsPoint worldPos = QgsCoordinateTransformCache::instance()->transform( mGeoPosCrs.authid(), "EPSG:4326" )->transform( mGeoPos );
200  geoImageAnnotationElem.setAttribute( "lon", worldPos.x() );
201  geoImageAnnotationElem.setAttribute( "lat", worldPos.y() );
202  _writeXML( doc, geoImageAnnotationElem );
203  documentElem.appendChild( geoImageAnnotationElem );

207 {
208  QString filePath = QgsProject::instance()->readPath( itemElem.attribute( "file" ) );
209  QString lon = itemElem.attribute( "lon" );
210  QString lat = itemElem.attribute( "lat" );

212 
213  QDomElement annotationElem = itemElem.firstChildElement( "AnnotationItem" );
214  if ( !annotationElem.isNull() )
215  {
216  _readXML( doc, annotationElem );

218
