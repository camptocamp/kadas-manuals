399 void QgsMapCanvas::setLayerSet( QList<QgsMapCanvasLayer> &layers )
400 {
401  // create layer set
402  QStringList layerSet, layerSetOverview;
403 
404  int i;
405  for ( i = 0; i < layers.size(); i++ )
406  {
407  QgsMapCanvasLayer &lyr = layers[i];
408  if ( !lyr.layer() )
409  {
410  continue;
411  }
412 
413  if ( lyr.isVisible() )
414  {
415  layerSet.push_back( lyr.layer()->id() );
416  }
417 
418  if ( lyr.isInOverview() )
419  {
420  layerSetOverview.push_back( lyr.layer()->id() );
421  }
422  }
423 
424  const QStringList& layerSetOld = mapSettings().layers();
425 
426  bool layerSetChanged = layerSetOld != layerSet;
427 
428  // update only if needed
429  if ( layerSetChanged )
430  {
431  QgsDebugMsg( "Layers changed to: " + layerSet.join( ", " ) );
432 
433  for ( i = 0; i < layerCount(); i++ )
434  {
435  // Add check if vector layer when disconnecting from selectionChanged slot
436  // Ticket #811 - racicot
437  QgsMapLayer *currentLayer = layer( i );
438  if ( !currentLayer )
439  continue;
440  disconnect( currentLayer, SIGNAL( repaintRequested() ), this, SLOT( refresh() ) );
441  disconnect( currentLayer, SIGNAL( layerCrsChanged() ), this, SLOT( layerCrsChange() ) );
442  QgsVectorLayer *isVectLyr = qobject_cast<QgsVectorLayer *>( currentLayer );
443  if ( isVectLyr )
444  {
445  disconnect( currentLayer, SIGNAL( selectionChanged() ), this, SLOT( selectionChangedSlot() ) );
446  }
447  }
448 
449  mSettings.setLayers( layerSet );
450 
451  for ( i = 0; i < layerCount(); i++ )
452  {
453  // Add check if vector layer when connecting to selectionChanged slot
454  // Ticket #811 - racicot
455  QgsMapLayer *currentLayer = layer( i );
456  connect( currentLayer, SIGNAL( repaintRequested() ), this, SLOT( refresh() ) );
457  connect( currentLayer, SIGNAL( layerCrsChanged() ), this, SLOT( layerCrsChange() ) );
458  QgsVectorLayer *isVectLyr = qobject_cast<QgsVectorLayer *>( currentLayer );
459  if ( isVectLyr )
460  {
461  connect( currentLayer, SIGNAL( selectionChanged() ), this, SLOT( selectionChangedSlot() ) );
462  }
463  }
464 
465  updateDatumTransformEntries();
466 
467  QgsDebugMsg( "Layers have changed, refreshing" );
468  emit layersChanged( layerSetOld );
469 
470  refresh();
471  }
472 
473  if ( mMapOverview )
474  {
475  const QStringList& layerSetOvOld = mMapOverview->layerSet();
476  if ( layerSetOvOld != layerSetOverview )
477  {
478  mMapOverview->setLayerSet( layerSetOverview );
479  }
480 
481  // refresh overview maplayers even if layer set is the same
482  // because full extent might have changed
483  updateOverview();
484  }

531 void QgsMapCanvas::setDestinationCrs( const QgsCoordinateReferenceSystem &crs )
532 {
533  if ( mSettings.destinationCrs() == crs )
534  return;
535 
536  if ( mSettings.hasCrsTransformEnabled() )
537  {
538  // try to reproject current extent to the new one
539  QgsRectangle rect;
540  if ( !mSettings.visibleExtent().isEmpty() )
541  {
542  QgsCoordinateTransform transform( mSettings.destinationCrs(), crs );
543  try
544  {
545  rect = transform.transformBoundingBox( mSettings.visibleExtent() );
546  }
547  catch ( QgsCsException &e )
548  {
549  QgsDebugMsg( QString( "Transform error caught: %1" ).arg( e.what() ) );
550  }
551  }
552  if ( !rect.isEmpty() )
553  {
554  setExtent( rect );
555  }
556 
557  QgsDebugMsg( "refreshing after destination CRS changed" );
558  refresh();
559  }
560 
561  mSettings.setDestinationCrs( crs );
562 
563  updateDatumTransformEntries();
564 
565  emit destinationCrsChanged();
566 }

677 void QgsMapCanvas::refreshMap()
678 {
679  Q_ASSERT( mRefreshScheduled );
680 
681  QgsDebugMsg( "CANVAS refresh!" );
682 
683  stopRendering(); // if any...
684  stopPreviewJobs();
685 
686  // from now on we can accept refresh requests again
687  mRefreshScheduled = false;
688 
689  //update $map variable to canvas
690  QgsExpression::setSpecialColumn( "$map", tr( "canvas" ) );
691 
692  // create the renderer job
693  Q_ASSERT( mJob == 0 );
694  mJobCancelled = false;
695  if ( mUseParallelRendering )
696  mJob = new QgsMapRendererParallelJob( mSettings );
697  else
698  mJob = new QgsMapRendererSequentialJob( mSettings );
699  connect( mJob, SIGNAL( finished() ), SLOT( rendererJobFinished() ) );
700  mJob->setCache( mCache );
701 
702  QStringList layersForGeometryCache;
703  foreach ( QString id, mSettings.layers() )
704  {
705  if ( QgsVectorLayer* vl = qobject_cast<QgsVectorLayer*>( QgsMapLayerRegistry::instance()->mapLayer( id ) ) )
706  {
707  if ( vl->isEditable() )
708  layersForGeometryCache << id;
709  }
710  }
711  mJob->setRequestedGeometryCacheForLayers( layersForGeometryCache );
712 
713  mJob->start();
714 
715  mMapUpdateTimer.start();
716 
717  emit renderStarting();
718 }

721 void QgsMapCanvas::rendererJobFinished()
722 {
723  QgsDebugMsg( QString( "CANVAS finish! %1" ).arg( !mJobCancelled ) );
724 
725  mMapUpdateTimer.stop();
726 
727  // TODO: would be better to show the errors in message bar
728  foreach ( const QgsMapRendererJob::Error& error, mJob->errors() )
729  {
730  QgsMessageLog::logMessage( error.layerID + " :: " + error.message, tr( "Rendering" ) );
731  }
732 
733  if ( !mJobCancelled )
734  {
735  // take labeling results before emitting renderComplete, so labeling map tools
736  // connected to signal work with correct results
737  delete mLabelingResults;
738  mLabelingResults = mJob->takeLabelingResults();
739 
740  QImage img = mJob->renderedImage();
741 
742  // emit renderComplete to get our decorations drawn
743  QPainter p( &img );
744  emit renderComplete( &p );
745 
746  QSettings settings;
747  if ( settings.value( "/Map/logCanvasRefreshEvent", false ).toBool() )
748  {
749  QString logMsg = tr( "Canvas refresh: %1 ms" ).arg( mJob->renderingTime() );
750  QgsMessageLog::logMessage( logMsg, tr( "Rendering" ) );
751  }
752 
753  if ( mDrawRenderingStats )
754  {
755  int w = img.width(), h = img.height();
756  QFont fnt = p.font();
757  fnt.setBold( true );
758  p.setFont( fnt );
759  int lh = p.fontMetrics().height() * 2;
760  QRect r( 0, h - lh, w, lh );
761  p.setPen( Qt::NoPen );
762  p.setBrush( QColor( 0, 0, 0, 110 ) );
763  p.drawRect( r );
764  p.setPen( Qt::white );
765  QString msg = QString( "%1 :: %2 ms" ).arg( mUseParallelRendering ? "PARALLEL" : "SEQUENTIAL" ).arg( mJob->renderingTime() );
766  p.drawText( r, msg, QTextOption( Qt::AlignCenter ) );
767  }
768

915 void QgsMapCanvas::setExtent( QgsRectangle const & r, bool adjustForOptimalScale )
916 {
917  QgsRectangle current = extent();
918 
919  if ( r == current )
920  return;
921 
922  // Don't allow too small extents
923  if ( !r.isEmpty() && mapSettings().computeScaleForExtent( r ) < 0.003 )
924  return;
925 
926  if ( r.isEmpty() )
927  {
928  if ( !mSettings.hasValidSettings() )
929  {
930  // we can't even just move the map center
931  QgsDebugMsg( "Empty extent - ignoring" );
932  return;
933  }
934

936  QgsDebugMsg( "Empty extent - keeping old scale with new center!" );
937  setCenter( r.center() );
938  }
939  else
940  {
941  mSettings.setExtent( r );

954  scaledRect.scale( scaleFactor );
955  mSettings.setExtent( scaledRect );
956  }
957  }

961  updateScale();
962  if ( mLastExtent.size() > 20 )

980  mLastExtentIndex = mLastExtent.size() - 1;
