191 // code adapted from
192 // http://en.wikipedia.org/wiki/Cohen%E2%80%93Sutherland_algorithm
193 struct _CohenSutherland
194 {
195  _CohenSutherland( const QgsRectangle& rect ) : mRect( rect ) {}
196 
197  typedef int OutCode;
198 
199  static const int INSIDE = 0; // 0000
200  static const int LEFT = 1; // 0001
201  static const int RIGHT = 2; // 0010
202  static const int BOTTOM = 4; // 0100
203  static const int TOP = 8; // 1000
204 
205  QgsRectangle mRect;
206 
207  OutCode computeOutCode( double x, double y )
208  {
209  OutCode code = INSIDE; // initialised as being inside of clip window
210  if ( x < mRect.xMinimum() ) // to the left of clip window
211  code |= LEFT;
212  else if ( x > mRect.xMaximum() ) // to the right of clip window
213  code |= RIGHT;
214  if ( y < mRect.yMinimum() ) // below the clip window
215  code |= BOTTOM;
216  else if ( y > mRect.yMaximum() ) // above the clip window
217  code |= TOP;
218  return code;
219  }
220 
221  bool isSegmentInRect( double x0, double y0, double x1, double y1 )
222  {
223  // compute outcodes for P0, P1, and whatever point lies outside the clip rectangle
224  OutCode outcode0 = computeOutCode( x0, y0 );
225  OutCode outcode1 = computeOutCode( x1, y1 );
226  bool accept = false;
227 
228  while ( true )
229  {
230  if ( !( outcode0 | outcode1 ) )
231  {
232  // Bitwise OR is 0. Trivially accept and get out of loop
233  accept = true;
234  break;
235  }
236  else if ( outcode0 & outcode1 )
237  {
238  // Bitwise AND is not 0. Trivially reject and get out of loop
239  break;
240  }
241  else
242  {
243  // failed both tests, so calculate the line segment to clip
244  // from an outside point to an intersection with clip edge
245  double x = 0, y = 0;
246 
247  // At least one endpoint is outside the clip rectangle; pick it.
248  OutCode outcodeOut = outcode0 ? outcode0 : outcode1;
249 
250  // Now find the intersection point;
251  // use formulas y = y0 + slope * (x - x0), x = x0 + (1 / slope) * (y - y0)
252  if ( outcodeOut & TOP )
253  { // point is above the clip rectangle
254  x = x0 + ( x1 - x0 ) * ( mRect.yMaximum() - y0 ) / ( y1 - y0 );
255  y = mRect.yMaximum();
256  }
257  else if ( outcodeOut & BOTTOM )
258  { // point is below the clip rectangle
259  x = x0 + ( x1 - x0 ) * ( mRect.yMinimum() - y0 ) / ( y1 - y0 );
260  y = mRect.yMinimum();
261  }
262  else if ( outcodeOut & RIGHT )
263  { // point is to the right of clip rectangle
264  y = y0 + ( y1 - y0 ) * ( mRect.xMaximum() - x0 ) / ( x1 - x0 );
265  x = mRect.xMaximum();
266  }
267  else if ( outcodeOut & LEFT )
268  { // point is to the left of clip rectangle
269  y = y0 + ( y1 - y0 ) * ( mRect.xMinimum() - x0 ) / ( x1 - x0 );
270  x = mRect.xMinimum();
271  }
272 
273  // Now we move outside point to intersection point to clip
274  // and get ready for next pass.
275  if ( outcodeOut == outcode0 )
276  {
277  x0 = x;
278  y0 = y;
279  outcode0 = computeOutCode( x0, y0 );
280  }
281  else
282  {
283  x1 = x;
284  y1 = y;
285  outcode1 = computeOutCode( x1, y1 );
286  }
287  }
288  }
289  return accept;
290  }
291 };

294 static QgsPointLocator::MatchList _geometrySegmentsInRect( QgsGeometry* geom, const QgsRectangle& rect, QgsVectorLayer* vl, QgsFeatureId fid )
295 {
296  // this code is stupidly based on QgsGeometry::closestSegmentWithContext
297  // we need iterator for segments...
298 
299  QgsPointLocator::MatchList lst;
300  unsigned char* wkb = const_cast<unsigned char*>( geom->asWkb() ); // we're not changing wkb, just need non-const for QgsWkbPtr
301  if ( !wkb )
302  return lst;
303 
304  _CohenSutherland cs( rect );
305 
306  QgsWkbPtr wkbPtr( wkb + 1 );
307  QGis::WkbType wkbType;
308  wkbPtr >> wkbType;
309 
310  bool hasZValue = false;
311  switch ( wkbType )
312  {
313  case QGis::WKBPoint25D:
314  case QGis::WKBPoint:
315  case QGis::WKBMultiPoint25D:
316  case QGis::WKBMultiPoint:
317  {
318  // Points have no lines
319  return lst;
320  }
321 
322  case QGis::WKBLineString25D:
323  hasZValue = true;
324  //intentional fall-through
325  case QGis::WKBLineString:
326  {
327  int nPoints;
328  wkbPtr >> nPoints;
329 
330  double prevx = 0.0, prevy = 0.0;
331  for ( int index = 0; index < nPoints; ++index )
332  {
333  double thisx, thisy;
334  wkbPtr >> thisx >> thisy;
335  if ( hasZValue )
336  wkbPtr += sizeof( double );
337 
338  if ( index > 0 )
339  {
340  if ( cs.isSegmentInRect( prevx, prevy, thisx, thisy ) )
341  {
342  QgsPoint edgePoints[2];
343  edgePoints[0].set( prevx, prevy );
344  edgePoints[1].set( thisx, thisy );
345  lst << QgsPointLocator::Match( QgsPointLocator::Edge, vl, fid, 0, QgsPoint(), index - 1, edgePoints );
346  }
347  }
348 
349  prevx = thisx;
350  prevy = thisy;
351  }
352  break;
353  }
354 
355  case QGis::WKBMultiLineString25D:
356  hasZValue = true;
357  //intentional fall-through
358  case QGis::WKBMultiLineString:
359  {
360  int nLines;
361  wkbPtr >> nLines;
362  for ( int linenr = 0, pointIndex = 0; linenr < nLines; ++linenr )
363  {
364  wkbPtr += 1 + sizeof( int );
365  int nPoints;
366  wkbPtr >> nPoints;
367 
368  double prevx = 0.0, prevy = 0.0;
369  for ( int pointnr = 0; pointnr < nPoints; ++pointnr )
370  {
371  double thisx, thisy;
372  wkbPtr >> thisx >> thisy;
373  if ( hasZValue )
374  wkbPtr += sizeof( double );
375 
376  if ( pointnr > 0 )
377  {
378  if ( cs.isSegmentInRect( prevx, prevy, thisx, thisy ) )
379  {
380  QgsPoint edgePoints[2];
381  edgePoints[0].set( prevx, prevy );
382  edgePoints[1].set( thisx, thisy );
383  lst << QgsPointLocator::Match( QgsPointLocator::Edge, vl, fid, 0, QgsPoint(), pointIndex - 1, edgePoints );
384  }
385  }
386 
387  prevx = thisx;
388  prevy = thisy;
389  ++pointIndex;
390  }
391  }
392  break;
393  }
394 
395  case QGis::WKBPolygon25D:
396  hasZValue = true;
397  //intentional fall-through
398  case QGis::WKBPolygon:
399  {
400  int nRings;
401  wkbPtr >> nRings;
402 
403  for ( int ringnr = 0, pointIndex = 0; ringnr < nRings; ++ringnr )//loop over rings
404  {
405  int nPoints;
406  wkbPtr >> nPoints;
407 
408  double prevx = 0.0, prevy = 0.0;
409  for ( int pointnr = 0; pointnr < nPoints; ++pointnr )//loop over points in a ring
410  {
411  double thisx, thisy;
412  wkbPtr >> thisx >> thisy;
413  if ( hasZValue )
414  wkbPtr += sizeof( double );
415 
416  if ( pointnr > 0 )
417  {
418  if ( cs.isSegmentInRect( prevx, prevy, thisx, thisy ) )
419  {
420  QgsPoint edgePoints[2];
421  edgePoints[0].set( prevx, prevy );
422  edgePoints[1].set( thisx, thisy );
423  lst << QgsPointLocator::Match( QgsPointLocator::Edge, vl, fid, 0, QgsPoint(), pointIndex - 1, edgePoints );
424  }
425  }
426 
427  prevx = thisx;
428  prevy = thisy;
429  ++pointIndex;
430  }
431  }
432  break;
433  }
434 
435  case QGis::WKBMultiPolygon25D:
436  hasZValue = true;
437  //intentional fall-through
438  case QGis::WKBMultiPolygon:
439  {
440  int nPolygons;
441  wkbPtr >> nPolygons;
442  for ( int polynr = 0, pointIndex = 0; polynr < nPolygons; ++polynr )
443  {
444  wkbPtr += 1 + sizeof( int );
445  int nRings;
446  wkbPtr >> nRings;
447  for ( int ringnr = 0; ringnr < nRings; ++ringnr )
448  {
449  int nPoints;
450  wkbPtr >> nPoints;
451 
452  double prevx = 0.0, prevy = 0.0;
453  for ( int pointnr = 0; pointnr < nPoints; ++pointnr )
454  {
455  double thisx, thisy;
456  wkbPtr >> thisx >> thisy;
457  if ( hasZValue )
458  wkbPtr += sizeof( double );
459 
460  if ( pointnr > 0 )
461  {
462  if ( cs.isSegmentInRect( prevx, prevy, thisx, thisy ) )
463  {
464  QgsPoint edgePoints[2];
465  edgePoints[0].set( prevx, prevy );
466  edgePoints[1].set( thisx, thisy );
467  lst << QgsPointLocator::Match( QgsPointLocator::Edge, vl, fid, 0, QgsPoint(), pointIndex - 1, edgePoints );
468  }
469  }
470 
471  prevx = thisx;
472  prevy = thisy;
473  ++pointIndex;
474  }
475  }
476  }
477  break;
478  }
479 
480  case QGis::WKBUnknown:
481  default:
482  return lst;
483  break;
484  } // switch (wkbType)
485 
486  return lst;

619 bool QgsPointLocator::rebuildIndex( int maxFeaturesToIndex )
620 {
621  destroyIndex();
622 
623  QLinkedList<RTree::Data*> dataList;
624  QgsFeature f;
625  QGis::GeometryType geomType = mLayer->geometryType();
626  if ( geomType == QGis::NoGeometry )
627  return true; // nothing to index
628 
629  QgsFeatureRequest request;
630  request.setSubsetOfAttributes( QgsAttributeList() );
631  if ( mExtent )
632  {
633  QgsRectangle rect = *mExtent;
634  if ( mTransform )
635  {
636  try
637  {
638  rect = mTransform->transformBoundingBox( rect, QgsCoordinateTransform::ReverseTransform );
639  }
640  catch ( const QgsException& e )
641  {
642  // See http://hub.qgis.org/issues/12634
643  QgsDebugMsg( QString( "could not transform bounding box to map, skipping the snap filter (%1)" ).arg( e.what() ) );
644  }
645  }
646  request.setFilterRect( rect );
647  }
648  QgsFeatureIterator fi = mLayer->getFeatures( request );
649  int indexedCount = 0;
650  while ( fi.nextFeature( f ) )
651  {
652  if ( !f.geometry() )
653  continue;
654 
655  if ( mTransform )
656  {
657  try
658  {
659  f.geometry()->transform( *mTransform );
660  }
661  catch ( const QgsException& e )
662  {
663  // See http://hub.qgis.org/issues/12634
664  QgsDebugMsg( QString( "could not transform geometry to map, skipping the snap for it (%1)" ).arg( e.what() ) );
665  continue;
666  }
667  }
668 
669  SpatialIndex::Region r( rect2region( f.geometry()->boundingBox() ) );
670  dataList << new RTree::Data( 0, 0, r, f.id() );
671  mGeoms[f.id()] = new QgsGeometry( *f.geometry() );
672  ++indexedCount;
673 
674  if ( maxFeaturesToIndex != -1 && indexedCount > maxFeaturesToIndex )
675  {
676  qDeleteAll( dataList );
677  destroyIndex();
678  return false;
679  }
680  }
681 
682  // R-Tree parameters
683  double fillFactor = 0.7;
684  unsigned long indexCapacity = 10;
685  unsigned long leafCapacity = 10;
686  unsigned long dimension = 2;
687  RTree::RTreeVariant variant = RTree::RV_RSTAR;
688  SpatialIndex::id_type indexId;
689 
690  if ( dataList.isEmpty() )
691  {
692  mIsEmptyLayer = true;
693  return true; // no features
694  }
695 
696  QgsPointLocator_Stream stream( dataList );
697  mRTree = RTree::createAndBulkLoadNewRTree( RTree::BLM_STR, stream, *mStorage, fillFactor, indexCapacity,
698  leafCapacity, dimension, variant, indexId );
699  return true;
700 }
