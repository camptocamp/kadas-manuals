43 void QgsMapRendererParallelJob::start()
44 {
45  if ( isActive() )
46  return;
47 
48  mRenderingStart.start();
49 
50  mStatus = RenderingLayers;
51 
52  delete mLabelingEngine;
53  mLabelingEngine = 0;
54 
55  if ( mSettings.testFlag( QgsMapSettings::DrawLabeling ) )
56  {
57  mLabelingEngine = new QgsPalLabeling;
58  mLabelingEngine->loadEngineSettings();
59  mLabelingEngine->init( mSettings );
60  }
61 
62 
63  mLayerJobs = prepareJobs( 0, mLabelingEngine );
64  // prepareJobs calls mapLayer->createMapRenderer may involve cloning a RasterDataProvider,
65  // whose constructor may need to download some data (i.e. WMS, AMS) and doing so runs a
66  // QEventLoop waiting for the network request to complete. If unluckily someone calls
67  // mapCanvas->refresh() while this is happening, QgsMapRendererCustomPainterJob::cancel is
68  // called, deleting the QgsMapRendererCustomPainterJob while this function is running.
69  // Hence we need to check whether the job is still active before proceeding
70  if ( !isActive() )
71  return;
72 
73  QgsDebugMsg( QString( "QThreadPool max thread count is %1" ).arg( QThreadPool::globalInstance()->maxThreadCount() ) );
74 
75  // start async job
76 
77  connect( &mFutureWatcher, SIGNAL( finished() ), SLOT( renderLayersFinished() ) );
78 
79  mFuture = QtConcurrent::map( mLayerJobs, renderLayerStatic );
80  mFutureWatcher.setFuture( mFuture );
81 }

83 void QgsMapRendererParallelJob::cancel()
84 {
85  if ( !isActive() )
86  return;
87 
88  QgsDebugMsg( QString( "PARALLEL cancel at status %1" ).arg( mStatus ) );
89 
90  mLabelingRenderContext.setRenderingStopped( true );
91  for ( LayerRenderJobs::iterator it = mLayerJobs.begin(); it != mLayerJobs.end(); ++it )
92  {
93  it->context.setRenderingStopped( true );
94  if ( it->renderer && it->renderer->feedback() )
95  {
96  it->renderer->feedback()->cancel();
97  }
98  }
99 
100  if ( mStatus == RenderingLayers )
101  {
102  disconnect( &mFutureWatcher, SIGNAL( finished() ), this, SLOT( renderLayersFinished() ) );
103 
104  mFutureWatcher.waitForFinished();
105 
106  renderLayersFinished();
107  }
108 
109  if ( mStatus == RenderingLabels )
110  {
111  disconnect( &mLabelingFutureWatcher, SIGNAL( finished() ), this, SLOT( renderingFinished() ) );
112 
113  mLabelingFutureWatcher.waitForFinished();
114 
115  renderingFinished();
116  }
117 
118  Q_ASSERT( mStatus == Idle );
119 }

121 void QgsMapRendererParallelJob::waitForFinished()
122 {
123  if ( !isActive() )
124  return;
125 
126  if ( mStatus == RenderingLayers )
127  {
128  disconnect( &mFutureWatcher, SIGNAL( finished() ), this, SLOT( renderLayersFinished() ) );
129 
130  QTime t;
131  t.start();
132 
133  mFutureWatcher.waitForFinished();
134 
135  QgsDebugMsg( QString( "waitForFinished (1): %1 ms" ).arg( t.elapsed() / 1000.0 ) );
136 
137  renderLayersFinished();
138  }
139 
140  if ( mStatus == RenderingLabels )
141  {
142  disconnect( &mLabelingFutureWatcher, SIGNAL( finished() ), this, SLOT( renderingFinished() ) );
143 
144  QTime t;
145  t.start();
146 
147  mLabelingFutureWatcher.waitForFinished();
148 
149  QgsDebugMsg( QString( "waitForFinished (2): %1 ms" ).arg( t.elapsed() / 1000.0 ) );
150 
151  renderingFinished();
152  }
153 
154  Q_ASSERT( mStatus == Idle );
155 }

175 void QgsMapRendererParallelJob::renderLayersFinished()
176 {
177  Q_ASSERT( mStatus == RenderingLayers );
178 
179  // compose final image
180  mFinalImage = composeImage( mSettings, mLayerJobs );
181 
182  cleanupJobs( mLayerJobs );
183 
184  QgsDebugMsg( "PARALLEL layers finished" );
185 
186  if ( mSettings.testFlag( QgsMapSettings::DrawLabeling ) && !mLabelingRenderContext.renderingStopped() )
187  {
188  mStatus = RenderingLabels;
189 
190  connect( &mLabelingFutureWatcher, SIGNAL( finished() ), this, SLOT( renderingFinished() ) );
191 
192  // now start rendering of labeling!
193  mLabelingFuture = QtConcurrent::run( renderLabelsStatic, this );
194  mLabelingFutureWatcher.setFuture( mLabelingFuture );
195  }
196  else
197  {
198  renderingFinished();
199  }
200 }

213 void QgsMapRendererParallelJob::renderLayerStatic( LayerRenderJob& job )
214 {
215  if ( job.context.renderingStopped() )
216  return;
217 
218  if ( job.cached )
219  return;
220 
221  QTime t;
222  t.start();
223  QgsDebugMsg( QString( "job %1 start" ).arg(( intptr_t ) &job, 0, 16 ) );
224 
225  try
226  {
227  job.renderer->render();
228  }
229  catch ( QgsException & e )
230  {
231  QgsDebugMsg( "Caught unhandled QgsException: " + e.what() );
232  }
233  catch ( std::exception & e )
234  {
235  QgsDebugMsg( "Caught unhandled std::exception: " + QString::fromAscii( e.what() ) );
236  }
237  catch ( ... )
238  {
239  QgsDebugMsg( "Caught unhandled unknown exception" );
240  }
241 
242  int tt = t.elapsed();
243  QgsDebugMsg( QString( "job %1 end [%2 ms]" ).arg(( intptr_t ) &job, 0, 16 ).arg( tt ) );
244  Q_UNUSED( tt );
245 }
