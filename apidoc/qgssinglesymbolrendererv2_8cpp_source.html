47 QgsSymbolV2* QgsSingleSymbolRendererV2::symbolForFeature( QgsFeature& feature )
48 {
49  if ( !mRotation.data() && !mSizeScale.data() ) return mSymbol.data();
50 
51  const double rotation = mRotation.data() ? mRotation->evaluate( feature ).toDouble() : 0;
52  const double sizeScale = mSizeScale.data() ? mSizeScale->evaluate( feature ).toDouble() : 1.;
53 
54  if ( mTempSymbol->type() == QgsSymbolV2::Marker )
55  {
56  QgsMarkerSymbolV2* markerSymbol = static_cast<QgsMarkerSymbolV2*>( mTempSymbol.data() );
57  if ( mRotation.data() ) markerSymbol->setAngle( rotation );
58  markerSymbol->setSize( sizeScale * mOrigSize );
59  markerSymbol->setScaleMethod( mScaleMethod );
60  }
61  else if ( mTempSymbol->type() == QgsSymbolV2::Line )
62  {
63  QgsLineSymbolV2* lineSymbol = static_cast<QgsLineSymbolV2*>( mTempSymbol.data() );
64  lineSymbol->setWidth( sizeScale * mOrigSize );
65  }
66  else if ( mTempSymbol->type() == QgsSymbolV2::Fill )
67  {
68  QgsFillSymbolV2* fillSymbol = static_cast<QgsFillSymbolV2*>( mTempSymbol.data() );
69  if ( mRotation.data() ) fillSymbol->setAngle( rotation );
70  }
71 
72  return mTempSymbol.data();
73 }

81 void QgsSingleSymbolRendererV2::startRender( QgsRenderContext& context, const QgsFields& fields )
82 {
83  if ( !mSymbol.data() ) return;
84 
85  mSymbol->startRender( context, &fields );
86 
87  if ( mRotation.data() || mSizeScale.data() )
88  {
89  // we are going to need a temporary symbol
90  mTempSymbol.reset( mSymbol->clone() );
91 
92  int hints = 0;
93  if ( mRotation.data() )
94  hints |= QgsSymbolV2::DataDefinedRotation;
95  if ( mSizeScale.data() )
96  hints |= QgsSymbolV2::DataDefinedSizeScale;
97  mTempSymbol->setRenderHints( hints );
98 
99  mTempSymbol->startRender( context, &fields );
100 
101  if ( mSymbol->type() == QgsSymbolV2::Marker )
102  {
103  mOrigSize = static_cast<QgsMarkerSymbolV2*>( mSymbol.data() )->size();
104  }
105  else if ( mSymbol->type() == QgsSymbolV2::Line )
106  {
107  mOrigSize = static_cast<QgsLineSymbolV2*>( mSymbol.data() )->width();
108  }
109  else

222 QgsFeatureRendererV2* QgsSingleSymbolRendererV2::create( QDomElement& element )
223 {
224  QDomElement symbolsElem = element.firstChildElement( "symbols" );
225  if ( symbolsElem.isNull() )
226  return NULL;
227 
228  QgsSymbolV2Map symbolMap = QgsSymbolLayerV2Utils::loadSymbols( symbolsElem );
229 
230  if ( !symbolMap.contains( "0" ) )
231  return NULL;
232 
233  QgsSingleSymbolRendererV2* r = new QgsSingleSymbolRendererV2( symbolMap.take( "0" ) );
234 
235  QDomElement legendSymbolElem = element.firstChildElement( "legendsymbol" );
236  if ( !legendSymbolElem.isNull() )
237  {
238  QgsSymbolV2Map legendSymbolMap = QgsSymbolLayerV2Utils::loadSymbols( legendSymbolElem );
239  if ( legendSymbolMap.contains( "0" ) )
240  {
241  r->setLegendSymbol( legendSymbolMap.take( "0" ) );
242  }
243  }
244 
245  r->setHtml( element.attribute( "html" ) );
246 
247  // delete symbols if there are any more
248  QgsSymbolLayerV2Utils::clearSymbolMap( symbolMap );
249 
250  QDomElement rotationElem = element.firstChildElement( "rotation" );
251  if ( !rotationElem.isNull() )
252  r->setRotationField( rotationElem.attribute( "field" ) );
253 
254  QDomElement sizeScaleElem = element.firstChildElement( "sizescale" );
255  if ( !sizeScaleElem.isNull() )
256  {
257  r->setSizeScaleField( sizeScaleElem.attribute( "field" ) );
258  r->setScaleMethod( QgsSymbolLayerV2Utils::decodeScaleMethod( sizeScaleElem.attribute( "scalemethod" ) ) );
259  }
260 
261  // TODO: symbol levels
262  return r;
263 }

265 QgsFeatureRendererV2* QgsSingleSymbolRendererV2::createFromSld( QDomElement& element, QGis::GeometryType geomType )
266 {
267  // XXX this renderer can handle only one Rule!
268 
269  // get the first Rule element
270  QDomElement ruleElem = element.firstChildElement( "Rule" );
271  if ( ruleElem.isNull() )
272  {
273  QgsDebugMsg( "no Rule elements found!" );
274  return NULL;
275  }
276 
277  QString label, description;
278  QgsSymbolLayerV2List layers;
279 
280  // retrieve the Rule element child nodes
281  QDomElement childElem = ruleElem.firstChildElement();
282  while ( !childElem.isNull() )
283  {
284  if ( childElem.localName() == "Name" )
285  {
286  // <se:Name> tag contains the rule identifier,
287  // so prefer title tag for the label property value
288  if ( label.isEmpty() )
289  label = childElem.firstChild().nodeValue();
290  }
291  else if ( childElem.localName() == "Description" )
292  {
293  // <se:Description> can contains a title and an abstract
294  QDomElement titleElem = childElem.firstChildElement( "Title" );
295  if ( !titleElem.isNull() )
296  {
297  label = titleElem.firstChild().nodeValue();
298  }
299 
300  QDomElement abstractElem = childElem.firstChildElement( "Abstract" );
301  if ( !abstractElem.isNull() )
302  {
303  description = abstractElem.firstChild().nodeValue();
304  }
305  }
306  else if ( childElem.localName() == "Abstract" )
307  {
308  // <sld:Abstract> (v1.0)
309  description = childElem.firstChild().nodeValue();
310  }
311  else if ( childElem.localName() == "Title" )
312  {
313  // <sld:Title> (v1.0)
314  label = childElem.firstChild().nodeValue();
315  }
316  else if ( childElem.localName().endsWith( "Symbolizer" ) )
317  {
318  // create symbol layers for this symbolizer
319  QgsSymbolLayerV2Utils::createSymbolLayerV2ListFromSld( childElem, geomType, layers );
320  }
321 
322  childElem = childElem.nextSiblingElement();
323  }
324 
325  if ( layers.size() == 0 )
326  return NULL;
327 
328  // now create the symbol
329  QgsSymbolV2 *symbol;
330  switch ( geomType )
331  {
332  case QGis::Line:
333  symbol = new QgsLineSymbolV2( layers );
334  break;
335 
336  case QGis::Polygon:
337  symbol = new QgsFillSymbolV2( layers );
338  break;
339 
340  case QGis::Point:
341  symbol = new QgsMarkerSymbolV2( layers );
342  break;
343 
344  default:
345  QgsDebugMsg( QString( "invalid geometry type: found %1" ).arg( geomType ) );
346  return NULL;
347  }
348 
349  // and finally return the new renderer
350  return new QgsSingleSymbolRendererV2( symbol );
351 }

363  rendererElem.appendChild( symbolsElem );

370  rendererElem.appendChild( legendSymbolElem );
371  }
372 
373  QDomElement rotationElem = doc.createElement( "rotation" );
374  if ( mRotation.data() )
375  rotationElem.setAttribute( "field", QgsSymbolLayerV2Utils::fieldOrExpressionFromExpression( mRotation.data() ) );
376  rendererElem.appendChild( rotationElem );
377 
378  QDomElement sizeScaleElem = doc.createElement( "sizescale" );
379  if ( mSizeScale.data() )
380  sizeScaleElem.setAttribute( "field", QgsSymbolLayerV2Utils::fieldOrExpressionFromExpression( mSizeScale.data() ) );
381  sizeScaleElem.setAttribute( "scalemethod", QgsSymbolLayerV2Utils::encodeScaleMethod( mScaleMethod ) );
382  rendererElem.appendChild( sizeScaleElem );

384 
385

422 QgsSingleSymbolRendererV2* QgsSingleSymbolRendererV2::convertFromRenderer( const QgsFeatureRendererV2 *renderer )
423 {
424  if ( renderer->type() == "singleSymbol" )
425  {
426  return dynamic_cast<QgsSingleSymbolRendererV2*>( renderer->clone() );
427  }
428  if ( renderer->type() == "pointDisplacement" )
429  {
430  const QgsPointDisplacementRenderer* pointDisplacementRenderer = dynamic_cast<const QgsPointDisplacementRenderer*>( renderer );
431  if ( pointDisplacementRenderer )
432  return convertFromRenderer( pointDisplacementRenderer->embeddedRenderer() );
433  }
434  if ( renderer->type() == "invertedPolygonRenderer" )
435  {
436  const QgsInvertedPolygonRenderer* invertedPolygonRenderer = dynamic_cast<const QgsInvertedPolygonRenderer*>( renderer );
437  if ( invertedPolygonRenderer )
438  return convertFromRenderer( invertedPolygonRenderer->embeddedRenderer() );
439  }
440 
441  QgsSymbolV2List symbols = const_cast<QgsFeatureRendererV2 *>( renderer )->symbols();
442  if ( symbols.size() > 0 )
443  {
444  return new QgsSingleSymbolRendererV2( symbols.at( 0 )->clone() );
445  }
446  return 0;
447 }
