25 static QString alphaLabel( int i )
26 {
27  QString label;
28  do
29  {
30  i -= 1;
31  int res = i % 26;
32  label.prepend( QChar( 'A' + res ) );
33  i /= 26;
34  }
35  while ( i > 0 );
36  return label;
37 }

39 class QgsGuideGridLayer::Renderer : public QgsMapLayerRenderer
40 {
41  public:
42  Renderer( QgsGuideGridLayer* layer, QgsRenderContext& rendererContext )
43  : QgsMapLayerRenderer( layer->id() )
44  , mLayer( layer )
45  , mRendererContext( rendererContext )
46  {}
47 
48  bool render() override
49  {
50  if ( mLayer->mRows == 0 || mLayer->mCols == 0 )
51  {
52  return true;
53  }
54 
55  static int labelBoxSize = mLayer->mFontSize + 5;
56  mRendererContext.painter()->save();
57  mRendererContext.painter()->setOpacity(( 100. - mLayer->mTransparency ) / 100. );
58  mRendererContext.painter()->setPen( QPen( mLayer->mColor, 1. ) );
59 
60  QFont font;
61  font.setPixelSize( mLayer->mFontSize );
62  mRendererContext.painter()->setFont( font );
63  const QgsCoordinateTransform* crst = mRendererContext.coordinateTransform();
64  const QgsMapToPixel& mapToPixel = mRendererContext.mapToPixel();
65  const QgsRectangle& gridRect = mLayer->mGridRect;
66  QgsPoint pTL = QgsPoint( mRendererContext.extent().xMinimum(), mRendererContext.extent().yMaximum() );
67  QgsPoint pBR = QgsPoint( mRendererContext.extent().xMaximum(), mRendererContext.extent().yMinimum() );
68  QPointF screenTL = mapToPixel.transform( crst ? crst->transform( pTL ) : pTL ).toQPointF();
69  QPointF screenBR = mapToPixel.transform( crst ? crst->transform( pBR ) : pBR ).toQPointF();
70  QRectF screenRect( screenTL, screenBR );
71 
72  // Draw vertical lines
73  double ix = gridRect.width() / mLayer->mCols;
74  QPair<QPointF, QPointF> vLine1 = screenLine( QgsPoint( gridRect.xMinimum(), gridRect.yMaximum() ), QgsPoint( gridRect.xMinimum(), gridRect.yMinimum() ) );
75  mRendererContext.painter()->drawLine( vLine1.first, vLine1.second );
76  double sy1 = qMax( vLine1.first.y(), screenRect.top() );
77  double sy2 = qMin( vLine1.second.y(), screenRect.bottom() );
78  for ( int col = 1; col <= mLayer->mCols; ++col )
79  {
80  double x2 = gridRect.xMinimum() + col * ix;
81  QPair<QPointF, QPointF> vLine2 = screenLine( QgsPoint( x2, gridRect.yMaximum() ), QgsPoint( x2, gridRect.yMinimum() ) );
82  mRendererContext.painter()->drawLine( vLine2.first, vLine2.second );
83 
84  double sx1 = vLine1.first.x();
85  double sx2 = vLine2.first.x();
86  QString label = mLayer->mLabelingMode == QgsGuideGridLayer::LABEL_A_1 ? alphaLabel( col ) : QString( "%1" ).arg( col );
87  if ( sy1 < vLine1.second.y() - 2 * labelBoxSize )
88  {
89  mRendererContext.painter()->drawText( QRectF( sx1, sy1, sx2 - sx1, labelBoxSize ), Qt::AlignHCenter | Qt::AlignVCenter, label );
90  }
91  if ( sy2 > vLine1.first.y() + 2 * labelBoxSize )
92  {
93  mRendererContext.painter()->drawText( QRectF( sx1, sy2 - labelBoxSize, sx2 - sx1, labelBoxSize ), Qt::AlignHCenter | Qt::AlignVCenter, label );
94  }
95 
96  vLine1 = vLine2;
97  }
98 
99  // Draw horizontal lines
100  double iy = gridRect.height() / mLayer->mRows;
101  QPair<QPointF, QPointF> hLine1 = screenLine( QgsPoint( gridRect.xMinimum(), gridRect.yMaximum() ), QgsPoint( gridRect.xMaximum(), gridRect.yMaximum() ) );
102  mRendererContext.painter()->drawLine( hLine1.first, hLine1.second );
103  double sx1 = qMax( hLine1.first.x(), screenRect.left() );
104  double sx2 = qMin( vLine1.second.x(), screenRect.right() );;
105  for ( int row = 1; row <= mLayer->mRows; ++row )
106  {
107  double y = gridRect.yMaximum() - row * iy;
108  QPair<QPointF, QPointF> hLine2 = screenLine( QgsPoint( gridRect.xMinimum(), y ), QgsPoint( gridRect.xMaximum(), y ) );
109  mRendererContext.painter()->drawLine( hLine2.first, hLine2.second );
110 
111  double sy1 = hLine1.first.y();
112  double sy2 = hLine2.first.y();
113  QString label = mLayer->mLabelingMode == QgsGuideGridLayer::LABEL_1_A ? alphaLabel( row ) : QString( "%1" ).arg( row );
114  if ( sx1 < vLine1.second.x() - 2 * labelBoxSize )
115  {
116  mRendererContext.painter()->drawText( QRectF( sx1, sy1, labelBoxSize, sy2 - sy1 ), Qt::AlignHCenter | Qt::AlignVCenter, label );
117  }
118  if ( sx2 > hLine1.first.x() + 2 * labelBoxSize )
119  {
120  mRendererContext.painter()->drawText( QRectF( sx2 - labelBoxSize, sy1, labelBoxSize, sy2 - sy1 ), Qt::AlignHCenter | Qt::AlignVCenter, label );
121  }
122 
123  hLine1 = hLine2;
124  }
125  mRendererContext.painter()->restore();
126  return true;
127  }

133  QPair<QPointF, QPointF> screenLine( const QgsPoint& p1, const QgsPoint& p2 ) const
134  {
135  const QgsCoordinateTransform* crst = mRendererContext.coordinateTransform();
136  const QgsMapToPixel& mapToPixel = mRendererContext.mapToPixel();
137  QPointF sp1 = mapToPixel.transform( crst ? crst->transform( p1 ) : p1 ).toQPointF();
138  QPointF sp2 = mapToPixel.transform( crst ? crst->transform( p2 ) : p2 ).toQPointF();
139  return qMakePair( sp1, sp2 );
140  }

182 QList<QgsGuideGridLayer::IdentifyResult> QgsGuideGridLayer::identify( const QgsPoint& mapPos, const QgsMapSettings& mapSettings )
183 {
184  const QgsCoordinateTransform* crst = QgsCoordinateTransformCache::instance()->transform( mapSettings.destinationCrs().authid(), crs().authid() );
185  QgsPoint pos = crst->transform( mapPos );
186 
187  double colWidth = ( mGridRect.xMaximum() - mGridRect.xMinimum() ) / mCols;
188  double rowHeight = ( mGridRect.yMaximum() - mGridRect.yMinimum() ) / mRows;
189 
190  int i = std::floor(( pos.x() - mGridRect.xMinimum() ) / colWidth );
191  int j = std::floor(( mGridRect.yMaximum() - pos.y() ) / rowHeight );
192 
193  QgsPolygonV2* bbox = new QgsPolygonV2();
194  QgsLineStringV2* ring = new QgsLineStringV2();
195  ring->setPoints(
196  QList<QgsPointV2>()
197  << QgsPointV2( mGridRect.xMinimum() + i * colWidth, mGridRect.yMaximum() - j * rowHeight )
198  << QgsPointV2( mGridRect.xMinimum() + ( i + 1 ) * colWidth, mGridRect.yMaximum() - j * rowHeight )
199  << QgsPointV2( mGridRect.xMinimum() + ( i + 1 ) * colWidth, mGridRect.yMaximum() - ( j + 1 ) * rowHeight )
200  << QgsPointV2( mGridRect.xMinimum() + i * colWidth, mGridRect.yMaximum() - ( j + 1 ) * rowHeight )
201  << QgsPointV2( mGridRect.xMinimum() + i * colWidth, mGridRect.yMaximum() - j * rowHeight )
202  );
203  bbox->setExteriorRing( ring );
204  QMap<QString, QVariant> attrs;
205 
206  if ( mLabelingMode == QgsGuideGridLayer::LABEL_1_A )
207  {
208  return QList<IdentifyResult>() << IdentifyResult( tr( "Cell %1, %2" ).arg( alphaLabel( 1 + j ) ).arg( 1 + i ), attrs, QgsGeometry( bbox ) );
209  }
210  else
211  {
212  return QList<IdentifyResult>() << IdentifyResult( tr( "Cell %1, %2" ).arg( 1 + j ).arg( alphaLabel( 1 + i ) ), attrs, QgsGeometry( bbox ) );
213  }
214 }
