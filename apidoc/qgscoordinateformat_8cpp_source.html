68 QString QgsCoordinateFormat::getDisplayString( const QgsPoint& p, const QgsCoordinateReferenceSystem& sSrs , Format format, const QString &epsg )
69 {
70  QgsPoint pTrans = QgsCoordinateTransformCache::instance()->transform( sSrs.authid(), epsg )->transform( p );
71  switch ( format )
72  {
73  case Default:
74  {
75  const QgsCoordinateReferenceSystem& crs = QgsCRSCache::instance()->crsByAuthId( epsg );
76  int prec = crs.mapUnits() == QGis::Degrees ? 4 : 0;
77  return QString( "%1, %2" ).arg( pTrans.x(), 0, 'f', prec ).arg( pTrans.y(), 0, 'f', prec );
78  }
79  case DegMinSec:
80  {
81  return pTrans.toDegreesMinutesSeconds( 1 );
82  }
83  case DegMin:
84  {
85  return pTrans.toDegreesMinutes( 3 );
86  }
87  case DecDeg:
88  {
89  return QString( "%1%2,%3%4" ).arg( pTrans.x(), 0, 'f', 5 ).arg( QChar( 176 ) )
90  .arg( pTrans.y(), 0, 'f', 5 ).arg( QChar( 176 ) );
91  }
92  case UTM:
93  {
94  QgsLatLonToUTM::UTMCoo coo = QgsLatLonToUTM::LL2UTM( pTrans );
95  return QString( "%1, %2 (zone %3%4)" ).arg( coo.easting ).arg( coo.northing ).arg( coo.zoneNumber ).arg( coo.zoneLetter );
96  }
97  case MGRS:
98  {
99  QgsLatLonToUTM::UTMCoo utm = QgsLatLonToUTM::LL2UTM( pTrans );
100  QgsLatLonToUTM::MGRSCoo mgrs = QgsLatLonToUTM::UTM2MGRS( utm );
101  if ( mgrs.letter100kID.isEmpty() )
102  return QString();
103 
104  return QString( "%1%2%3 %4 %5" ).arg( mgrs.zoneNumber ).arg( mgrs.zoneLetter ).arg( mgrs.letter100kID ).arg( mgrs.easting, 5, 10, QChar( '0' ) ).arg( mgrs.northing, 5, 10, QChar( '0' ) );
105  }
106  }
107  return "";
108 }

115 double QgsCoordinateFormat::getHeightAtPos( const QgsPoint& p, const QgsCoordinateReferenceSystem& crs, QGis::UnitType unit, QString* errMsg )
116 {
117  QString layerid = QgsProject::instance()->readEntry( "Heightmap", "layer" );
118  QgsMapLayer* layer = QgsMapLayerRegistry::instance()->mapLayer( layerid );
119  if ( !layer || layer->type() != QgsMapLayer::RasterLayer )
120  {
121  if ( errMsg )
122  *errMsg = tr( "No heightmap is defined in the project." ), tr( "Right-click a raster layer in the layer tree and select it to be used as heightmap." );
123  return 0;
124  }
125  QString rasterFile = layer->source();
126  GDALDatasetH raster = GDALOpen( rasterFile.toLocal8Bit().data(), GA_ReadOnly );
127  if ( !raster )
128  {
129  if ( errMsg )
130  *errMsg = tr( "Failed to open raster file: %1" ).arg( rasterFile );
131  return 0;
132  }
133 
134  double gtrans[6] = {};
135  if ( GDALGetGeoTransform( raster, &gtrans[0] ) != CE_None )
136  {
137  if ( errMsg )
138  *errMsg = tr( "Failed to get raster geotransform" );
139  GDALClose( raster );
140  return 0;
141  }
142 
143  QString proj( GDALGetProjectionRef( raster ) );
144  const QgsCoordinateReferenceSystem& rasterCrs = QgsCRSCache::instance()->crsByWkt( proj );
145  if ( !rasterCrs.isValid() )
146  {
147  if ( errMsg )
148  *errMsg = tr( "Failed to get raster CRS" );
149  GDALClose( raster );
150  return 0;
151  }
152 
153  GDALRasterBandH band = GDALGetRasterBand( raster, 1 );
154  if ( !raster )
155  {
156  if ( errMsg )
157  *errMsg = tr( "Failed to open raster band 0" );
158  GDALClose( raster );
159  return 0;
160  }
161 
162  // Get vertical unit
163  QGis::UnitType vertUnit = strcmp( GDALGetRasterUnitType( band ), "ft" ) == 0 ? QGis::Feet : QGis::Meters;
164 
165  // Transform geo position to raster CRS
166  QgsPoint pRaster = QgsCoordinateTransformCache::instance()->transform( crs.authid(), rasterCrs.authid() )->transform( p );
167  QgsDebugMsg( QString( "Transform %1 from %2 to %3 gives %4" ).arg( p.toString() )
168  .arg( crs.authid() ).arg( rasterCrs.authid() ).arg( pRaster.toString() ) );
169 
170  // Transform raster geo position to pixel coordinates
171  double row = ( -gtrans[0] * gtrans[4] + gtrans[1] * gtrans[3] - gtrans[1] * pRaster.y() + gtrans[4] * pRaster.x() ) / ( gtrans[2] * gtrans[4] - gtrans[1] * gtrans[5] );
172  double col = ( -gtrans[0] * gtrans[5] + gtrans[2] * gtrans[3] - gtrans[2] * pRaster.y() + gtrans[5] * pRaster.x() ) / ( gtrans[1] * gtrans[5] - gtrans[2] * gtrans[4] );
173 
174  double pixValues[4] = {};
175  if ( CE_None != GDALRasterIO( band, GF_Read,
176  qFloor( col ), qFloor( row ), 2, 2, &pixValues[0], 2, 2, GDT_Float64, 0, 0 ) )
177  {
178  if ( errMsg )
179  *errMsg = tr( "Failed to read pixel values" );
180  GDALClose( raster );
181  return 0;
182  }
183 
184  GDALClose( raster );
185 
186  // Interpolate values
187  double lambdaR = row - qFloor( row );
188  double lambdaC = col - qFloor( col );
189 
190  double value = ( pixValues[0] * ( 1. - lambdaC ) + pixValues[1] * lambdaC ) * ( 1. - lambdaR )
191  + ( pixValues[2] * ( 1. - lambdaC ) + pixValues[3] * lambdaC ) * ( lambdaR );
192  if ( rasterCrs.mapUnits() != unit )
193  {
194  value *= QGis::fromUnitToUnitFactor( vertUnit, unit );
195  }
196  return value;
197 }
