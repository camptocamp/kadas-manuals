777 void QgsComposerMapGrid::calculateCRSTransformLines()
778 {
779  QgsRectangle crsBoundingRect;
780  QgsCoordinateTransform inverseTr;
781  if ( crsGridParams( crsBoundingRect, inverseTr ) != 0 )
782  {
783  return;
784  }
785 
786  //calculate x grid lines
787  mTransformedXLines.clear();
788  xGridLinesCRSTransform( crsBoundingRect, inverseTr, mTransformedXLines );
789 
790  //calculate y grid lines
791  mTransformedYLines.clear();
792  yGridLinesCRSTransform( crsBoundingRect, inverseTr, mTransformedYLines );
793 
794  if ( mGridStyle == QgsComposerMapGrid::Cross || mGridStyle == QgsComposerMapGrid::Markers )
795  {
796  //cross or markers style - we also need to calculate intersections of lines
797 
798  //first convert lines to QgsGeometry
799  QList< QgsGeometry* > yLines;
800  QList< QPair< double, QPolygonF > >::const_iterator yGridIt = mTransformedYLines.constBegin();
801  for ( ; yGridIt != mTransformedYLines.constEnd(); ++yGridIt )
802  {
803  QgsPolyline yLine;
804  for ( int i = 0; i < ( *yGridIt ).second.size(); ++i )
805  {
806  yLine.append( QgsPoint(( *yGridIt ).second.at( i ).x(), ( *yGridIt ).second.at( i ).y() ) );
807  }
808  yLines << QgsGeometry::fromPolyline( yLine );
809  }
810  QList< QgsGeometry* > xLines;
811  QList< QPair< double, QPolygonF > >::const_iterator xGridIt = mTransformedXLines.constBegin();
812  for ( ; xGridIt != mTransformedXLines.constEnd(); ++xGridIt )
813  {
814  QgsPolyline xLine;
815  for ( int i = 0; i < ( *xGridIt ).second.size(); ++i )
816  {
817  xLine.append( QgsPoint(( *xGridIt ).second.at( i ).x(), ( *xGridIt ).second.at( i ).y() ) );
818  }
819  xLines << QgsGeometry::fromPolyline( xLine );
820  }
821 
822  //now, loop through geometries and calculate intersection points
823  mTransformedIntersections.clear();
824  QList< QgsGeometry* >::const_iterator yLineIt = yLines.constBegin();
825  for ( ; yLineIt != yLines.constEnd(); ++yLineIt )
826  {
827  QList< QgsGeometry* >::const_iterator xLineIt = xLines.constBegin();
828  for ( ; xLineIt != xLines.constEnd(); ++xLineIt )
829  {
830  //look for intersections between lines
831  QgsGeometry* intersects = ( *yLineIt )->intersection(( *xLineIt ) );
832 
833  //go through all intersections and draw grid markers/crosses
834  int i = 0;
835  QgsPoint vertex = intersects->vertexAt( i );
836  while ( vertex != QgsPoint( 0, 0 ) )
837  {
838  mTransformedIntersections << vertex;
839  i = i + 1;
840  vertex = intersects->vertexAt( i );
841  }
842  }
843  }
844  //clean up
845  qDeleteAll( yLines );
846  yLines.clear();
847  qDeleteAll( xLines );
848  xLines.clear();
849  }
850 
851  mTransformDirty = false;
852 }

931 void QgsComposerMapGrid::drawGridNoTransform( QgsRenderContext &context, double dotsPerMM, QList< QPair< double, QLineF > > &horizontalLines,
932  QList< QPair< double, QLineF > > &verticalLines ) const
933 {
934  //get line positions
935  yGridLines( verticalLines );
936  QList< QPair< double, QLineF > >::const_iterator vIt = verticalLines.constBegin();
937  xGridLines( horizontalLines );
938  QList< QPair< double, QLineF > >::const_iterator hIt = horizontalLines.constBegin();
939 
940  //simple approach: draw vertical lines first, then horizontal ones
941  if ( mGridStyle == QgsComposerMapGrid::Solid )
942  {
943  //we need to scale line coordinates to dots, rather than mm, since the painter has already been scaled to dots
944  //this is done by multiplying each line coordinate by dotsPerMM
945  QLineF line;
946  for ( ; vIt != verticalLines.constEnd(); ++vIt )
947  {
948  line = QLineF( vIt->second.p1() * dotsPerMM, vIt->second.p2() * dotsPerMM );
949  drawGridLine( line, context );
950  }
951 
952  for ( ; hIt != horizontalLines.constEnd(); ++hIt )
953  {
954  line = QLineF( hIt->second.p1() * dotsPerMM, hIt->second.p2() * dotsPerMM );
955  drawGridLine( line, context );
956  }
957  }
958  else if ( mGridStyle != QgsComposerMapGrid::FrameAnnotationsOnly ) //cross or markers
959  {
960  QPointF intersectionPoint, crossEnd1, crossEnd2;
961  for ( ; vIt != verticalLines.constEnd(); ++vIt )
962  {
963  //test for intersection with every horizontal line
964  hIt = horizontalLines.constBegin();
965  for ( ; hIt != horizontalLines.constEnd(); ++hIt )
966  {
967  if ( hIt->second.intersect( vIt->second, &intersectionPoint ) == QLineF::BoundedIntersection )
968  {
969  if ( mGridStyle == QgsComposerMapGrid::Cross )
970  {
971  //apply a threshold to avoid calculate point if the two points are very close together (can lead to artifacts)
972  crossEnd1 = (( intersectionPoint - vIt->second.p1() ).manhattanLength() > 0.01 ) ?
973  QgsSymbolLayerV2Utils::pointOnLineWithDistance( intersectionPoint, vIt->second.p1(), mCrossLength ) : intersectionPoint;
974  crossEnd2 = (( intersectionPoint - vIt->second.p2() ).manhattanLength() > 0.01 ) ?
975  QgsSymbolLayerV2Utils::pointOnLineWithDistance( intersectionPoint, vIt->second.p2(), mCrossLength ) : intersectionPoint;
976  //draw line using coordinates scaled to dots
977  drawGridLine( QLineF( crossEnd1 * dotsPerMM, crossEnd2 * dotsPerMM ), context );
978  }
979  else if ( mGridStyle == QgsComposerMapGrid::Markers )
980  {
981  drawGridMarker( intersectionPoint * dotsPerMM, context );
982  }
983  }
984  }
985  }
986  if ( mGridStyle == QgsComposerMapGrid::Markers )
987  {
988  //markers mode, so we have no need to process horizontal lines (we've already
989  //drawn markers on the intersections between horizontal and vertical lines)
990  return;
991  }
992 
993  hIt = horizontalLines.constBegin();
994  for ( ; hIt != horizontalLines.constEnd(); ++hIt )
995  {
996  vIt = verticalLines.constBegin();
997  for ( ; vIt != verticalLines.constEnd(); ++vIt )
998  {
999  if ( vIt->second.intersect( hIt->second, &intersectionPoint ) == QLineF::BoundedIntersection )
1000  {
1001  //apply a threshold to avoid calculate point if the two points are very close together (can lead to artifacts)
1002  crossEnd1 = (( intersectionPoint - hIt->second.p1() ).manhattanLength() > 0.01 ) ?
1003  QgsSymbolLayerV2Utils::pointOnLineWithDistance( intersectionPoint, hIt->second.p1(), mCrossLength ) : intersectionPoint;
1004  crossEnd2 = (( intersectionPoint - hIt->second.p2() ).manhattanLength() > 0.01 ) ?
1005  QgsSymbolLayerV2Utils::pointOnLineWithDistance( intersectionPoint, hIt->second.p2(), mCrossLength ) : intersectionPoint;
1006  //draw line using coordinates scaled to dots
1007  drawGridLine( QLineF( crossEnd1 * dotsPerMM, crossEnd2 * dotsPerMM ), context );
