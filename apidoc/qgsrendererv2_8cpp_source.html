40 const unsigned char* QgsFeatureRendererV2::_getPoint( QPointF& pt, QgsRenderContext& context, const unsigned char* wkb )
41 {
42  QgsConstWkbPtr wkbPtr( wkb + 1 );
43  unsigned int wkbType;
44  wkbPtr >> wkbType;
45 
46  double x, y;
47  wkbPtr >> x >> y;
48 
49  if ( QgsWKBTypes::hasZ(( QgsWKBTypes::Type )wkbType ) )
50  {
51  wkbPtr += sizeof( double );
52  }
53  if ( QgsWKBTypes::hasM(( QgsWKBTypes::Type )wkbType ) )
54  {
55  wkbPtr += sizeof( double );
56  }
57 
58  if ( context.coordinateTransform() )
59  {
60  double z = 0; // dummy variable for coordiante transform
61  context.coordinateTransform()->transformInPlace( x, y, z );
62  }
63 
64  context.mapToPixel().transformInPlace( x, y );
65 
66  pt = QPointF( x, y );
67  return wkbPtr;
68 }

70 const unsigned char* QgsFeatureRendererV2::_getLineString( QPolygonF& pts, QgsRenderContext& context, const unsigned char* wkb )
71 {
72  QgsConstWkbPtr wkbPtr( wkb + 1 );
73  unsigned int wkbType, nPoints;
74  wkbPtr >> wkbType >> nPoints;
75 
76  bool hasZValue = QgsWKBTypes::hasZ(( QgsWKBTypes::Type )wkbType );
77  bool hasMValue = QgsWKBTypes::hasM(( QgsWKBTypes::Type )wkbType );
78 
79  double x, y;
80  const QgsCoordinateTransform* ct = context.coordinateTransform();
81  const QgsMapToPixel& mtp = context.mapToPixel();
82 
83  //apply clipping for large lines to achieve a better rendering performance
84  if ( nPoints > 1 && !context.renderMapTile() )
85  {
86  const QgsRectangle& e = context.extent();
87  double cw = e.width() / 10; double ch = e.height() / 10;
88  QgsRectangle clipRect( e.xMinimum() - cw, e.yMinimum() - ch, e.xMaximum() + cw, e.yMaximum() + ch );
89  wkbPtr = QgsConstWkbPtr( QgsClipper::clippedLineWKB( wkb, clipRect, pts ) );
90  }
91  else
92  {
93  pts.resize( nPoints );
94 
95  QPointF* ptr = pts.data();
96  for ( unsigned int i = 0; i < nPoints; ++i, ++ptr )
97  {
98  wkbPtr >> x >> y;
99  if ( hasZValue )
100  wkbPtr += sizeof( double );
101  if ( hasMValue )
102  wkbPtr += sizeof( double );
103 
104  *ptr = QPointF( x, y );
105  }
106  }
107 
108  //transform the QPolygonF to screen coordinates
109  if ( ct )
110  {
111  ct->transformPolygon( pts );
112  }
113 
114  QPointF* ptr = pts.data();
115  for ( int i = 0; i < pts.size(); ++i, ++ptr )
116  {
117  mtp.transformInPlace( ptr->rx(), ptr->ry() );
118  }
119 
120  return wkbPtr;
121 }

123 const unsigned char* QgsFeatureRendererV2::_getPolygon( QPolygonF& pts, QList<QPolygonF>& holes, QgsRenderContext& context, const unsigned char* wkb )
124 {
125  QgsConstWkbPtr wkbPtr( wkb + 1 );
126 
127  unsigned int wkbType, numRings;
128  wkbPtr >> wkbType >> numRings;
129 
130  if ( numRings == 0 ) // sanity check for zero rings in polygon
131  return wkbPtr;
132 
133  bool hasZValue = QgsWKBTypes::hasZ(( QgsWKBTypes::Type )wkbType );
134  bool hasMValue = QgsWKBTypes::hasM(( QgsWKBTypes::Type )wkbType );
135 
136  double x, y;
137  holes.clear();
138 
139  const QgsCoordinateTransform* ct = context.coordinateTransform();
140  const QgsMapToPixel& mtp = context.mapToPixel();
141  const QgsRectangle& e = context.extent();
142  double cw = e.width() / 10; double ch = e.height() / 10;
143  QgsRectangle clipRect( e.xMinimum() - cw, e.yMinimum() - ch, e.xMaximum() + cw, e.yMaximum() + ch );
144 
145  for ( unsigned int idx = 0; idx < numRings; idx++ )
146  {
147  unsigned int nPoints;
148  wkbPtr >> nPoints;
149 
150  QPolygonF poly( nPoints );
151 
152  // Extract the points from the WKB and store in a pair of vectors.
153  QPointF* ptr = poly.data();
154  for ( unsigned int jdx = 0; jdx < nPoints; ++jdx, ++ptr )
155  {
156  wkbPtr >> x >> y;
157  if ( hasZValue )
158  wkbPtr += sizeof( double );
159  if ( hasMValue )
160  wkbPtr += sizeof( double );
161 
162  *ptr = QPointF( x, y );
163  }
164 
165  if ( nPoints < 1 )
166  continue;
167 
168  //clip close to view extent, if needed
169  QRectF ptsRect = poly.boundingRect();
170  if ( !context.extent().contains( ptsRect ) && !context.renderMapTile() ) QgsClipper::trimPolygon( poly, clipRect );
171 
172  //transform the QPolygonF to screen coordinates
173  if ( ct )
174  {
175  ct->transformPolygon( poly );
176  }
177 
178 
179  ptr = poly.data();
180  for ( int i = 0; i < poly.size(); ++i, ++ptr )
181  {
182  mtp.transformInPlace( ptr->rx(), ptr->ry() );
183  }
184 
185  if ( idx == 0 )
186  pts = poly;
187  else
188  holes.append( poly );
189  }
190 
191  return wkbPtr;
192 }

239 void QgsFeatureRendererV2::renderFeatureWithSymbol( QgsFeature& feature, QgsSymbolV2* symbol, QgsRenderContext& context, int layer, bool selected, bool drawVertexMarker )
240 {
241  QgsSymbolV2::SymbolType symbolType = symbol->type();
242 
243  const QgsGeometry* geom = feature.constGeometry();
244  if ( !geom || !geom->geometry() )
245  {
246  return;
247  }
248 
249  const QgsGeometry* segmentizedGeometry = geom;
250  bool deleteSegmentizedGeometry = false;
251  context.setGeometry( geom->geometry() );
252 
253  //convert curve types to normal point/line/polygon ones
254  switch ( QgsWKBTypes::flatType( geom->geometry()->wkbType() ) )
255  {
256  case QgsWKBTypes::CurvePolygon:
257  case QgsWKBTypes::CircularString:
258  case QgsWKBTypes::CompoundCurve:
259  case QgsWKBTypes::MultiSurface:
260  case QgsWKBTypes::MultiCurve:
261  {
262  QgsAbstractGeometryV2* g = geom->geometry()->segmentize();
263  if ( !g )
264  {
265  return;
266  }
267  segmentizedGeometry = new QgsGeometry( g );
268  deleteSegmentizedGeometry = true;
269  }
270  default:
271  break;
272  }
273 
274  switch ( QgsWKBTypes::flatType( segmentizedGeometry->geometry()->wkbType() ) )
275  {
276  case QgsWKBTypes::Point:
277  {
278  if ( symbolType != QgsSymbolV2::Marker )
279  {
280  QgsDebugMsg( "point can be drawn only with marker symbol!" );
281  break;
282  }
283  QPointF pt;
284  _getPoint( pt, context, segmentizedGeometry->asWkb() );
285  (( QgsMarkerSymbolV2* )symbol )->renderPoint( pt, &feature, context, layer, selected );
286  }
287  break;
288  case QgsWKBTypes::LineString:
289  {
290  if ( symbolType != QgsSymbolV2::Line )
291  {
292  QgsDebugMsg( "linestring can be drawn only with line symbol!" );
293  break;
294  }
295  QPolygonF pts;
296  _getLineString( pts, context, segmentizedGeometry->asWkb() );
297  (( QgsLineSymbolV2* )symbol )->renderPolyline( pts, &feature, context, layer, selected );
298  }
299  break;
300  case QgsWKBTypes::Polygon:
301  {
302  if ( symbolType != QgsSymbolV2::Fill )
303  {
304  QgsDebugMsg( "polygon can be drawn only with fill symbol!" );
305  break;
306  }
307  QPolygonF pts;
308  QList<QPolygonF> holes;
309  _getPolygon( pts, holes, context, segmentizedGeometry->asWkb() );
310  (( QgsFillSymbolV2* )symbol )->renderPolygon( pts, ( holes.count() ? &holes : NULL ), &feature, context, layer, selected );
311  }
312  break;
313 
314  case QgsWKBTypes::MultiPoint:
315  {
316  if ( symbolType != QgsSymbolV2::Marker )
317  {
318  QgsDebugMsg( "multi-point can be drawn only with marker symbol!" );
319  break;
320  }
321 
322  QgsConstWkbPtr wkbPtr( segmentizedGeometry->asWkb() + 1 + sizeof( int ) );
323  unsigned int num;
324  wkbPtr >> num;
325  const unsigned char* ptr = wkbPtr;
326  QPointF pt;
327 
328  for ( unsigned int i = 0; i < num; ++i )
329  {
330  ptr = QgsConstWkbPtr( _getPoint( pt, context, ptr ) );
331  (( QgsMarkerSymbolV2* )symbol )->renderPoint( pt, &feature, context, layer, selected );
332  }
333  }
334  break;
335 
336  case QgsWKBTypes::MultiCurve:
337  case QgsWKBTypes::MultiLineString:
338  {
339  if ( symbolType != QgsSymbolV2::Line )
340  {
341  QgsDebugMsg( "multi-linestring can be drawn only with line symbol!" );
342  break;
343  }
344 
345  QgsConstWkbPtr wkbPtr( segmentizedGeometry->asWkb() + 1 + sizeof( int ) );
346  unsigned int num;
347  wkbPtr >> num;
348  const unsigned char* ptr = wkbPtr;
349  QPolygonF pts;
350 
351  const QgsGeometryCollectionV2* geomCollection = dynamic_cast<const QgsGeometryCollectionV2*>( geom->geometry() );
352 
353  for ( unsigned int i = 0; i < num; ++i )
354  {
355  if ( geomCollection )
356  {
357  context.setGeometry( geomCollection->geometryN( i ) );
358  }
359  ptr = QgsConstWkbPtr( _getLineString( pts, context, ptr ) );
360  (( QgsLineSymbolV2* )symbol )->renderPolyline( pts, &feature, context, layer, selected );
361  }
362  }
363  break;
364 
365  case QgsWKBTypes::MultiSurface:
366  case QgsWKBTypes::MultiPolygon:
367  {
368  if ( symbolType != QgsSymbolV2::Fill )
369  {
370  QgsDebugMsg( "multi-polygon can be drawn only with fill symbol!" );
371  break;
372  }
373 
374  QgsConstWkbPtr wkbPtr( segmentizedGeometry->asWkb() + 1 + sizeof( int ) );
375  unsigned int num;
376  wkbPtr >> num;
377  const unsigned char* ptr = wkbPtr;
378  QPolygonF pts;
379  QList<QPolygonF> holes;
380 
381  const QgsGeometryCollectionV2* geomCollection = dynamic_cast<const QgsGeometryCollectionV2*>( geom->geometry() );
382 
383  for ( unsigned int i = 0; i < num; ++i )
384  {
385  if ( geomCollection )
386  {
387  context.setGeometry( geomCollection->geometryN( i ) );
388  }
389  ptr = _getPolygon( pts, holes, context, ptr );
390  (( QgsFillSymbolV2* )symbol )->renderPolygon( pts, ( holes.count() ? &holes : NULL ), &feature, context, layer, selected );
391  }
392  break;
393  }
394  default:
395  QgsDebugMsg( QString( "feature %1: unsupported wkb type 0x%2 for rendering" ).arg( feature.id() ).arg( geom->wkbType(), 0, 16 ) );
396  }
397 
398  if ( drawVertexMarker )
399  {
400  const QgsCoordinateTransform* ct = context.coordinateTransform();
401  const QgsMapToPixel& mtp = context.mapToPixel();
402 
403  QgsPointV2 vertexPoint;
404  QgsVertexId vertexId;
405  double x, y, z;
406  QPointF mapPoint;
407  while ( geom->geometry()->nextVertex( vertexId, vertexPoint ) )
408  {
409  //transform
410  x = vertexPoint.x(); y = vertexPoint.y(); z = vertexPoint.z();
411  if ( ct )
412  {
413  ct->transformInPlace( x, y, z );
414  }
415  mapPoint.setX( x ); mapPoint.setY( y );
416  mtp.transformInPlace( mapPoint.rx(), mapPoint.ry() );
417  renderVertexMarker( mapPoint, context );
418  }
419  }
420 
421  if ( deleteSegmentizedGeometry )
422  {
423  delete segmentizedGeometry;
424  }
425 }

461 QgsFeatureRendererV2* QgsFeatureRendererV2::loadSld( const QDomNode &node, QGis::GeometryType geomType, QString &errorMessage )
462 {
463  QDomElement element = node.toElement();
464  if ( element.isNull() )
465  return NULL;
466 
467  // get the UserStyle element
468  QDomElement userStyleElem = element.firstChildElement( "UserStyle" );
469  if ( userStyleElem.isNull() )
470  {
471  // UserStyle element not found, nothing will be rendered
472  errorMessage = "Info: UserStyle element not found.";
473  return NULL;
474  }
475 
476  // get the FeatureTypeStyle element
477  QDomElement featTypeStyleElem = userStyleElem.firstChildElement( "FeatureTypeStyle" );
478  if ( featTypeStyleElem.isNull() )
479  {
480  errorMessage = "Info: FeatureTypeStyle element not found.";
481  return NULL;
482  }
483 
484  // use the RuleRenderer when more rules are present or the rule
485  // has filters or min/max scale denominators set,
486  // otherwise use the SingleSymbol renderer
487  bool needRuleRenderer = false;
488  int ruleCount = 0;
489 
490  QDomElement ruleElem = featTypeStyleElem.firstChildElement( "Rule" );
491  while ( !ruleElem.isNull() )
492  {
493  ruleCount++;
494 
495  // more rules present, use the RuleRenderer
496  if ( ruleCount > 1 )
497  {
498  QgsDebugMsg( "more Rule elements found: need a RuleRenderer" );
499  needRuleRenderer = true;
500  break;
501  }
502 
503  QDomElement ruleChildElem = ruleElem.firstChildElement();
504  while ( !ruleChildElem.isNull() )
505  {
506  // rule has filter or min/max scale denominator, use the RuleRenderer
507  if ( ruleChildElem.localName() == "Filter" ||
508  ruleChildElem.localName() == "MinScaleDenominator" ||
509  ruleChildElem.localName() == "MaxScaleDenominator" )
510  {
511  QgsDebugMsg( "Filter or Min/MaxScaleDenominator element found: need a RuleRenderer" );
512  needRuleRenderer = true;
513  break;
514  }
515 
516  ruleChildElem = ruleChildElem.nextSiblingElement();
517  }
518 
519  if ( needRuleRenderer )
520  {
521  break;
522  }
523 
524  ruleElem = ruleElem.nextSiblingElement( "Rule" );
525  }
526 
527  QString rendererType;
528  if ( needRuleRenderer )
529  {
530  rendererType = "RuleRenderer";
531  }
532  else
533  {
534  rendererType = "singleSymbol";
535  }
536  QgsDebugMsg( QString( "Instantiating a '%1' renderer..." ).arg( rendererType ) );
537 
538  // create the renderer and return it
539  QgsRendererV2AbstractMetadata* m = QgsRendererV2Registry::instance()->rendererMetadata( rendererType );
540  if ( m == NULL )
541  {
542  errorMessage = QString( "Error: Unable to get metadata for '%1' renderer." ).arg( rendererType );
543  return NULL;
544  }
545 
546  QgsFeatureRendererV2* r = m->createRendererFromSld( featTypeStyleElem, geomType );
547  return r;
548 }
