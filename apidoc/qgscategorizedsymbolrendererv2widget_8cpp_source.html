163 bool QgsCategorizedSymbolRendererV2Model::setData( const QModelIndex & index, const QVariant & value, int role )
164 {
165  if ( !index.isValid() )
166  return false;
167 
168  if ( index.column() == 0 && role == Qt::CheckStateRole )
169  {
170  mRenderer->updateCategoryRenderState( index.row(), value == Qt::Checked );
171  emit dataChanged( index, index );
172  return true;
173  }
174 
175  if ( role != Qt::EditRole )
176  return false;
177 
178  switch ( index.column() )
179  {
180  case 1: // value
181  {
182  // try to preserve variant type for this value
183  QVariant val;
184  switch ( mRenderer->categories().value( index.row() ).value().type() )
185  {
186  case QVariant::Int:
187  val = value.toInt();
188  break;
189  case QVariant::Double:
190  val = value.toDouble();
191  break;
192  default:
193  val = value.toString();
194  break;
195  }
196  mRenderer->updateCategoryValue( index.row(), val );
197  break;
198  }
199  case 2: // label
200  mRenderer->updateCategoryLabel( index.row(), value.toString() );
201  break;
202  case 3: //html
203  mRenderer->updateCategoryHtml( index.row(), value.toString() );
204  break;
205  default:
206  return false;
207  }
208 
209  emit dataChanged( index, index );
210  return true;

279 bool QgsCategorizedSymbolRendererV2Model::dropMimeData( const QMimeData *data, Qt::DropAction action, int row, int column, const QModelIndex &parent )
280 {
281  Q_UNUSED( row );
282  Q_UNUSED( column );
283  if ( action != Qt::MoveAction ) return true;
284 
285  if ( !data->hasFormat( mMimeFormat ) ) return false;
286 
287  QByteArray encodedData = data->data( mMimeFormat );
288  QDataStream stream( &encodedData, QIODevice::ReadOnly );
289 
290  QVector<int> rows;
291  while ( !stream.atEnd() )
292  {
293  int r;
294  stream >> r;
295  rows.append( r );
296  }
297 
298  int to = parent.row();
299  // to is -1 if dragged outside items, i.e. below any item,
300  // then move to the last position
301  if ( to == -1 ) to = mRenderer->categories().size(); // out of rang ok, will be decreased
302  for ( int i = rows.size() - 1; i >= 0; i-- )
303  {
304  QgsDebugMsg( QString( "move %1 to %2" ).arg( rows[i] ).arg( to ) );
305  int t = to;
306  // moveCategory first removes and then inserts
307  if ( rows[i] < t ) t--;
308  mRenderer->moveCategory( rows[i], t );
309  // current moved under another, shift its index up
310  for ( int j = 0; j < i; j++ )
311  {
312  if ( to < rows[j] && rows[i] > rows[j] ) rows[j] += 1;
313  }
314  // removed under 'to' so the target shifted down
315  if ( rows[i] < to ) to--;
316  }
317  emit dataChanged( createIndex( 0, 0 ), createIndex( mRenderer->categories().size(), 0 ) );
318  emit rowsMoved();
319  return false;
320 }

664 static void _createCategories( QgsCategoryList& cats, QList<QVariant>& values, QgsSymbolV2* symbol )
665 {
666  // sort the categories first
667  QgsSymbolLayerV2Utils::sortVariantList( values, Qt::AscendingOrder );
668 
669  int num = values.count();
670 
671  bool hasNull = false;
672 
673  for ( int i = 0; i < num; i++ )
674  {
675  QVariant value = values[i];
676  if ( value.toString().isNull() )
677  {
678  hasNull = true;
679  }
680  QgsSymbolV2* newSymbol = symbol->clone();
681 
682  cats.append( QgsRendererCategoryV2( value, newSymbol, value.toString(), true ) );
683  }
684 
685  // add null (default) value if not exists
