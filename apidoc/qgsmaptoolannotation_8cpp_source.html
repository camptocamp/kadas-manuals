97 void QgsMapToolEditAnnotation::canvasPressEvent( QMouseEvent *e )
98 {
99  int nItems = mItems.size();
100  if ( e->button() == Qt::LeftButton && ( e->modifiers() & Qt::ControlModifier ) == 0 )
101  {
102  mMouseMoveLastXY = e->pos();
103  if ( nItems == 1 )
104  {
105  mAnnotationMoveAction = mItems.front()->moveActionForPosition( e->pos() );
106  }
107  else if ( nItems > 1 && mRectItem->contains( canvas()->mapToScene( e->pos() ) ) )
108  {
109  mDraggingRect = true;
110  mCanvas->setCursor( Qt::SizeAllCursor );
111  }
112  }
113  else if ( e->button() == Qt::RightButton )
114  {
115  // If annotation item is selected, show its context menu
116  if ( nItems == 1 && mItems.front()->hitTest( e->pos() ) )
117  {
118  mItems.front()->showContextMenu( canvas()->mapToGlobal( e->pos() ) );
119  }
120  else if ( nItems > 1 && mRectItem->contains( canvas()->mapToScene( e->pos() ) ) )
121  {
122  QMenu menu;
123  // If all items are pins, allow converting to waypoints
124  bool allPins = true;
125  for ( QgsAnnotationItem* item : mItems )
126  {
127  if ( !dynamic_cast<QgsPinAnnotationItem*>( item ) )
128  {
129  allPins = false;
130  break;
131  }
132  }
133  if ( allPins )
134  {
135  menu.addAction( QIcon( ":/images/themes/default/mIconPointLayer.svg" ), tr( "Convert to waypoints" ), this, SLOT( convertToWaypoints() ) );
136  }
137  menu.addAction( QIcon( ":/images/themes/default/mActionDeleteSelected.svg" ), tr( "Delete" ), this, SLOT( deleteAll() ) );
138  menu.exec( e->globalPos() );
139  }
140  }
141 }

143 void QgsMapToolEditAnnotation::canvasMoveEvent( QMouseEvent *e )
144 {
145  int nItems = mItems.size();
146  if ( e->buttons() & Qt::LeftButton )
147  {
148  if ( nItems == 1 && mAnnotationMoveAction != QgsAnnotationItem::NoAction )
149  {
150  mItems.front()->handleMoveAction( mAnnotationMoveAction, e->posF(), mMouseMoveLastXY );
151  mMouseMoveLastXY = e->pos();
152  }
153  else if ( nItems > 1 && mDraggingRect )
154  {
155  for ( QgsAnnotationItem* item : mItems )
156  {
157  if (( item->itemFlags() & QgsAnnotationItem::ItemMapPositionLocked ) == 0 )
158  {
159  item->handleMoveAction( QgsAnnotationItem::MoveMapPosition, e->posF(), mMouseMoveLastXY );
160  }
161  }
162  mMouseMoveLastXY = e->pos();
163  updateRect();
164  }
165  }
166  else if ( nItems == 1 )
167  {
168  int moveAction = mItems.front()->moveActionForPosition( e->pos() );
169  if ( moveAction != QgsAnnotationItem::NoAction )
170  mCanvas->setCursor( QCursor( mItems.front()->cursorShapeForAction( moveAction ) ) );
171  else
172  mCanvas->setCursor( mCursor );
173  }
174 }

176 void QgsMapToolEditAnnotation::canvasReleaseEvent( QMouseEvent *e )
177 {
178  if ( mAnnotationMoveAction != QgsAnnotationItem::NoAction || mDraggingRect )
179  {
180  mAnnotationMoveAction = QgsAnnotationItem::NoAction;
181  if ( mDraggingRect )
182  {
183  mCanvas->setCursor( mCursor );
184  }
185  mDraggingRect = false;
186  }
187  else if (( e->modifiers() & Qt::ControlModifier ) == 0 )
188  {
189  if (
190  ( mItems.size() == 1 && mItems.front() != mCanvas->annotationItemAtPos( e->pos() ) )
191  ||
192  ( mRectItem && !mRectItem->contains( canvas()->mapToScene( e->pos() ) ) )
193  )
194  {
195  deleteLater();
196  }
197  }
198  else if ( QgsAnnotationItem* clickedItem = mCanvas->annotationItemAtPos( e->pos() ) )
199  {
200  int pos = 0, n = mItems.size();
201  for ( ; pos < n && mItems[pos].data() != clickedItem; ++pos );
202  if ( pos >= n ) // Not already in list
203  {
204  if ( mItems.size() == 1 )
205  {
206  disconnect( mItems.front().data(), SIGNAL( destroyed( QObject* ) ), this, SLOT( deleteLater() ) );
207  mRectItem = new QGraphicsRectItem();
208  mRectItem->setPen( QPen( Qt::black, 2, Qt::DashLine ) );
209  mCanvas->scene()->addItem( mRectItem );
210  }
211  mItems.append( QPointer<QgsAnnotationItem>( clickedItem ) );
212  clickedItem->setSelected( true );
213  }
214  else if ( mItems.size() > 1 )
215  {
216  mItems[pos]->setSelected( false );
217  mItems.removeAt( pos );
218  if ( mItems.size() == 1 )
219  {
220  connect( mItems.front().data(), SIGNAL( destroyed( QObject* ) ), this, SLOT( deleteLater() ) );
221  delete mRectItem;
222  mRectItem = nullptr;
223  }
224  }
225  updateRect();
226  }
227 }

247 void QgsMapToolEditAnnotation::convertToWaypoints()
248 {
249  QgsRedliningLayer* gpsLayer = QgsMapLayerRegistry::instance()->getOrCreateGpsRoutesLayer();
250 for ( QgsAnnotationItem* item : mItems )
251  {
252  QgsPoint wgsPos = QgsCoordinateTransformCache::instance()->transform( item->mapGeoPosCrs().authid(), "EPSG:4326" )->transform( item->mapGeoPos() );
253  QString name = item->getName();
254  QString flags( "shape=point,symbol=circle,r=0,bold=1" );
255  gpsLayer->addShape( new QgsGeometry( new QgsPointV2( wgsPos ) ), Qt::yellow, Qt::yellow, 2, Qt::SolidLine, Qt::SolidPattern, flags, QString(), name );
256  }
257  deleteAll();
258  gpsLayer->triggerRepaint();
259 }
