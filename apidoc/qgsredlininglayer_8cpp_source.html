104 QgsFeatureList QgsRedliningLayer::pasteFeatures( const QgsFeatureStore& featureStore, QgsPoint* targetPos )
105 {
106  const QgsCoordinateTransform* ct = QgsCoordinateTransformCache::instance()->transform( featureStore.crs().authid(), crs().authid() );
107  QgsFeatureList newFeatures;
108  QgsPoint oldCenter;
109  foreach ( const QgsFeature& feature, featureStore.features() )
110  {
111  QString flags = feature.attribute( "flags" ).toString();
112  if ( flags.isEmpty() )
113  {
114  if ( feature.geometry()->type() == QGis::Point )
115  {
116  flags = "shape=point,symbol=circle";
117  }
118  else if ( feature.geometry()->type() == QGis::Line )
119  {
120  flags = "shape=line";
121  }
122  else if ( feature.geometry()->type() == QGis::Polygon )
123  {
124  flags = "shape=polygon";
125  }
126  else
127  {
128  continue;
129  }
130  }
131  QgsFeature newFeature( dataProvider()->fields() );
132  newFeature.setGeometry( QgsGeometry( feature.geometry()->geometry()->transformed( *ct ) ) );
133  oldCenter += newFeature.geometry()->boundingBox().center();
134 
135  newFeature.setAttributes( feature.attributes() );
136 
137  QMap<QString, QVariant> attribs;
138  attribs["flags"] = flags;
139  attribs["size"] = 1;
140  attribs["outline"] = QgsSymbolLayerV2Utils::encodeColor( Qt::black );
141  attribs["fill"] = QgsSymbolLayerV2Utils::encodeColor( Qt::yellow );
142  attribs["outline_style"] = QgsSymbolLayerV2Utils::encodePenStyle( Qt::SolidLine );
143  attribs["fill_style"] = QgsSymbolLayerV2Utils::encodeBrushStyle( Qt::SolidPattern );
144 
145  foreach ( const QString& key, attribs.keys() )
146  {
147  QVariant srcValue = feature.attribute( key );
148  newFeature.setAttribute( key, srcValue.isNull() ? attribs[key] : srcValue );
149  }
150  newFeatures.append( newFeature );
151  }
152  if ( targetPos )
153  {
154  int n = newFeatures.size();
155  oldCenter = QgsPoint( oldCenter.x() / n, oldCenter.y() / n );
156  QgsVector delta = *targetPos - oldCenter;
157  for ( int i = 0; i < n; ++i )
158  {
159  newFeatures[i].geometry()->translate( delta.x(), delta.y() );
160  }
161  }
162 
163  dataProvider()->addFeatures( newFeatures );
164  return newFeatures;
165 }

209 QMap<QString, QString> QgsRedliningLayer::deserializeFlags( const QString& flagsStr )
210 {

212  foreach ( const QString& flag, flagsStr.split( ",", QString::SkipEmptyParts ) )
213  {

230 bool QgsRedliningLayer::readXml( const QDomNode& layer_node )
231 {
232  QgsFeatureList features;
233  QDomNodeList nodes = layer_node.toElement().childNodes();
234  setLayerTransparency( layer_node.firstChildElement( "layerTransparency" ).text().toInt() );
235  for ( int iNode = 0, nNodes = nodes.size(); iNode < nNodes; ++iNode )
236  {
237  QDomElement redliningItemElem = nodes.at( iNode ).toElement();
238  if ( redliningItemElem.nodeName() == "RedliningItem" )
239  {
240  QgsFeature feature( dataProvider()->fields() );
241  feature.setAttribute( "size", redliningItemElem.attribute( "size", "1" ) );
242  feature.setAttribute( "outline", redliningItemElem.attribute( "outline", "255,0,0,255" ) );
243  feature.setAttribute( "fill", redliningItemElem.attribute( "fill", "0,0,255,255" ) );
244  feature.setAttribute( "outline_style", redliningItemElem.attribute( "outline_style", "1" ) );
245  feature.setAttribute( "fill_style", redliningItemElem.attribute( "fill_style", "1" ) );
246  feature.setAttribute( "text", redliningItemElem.attribute( "text", "" ) );
247  feature.setAttribute( "flags", redliningItemElem.attribute( "flags", "" ) );
248  feature.setAttribute( "tooltip", redliningItemElem.attribute( "tooltip" ) );
249  feature.setAttribute( "attributes", redliningItemElem.attribute( "attributes", "" ) );
250  feature.setGeometry( QgsGeometry::fromWkt( redliningItemElem.attribute( "geometry", "" ) ) );
251  features.append( feature );
252  }
253  }
254  dataProvider()->addFeatures( features );
255  updateFields();
256  emit layerModified();
257  return true;
