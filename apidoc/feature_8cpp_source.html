143  void FeaturePart::extractCoords( const GEOSGeometry* geom )
144  {
145  int i, j;
146 
147  const GEOSCoordSequence *coordSeq;
148  GEOSContextHandle_t geosctxt = geosContext();
149 
150  type = GEOSGeomTypeId_r( geosctxt, geom );
151 
152  if ( type == GEOS_POLYGON )
153  {
154  if ( GEOSGetNumInteriorRings_r( geosctxt, geom ) > 0 )
155  {
156  // set nbHoles, holes member variables
157  nbHoles = GEOSGetNumInteriorRings_r( geosctxt, geom );
158  holes = new PointSet*[nbHoles];
159 
160  for ( i = 0; i < nbHoles; i++ )
161  {
162  holes[i] = new PointSet();
163  holes[i]->holeOf = NULL;
164 
165  const GEOSGeometry* interior = GEOSGetInteriorRingN_r( geosctxt, geom, i );
166  holes[i]->nbPoints = GEOSGetNumCoordinates_r( geosctxt, interior );
167  holes[i]->x = new double[holes[i]->nbPoints];
168  holes[i]->y = new double[holes[i]->nbPoints];
169 
170  holes[i]->xmin = holes[i]->ymin = std::numeric_limits<double>::max();
171  holes[i]->xmax = holes[i]->ymax = -std::numeric_limits<double>::max();
172 
173  coordSeq = GEOSGeom_getCoordSeq_r( geosctxt, interior );
174 
175  for ( j = 0; j < holes[i]->nbPoints; j++ )
176  {
177  GEOSCoordSeq_getX_r( geosctxt, coordSeq, j, &holes[i]->x[j] );
178  GEOSCoordSeq_getY_r( geosctxt, coordSeq, j, &holes[i]->y[j] );
179 
180  holes[i]->xmax = holes[i]->x[j] > holes[i]->xmax ? holes[i]->x[j] : holes[i]->xmax;
181  holes[i]->xmin = holes[i]->x[j] < holes[i]->xmin ? holes[i]->x[j] : holes[i]->xmin;
182 
183  holes[i]->ymax = holes[i]->y[j] > holes[i]->ymax ? holes[i]->y[j] : holes[i]->ymax;
184  holes[i]->ymin = holes[i]->y[j] < holes[i]->ymin ? holes[i]->y[j] : holes[i]->ymin;
185  }
186 
187  reorderPolygon( holes[i]->nbPoints, holes[i]->x, holes[i]->y );
188  }
189  }
190 
191  // use exterior ring for the extraction of coordinates that follows
192  geom = GEOSGetExteriorRing_r( geosctxt, geom );
193  }
194  else
195  {
196  nbHoles = 0;
197  holes = NULL;
198  }
199 
200  // find out number of points
201  nbPoints = GEOSGetNumCoordinates_r( geosctxt, geom );
202  coordSeq = GEOSGeom_getCoordSeq_r( geosctxt, geom );
203 
204  // initialize bounding box
205  xmin = ymin = std::numeric_limits<double>::max();
206  xmax = ymax = -std::numeric_limits<double>::max();
207 
208  // initialize coordinate arrays
209  x = new double[nbPoints];
210  y = new double[nbPoints];
211 
212  for ( i = 0; i < nbPoints; i++ )
213  {
214  GEOSCoordSeq_getX_r( geosctxt, coordSeq, i, &x[i] );
215  GEOSCoordSeq_getY_r( geosctxt, coordSeq, i, &y[i] );
216 
217  xmax = x[i] > xmax ? x[i] : xmax;
218  xmin = x[i] < xmin ? x[i] : xmin;
219 
220  ymax = y[i] > ymax ? y[i] : ymax;
221  ymin = y[i] < ymin ? y[i] : ymin;
222  }

361  int FeaturePart::setPositionForPoint( double x, double y, double scale, LabelPosition ***lPos, double delta_width, double angle )
362  {
363 
364 #ifdef _DEBUG_

386 
387  //std::cout << "Nbp : " << nbp << std::endl;
388 
389  int i;
390  int icost = 0;
391  int inc = 2;
392 
393  double alpha;
394  double beta = 2 * M_PI / nbp; /* angle bw 2 pos */
395 
396  // uncomment for Wolff 2 position model test on RailwayStation
397  //if (nbp==2)
398  // beta = M_PI/2;
399 
400 #if 0
401  double distlabel = unit_convert( this->distlabel,
402  pal::PIXEL,
403  layer->pal->map_unit,
404  dpi, scale, delta_width );
405 #endif
406 
407  double lx, ly; /* label pos */
408 
409  /* various alpha */
410  double a90 = M_PI / 2;
411  double a180 = M_PI;
412  double a270 = a180 + a90;
413  double a360 = 2 * M_PI;
414 
415 
416  double gamma1, gamma2;
417 
418  if ( distlabel > 0 )
419  {
420  gamma1 = atan2( yrm / 2, distlabel + xrm / 2 );
421  gamma2 = atan2( xrm / 2, distlabel + yrm / 2 );
422  }
423  else
424  {
425  gamma1 = gamma2 = a90 / 3.0;
426  }
427 
428 
429  if ( gamma1 > a90 / 3.0 )
430  gamma1 = a90 / 3.0;
431 
432  if ( gamma2 > a90 / 3.0 )
433  gamma2 = a90 / 3.0;
434 
435 
436  if ( gamma1 == 0 || gamma2 == 0 )
437  {
438  std::cout << "Oups... label size error..." << std::endl;
439  }
440 
441  *lPos = new LabelPosition *[nbp];
442 
443  for ( i = 0, alpha = M_PI / 4; i < nbp; i++, alpha += beta )
444  {
445  lx = x;
446  ly = y;
447 
448  if ( alpha > a360 )
449  alpha -= a360;
450 
451  if ( alpha < gamma1 || alpha > a360 - gamma1 ) // on the right
452  {
453  lx += distlabel;
454  double iota = ( alpha + gamma1 );
455  if ( iota > a360 - gamma1 )
456  iota -= a360;
457 
458  //ly += -yrm/2.0 + tan(alpha)*(distlabel + xrm/2);
459  ly += -yrm + yrm * iota / ( 2 * gamma1 );
460  }
461  else if ( alpha < a90 - gamma2 ) // top-right
462  {
463  lx += distlabel * cos( alpha );
464  ly += distlabel * sin( alpha );
465  }
466  else if ( alpha < a90 + gamma2 ) // top
467  {
468  //lx += -xrm/2.0 - tan(alpha+a90)*(distlabel + yrm/2);
469  lx += -xrm * ( alpha - a90 + gamma2 ) / ( 2 * gamma2 );
470  ly += distlabel;
471  }
472  else if ( alpha < a180 - gamma1 ) // top left
473  {
474  lx += distlabel * cos( alpha ) - xrm;
475  ly += distlabel * sin( alpha );
476  }
477  else if ( alpha < a180 + gamma1 ) // left
478  {
479  lx += -distlabel - xrm;
480  //ly += -yrm/2.0 - tan(alpha)*(distlabel + xrm/2);
481  ly += - ( alpha - a180 + gamma1 ) * yrm / ( 2 * gamma1 );
482  }
483  else if ( alpha < a270 - gamma2 ) // down - left
484  {
485  lx += distlabel * cos( alpha ) - xrm;
486  ly += distlabel * sin( alpha ) - yrm;
487  }
488  else if ( alpha < a270 + gamma2 ) // down
489  {
490  ly += -distlabel - yrm;
491  //lx += -xrm/2.0 + tan(alpha+a90)*(distlabel + yrm/2);
492  lx += -xrm + ( alpha - a270 + gamma2 ) * xrm / ( 2 * gamma2 );
493  }
494  else if ( alpha < a360 )
495  {
496  lx += distlabel * cos( alpha );
497  ly += distlabel * sin( alpha ) - yrm;
498  }
499 
500  double cost;
501 
502  if ( nbp == 1 )
503  cost = 0.0001;
504  else
505  cost = 0.0001 + 0.0020 * double( icost ) / double( nbp - 1 );
506 
507  ( *lPos )[i] = new LabelPosition( i, lx, ly, xrm, yrm, angle, cost, this );
508 
509  icost += inc;
510 
511  if ( icost == nbp )
512  {
513  icost = nbp - 1;
514  inc = -2;
515  }
516  else if ( icost > nbp )
517  {
518  icost = nbp - 2;
519  inc = -2;
520  }
521 
522  }
523 
524  return nbp;
525  }
