58  LabelPosition::LabelPosition( int id, double x1, double y1, double w, double h, double alpha, double cost, FeaturePart *feature, bool isReversed )
59  : id( id ), cost( cost ), feature( feature ), probFeat( 0 ), nbOverlap( 0 ), alpha( alpha ), w( w ), h( h ), nextPart( NULL ), partId( -1 ), reversed( isReversed ), upsideDown( false )
60  {
61 
62  // alpha take his value bw 0 and 2*pi rad
63  while ( this->alpha > 2*M_PI )
64  this->alpha -= 2 * M_PI;
65 
66  while ( this->alpha < 0 )
67  this->alpha += 2 * M_PI;
68 
69  double beta = this->alpha + ( M_PI / 2 );
70 
71  double dx1, dx2, dy1, dy2;
72 
73  double tx, ty;
74 
75  dx1 = cos( this->alpha ) * w;
76  dy1 = sin( this->alpha ) * w;
77 
78  dx2 = cos( beta ) * h;
79  dy2 = sin( beta ) * h;
80 
81  x[0] = x1;
82  y[0] = y1;
83 
84  x[1] = x1 + dx1;
85  y[1] = y1 + dy1;
86 
87  x[2] = x1 + dx1 + dx2;
88  y[2] = y1 + dy1 + dy2;
89 
90  x[3] = x1 + dx2;
91  y[3] = y1 + dy2;
92 
93  // upside down ? (curved labels are always correct)
94  if ( feature->getLayer()->getArrangement() != P_CURVED &&
95  this->alpha > M_PI / 2 && this->alpha <= 3*M_PI / 2 )
96  {
97  bool uprightLabel = false;
98 
99  switch ( feature->getLayer()->getUpsidedownLabels() )
100  {
101  case Layer::Upright:
102  uprightLabel = true;
103  break;
104  case Layer::ShowDefined:
105  // upright only dynamic labels
106  if ( !feature->getFixedRotation() || ( !feature->getFixedPosition() && feature->getLabelAngle() == 0.0 ) )
107  {
108  uprightLabel = true;
109  }
110  break;
111  case Layer::ShowAll:
112  break;
113  default:
114  uprightLabel = true;
115  }
116 
117  if ( uprightLabel )
118  {
119  tx = x[0];
120  ty = y[0];
121 
122  x[0] = x[2];
123  y[0] = y[2];
124 
125  x[2] = tx;
126  y[2] = ty;
127 
128  tx = x[1];
129  ty = y[1];
130 
131  x[1] = x[3];
132  y[1] = y[3];
133 
134  x[3] = tx;
135  y[3] = ty;
136 
137  if ( this->alpha < M_PI )
138  this->alpha += M_PI;
139  else
140  this->alpha -= M_PI;
141 
142  // labels with text shown upside down are not classified as upsideDown,
143  // only those whose boundary points have been inverted
144  upsideDown = true;
145  }
146  }
147  }

246  bool LabelPosition::isInConflictSinglePart( LabelPosition* lp )
247  {
248  // TODO: add bounding box test to possibly avoid cross product calculation
249 
250  int i, i2, j;
251  int d1, d2;
252  double cp1, cp2;
253 
254  for ( i = 0; i < 4; i++ )
255  {
256  i2 = ( i + 1 ) % 4;
257  d1 = -1;
258  d2 = -1;
259 
260  for ( j = 0; j < 4; j++ )
261  {
262  cp1 = cross_product( x[i], y[i], x[i2], y[i2], lp->x[j], lp->y[j] );
263  if ( cp1 > 0 )
264  {
265  d1 = 1;
266  }
267  cp2 = cross_product( lp->x[i], lp->y[i],
268  lp->x[i2], lp->y[i2],
269  x[j], y[j] );
270 
271  if ( cp2 > 0 )
272  {
273  d2 = 1;
274  }
275  }
276 
277  if ( d1 == -1 || d2 == -1 ) // disjoint
278  return false;
279  }
280  return true;
281  }

283  bool LabelPosition::isInConflictMultiPart( LabelPosition* lp )
284  {
285  // check all parts against all parts of other one
286  LabelPosition* tmp1 = this;
287  while ( tmp1 )
288  {
289  // check tmp1 against parts of other label
290  LabelPosition* tmp2 = lp;
291  while ( tmp2 )
292  {
293  if ( tmp1->isInConflictSinglePart( tmp2 ) )
294  return true;
295  tmp2 = tmp2->nextPart;
296  }
297 
298  tmp1 = tmp1->nextPart;
299  }
300  return false; // no conflict found

500  double LabelPosition::getDistanceToPoint( double xp, double yp )
501  {
502  int i;
503  int j;
504 
505  double mx[4];
506  double my[4];
507 
508  double dist_min = std::numeric_limits<double>::max();
509  double dist;
510 
511  for ( i = 0; i < 4; i++ )
512  {
513  j = ( i + 1 ) % 4;
514  mx[i] = ( x[i] + x[j] ) / 2.0;
515  my[i] = ( y[i] + y[j] ) / 2.0;
516  }
517 
518  if ( vabs( cross_product( mx[0], my[0], mx[2], my[2], xp, yp ) / h ) < w / 2 )
519  {
520  dist = cross_product( x[1], y[1], x[0], y[0], xp, yp ) / w;
521  if ( vabs( dist ) < vabs( dist_min ) )
522  dist_min = dist;
523 
524  dist = cross_product( x[3], y[3], x[2], y[2], xp, yp ) / w;
525  if ( vabs( dist ) < vabs( dist_min ) )
526  dist_min = dist;
527  }
528 
529  if ( vabs( cross_product( mx[1], my[1], mx[3], my[3], xp, yp ) / w ) < h / 2 )
530  {
531  dist = cross_product( x[2], y[2], x[1], y[1], xp, yp ) / h;
532  if ( vabs( dist ) < vabs( dist_min ) )
533  dist_min = dist;
534 
535  dist = cross_product( x[0], y[0], x[3], y[3], xp, yp ) / h;
536  if ( vabs( dist ) < vabs( dist_min ) )
537  dist_min = dist;
538  }
539 
540  for ( i = 0; i < 4; i++ )
541  {
542  dist = dist_euc2d( x[i], y[i], xp, yp );
543  if ( vabs( dist ) < vabs( dist_min ) )
544  dist_min = dist;
545  }
546 
547  if ( nextPart && dist_min > 0 )
548  return min( dist_min, nextPart->getDistanceToPoint( xp, yp ) );
549 
550  return dist_min;
551  }

555  {
556  double ca, cb;
557  for ( int i = 0; i < 4; i++ )
558  {
559  for ( int j = 0; j < feat->getNumPoints() - 1; j++ )
560  {
561  ca = cross_product( x[i], y[i], x[( i+1 ) %4], y[( i+1 ) %4],
562  feat->x[j], feat->y[j] );

565

567  {
568  ca = cross_product( feat->x[j], feat->y[j], feat->x[j+1], feat->y[j+1],
