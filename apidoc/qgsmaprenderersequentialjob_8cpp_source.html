54 void QgsMapRendererSequentialJob::start()
55 {
56  if ( isActive() )
57  return; // do nothing if we are already running
58 
59  mRenderingStart.start();
60 
61  mErrors.clear();
62 
63  QgsDebugMsg( "SEQUENTIAL START" );
64 
65  Q_ASSERT( mInternalJob == 0 && mPainter == 0 );
66 
67  mPainter = new QPainter( &mImage );
68 
69  mInternalJob = new QgsMapRendererCustomPainterJob( mSettings, mPainter );
70  mInternalJob->setCache( mCache );
71 
72  connect( mInternalJob, SIGNAL( finished() ), SLOT( internalFinished() ) );
73 
74  mInternalJob->start();
75 }

120 void QgsMapRendererSequentialJob::internalFinished()
121 {
122  QgsDebugMsg( "SEQUENTIAL finished" );
123 
124  mPainter->end();
125  delete mPainter;
126  mPainter = 0;
127 
128  mLabelingResults = mInternalJob->takeLabelingResults();
129 
130  mErrors = mInternalJob->errors();
131 
132  // now we are in a slot called from mInternalJob - do not delete it immediately
133  // so the class is still valid when the execution returns to the class
134  mInternalJob->deleteLater();
135  mInternalJob = 0;
136 
137  mRenderingTime = mRenderingStart.elapsed();
138 
139  emit finished();
140 }
