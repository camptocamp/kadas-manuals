87 void QgsAbstractFeatureIterator::ref()
88 {
89  // Prepare if required the simplification of geometries to fetch:
90  // This code runs here because of 'prepareSimplification()' is virtual and it can be overrided

108 bool QgsAbstractFeatureIterator::prepareSimplification( const QgsSimplifyMethod& simplifyMethod )
109 {
110  mLocalSimplification = false;
111 
112  delete mGeometrySimplifier;
113  mGeometrySimplifier = NULL;
114 
115  // setup the simplification of geometries to fetch
116  if ( !( mRequest.flags() & QgsFeatureRequest::NoGeometry ) && simplifyMethod.methodType() != QgsSimplifyMethod::NoSimplification && ( simplifyMethod.forceLocalOptimization() || !providerCanSimplify( simplifyMethod.methodType() ) ) )
117  {
118  mGeometrySimplifier = QgsSimplifyMethod::createGeometrySimplifier( simplifyMethod );
119  mLocalSimplification = mGeometrySimplifier != NULL;
120  return mLocalSimplification;
121  }
122  return false;
123 }

131 bool QgsAbstractFeatureIterator::simplify( QgsFeature& feature )
132 {
133  // simplify locally the geometry using the configured simplifier
134  if ( mGeometrySimplifier )
135  {
136  QgsGeometry* geometry = feature.geometry();
137 
138  QGis::GeometryType geometryType = geometry->type();
139  if ( geometryType == QGis::Line || geometryType == QGis::Polygon )
140  return mGeometrySimplifier->simplifyGeometry( geometry );
141  }
142  return false;
143 }
