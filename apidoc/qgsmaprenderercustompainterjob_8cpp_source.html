44 void QgsMapRendererCustomPainterJob::start()
45 {
46  if ( isActive() )
47  return;
48 
49  mRenderingStart.start();
50 
51  mActive = true;
52 
53  mErrors.clear();
54 
55  QgsDebugMsg( QString( "0x%1: QPAINTER run!" ).arg(( quintptr )this, QT_POINTER_SIZE * 2, 16, QChar( '0' ) ) );
56 
57  QgsDebugMsg( QString( "0x%1: Preparing list of layer jobs for rendering" ).arg(( quintptr )this, QT_POINTER_SIZE * 2, 16, QChar( '0' ) ) );
58  QTime prepareTime;
59  prepareTime.start();
60 
61  // clear the background
62  if ( mSettings.backgroundColor() != QColor( Qt::transparent ) )
63  mPainter->fillRect( 0, 0, mSettings.outputSize().width(), mSettings.outputSize().height(), mSettings.backgroundColor() );
64 
65  mPainter->setRenderHint( QPainter::Antialiasing, mSettings.testFlag( QgsMapSettings::Antialiasing ) );
66 
67  QPaintDevice* thePaintDevice = mPainter->device();
68 
69  QString errMsg = QString( "pre-set DPI not equal to painter's DPI (%1 vs %2)" ).arg( thePaintDevice->logicalDpiX() ).arg( mSettings.outputDpi() );
70  Q_ASSERT_X( thePaintDevice->logicalDpiX() == mSettings.outputDpi(), "Job::startRender()", errMsg.toAscii().data() );
71 
72  delete mLabelingEngine;
73  mLabelingEngine = 0;
74 
75  if ( mSettings.testFlag( QgsMapSettings::DrawLabeling ) )
76  {
77  mLabelingEngine = new QgsPalLabeling;
78  mLabelingEngine->loadEngineSettings();
79  mLabelingEngine->init( mSettings );
80  }
81 
82  mLayerJobs = prepareJobs( mPainter, mLabelingEngine );
83  // prepareJobs calls mapLayer->createMapRenderer may involve cloning a RasterDataProvider,
84  // whose constructor may need to download some data (i.e. WMS, AMS) and doing so runs a
85  // QEventLoop waiting for the network request to complete. If unluckily someone calls
86  // mapCanvas->refresh() while this is happening, QgsMapRendererCustomPainterJob::cancel is
87  // called, deleting the QgsMapRendererCustomPainterJob while this function is running.
88  // Hence we need to check whether the job is still active before proceeding
89  if ( !isActive() )
90  return;
91 
92  QgsDebugMsg( QString( "0x%1: Rendering prepared in (seconds): %2" ).arg(( quintptr )this, QT_POINTER_SIZE * 2, 16, QChar( '0' ) ).arg( prepareTime.elapsed() / 1000.0 ) );
93 
94  if ( mRenderSynchronously )
95  {
96  // do the rendering right now!
97  doRender();
98  return;
99  }
100 
101  // now we are ready to start rendering!
102  connect( &mFutureWatcher, SIGNAL( finished() ), SLOT( futureFinished() ) );
103 
104  mFuture = QtConcurrent::run( staticRender, this );
105  mFutureWatcher.setFuture( mFuture );
106 }

217 void QgsMapRendererCustomPainterJob::doRender()
218 {
219  QgsDebugMsg( QString( "0x%1: Starting to render layer stack." ).arg(( quintptr )this, QT_POINTER_SIZE * 2, 16, QChar( '0' ) ) );
220  QTime renderTime;
221  renderTime.start();
222 
223  for ( LayerRenderJobs::iterator it = mLayerJobs.begin(); it != mLayerJobs.end(); ++it )
224  {
225  LayerRenderJob& job = *it;
226 
227  if ( job.context.renderingStopped() )
228  break;
229 
230  if ( job.context.useAdvancedEffects() )
231  {
232  // Set the QPainter composition mode so that this layer is rendered using
233  // the desired blending mode
234  mPainter->setCompositionMode( job.blendMode );
235  }
236 
237  if ( !job.cached )
238  job.renderer->render();
239 
240  if ( job.img )
241  {
242  // If we flattened this layer for alternate blend modes, composite it now
243  mPainter->drawImage( 0, 0, *job.img );
244  }
245 
246  }
247 
248  QgsDebugMsg( QString( "0x%1: Done rendering map layers" ).arg(( quintptr )this, QT_POINTER_SIZE * 2, 16, QChar( '0' ) ) );
249 
250  if ( mSettings.testFlag( QgsMapSettings::DrawLabeling ) && !mLabelingRenderContext.renderingStopped() )
251  drawLabeling( mSettings, mLabelingRenderContext, mLabelingEngine, mPainter );
252 
253  QgsDebugMsg( QString( "0x%1: Rendering completed in (seconds): %2" ).arg(( quintptr )this, QT_POINTER_SIZE * 2, 16, QChar( '0' ) ).arg( renderTime.elapsed() / 1000.0 ) );
254 }

281 void QgsMapRendererJob::drawOldLabeling( const QgsMapSettings& settings, QgsRenderContext& renderContext )
282 {
283  // render all labels for vector layers in the stack, starting at the base
284  QListIterator<QString> li( settings.layers() );
285  li.toBack();
286  while ( li.hasPrevious() )
287  {
288  if ( renderContext.renderingStopped() )
289  {
290  break;
291  }
292 
293  QString layerId = li.previous();
294 
295  QgsMapLayer *ml = QgsMapLayerRegistry::instance()->mapLayer( layerId );
296 
297  if ( !ml || ( ml->type() != QgsMapLayer::VectorLayer ) )
298  continue;
299 
300  // only make labels if the layer is visible
301  // after scale dep viewing settings are checked
302  if ( ml->hasScaleBasedVisibility() && ( settings.scale() < ml->minimumScale() || settings.scale() > ml->maximumScale() ) )
303  continue;
304 
305  const QgsCoordinateTransform* ct = 0;
306  QgsRectangle r1 = settings.visibleExtent(), r2;
307 
308  if ( settings.hasCrsTransformEnabled() )
309  {
310  ct = settings.layerTransform( ml );
311  if ( ct )
312  reprojectToLayerExtent( ct, ml->crs().geographicFlag(), r1, r2 );
313  }
314 
315  renderContext.setCoordinateTransform( ct );
316  renderContext.setExtent( r1 );
317 
318  ml->drawLabels( renderContext );
319  }

348 bool QgsMapRendererJob::needTemporaryImage( QgsMapLayer* ml )
349 {
350  if ( ml->type() == QgsMapLayer::VectorLayer || ml->type() == QgsMapLayer::RedliningLayer )
351  {
352  if ( mSettings.testFlag( QgsMapSettings::UseAdvancedEffects ) && ( ml->type() == QgsMapLayer::VectorLayer || ml->type() == QgsMapLayer::RedliningLayer ) )
353  {
354  QgsVectorLayer* vl = qobject_cast<QgsVectorLayer *>( ml );
355  if ((( vl->blendMode() != QPainter::CompositionMode_SourceOver )
356  || ( vl->featureBlendMode() != QPainter::CompositionMode_SourceOver )
357  || ( vl->layerTransparency() != 0 ) ) )
358  return true;
359  }
360  }
361  else if ( ml->type() == QgsMapLayer::RasterLayer )
362  {
363  // preview of intermediate raster rendering results requires a temporary output image
364  if ( mSettings.testFlag( QgsMapSettings::RenderPartialOutput ) )
365  return true;
366  }
367 
368  return false;
369 }
