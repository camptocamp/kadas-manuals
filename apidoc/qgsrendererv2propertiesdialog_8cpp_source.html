36 static bool _initRenderer( QString name, QgsRendererV2WidgetFunc f, QString iconName = QString() )
37 {
38  QgsRendererV2Registry* reg = QgsRendererV2Registry::instance();
39  QgsRendererV2AbstractMetadata* am = reg->rendererMetadata( name );
40  if ( am == NULL )
41  return false;
42  QgsRendererV2Metadata* m = dynamic_cast<QgsRendererV2Metadata*>( am );
43  if ( m == NULL )
44  return false;
45 
46  m->setWidgetFunction( f );
47 
48  if ( !iconName.isEmpty() )
49  {
50  QString iconPath = QgsApplication::defaultThemePath() + iconName;
51  QPixmap pix;
52  if ( pix.load( iconPath, "png" ) )
53  m->setIcon( pix );
54  }
55 
56  QgsDebugMsg( "Set for " + name );
57  return true;
58 }

137 void QgsRendererV2PropertiesDialog::rendererChanged()
138 {
139 
140  if ( cboRenderers->currentIndex() == -1 )
141  {
142  QgsDebugMsg( "No current item -- this should never happen!" );
143  return;
144  }
145 
146  QString rendererName = cboRenderers->itemData( cboRenderers->currentIndex() ).toString();
147 
148  //Retrieve the previous renderer: from the old active widget if possible, otherwise from the layer
149  QgsFeatureRendererV2* oldRenderer;
150  if ( mActiveWidget && mActiveWidget->renderer() )
151  {
152  oldRenderer = mActiveWidget->renderer()->clone();
153  }
154  else
155  {
156  oldRenderer = mLayer->rendererV2()->clone();
157  }
158 
159  // get rid of old active widget (if any)
160  if ( mActiveWidget )
161  {
162  stackedWidget->removeWidget( mActiveWidget );
163 
164  delete mActiveWidget;
165  mActiveWidget = NULL;
166  }
167 
168  QgsRendererV2Widget* w = NULL;
169  QgsRendererV2AbstractMetadata* m = QgsRendererV2Registry::instance()->rendererMetadata( rendererName );
170  if ( m != NULL )
171  w = m->createRendererWidget( mLayer, mStyle, oldRenderer );
172 
173  if ( w != NULL )
174  {
175  // instantiate the widget and set as active
176  mActiveWidget = w;
177  stackedWidget->addWidget( mActiveWidget );
178  stackedWidget->setCurrentWidget( mActiveWidget );
179  }
180  else
181  {
182  // set default "no edit widget available" page
183  stackedWidget->setCurrentWidget( pageNoWidget );
184  }
185 
186 }

188 void QgsRendererV2PropertiesDialog::apply()
189 {
190  if ( !mActiveWidget || !mLayer )
191  {
192  return;
193  }
194 
195  QgsFeatureRendererV2* renderer = mActiveWidget->renderer();
196  if ( renderer )
197  {
198  mLayer->setRendererV2( renderer->clone() );
199  }
200 
201  // set the blend modes for the layer
202  mLayer->setBlendMode( mBlendModeComboBox->blendMode() );
203  mLayer->setFeatureBlendMode( mFeatureBlendComboBox->blendMode() );
204 
205  // set transparency for the layer
206  mLayer->setLayerTransparency( mLayerTransparencySlider->value() );
207 }
