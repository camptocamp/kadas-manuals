96 void QgsApplication::init( QString customConfigPath )
97 {
98  if ( customConfigPath.isEmpty() )
99  {
100  if ( getenv( "QGIS_CUSTOM_CONFIG_PATH" ) )
101  {
102  customConfigPath = getenv( "QGIS_CUSTOM_CONFIG_PATH" );
103  }
104  else
105  {
106  customConfigPath = QString( "%1/.qgis%2/" ).arg( QDir::homePath() ).arg( VERSION_INT / 10000 );
107  }
108  }
109 
110 #ifndef COMPILING_CRSSYNC
111  qRegisterMetaType<QgsGeometry::Error>( "QgsGeometry::Error" );
112  qRegisterMetaType<QgsFeatureId>( "QgsFeatureId" );
113 #endif
114 
115  QString prefixPath( getenv( "QGIS_PREFIX_PATH" ) ? getenv( "QGIS_PREFIX_PATH" ) : applicationDirPath() );
116  // QgsDebugMsg( QString( "prefixPath(): %1" ).arg( prefixPath ) );
117 
118  // check if QGIS is run from build directory (not the install directory)
119  QFile f;
120  // "/../../.." is for Mac bundled app in build directory
121  foreach ( QString path, QStringList() << "" << "/.." << "/bin" << "/../../.." )
122  {
123  f.setFileName( prefixPath + path + "/qgisbuildpath.txt" );
124  if ( f.exists() )
125  break;
126  }
127  if ( f.exists() && f.open( QIODevice::ReadOnly ) )
128  {
129  ABISYM( mRunningFromBuildDir ) = true;
130  ABISYM( mBuildSourcePath ) = f.readLine().trimmed();
131  ABISYM( mBuildOutputPath ) = f.readLine().trimmed();
132  qDebug( "Running from build directory!" );
133  qDebug( "- source directory: %s", ABISYM( mBuildSourcePath ).toUtf8().data() );
134  qDebug( "- output directory of the build: %s", ABISYM( mBuildOutputPath ).toUtf8().data() );
135 #ifdef _MSC_VER
136  ABISYM( mCfgIntDir ) = prefixPath.split( "/", QString::SkipEmptyParts ).last();
137  qDebug( "- cfg: %s", ABISYM( mCfgIntDir ).toUtf8().data() );
138 #endif
139  }
140 
141  if ( ABISYM( mRunningFromBuildDir ) )
142  {
143  // we run from source directory - not installed to destination (specified prefix)
144  ABISYM( mPrefixPath ) = QString(); // set invalid path
145 #if defined(_MSC_VER) && !defined(USING_NMAKE)
146  setPluginPath( ABISYM( mBuildOutputPath ) + "/" + QString( QGIS_PLUGIN_SUBDIR ) + "/" + ABISYM( mCfgIntDir ) );
147 #else
148  setPluginPath( ABISYM( mBuildOutputPath ) + "/" + QString( QGIS_PLUGIN_SUBDIR ) );
149 #endif
150  setPkgDataPath( ABISYM( mBuildSourcePath ) ); // directly source path - used for: doc, resources, svg
151  ABISYM( mLibraryPath ) = ABISYM( mBuildOutputPath ) + "/" + QGIS_LIB_SUBDIR + "/";
152 #if defined(_MSC_VER) && !defined(USING_NMAKE)
153  ABISYM( mLibexecPath ) = ABISYM( mBuildOutputPath ) + "/" + QGIS_LIBEXEC_SUBDIR + "/" + ABISYM( mCfgIntDir ) + "/";
154 #else
155  ABISYM( mLibexecPath ) = ABISYM( mBuildOutputPath ) + "/" + QGIS_LIBEXEC_SUBDIR + "/";
156 #endif
157  }
158  else
159  {
160  char *prefixPath = getenv( "QGIS_PREFIX_PATH" );
161  if ( !prefixPath )
162  {
163 #if defined(ANDROID)
164  // this is "/data/data/org.qgis.qgis" in android
165  QDir myDir( QDir::homePath() );
166  myDir.cdUp();
167  QString myPrefix = myDir.absolutePath();
168  setPrefixPath( myPrefix, true );
169 #else
170  QString appDirPath = applicationDirPath();
171  if ( appDirPath.endsWith( "/" ) )
172  {
173  appDirPath.chop( 1 );
174  }
175  QString binSubDir( QGIS_BIN_SUBDIR );
176  if ( binSubDir.endsWith( "/" ) )
177  {
178  appDirPath.chop( 1 );
179  }
180  if ( appDirPath.endsWith( binSubDir ) )
181  {
182  appDirPath.chop( binSubDir.length() );
183  }
184  setPrefixPath( QDir( appDirPath ).absolutePath(), true );
185 #endif
186  }
187  else
188  {
189  setPrefixPath( prefixPath, true );
190  }
191  }
192 
193  if ( !customConfigPath.isEmpty() )
194  {
195  ABISYM( mConfigPath ) = customConfigPath + "/"; // make sure trailing slash is included
196  }
197 
198  ABISYM( mDefaultSvgPaths ) << qgisSettingsDirPath() + QString( "svg/" );
199 
200  // store system environment variables passed to application, before they are adjusted
201  QMap<QString, QString> systemEnvVarMap;
202  foreach ( const QString &varStr, QProcess::systemEnvironment() )
203  {
204  int pos = varStr.indexOf( QLatin1Char( '=' ) );
205  if ( pos == -1 )
206  continue;
207  QString varStrName = varStr.left( pos );
208  QString varStrValue = varStr.mid( pos + 1 );
209  systemEnvVarMap.insert( varStrName, varStrValue );
210  }
211  ABISYM( mSystemEnvVars ) = systemEnvVarMap;
212 
213  // allow Qt to search for Qt plugins (e.g. sqldrivers) in our plugin directory
214  QCoreApplication::addLibraryPath( pluginPath() );
215 
216  // set max. thread count to -1
217  // this should be read from QSettings but we don't know where they are at this point
218  // so we read actual value in main.cpp
219  ABISYM( mMaxThreads ) = -1;
220 }

226 bool QgsApplication::event( QEvent * event )
227 {
228  bool done = false;
229  if ( event->type() == QEvent::FileOpen )
230  {
231  // handle FileOpen event (double clicking a file icon in Mac OS X Finder)
232  if ( ABISYM( mFileOpenEventReceiver ) )
233  {
234  // Forward event to main window.
235  done = notify( ABISYM( mFileOpenEventReceiver ), event );
236  }
237  else
238  {
239  // Store filename because receiver has not registered yet.
240  // If QGIS has been launched by double clicking a file icon, FileOpen will be
241  // the first event; the main window is not yet ready to handle the event.
242  ABISYM( mFileOpenEventList ).append( static_cast<QFileOpenEvent *>( event )->file() );
243  done = true;
244  }
245  }
246  else
247  {
248  // pass other events to base class
249  done = QApplication::event( event );
250  }
251  return done;
252 }

254 bool QgsApplication::notify( QObject * receiver, QEvent * event )
255 {
256  bool done = false;
257  // Crashes in customization (especially on Mac), if we're not in the main/UI thread, see #5597
258  if ( thread() == receiver->thread() )
259  emit preNotify( receiver, event, &done );
260 
261  if ( done )
262  return true;
263 
264  // Send event to receiver and catch unhandled exceptions
265  done = true;
266  try
267  {
268  done = QApplication::notify( receiver, event );
269  }
270  catch ( QgsException & e )
271  {
272  QgsDebugMsg( "Caught unhandled QgsException: " + e.what() );
273  if ( qApp->thread() == QThread::currentThread() )
274  QMessageBox::critical( activeWindow(), tr( "Exception" ), e.what() );
275  }
276  catch ( std::exception & e )
277  {
278  QgsDebugMsg( "Caught unhandled std::exception: " + QString::fromAscii( e.what() ) );
279  if ( qApp->thread() == QThread::currentThread() )
280  QMessageBox::critical( activeWindow(), tr( "Exception" ), e.what() );
281  }
282  catch ( ... )
283  {
284  QgsDebugMsg( "Caught unhandled unknown exception" );
285  if ( qApp->thread() == QThread::currentThread() )
286  QMessageBox::critical( activeWindow(), tr( "Exception" ), tr( "unknown exception" ) );
287  }
288 
289  return done;
290 }

292 void QgsApplication::setFileOpenEventReceiver( QObject * receiver )
293 {
294  // Set receiver for FileOpen events
295  ABISYM( mFileOpenEventReceiver ) = receiver;
296  // Propagate any events collected before the receiver has registered.
297  if ( ABISYM( mFileOpenEventList ).count() > 0 )
298  {
299  QStringListIterator i( ABISYM( mFileOpenEventList ) );
300  while ( i.hasNext() )
301  {
302  QFileOpenEvent foe( i.next() );
303  QgsApplication::sendEvent( ABISYM( mFileOpenEventReceiver ), &foe );
304  }
305  ABISYM( mFileOpenEventList ).clear();
306  }

826 QString QgsApplication::absolutePathToRelativePath( QString aPath, QString targetPath )
827 {
828 #if defined( Q_OS_WIN )
829  const Qt::CaseSensitivity cs = Qt::CaseInsensitive;
830 
831  aPath.replace( "\\", "/" );
832  if ( aPath.startsWith( "//" ) )
833  {
834  // keep UNC prefix
835  aPath = "\\\\" + aPath.mid( 2 );
836  }
837 
838  targetPath.replace( "\\", "/" );
839  if ( targetPath.startsWith( "//" ) )
840  {
841  // keep UNC prefix
842  targetPath = "\\\\" + targetPath.mid( 2 );
843  }
844 #else
845  const Qt::CaseSensitivity cs = Qt::CaseSensitive;
846 #endif
847 
848  QStringList targetElems = targetPath.split( "/", QString::SkipEmptyParts );
849  QStringList aPathElems = aPath.split( "/", QString::SkipEmptyParts );
850 
851  targetElems.removeAll( "." );
852  aPathElems.removeAll( "." );
853 
854  // remove common part
855  int n = 0;
856  while ( aPathElems.size() > 0 &&
857  targetElems.size() > 0 &&
858  aPathElems[0].compare( targetElems[0], cs ) == 0 )
859  {
860  aPathElems.removeFirst();
861  targetElems.removeFirst();
862  n++;
863  }
864 
865  if ( n == 0 )
866  {
867  // no common parts; might not even be a file
868  return aPath;
869  }
870 
871  if ( targetElems.size() > 0 )
872  {
873  // go up to the common directory
874  for ( int i = 0; i < targetElems.size(); i++ )
875  {
876  aPathElems.insert( 0, ".." );
877  }
878  }
879  else
880  {
881  // let it start with . nevertheless,
882  // so relative path always start with either ./ or ../
883  aPathElems.insert( 0, "." );
884  }
885 
886  return aPathElems.join( "/" );
887 }

893  {
894  return rpath;

902 #endif

905  QStringList targetElems = targetPath.split( "/", QString::SkipEmptyParts );
906

910  targetElems.insert( 0, "" );
911  targetElems.insert( 0, "" );
912  }

970 bool QgsApplication::createDB( QString *errorMessage )
971 {
972  // set a working directory up for gdal to write .aux.xml files into
973  // for cases where the raster dir is read only to the user
974  // if the env var is already set it will be used preferentially
975  QString myPamPath = qgisSettingsDirPath() + QString( "gdal_pam/" );
976  QDir myDir( myPamPath );
977  if ( !myDir.exists() )
978  {
979  myDir.mkpath( myPamPath ); //fail silently
980  }
981 
982 #if defined(Q_OS_WIN) || defined(WIN32)
983  CPLSetConfigOption( "GDAL_PAM_PROXY_DIR", myPamPath.toUtf8() );
984 #else
985  //under other OS's we use an environment var so the user can
986  //override the path if he likes
987  int myChangeFlag = 0; //whether we want to force the env var to change
988  setenv( "GDAL_PAM_PROXY_DIR", myPamPath.toUtf8(), myChangeFlag );
989 #endif
990 
991  // Check qgis.db and make private copy if necessary
992  QFile qgisPrivateDbFile( QgsApplication::qgisUserDbFilePath() );
993 
994  // first we look for ~/.qgis/qgis.db
995  if ( !qgisPrivateDbFile.exists() )
996  {
997  // if it doesnt exist we copy it in from the global resources dir
998  QString qgisMasterDbFileName = QgsApplication::qgisMasterDbFilePath();
999  QFile masterFile( qgisMasterDbFileName );
1000 
1001  // Must be sure there is destination directory ~/.qgis
1002  QDir().mkpath( QgsApplication::qgisSettingsDirPath() );
1003 
1004  //now copy the master file into the users .qgis dir
1005  bool isDbFileCopied = masterFile.copy( qgisPrivateDbFile.fileName() );
1006 
1007  if ( !isDbFileCopied )
1008  {
1009  if ( errorMessage )
1010  {
1011  *errorMessage = tr( "[ERROR] Can not make qgis.db private copy" );
1012  }
1013  return false;
1014  }
1015  }
1016  else
1017  {
1018  // migrate if necessary
1019  sqlite3 *db;
1020  if ( sqlite3_open( QgsApplication::qgisUserDbFilePath().toUtf8().constData(), &db ) != SQLITE_OK )
1021  {
1022  if ( errorMessage )
1023  {
1024  *errorMessage = tr( "Could not open qgis.db" );
1025  }
1026  return false;
1027  }
1028 
1029  char *errmsg;
1030  int res = sqlite3_exec( db, "SELECT epsg FROM tbl_srs LIMIT 0", 0, 0, &errmsg );
1031  if ( res == SQLITE_OK )
1032  {
1033  // epsg column exists => need migration
1034  if ( sqlite3_exec( db,
1035  "ALTER TABLE tbl_srs RENAME TO tbl_srs_bak;"
1036  "CREATE TABLE tbl_srs ("
1037  "srs_id INTEGER PRIMARY KEY,"
1038  "description text NOT NULL,"
1039  "projection_acronym text NOT NULL,"
1040  "ellipsoid_acronym NOT NULL,"
1041  "parameters text NOT NULL,"
1042  "srid integer,"
1043  "auth_name varchar,"
1044  "auth_id varchar,"
1045  "is_geo integer NOT NULL,"
1046  "deprecated boolean);"
1047  "CREATE INDEX idx_srsauthid on tbl_srs(auth_name,auth_id);"
1048  "INSERT INTO tbl_srs(srs_id,description,projection_acronym,ellipsoid_acronym,parameters,srid,auth_name,auth_id,is_geo,deprecated) SELECT srs_id,description,projection_acronym,ellipsoid_acronym,parameters,srid,'','',is_geo,0 FROM tbl_srs_bak;"
1049  "DROP TABLE tbl_srs_bak", 0, 0, &errmsg ) != SQLITE_OK
1050  )
1051  {
1052  if ( errorMessage )
1053  {
1054  *errorMessage = tr( "Migration of private qgis.db failed.\n%1" ).arg( QString::fromUtf8( errmsg ) );
1055  }
1056  sqlite3_free( errmsg );
1057  sqlite3_close( db );
1058  return false;
1059  }
1060  }
1061  else
1062  {
1063  sqlite3_free( errmsg );
1064  }
1065 
1066  if ( sqlite3_exec( db, "DROP VIEW vw_srs", 0, 0, &errmsg ) != SQLITE_OK )
1067  {
1068  QgsDebugMsg( QString( "vw_srs didn't exists in private qgis.db: %1" ).arg( errmsg ) );
1069  }
1070 
1071  if ( sqlite3_exec( db,
1072  "CREATE VIEW vw_srs AS"
1073  " SELECT"
1074  " a.description AS description"
1075  ",a.srs_id AS srs_id"
1076  ",a.is_geo AS is_geo"
1077  ",coalesce(b.name,a.projection_acronym) AS name"
1078  ",a.parameters AS parameters"
1079  ",a.auth_name AS auth_name"
1080  ",a.auth_id AS auth_id"
1081  ",a.deprecated AS deprecated"
1082  " FROM tbl_srs a"
1083  " LEFT OUTER JOIN tbl_projection b ON a.projection_acronym=b.acronym"
1084  " ORDER BY coalesce(b.name,a.projection_acronym),a.description", 0, 0, &errmsg ) != SQLITE_OK
1085  )
1086  {
1087  if ( errorMessage )
1088  {
1089  *errorMessage = tr( "Update of view in private qgis.db failed.\n%1" ).arg( QString::fromUtf8( errmsg ) );
1090  }
1091  sqlite3_free( errmsg );
1092  sqlite3_close( db );
1093  return false;
1094  }
1095 
1096  sqlite3_close( db );
1097  }
1098  return true;
1099 }
