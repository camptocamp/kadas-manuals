226 // nextNiceNumber(4573.23, d) = 5000 (d=1) -> 4600 (d=10) -> 4580 (d=100) -> 4574 (d=1000) -> etc
227 inline double nextNiceNumber( double a, double d = 1 )
228 {
229  double s = qPow( 10.0, floor( log10( a ) ) ) / d;
230  return ceil( a / s ) * s;
231 }
232 
233 // prevNiceNumber(4573.23, d) = 4000 (d=1) -> 4500 (d=10) -> 4570 (d=100) -> 4573 (d=1000) -> etc
234 inline double prevNiceNumber( double a, double d = 1 )
235 {
236  double s = qPow( 10.0, floor( log10( a ) ) ) / d;
237  return floor( a / s ) * s;
238 }

240 void QgsComposerScaleBar::refreshSegmentMillimeters()
241 {
242  if ( mComposerMap )
243  {
244  //get mm dimension of composer map
245  QRectF composerItemRect = mComposerMap->rect();
246 
247  if ( mSegmentSizeMode == SegmentSizeFixed )
248  {
249  //calculate size depending on mNumUnitsPerSegment
250  mSegmentMillimeters = composerItemRect.width() / mapWidth() * mNumUnitsPerSegment;
251  }
252  else /*if(mSegmentSizeMode == SegmentSizeFitWidth)*/
253  {
254  if ( mMaxBarWidth < mMinBarWidth )
255  {
256  mSegmentMillimeters = 0;
257  }
258  else
259  {
260  double nSegments = ( mNumSegmentsLeft != 0 ) + mNumSegments;
261  // unitsPerSegments which fit minBarWidth resp. maxBarWidth
262  double minUnitsPerSeg = ( mMinBarWidth * mapWidth() ) / ( nSegments * composerItemRect.width() );
263  double maxUnitsPerSeg = ( mMaxBarWidth * mapWidth() ) / ( nSegments * composerItemRect.width() );
264 
265  // Start with coarsest "nice" number closest to minUnitsPerSeg resp
266  // maxUnitsPerSeg, then proceed to finer numbers as long as neither
267  // lowerNiceUnitsPerSeg nor upperNiceUnitsPerSeg are are in
268  // [minUnitsPerSeg, maxUnitsPerSeg]
269  double lowerNiceUnitsPerSeg = nextNiceNumber( minUnitsPerSeg );
270  double upperNiceUnitsPerSeg = prevNiceNumber( maxUnitsPerSeg );
271 
272  double d = 1;
273  while ( lowerNiceUnitsPerSeg > maxUnitsPerSeg && upperNiceUnitsPerSeg < minUnitsPerSeg )
274  {
275  d *= 10;
276  lowerNiceUnitsPerSeg = nextNiceNumber( minUnitsPerSeg, d );
277  upperNiceUnitsPerSeg = prevNiceNumber( maxUnitsPerSeg, d );
278  }
279 
280  // Pick mNumUnitsPerSegment from {lowerNiceUnitsPerSeg, upperNiceUnitsPerSeg}, use the larger if possible
281  mNumUnitsPerSegment = upperNiceUnitsPerSeg < minUnitsPerSeg ? lowerNiceUnitsPerSeg : upperNiceUnitsPerSeg;
282  mSegmentMillimeters = composerItemRect.width() / mapWidth() * mNumUnitsPerSegment;
283  }
284  }
285  }

288 double QgsComposerScaleBar::mapWidth() const
289 {
290  if ( !mComposerMap )
291  {
292  return 0.0;
293  }
294 
295  QgsRectangle composerMapRect = *( mComposerMap->currentMapExtent() );
296  if ( mUnits == MapUnits )
297  {
298  return composerMapRect.width();
299  }
300  else
301  {
302  QgsDistanceArea da;
303  da.setEllipsoid( QgsProject::instance()->readEntry( "Measure", "/Ellipsoid", "WGS84" ) );
304  da.setEllipsoidalMode( mComposition->mapSettings().hasCrsTransformEnabled() );
305  da.setSourceCrs( mComposition->mapSettings().destinationCrs() );
306 
307  double measure = da.measureLine( QgsPoint( composerMapRect.xMinimum(), composerMapRect.yMinimum() ), QgsPoint( composerMapRect.xMaximum(), composerMapRect.yMinimum() ) );
308  QGis::UnitType displayUnit;
309  switch ( mUnits )
310  {
311  case ScaleBarUnits::Meters: displayUnit = QGis::Meters; break;
312  case ScaleBarUnits::Feet: displayUnit = QGis::Feet; break;
313  case ScaleBarUnits::NauticalMiles: displayUnit = QGis::NauticalMiles; break;
314  default: displayUnit = QGis::UnknownUnit; break;
315  }
316 
317  if ( displayUnit != QGis::UnknownUnit )
318  {
319  QGis::UnitType measureUnit = mComposition->mapSettings().mapUnits();
320  da.convertMeasurement( measure, measureUnit, displayUnit, false );
321  }
322  return measure;
323  }
324 }

408 void QgsComposerScaleBar::applyDefaultSize( QgsComposerScaleBar::ScaleBarUnits u )
409 {
410  if ( mComposerMap )
411  {
412  setUnits( u );
413  double upperMagnitudeMultiplier = 1.0;
414  double widthInSelectedUnits = mapWidth();
415  double initialUnitsPerSegment = widthInSelectedUnits / 10.0; //default scalebar width equals half the map width
416  setNumUnitsPerSegment( initialUnitsPerSegment );
417 
418  switch ( mUnits )
419  {
420  case MapUnits:
421  {
422  upperMagnitudeMultiplier = 1.0;
423  setUnitLabeling( tr( "units" ) );
424  break;
425  }
426  case Meters:
427  {
428  if ( initialUnitsPerSegment > 1000.0 )
429  {
430  upperMagnitudeMultiplier = 1000.0;
431  setUnitLabeling( tr( "km" ) );
432  }
433  else
434  {
435  upperMagnitudeMultiplier = 1.0;
436  setUnitLabeling( tr( "m" ) );
437  }
438  break;
439  }
440  case Feet:
441  {
442  if ( initialUnitsPerSegment > 5419.95 )
443  {
444  upperMagnitudeMultiplier = 5419.95;
445  setUnitLabeling( tr( "miles" ) );
446  }
447  else
448  {
449  upperMagnitudeMultiplier = 1.0;
450  setUnitLabeling( tr( "ft" ) );
451  }
452  break;
453  }
454  case NauticalMiles:
455  {
456  upperMagnitudeMultiplier = 1;
457  setUnitLabeling( tr( "Nm" ) );
458  break;
459  }
460  }
461 
462  double segmentWidth = initialUnitsPerSegment / upperMagnitudeMultiplier;
463  int segmentMagnitude = floor( log10( segmentWidth ) );
464  double unitsPerSegment = upperMagnitudeMultiplier * ( qPow( 10.0, segmentMagnitude ) );
465  double multiplier = floor(( widthInSelectedUnits / ( unitsPerSegment * 10.0 ) ) / 2.5 ) * 2.5;
466 
467  if ( multiplier > 0 )
468  {
469  unitsPerSegment = unitsPerSegment * multiplier;
470  }
471  setNumUnitsPerSegment( unitsPerSegment );
472  setNumMapUnitsPerScaleBarUnit( upperMagnitudeMultiplier );
473 
474  setNumSegments( 4 );
475  setNumSegmentsLeft( 2 );
476  }
477 
478  refreshSegmentMillimeters();
479  adjustBoxSize();
480  emit itemChanged();
481 }

484 {
485  if ( !mStyle )
486  {
487  return;
488  }
489 
490  QRectF box = mStyle->calculateBoxSize();
491  if ( rect().height() > box.height() )
492  {
493  //keep user specified item height if higher than minimum scale bar height
494  box.setHeight( rect().height() );
495  }
496 
497  //update rect for data defined size and position
498  QRectF newRect = evalItemRect( box, true );
499 
500  //scale bars have a minimum size, respect that regardless of data defined settings
501  if ( newRect.width() < box.width() )
502  {
503  newRect.setWidth( box.width() );
504  }
505  if ( newRect.height() < box.height() )
506  {
507  newRect.setHeight( box.height() );
508  }

563 void QgsComposerScaleBar::segmentPositions( QList<QPair<double, double> >& posWidthList ) const
564 {
565  posWidthList.clear();
566  double mCurrentXCoord = mPen.widthF() + mBoxContentSpace;
567 
568  //left segments
569  double leftSegmentSize = mSegmentMillimeters / mNumSegmentsLeft;
570  for ( int i = 0; i < mNumSegmentsLeft; ++i )
571  {
572  posWidthList.push_back( qMakePair( mCurrentXCoord, leftSegmentSize ) );
573  mCurrentXCoord += leftSegmentSize;
574  }
575 
576  //right segments
577  for ( int i = 0; i < mNumSegments; ++i )
578  {
579  posWidthList.push_back( qMakePair( mCurrentXCoord, mSegmentMillimeters ) );
580  mCurrentXCoord += mSegmentMillimeters;
581  }
582 }
