187 void QgsHtmlAnnotationItem::readXML( const QDomDocument& doc, const QDomElement& itemElem )
188 {
189  mVectorLayer = 0;
190  if ( itemElem.hasAttribute( "vectorLayer" ) )
191  {
192  mVectorLayer = dynamic_cast<QgsVectorLayer*>( QgsMapLayerRegistry::instance()->mapLayer( itemElem.attribute( "vectorLayer", "" ) ) );
193  if ( mVectorLayer )
194  {
195  connect( mVectorLayer, SIGNAL( layerModified() ), this, SLOT( setFeatureForMapPosition() ) );
196  connect( mMapCanvas, SIGNAL( renderComplete( QPainter* ) ), this, SLOT( setFeatureForMapPosition() ) );
197  connect( mMapCanvas, SIGNAL( layersChanged( QStringList ) ), this, SLOT( updateVisibility() ) );
198  }
199  }
200  mHasAssociatedFeature = itemElem.attribute( "hasFeature", "0" ).toInt();
201  mFeatureId = itemElem.attribute( "feature", "0" ).toInt();
202  mHtmlFile = itemElem.attribute( "htmlfile", "" );
203  QDomElement annotationElem = itemElem.firstChildElement( "AnnotationItem" );
204  if ( !annotationElem.isNull() )
205  {
206  _readXML( doc, annotationElem );
207  }
208 
209  if ( mWebView )
210  {
211  setHTMLPage( mHtmlFile );
212  }
213  updateVisibility();
214 }

221 void QgsHtmlAnnotationItem::setFeatureForMapPosition()
222 {
223  if ( !mVectorLayer || !mMapCanvas )
224  {
225  return;
226  }
227 
228  QSettings settings;
229  double halfIdentifyWidth = QgsMapTool::searchRadiusMU( mMapCanvas );
230  QgsRectangle searchRect( mMapPosition.x() - halfIdentifyWidth, mMapPosition.y() - halfIdentifyWidth,
231  mMapPosition.x() + halfIdentifyWidth, mMapPosition.y() + halfIdentifyWidth );
232 
233  QgsFeatureIterator fit = mVectorLayer->getFeatures( QgsFeatureRequest().setFilterRect( searchRect ).setFlags( QgsFeatureRequest::NoGeometry | QgsFeatureRequest::ExactIntersect ) );
234 
235  QgsFeature currentFeature;
236  QgsFeatureId currentFeatureId = 0;
237  bool featureFound = false;
238 
239  while ( fit.nextFeature( currentFeature ) )
240  {
241  currentFeatureId = currentFeature.id();
242  featureFound = true;
243  break;
244  }
245 
246  mHasAssociatedFeature = featureFound;
247  mFeatureId = currentFeatureId;
248  mFeature = currentFeature;
249 
250  QString newtext = QgsExpression::replaceExpressionText( mHtmlSource, &mFeature, vectorLayer() );
251  mWebView->setHtml( newtext );
252 }
