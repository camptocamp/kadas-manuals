230 QgsSymbolV2* QgsCategorizedSymbolRendererV2::symbolForFeature( QgsFeature& feature )
231 {
232  QgsSymbolV2* symbol = originalSymbolForFeature( feature );
233  if ( !symbol )
234  return 0;
235 
236  if ( !mRotation.data() && !mSizeScale.data() )
237  return symbol; // no data-defined rotation/scaling - just return the symbol
238 
239  // find out rotation, size scale
240  const double rotation = mRotation.data() ? mRotation->evaluate( feature ).toDouble() : 0;
241  const double sizeScale = mSizeScale.data() ? mSizeScale->evaluate( feature ).toDouble() : 1.;
242 
243  // take a temporary symbol (or create it if doesn't exist)
244  QgsSymbolV2* tempSymbol = mTempSymbols[symbol];
245 
246  // modify the temporary symbol and return it
247  if ( tempSymbol->type() == QgsSymbolV2::Marker )
248  {
249  QgsMarkerSymbolV2* markerSymbol = static_cast<QgsMarkerSymbolV2*>( tempSymbol );
250  if ( mRotation.data() ) markerSymbol->setAngle( rotation );
251  markerSymbol->setSize( sizeScale * static_cast<QgsMarkerSymbolV2*>( symbol )->size() );
252  markerSymbol->setScaleMethod( mScaleMethod );
253  }
254  else if ( tempSymbol->type() == QgsSymbolV2::Line )
255  {
256  QgsLineSymbolV2* lineSymbol = static_cast<QgsLineSymbolV2*>( tempSymbol );
257  lineSymbol->setWidth( sizeScale * static_cast<QgsLineSymbolV2*>( symbol )->width() );
258  }
259 
260  return tempSymbol;
261 }

265 {

441 void QgsCategorizedSymbolRendererV2::startRender( QgsRenderContext& context, const QgsFields& fields )
442 {
443  mCounting = context.rendererScale() == 0.0;
444 
445  // make sure that the hash table is up to date
446  rebuildHash();
447 
448  // find out classification attribute index from name
449  mAttrNum = fields.fieldNameIndex( mAttrName );
450  if ( mAttrNum == -1 )
451  {
452  mExpression.reset( new QgsExpression( mAttrName ) );
453  mExpression->prepare( fields );
454  }
455 
456  QgsCategoryList::iterator it = mCategories.begin();
457  for ( ; it != mCategories.end(); ++it )
458  {
459  if ( !it->symbol() )
460  {
461  continue;
462  }
463  it->symbol()->startRender( context, &fields );
464 
465  if ( mRotation.data() || mSizeScale.data() )
466  {
467  QgsSymbolV2* tempSymbol = it->symbol()->clone();
468  tempSymbol->setRenderHints(( mRotation.data() ? QgsSymbolV2::DataDefinedRotation : 0 ) |
469  ( mSizeScale.data() ? QgsSymbolV2::DataDefinedSizeScale : 0 ) );
470  tempSymbol->startRender( context, &fields );
471  mTempSymbols[ it->symbol()] = tempSymbol;
472  }
473  }

499 QList<QString> QgsCategorizedSymbolRendererV2::usedAttributes()
500 {
501  QSet<QString> attributes;
502 
503  // mAttrName can contain either attribute name or an expression.
504  // Sometimes it is not possible to distinguish between those two,
505  // e.g. "a - b" can be both a valid attribute name or expression.
506  // Since we do not have access to fields here, try both options.
507  attributes << mAttrName;
508 
509  QgsExpression testExpr( mAttrName );
510  if ( !testExpr.hasParserError() )
511  attributes.unite( testExpr.referencedColumns().toSet() );
512 
513  if ( mRotation.data() ) attributes.unite( mRotation->referencedColumns().toSet() );
514  if ( mSizeScale.data() ) attributes.unite( mSizeScale->referencedColumns().toSet() );
515 
516  QgsCategoryList::const_iterator catIt = mCategories.constBegin();
517  for ( ; catIt != mCategories.constEnd(); ++catIt )
518  {
519  QgsSymbolV2* catSymbol = catIt->symbol();
520  if ( catSymbol )
521  {
522  attributes.unite( catSymbol->usedAttributes() );
523  }
524  }
525  return attributes.toList();
526 }

578 QgsFeatureRendererV2* QgsCategorizedSymbolRendererV2::create( QDomElement& element )
579 {
580  QDomElement symbolsElem = element.firstChildElement( "symbols" );
581  if ( symbolsElem.isNull() )
582  return NULL;
583 
584  QDomElement catsElem = element.firstChildElement( "categories" );
585  if ( catsElem.isNull() )
586  return NULL;
587 
588  QgsSymbolV2Map symbolMap = QgsSymbolLayerV2Utils::loadSymbols( symbolsElem );
589  QgsCategoryList cats;
590 
591  //legend symbols
592  QgsSymbolV2Map legendSymbolsMap;
593  QDomElement legendSymbolsElem = element.firstChildElement( "legendsymbols" );
594  if ( !legendSymbolsElem.isNull() )
595  {
596  legendSymbolsMap = QgsSymbolLayerV2Utils::loadSymbols( legendSymbolsElem );
597  }
598 
599  QDomElement catElem = catsElem.firstChildElement();
600  while ( !catElem.isNull() )
601  {
602  if ( catElem.tagName() == "category" )
603  {
604  QVariant value = QVariant( catElem.attribute( "value" ) );
605  QString symbolName = catElem.attribute( "symbol" );
606  QString label = catElem.attribute( "label" );
607  QString html = catElem.attribute( "html" );
608  bool render = catElem.attribute( "render" ) != "false";
609  QgsSymbolV2* symbol = 0;
610  if ( symbolMap.contains( symbolName ) )
611  {
612  symbol = symbolMap.take( symbolName );
613  }
614  QgsSymbolV2* legendSymbol = 0;
615  if ( legendSymbolsMap.contains( symbolName ) )
616  {
617  legendSymbol = legendSymbolsMap.take( symbolName );
618  }
619  cats.append( QgsRendererCategoryV2( value, symbol, label, render, html, legendSymbol ) );
620  }
621  catElem = catElem.nextSiblingElement();
622  }
623 
624  QString attrName = element.attribute( "attr" );
625 
626  QgsCategorizedSymbolRendererV2* r = new QgsCategorizedSymbolRendererV2( attrName, cats );
627 
628  // delete symbols if there are any more
629  QgsSymbolLayerV2Utils::clearSymbolMap( symbolMap );
630 
631  // try to load source symbol (optional)
632  QDomElement sourceSymbolElem = element.firstChildElement( "source-symbol" );
633  if ( !sourceSymbolElem.isNull() )
634  {
635  QgsSymbolV2Map sourceSymbolMap = QgsSymbolLayerV2Utils::loadSymbols( sourceSymbolElem );
636  if ( sourceSymbolMap.contains( "0" ) )
637  {
638  r->setSourceSymbol( sourceSymbolMap.take( "0" ) );
639  }
640  QgsSymbolLayerV2Utils::clearSymbolMap( sourceSymbolMap );
641  }
642 
643  // try to load color ramp (optional)
644  QDomElement sourceColorRampElem = element.firstChildElement( "colorramp" );
645  if ( !sourceColorRampElem.isNull() && sourceColorRampElem.attribute( "name" ) == "[source]" )
646  {
647  r->setSourceColorRamp( QgsSymbolLayerV2Utils::loadColorRamp( sourceColorRampElem ) );
648  QDomElement invertedColorRampElem = element.firstChildElement( "invertedcolorramp" );
649  if ( !invertedColorRampElem.isNull() )
650  r->setInvertedColorRamp( invertedColorRampElem.attribute( "value" ) == "1" );
651  }
652 
653  QDomElement rotationElem = element.firstChildElement( "rotation" );
654  if ( !rotationElem.isNull() )
655  r->setRotationField( rotationElem.attribute( "field" ) );
656 
657  QDomElement sizeScaleElem = element.firstChildElement( "sizescale" );
658  if ( !sizeScaleElem.isNull() )
659  {
660  r->setSizeScaleField( sizeScaleElem.attribute( "field" ) );
661  r->setScaleMethod( QgsSymbolLayerV2Utils::decodeScaleMethod( sizeScaleElem.attribute( "scalemethod" ) ) );
662  }
663 
664  // TODO: symbol levels
665  return r;
666 }

922  if ( invertedPolygonRenderer )
