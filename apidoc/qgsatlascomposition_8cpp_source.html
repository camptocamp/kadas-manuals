172 //
173 // Private class only used for the sorting of features
174 class FieldSorter
175 {
176  public:
177  FieldSorter( QgsAtlasComposition::SorterKeys& keys, bool ascending = true ) : mKeys( keys ), mAscending( ascending ) {}
178 
179  bool operator()( const QgsFeatureId& id1, const QgsFeatureId& id2 )
180  {
181  bool result = true;
182 
183  if ( mKeys[ id1 ].type() == QVariant::Int )
184  {
185  result = mKeys[ id1 ].toInt() < mKeys[ id2 ].toInt();
186  }
187  else if ( mKeys[ id1 ].type() == QVariant::Double )
188  {
189  result = mKeys[ id1 ].toDouble() < mKeys[ id2 ].toDouble();
190  }
191  else if ( mKeys[ id1 ].type() == QVariant::String )
192  {
193  result = ( QString::localeAwareCompare( mKeys[ id1 ].toString(), mKeys[ id2 ].toString() ) < 0 );
194  }
195 
196  return mAscending ? result : !result;
197  }
198  private:
199  QgsAtlasComposition::SorterKeys& mKeys;
200  bool mAscending;
201 };

203 int QgsAtlasComposition::updateFeatures()
204 {
205  //needs to be called when layer, filter, sort changes
206 
207  if ( !mCoverageLayer )
208  {
209  return 0;
210  }
211 
212  updateFilenameExpression();
213 
214  // select all features with all attributes
215  QgsFeatureIterator fit = mCoverageLayer->getFeatures();
216 
217  QScopedPointer<QgsExpression> filterExpression;
218  if ( mFilterFeatures && !mFeatureFilter.isEmpty() )
219  {
220  filterExpression.reset( new QgsExpression( mFeatureFilter ) );
221  if ( filterExpression->hasParserError() )
222  {
223  mFilterParserError = filterExpression->parserErrorString();
224  return 0;
225  }
226  }
227  mFilterParserError = QString();
228 
229  // We cannot use nextFeature() directly since the feature pointer is rewinded by the rendering process
230  // We thus store the feature ids for future extraction
231  QgsFeature feat;
232  mFeatureIds.clear();
233  mFeatureKeys.clear();
234  int sortIdx = mCoverageLayer->fieldNameIndex( mSortKeyAttributeName );
235  while ( fit.nextFeature( feat ) )
236  {
237  if ( !filterExpression.isNull() )
238  {
239  QVariant result = filterExpression->evaluate( &feat, mCoverageLayer->pendingFields() );
240  if ( filterExpression->hasEvalError() )
241  {
242  QgsMessageLog::logMessage( tr( "Atlas filter eval error: %1" ).arg( filterExpression->evalErrorString() ), tr( "Composer" ) );
243  }
244 
245  // skip this feature if the filter evaluation if false
246  if ( !result.toBool() )
247  {
248  continue;
249  }
250  }
251  mFeatureIds.push_back( feat.id() );
252 
253  if ( mSortFeatures && sortIdx != -1 )
254  {
255  mFeatureKeys.insert( feat.id(), feat.attributes()[ sortIdx ] );
256  }
257  }
258 
259  // sort features, if asked for
260  if ( mFeatureKeys.count() )
261  {
262  FieldSorter sorter( mFeatureKeys, mSortAscending );
263  qSort( mFeatureIds.begin(), mFeatureIds.end(), sorter );
264  }
265 
266  QgsExpression::setSpecialColumn( "$numfeatures", QVariant(( int )mFeatureIds.size() ) );
267 
268  //jump to first feature if currently using an atlas preview
269  //need to do this in case filtering/layer change has altered matching features
270  if ( mComposition->atlasMode() == QgsComposition::PreviewAtlas )
271  {
272  firstFeature();
273  }
274 
275  return mFeatureIds.size();
276 }

387 bool QgsAtlasComposition::prepareForFeature( const int featureI, const bool updateMaps )
388 {
389  if ( !mCoverageLayer )
390  {
391  return false;
392  }
393 
394  if ( mFeatureIds.size() == 0 )
395  {
396  emit statusMsgChanged( tr( "No matching atlas features" ) );
397  return false;
398  }
399 
400  mCurrentFeatureNo = featureI;
401 
402  // retrieve the next feature, based on its id
403  mCoverageLayer->getFeatures( QgsFeatureRequest().setFilterFid( mFeatureIds[ featureI ] ) ).nextFeature( mCurrentFeature );
404 
405  QgsExpression::setSpecialColumn( "$atlasfeatureid", mCurrentFeature.id() );
406  QgsExpression::setSpecialColumn( "$atlasgeometry", QVariant::fromValue( *mCurrentFeature.geometry() ) );
407  QgsExpression::setSpecialColumn( "$atlasfeature", QVariant::fromValue( mCurrentFeature ) );
408  QgsExpression::setSpecialColumn( "$feature", QVariant(( int )featureI + 1 ) );
409 
410  // generate filename for current feature
411  if ( !evalFeatureFilename() )
412  {
413  //error evaluating filename
414  return false;
415  }
416 
417  emit featureChanged( &mCurrentFeature );
418  emit statusMsgChanged( QString( tr( "Atlas feature %1 of %2" ) ).arg( featureI + 1 ).arg( mFeatureIds.size() ) );
419 
420  if ( !mCurrentFeature.isValid() )
421  {
422  //bad feature
423  return true;
424  }
425 
426  if ( !updateMaps )
427  {
428  //nothing more to do
429  return true;
430  }
431 
432  //update composer maps
433 
434  //build a list of atlas-enabled composer maps
435  QList<QgsComposerMap*> maps;
436  QList<QgsComposerMap*> atlasMaps;
437  mComposition->composerItems( maps );
438  if ( maps.isEmpty() )
439  {
440  return true;
441  }
442  for ( QList<QgsComposerMap*>::iterator mit = maps.begin(); mit != maps.end(); ++mit )
443  {
444  QgsComposerMap* currentMap = ( *mit );
445  if ( !currentMap->atlasDriven() )
446  {
447  continue;
448  }
449  atlasMaps << currentMap;
450  }
451 
452  if ( atlasMaps.count() > 0 )
453  {
454  //clear the transformed bounds of the previous feature
455  mTransformedFeatureBounds = QgsRectangle();
456 
457  // compute extent of current feature in the map CRS. This should be set on a per-atlas map basis,
458  // but given that it's not currently possible to have maps with different CRSes we can just
459  // calculate it once based on the first atlas maps' CRS.
460  computeExtent( atlasMaps[0] );
461  }
462 
463  for ( QList<QgsComposerMap*>::iterator mit = maps.begin(); mit != maps.end(); ++mit )
464  {
465  if (( *mit )->atlasDriven() )
466  {
467  // map is atlas driven, so update it's bounds (causes a redraw)
468  prepareMap( *mit );
469  }
470  else
471  {
472  // map is not atlas driven, so manually force a redraw (to reflect possibly atlas
473  // dependent symbology)
474  ( *mit )->cache();
475  }
476  }
477 
478  return true;
479 }

481 void QgsAtlasComposition::computeExtent( QgsComposerMap* map )
482 {
483  // compute the extent of the current feature, in the crs of the specified map
484 
485  const QgsCoordinateReferenceSystem& coverage_crs = mCoverageLayer->crs();
486  // transformation needed for feature geometries
487  const QgsCoordinateReferenceSystem& destination_crs = map->composition()->mapSettings().destinationCrs();
488  mTransform.setSourceCrs( coverage_crs );
489  mTransform.setDestCRS( destination_crs );
490 
491  // QgsGeometry::boundingBox is expressed in the geometry"s native CRS
492  // We have to transform the grometry to the destination CRS and ask for the bounding box
493  // Note: we cannot directly take the transformation of the bounding box, since transformations are not linear
494  QgsGeometry tgeom( *mCurrentFeature.geometry() );
495  tgeom.transform( mTransform );
496  mTransformedFeatureBounds = tgeom.boundingBox();
497 }

499 void QgsAtlasComposition::prepareMap( QgsComposerMap* map )
500 {
501  if ( !map->atlasDriven() )
502  {
503  return;
504  }
505 
506  if ( mTransformedFeatureBounds.isEmpty() )
507  {
508  //transformed extent of current feature hasn't been calculated yet. This can happen if
509  //a map has been set to be atlas controlled after prepare feature was called
510  computeExtent( map );
511  }
512 
513  double xa1 = mTransformedFeatureBounds.xMinimum();
514  double xa2 = mTransformedFeatureBounds.xMaximum();
515  double ya1 = mTransformedFeatureBounds.yMinimum();
516  double ya2 = mTransformedFeatureBounds.yMaximum();
517  QgsRectangle newExtent = mTransformedFeatureBounds;
518  QgsRectangle mOrigExtent( map->extent() );
519 
520  //sanity check - only allow fixed scale mode for point layers
521  bool isPointLayer = false;
522  switch ( mCoverageLayer->wkbType() )
523  {
524  case QGis::WKBPoint:
525  case QGis::WKBPoint25D:
526  case QGis::WKBMultiPoint:
527  case QGis::WKBMultiPoint25D:
528  isPointLayer = true;
529  break;
530  default:
531  isPointLayer = false;
532  break;
533  }
534 
535  if ( map->atlasScalingMode() == QgsComposerMap::Fixed || map->atlasScalingMode() == QgsComposerMap::Predefined || isPointLayer )
536  {
537  QgsScaleCalculator calc;
538  calc.setMapUnits( composition()->mapSettings().mapUnits() );
539  calc.setDpi( 25.4 );
540  double originalScale = calc.calculate( mOrigExtent, map->rect().width() );
541  double geomCenterX = ( xa1 + xa2 ) / 2.0;
542  double geomCenterY = ( ya1 + ya2 ) / 2.0;
543 
544  if ( map->atlasScalingMode() == QgsComposerMap::Fixed || isPointLayer )
545  {
546  // only translate, keep the original scale (i.e. width x height)
547  double xMin = geomCenterX - mOrigExtent.width() / 2.0;
548  double yMin = geomCenterY - mOrigExtent.height() / 2.0;
549  newExtent = QgsRectangle( xMin,
550  yMin,
551  xMin + mOrigExtent.width(),
552  yMin + mOrigExtent.height() );
553 
554  //scale newExtent to match original scale of map
555  //this is required for geographic coordinate systems, where the scale varies by extent
556  double newScale = calc.calculate( newExtent, map->rect().width() );
557  newExtent.scale( originalScale / newScale );
558  }
559  else if ( map->atlasScalingMode() == QgsComposerMap::Predefined )
560  {
561  // choose one of the predefined scales
562  double newWidth = mOrigExtent.width();
563  double newHeight = mOrigExtent.height();
564  const QVector<qreal>& scales = mPredefinedScales;
565  for ( int i = 0; i < scales.size(); i++ )
566  {
567  double ratio = scales[i] / originalScale;
568  newWidth = mOrigExtent.width() * ratio;
569  newHeight = mOrigExtent.height() * ratio;
570 
571  // compute new extent, centered on feature
572  double xMin = geomCenterX - newWidth / 2.0;
573  double yMin = geomCenterY - newHeight / 2.0;
574  newExtent = QgsRectangle( xMin,
575  yMin,
576  xMin + newWidth,
577  yMin + newHeight );
578 
579  //scale newExtent to match desired map scale
580  //this is required for geographic coordinate systems, where the scale varies by extent
581  double newScale = calc.calculate( newExtent, map->rect().width() );
582  newExtent.scale( scales[i] / newScale );
583 
584  if (( newExtent.width() >= mTransformedFeatureBounds.width() ) && ( newExtent.height() >= mTransformedFeatureBounds.height() ) )
585  {
586  // this is the smallest extent that embeds the feature, stop here
587  break;
588  }
589  }
590  }
591  }
592  else if ( map->atlasScalingMode() == QgsComposerMap::Auto )
593  {
594  // auto scale
595 
596  double geomRatio = mTransformedFeatureBounds.width() / mTransformedFeatureBounds.height();
597  double mapRatio = mOrigExtent.width() / mOrigExtent.height();
598 
599  // geometry height is too big
600  if ( geomRatio < mapRatio )
601  {
602  // extent the bbox's width
603  double adjWidth = ( mapRatio * mTransformedFeatureBounds.height() - mTransformedFeatureBounds.width() ) / 2.0;
604  xa1 -= adjWidth;
605  xa2 += adjWidth;
606  }
607  // geometry width is too big
608  else if ( geomRatio > mapRatio )
609  {
610  // extent the bbox's height
611  double adjHeight = ( mTransformedFeatureBounds.width() / mapRatio - mTransformedFeatureBounds.height() ) / 2.0;
612  ya1 -= adjHeight;
613  ya2 += adjHeight;
614  }
615  newExtent = QgsRectangle( xa1, ya1, xa2, ya2 );
616 
617  if ( map->atlasMargin() > 0.0 )
618  {
619  newExtent.scale( 1 + map->atlasMargin() );
620  }
621  }
622 
623  // set the new extent (and render)
624  map->setNewAtlasFeatureExtent( newExtent );
625 }

668 void QgsAtlasComposition::readXML( const QDomElement& atlasElem )
669 {
670  mEnabled = atlasElem.attribute( "enabled", "false" ) == "true" ? true : false;
671  emit toggled( mEnabled );
672  if ( !mEnabled )
673  {
674  emit parameterChanged();
675  return;
676  }
677 
678  // look for stored layer name
679  mCoverageLayer = 0;
680  QMap<QString, QgsMapLayer*> layers = QgsMapLayerRegistry::instance()->mapLayers();
681  for ( QMap<QString, QgsMapLayer*>::const_iterator it = layers.begin(); it != layers.end(); ++it )
682  {
683  if ( it.key() == atlasElem.attribute( "coverageLayer" ) )
684  {
685  mCoverageLayer = dynamic_cast<QgsVectorLayer*>( it.value() );
686  break;
687  }
688  }
689 
690  mFilenamePattern = atlasElem.attribute( "filenamePattern", "" );
691 
692  mSortFeatures = atlasElem.attribute( "sortFeatures", "false" ) == "true" ? true : false;
693  if ( mSortFeatures )
694  {
695  mSortKeyAttributeName = atlasElem.attribute( "sortKey", "" );
696  // since 2.3, the field name is saved instead of the field index
697  // following code keeps compatibility with version 2.2 projects
698  // to be removed in QGIS 3.0
699  bool isIndex;
700  int idx = mSortKeyAttributeName.toInt( &isIndex );
701  if ( isIndex && mCoverageLayer )
702  {
703  const QgsFields fields = mCoverageLayer->pendingFields();
704  if ( idx >= 0 && idx < fields.count() )
705  {
706  mSortKeyAttributeName = fields[idx].name();
707  }
708  }
709  mSortAscending = atlasElem.attribute( "sortAscending", "true" ) == "true" ? true : false;
710  }
711  mFilterFeatures = atlasElem.attribute( "filterFeatures", "false" ) == "true" ? true : false;
712  if ( mFilterFeatures )
713  {
714  mFeatureFilter = atlasElem.attribute( "featureFilter", "" );
715  }
716 
717  mHideCoverage = atlasElem.attribute( "hideCoverage", "false" ) == "true" ? true : false;
718 
719  emit parameterChanged();
720 }
