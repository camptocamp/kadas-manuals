22 bool QgsScaleUtils::saveScaleList( const QString &fileName, const QStringList &scales, QString &errorMessage )
23 {
24  QDomDocument doc;

26  root.setAttribute( "version", "1.0" );
27  doc.appendChild( root );

29  for ( int i = 0; i < scales.count(); ++i )

31  QDomElement el = doc.createElement( "scale" );
