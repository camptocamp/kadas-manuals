70  virtual QList<QNetworkProxy> queryProxy( const QNetworkProxyQuery & query = QNetworkProxyQuery() ) override
71  {
72  QgsNetworkAccessManager *nam = QgsNetworkAccessManager::instance();
73 
74  // iterate proxies factories and take first non empty list
75  foreach ( QNetworkProxyFactory *f, nam->proxyFactories() )
76  {
77  QList<QNetworkProxy> systemproxies = f->systemProxyForQuery( query );
78  if ( systemproxies.size() > 0 )
79  return systemproxies;
80 
81  QList<QNetworkProxy> proxies = f->queryProxy( query );
82  if ( proxies.size() > 0 )
83  return proxies;
84  }
85 
86  // no proxies from the proxy factor list check for excludes
87  if ( query.queryType() != QNetworkProxyQuery::UrlRequest )
88  return QList<QNetworkProxy>() << nam->fallbackProxy();
89 
90  QString url = query.url().toString();
91 
92  foreach ( QString exclude, nam->excludeList() )
93  {
94  if ( url.startsWith( exclude ) )
95  {
96  QgsDebugMsg( QString( "using default proxy for %1 [exclude %2]" ).arg( url ).arg( exclude ) );
97  return QList<QNetworkProxy>() << QNetworkProxy();
98  }
99  }
100 
101  if ( nam->useSystemProxy() )
102  {
103  QgsDebugMsg( QString( "requesting system proxy for query %1" ).arg( url ) );
104  QList<QNetworkProxy> proxies = QNetworkProxyFactory::systemProxyForQuery( query );
105  if ( !proxies.isEmpty() )
106  {
107  QgsDebugMsg( QString( "using system proxy %1:%2 for query" )
108  .arg( proxies.first().hostName() ).arg( proxies.first().port() ) );
109  return proxies;
110  }
111  }
112 
113  QgsDebugMsg( QString( "using fallback proxy for %1" ).arg( url ) );
114  return QList<QNetworkProxy>() << nam->fallbackProxy();
115  }

236 void QgsNetworkAccessManager::getCredentials( QNetworkReply *reply, QAuthenticator * auth )
237 {
238  QgsCredentials::instance()->lock();
239  QString realm = QString( "%1 at %2" ).arg( auth->realm() ).arg( reply->url().host() );
240  QString username = auth->user();
241  QString password = auth->password();
242  bool isGuiThread = thread() == qApp->thread();
243 
244  if ( username.isEmpty() && password.isEmpty() && reply->request().hasRawHeader( "Authorization" ) )
245  {
246  QByteArray header( reply->request().rawHeader( "Authorization" ) );
247  if ( header.startsWith( "Basic " ) )
248  {
249  QByteArray auth( QByteArray::fromBase64( header.mid( 6 ) ) );
250  int pos = auth.indexOf( ":" );
251  if ( pos >= 0 )
252  {
253  username = auth.left( pos );
254  password = auth.mid( pos + 1 );
255  }
256  }
257  }
258 
259  if ( QgsCredentials::instance()->get( realm, username, password, tr( "Authentication required" ), isGuiThread ) )
260  {
261  auth->setUser( username );
262  auth->setPassword( password );
263  QgsCredentials::instance()->put( realm, username, password );
264  }
265  QgsCredentials::instance()->unlock();
266 }

294 void QgsNetworkAccessManager::handleSSLErrors( QNetworkReply *reply, const QList<QSslError> &errors )
295 {
296  QgsDebugMsg( "Handling ssl errors" );
297  QgsSSLIgnoreList::lock();
298  QList<QSslError> newErrors;
299  foreach ( QSslError error, errors )
300  {
301  if ( error.error() == QSslError::NoError )
302  continue;
303  if ( !QgsSSLIgnoreList::ignoreErrors().contains( error.error() ) )
304  {
305  newErrors.append( error );
306  }
307  }
308  if ( newErrors.isEmpty() )
309  {
310  reply->ignoreSslErrors();
311  }
312  else
313  {
314  bool ok = false;
315  emit sslErrorsConformationRequired( reply->url(), newErrors, &ok );
316  if ( ok )
317  {
318  foreach ( QSslError error, newErrors )
319  {
320  QgsSSLIgnoreList::ignoreErrors().insert( error.error() );
321  }
322  reply->ignoreSslErrors();
323  }
324  }
325  QgsSSLIgnoreList::unlock();
326 }

372 void QgsNetworkAccessManager::setupDefaultProxyAndCache()
373 {
374  mInitialized = true;
375  mUseSystemProxy = false;
376  Q_ASSERT( smMainNAM );
377 
378  if ( smMainNAM != this )
379  {
380  connect( this, SIGNAL( requestTimedOut( QUrl ) ),
381  smMainNAM, SIGNAL( requestTimedOut( QUrl ) ) );
382 
383 #ifndef QT_NO_OPENSSL
384  connect( this, SIGNAL( sslErrors( QNetworkReply *, const QList<QSslError> & ) ),
385  smMainNAM, SIGNAL( sslErrors( QNetworkReply *, const QList<QSslError> & ) ),
386  Qt::BlockingQueuedConnection );
387 #endif
388  connect( smMainNAM, SIGNAL( cookieAdded( QByteArray, QUrl ) ), this, SLOT( addCookie( QByteArray, QUrl ) ) );
389  }
390  else
391  {
392  // Look for certificates in <appDataDir>/certificates to add to the SSL socket CA certificate database
393 #ifdef _MSC_VER
394  QDir certDir( QDir( QApplication::applicationDirPath() ).absoluteFilePath( "certificates" ) );
395 #else
396  QDir certDir( QgsApplication::prefixPath() + "/share/qgis/certificates" );
397 #endif
398  QgsDebugMsg( QString( "Looking for certificates in %1" ).arg( certDir.absolutePath() ) );
399  foreach ( const QString& certFilename, certDir.entryList( QStringList() << "*.pem", QDir::Files ) )
400  {
401  QFile certFile( certDir.absoluteFilePath( certFilename ) );
402  if ( certFile.open( QIODevice::ReadOnly ) )
403  {
404  QgsDebugMsg( QString( "Reading certificate file %1" ).arg( certFile.fileName() ) );
405  QByteArray pem = certFile.readAll();
406  QList<QSslCertificate> certs = QSslCertificate::fromData( pem, QSsl::Pem );
407  QgsDebugMsg( QString( "Adding %1 certificates" ).arg( certs.size() ) );
408  foreach ( const QSslCertificate& cert, certs )
409  {
410  QSslSocket::addDefaultCaCertificate( cert );
411  }
412  }
413  }
414 
415  connect( this, SIGNAL( sslErrors( QNetworkReply*, QList<QSslError> ) ), this, SLOT( handleSSLErrors( QNetworkReply*, QList<QSslError> ) ) );
416  }
417  connect( this, SIGNAL( authenticationRequired( QNetworkReply *, QAuthenticator * ) ),
418  this, SLOT( getCredentials( QNetworkReply*, QAuthenticator* ) ) );
419  connect( this, SIGNAL( proxyAuthenticationRequired( const QNetworkProxy &, QAuthenticator * ) ),
420  this, SLOT( getProxyCredentials( QNetworkProxy, QAuthenticator* ) ) );
421 
422  // check if proxy is enabled
423  QNetworkProxy proxy;
424  QStringList excludes;
425  QSettings settings;
426 
427  bool proxyEnabled = settings.value( "proxy/proxyEnabled", false ).toBool();
428  if ( proxyEnabled )
429  {
430  excludes = settings.value( "proxy/proxyExcludedUrls", "" ).toString().split( "|", QString::SkipEmptyParts );
431 
432  //read type, host, port, user, passw from settings
433  QString proxyHost = settings.value( "proxy/proxyHost", "" ).toString();
434  int proxyPort = settings.value( "proxy/proxyPort", "" ).toString().toInt();
435  QString proxyUser = settings.value( "proxy/proxyUser", "" ).toString();
436  QString proxyPassword = settings.value( "proxy/proxyPassword", "" ).toString();
437 
438  QString proxyTypeString = settings.value( "proxy/proxyType", "" ).toString();
439 
440  if ( proxyTypeString == "DefaultProxy" )
441  {
442  mUseSystemProxy = true;
443  QNetworkProxyFactory::setUseSystemConfiguration( true );
444  QList<QNetworkProxy> proxies = QNetworkProxyFactory::systemProxyForQuery();
445  if ( !proxies.isEmpty() )
446  {
447  proxy = proxies.first();
448  }
449  QgsDebugMsg( "setting default proxy" );
450  }
451  else
452  {
453  QNetworkProxy::ProxyType proxyType = QNetworkProxy::DefaultProxy;
454  if ( proxyTypeString == "Socks5Proxy" )
455  {
456  proxyType = QNetworkProxy::Socks5Proxy;
457  }
458  else if ( proxyTypeString == "HttpProxy" )
459  {
460  proxyType = QNetworkProxy::HttpProxy;
461  }
462  else if ( proxyTypeString == "HttpCachingProxy" )
463  {
464  proxyType = QNetworkProxy::HttpCachingProxy;
465  }
466  else if ( proxyTypeString == "FtpCachingProxy" )
467  {
468  proxyType = QNetworkProxy::FtpCachingProxy;
469  }
470  QgsDebugMsg( QString( "setting proxy %1 %2:%3 %4/%5" )
471  .arg( proxyType )
472  .arg( proxyHost ).arg( proxyPort )
473  .arg( proxyUser ).arg( proxyPassword )
474  );
475  proxy = QNetworkProxy( proxyType, proxyHost, proxyPort, proxyUser, proxyPassword );
476  }
477  }
478 
479 #if QT_VERSION >= 0x40500
480  setFallbackProxyAndExcludes( proxy, excludes );
481 
482  QgsNetworkDiskCache *newcache = qobject_cast<QgsNetworkDiskCache*>( cache() );
483  if ( !newcache )
484  newcache = new QgsNetworkDiskCache( this );
485  qint64 cacheSize = settings.value( "cache/size", 50 * 1024 * 1024 ).toULongLong();
486  QgsDebugMsg( QString( "setCacheDirectory: %1" ).arg( QgsApplication::cacheDir() ) );
487  QgsDebugMsg( QString( "setMaximumCacheSize: %1" ).arg( cacheSize ) );
488  newcache->setCacheDirectory( QgsApplication::cacheDir() );
489  newcache->setMaximumCacheSize( cacheSize );
490  QgsDebugMsg( QString( "cacheDirectory: %1" ).arg( newcache->cacheDirectory() ) );
491  QgsDebugMsg( QString( "maximumCacheSize: %1" ).arg( newcache->maximumCacheSize() ) );
492 
493  if ( cache() != newcache )
494  setCache( newcache );
495 #else
496  setProxy( proxy );
497 #endif
498 
499  if ( smMainNAM )
500  {
501  smMainNAM->mCookieLock.lock();
502  static_cast<QgsNetworkCookieJar*>( cookieJar() )->setAllCookies( static_cast<QgsNetworkCookieJar*>( smMainNAM->cookieJar() )->allCookies() );
503  smMainNAM->mCookieLock.unlock();
504  }

516 QNetworkRequest QgsNetworkAccessManager::requestWithUserInfo( const QNetworkRequest& req )
517 {
518  // HACK for NTLM SSO: if no authorization info is provided, try setting an
519  // empty username and some random password (i.e. sso) as user info to fool
520  // QNetworkAccessManager for NTLM SSO authorization, see
521  // http://stackoverflow.com/questions/14706851/ntlmv2-authentication-in-qt
522  QNetworkRequest adjustedRequest( req );
523  QUrl url = req.url();
524 
525  bool trySSO = QSettings().value( "/Qgis/networkAndProxy/attemptSSO", false ).toBool();
526  if ( trySSO && url.userInfo().isEmpty() && req.rawHeader( "Authorization" ).isEmpty() )
527  {
528  url.setUserInfo( ":sso" );
529  }
530  adjustedRequest.setUrl( url );
531 
532  bool disablePeerVerification = QSettings().value( "/Qgis/networkAndProxy/disableSSLPeerVerification", false ).toBool();
533  if ( disablePeerVerification )
534  {
535  QgsDebugMsg( "Disabling SSL peer verification" );
536  QSslConfiguration config = req.sslConfiguration();
537  config.setPeerVerifyMode( QSslSocket::VerifyNone );
538  adjustedRequest.setSslConfiguration( config );
539  }
540 
541  return adjustedRequest;
542 }
