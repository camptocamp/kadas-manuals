34 QgsSingleSymbolRendererV2Widget::QgsSingleSymbolRendererV2Widget( QgsVectorLayer* layer, QgsStyleV2* style, QgsFeatureRendererV2* renderer )
35  : QgsRendererV2Widget( layer, style )
36  , mRenderer( NULL )
37 {
38  // try to recognize the previous renderer
39  // (null renderer means "no previous renderer")
40 
41  if ( renderer )
42  {
43  mRenderer = QgsSingleSymbolRendererV2::convertFromRenderer( renderer );
44  }
45  if ( !mRenderer )
46  {
47  QgsSymbolV2* symbol = QgsSymbolV2::defaultSymbol( mLayer->geometryType() );
48 
49  mRenderer = new QgsSingleSymbolRendererV2( symbol );
50  }
51 
52  // load symbol from it
53  mSingleSymbol = mRenderer->symbol()->clone();
54 
55  // setup ui
56  setupUi( this );
57  mHtmlLineEdit->setToolTip( QgsRendererV2Widget::htmlToolTip() );
58  mSelector = new QgsSymbolV2SelectorDialog( mSingleSymbol, mStyle, mLayer, NULL, true );
59  connect( mSelector, SIGNAL( symbolModified() ), this, SLOT( changeSingleSymbol() ) );
60  mVerticalLayout->addWidget( mSelector );
61 
62  mHtmlLineEdit->setText( mRenderer->html() );
63  mWMSLegendSettingsGroupBox->setCollapsed( true );
64 
65  // advanced actions - data defined rendering
66  QMenu* advMenu = mSelector->advancedMenu();
67 
68  advMenu->addAction( tr( "Symbol levels..." ), this, SLOT( showSymbolLevels() ) );
69 
70  mDataDefinedMenus = new QgsRendererV2DataDefinedMenus( advMenu, mLayer,
71  mRenderer->rotationField(), mRenderer->sizeScaleField(), mRenderer->scaleMethod() );
72  connect( mDataDefinedMenus, SIGNAL( rotationFieldChanged( QString ) ), this, SLOT( rotationFieldChanged( QString ) ) );
73  connect( mDataDefinedMenus, SIGNAL( sizeScaleFieldChanged( QString ) ), this, SLOT( sizeScaleFieldChanged( QString ) ) );
74  connect( mDataDefinedMenus, SIGNAL( scaleMethodChanged( QgsSymbolV2::ScaleMethod ) ), this, SLOT( scaleMethodChanged( QgsSymbolV2::ScaleMethod ) ) );
75 
76  //add legend symbol icon to button
77  if ( mRenderer->legendSymbol() )
78  {
79  QIcon icon = QgsSymbolLayerV2Utils::symbolPreviewIcon( mRenderer->legendSymbol(), mLegendIconButton->iconSize() );
80  mLegendIconButton->setIcon( icon );
81  }
82 }
