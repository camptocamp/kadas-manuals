62 QgsRasterBlock* QgsSingleBandColorDataRenderer::block2( int bandNo, QgsRectangle const & extent, int width, int height, QgsRasterBlockFeedback* feedback )
63 {
64  Q_UNUSED( bandNo );
65 
66  QgsRasterBlock *outputBlock = new QgsRasterBlock();
67  if ( !mInput )
68  {
69  return outputBlock;
70  }
71 
72  QgsRasterBlock *inputBlock = mInput->block2( mBand, extent, width, height, feedback );
73  if ( !inputBlock || inputBlock->isEmpty() )
74  {
75  QgsDebugMsg( "No raster data!" );
76  delete inputBlock;
77  return outputBlock;
78  }
79 
80  bool hasTransparency = usesTransparency();
81  if ( !hasTransparency )
82  {
83  // Nothing to do, just retype if necessary
84  inputBlock->convert( QGis::ARGB32_Premultiplied );
85  delete outputBlock;
86  return inputBlock;
87  }
88 
89  if ( !outputBlock->reset( QGis::ARGB32_Premultiplied, width, height ) )
90  {
91  delete inputBlock;
92  return outputBlock;
93  }
94 
95  for ( qgssize i = 0; i < ( qgssize )width*height; i++ )
96  {
97  QRgb pixelColor;
98  QRgb c = inputBlock->color( i );
99  double alpha = qAlpha( c );
100  pixelColor = qRgba( mOpacity * qRed( c ), mOpacity * qGreen( c ), mOpacity * qBlue( c ), mOpacity * alpha );
101  outputBlock->setColor( i, pixelColor );
102  }
103 
104  delete inputBlock;
105  return outputBlock;
106 }

129 bool QgsSingleBandColorDataRenderer::setInput( QgsRasterInterface* input )
130 {
131  // Renderer can only work with numerical values in at least 1 band
132  if ( !input ) return false;
133 
134  if ( !mOn )
135  {
136  // In off mode we can connect to anything
137  mInput = input;
138  return true;
139  }
140 
141  if ( input->dataType( 1 ) == QGis::ARGB32 ||
142  input->dataType( 1 ) == QGis::ARGB32_Premultiplied )
143  {
144  mInput = input;
145  return true;
146  }
147  return false;
