99  while ( item && item->data( 0, sGeometryRole ).isNull() )

103 
104  int idx = item->data( 0, sGeometryRole ).toInt();

122  }

134 void QgsMapIdentifyDialog::collectInfo( const QgsPoint &mapPos )
135 {
136  // Prepare for raster layers
137  const QgsCoordinateReferenceSystem& mapCrs = mCanvas->mapSettings().destinationCrs();
138  const QgsCoordinateTransform* crst = QgsCoordinateTransformCache::instance()->transform( mapCrs.authid(), "EPSG:4326" );
139  QgsPoint worldPos = crst->transform( mapPos );
140  QgsRectangle worldExtent = crst->transform( mCanvas->extent() );
141  QStringList rlayerIds;
142  QMap<QString, QVariant> rlayerMap;
143 
144  // Prepare for vector layers
145  QgsRenderContext renderContext = QgsRenderContext::fromMapSettings( mCanvas->mapSettings() );
146  double radiusmm = QSettings().value( "/Map/searchRadiusMM", QGis::DEFAULT_SEARCH_RADIUS_MM ).toDouble();
147  radiusmm = radiusmm > 0 ? radiusmm : QGis::DEFAULT_SEARCH_RADIUS_MM;
148  double radiusmu = radiusmm * renderContext.scaleFactor() * renderContext.mapToPixel().mapUnitsPerPixel();
149  QgsRectangle filterRect;
150  filterRect.setXMinimum( mapPos.x() - radiusmu );
151  filterRect.setXMaximum( mapPos.x() + radiusmu );
152  filterRect.setYMinimum( mapPos.y() - radiusmu );
153  filterRect.setYMaximum( mapPos.y() + radiusmu );
154 
155  foreach ( QgsMapLayer* layer, mCanvas->layers() )
156  {
157  if ( dynamic_cast<QgsPluginLayer*>( layer ) )
158  {
159  QgsPluginLayer* pluginLayer = static_cast<QgsPluginLayer*>( layer );
160  QList<QgsPluginLayer::IdentifyResult> results = pluginLayer->identify( mapPos, mCanvas->mapSettings() );
161  if ( !results.isEmpty() )
162  {
163  addPluginLayerResults( pluginLayer, results );
164  }
165  }
166  else if ( dynamic_cast<QgsRasterLayer*>( layer ) )
167  {
168  const QgsRasterLayer* rlayer = static_cast<const QgsRasterLayer*>( layer );
169 
170  // Detect ArcGIS Rest MapServer layers
171  QgsDataSourceURI dataSource( rlayer->dataProvider()->dataSourceUri() );
172  QStringList urlParts = dataSource.param( "url" ).split( "/", QString::SkipEmptyParts );
173  int nParts = urlParts.size();
174  if ( nParts > 4 && urlParts[nParts - 1] == "MapServer" && urlParts[nParts - 4] == "services" )
175  {
176  rlayerIds.append( urlParts[nParts - 3] + ":" + urlParts[nParts - 2] );
177  rlayerMap.insert( urlParts[nParts - 2], rlayer->id() );
178  }
179 
180  // Detect geo.admin.ch layers
181  dataSource.setEncodedUri( rlayer->dataProvider()->dataSourceUri() );
182  if ( dataSource.param( "url" ).contains( "geo.admin.ch" ) )
183  {
184  QStringList sublayerIds = dataSource.params( "layers" );
185  rlayerIds.append( sublayerIds );
186  foreach ( const QString& id, sublayerIds )
187  {
188  rlayerMap.insert( id, rlayer->id() );
189  }
190  }
191  }
192  else if ( dynamic_cast<QgsVectorLayer*>( layer ) )
193  {
194  QgsVectorLayer* vlayer = static_cast<QgsVectorLayer*>( layer );
195  if ( vlayer->hasScaleBasedVisibility() &&
196  ( vlayer->minimumScale() > mCanvas->mapSettings().scale() ||
197  vlayer->maximumScale() <= mCanvas->mapSettings().scale() ) )
198  {
199  continue;
200  }
201 
202  QgsFeatureRendererV2* renderer = vlayer->rendererV2();
203  bool filteredRendering = false;
204  if ( renderer && renderer->capabilities() & QgsFeatureRendererV2::ScaleDependent )
205  {
206  // setup scale for scale dependent visibility (rule based)
207  renderer->startRender( renderContext, vlayer->pendingFields() );
208  filteredRendering = renderer->capabilities() & QgsFeatureRendererV2::Filter;
209  }
210 
211  QgsRectangle layerFilterRect = mCanvas->mapSettings().mapToLayerCoordinates( vlayer, filterRect );
212  QgsFeatureIterator fit = vlayer->getFeatures( QgsFeatureRequest( layerFilterRect ).setFlags( QgsFeatureRequest::ExactIntersect ) );
213  QgsFeature feature;
214  while ( fit.nextFeature( feature ) )
215  {
216  if ( filteredRendering && !renderer->willRenderFeature( feature ) )
217  {
218  continue;
219  }
220  addVectorLayerResult( vlayer, feature );
221  }
222  if ( renderer && renderer->capabilities() & QgsFeatureRendererV2::ScaleDependent )
223  {
224  renderer->stopRender( renderContext );
225  }
226  }
227  }
228 
229  // Raster layer query
230  if ( !rlayerIds.isEmpty() )
231  {
232  QUrl identifyUrl( QSettings().value( "vbs/identifyurl", "" ).toString() );
233  identifyUrl.addQueryItem( "geometryType", "esriGeometryPoint" );
234  identifyUrl.addQueryItem( "geometry", QString( "%1,%2" ).arg( worldPos.x(), 0, 'f', 10 ).arg( worldPos.y(), 0, 'f', 10 ) );
235  identifyUrl.addQueryItem( "imageDisplay", QString( "%1,%2,%3" ).arg( mCanvas->width() ).arg( mCanvas->height() ).arg( mCanvas->mapSettings().outputDpi() ) );
236  identifyUrl.addQueryItem( "mapExtent", QString( "%1,%2,%3,%4" ).arg( worldExtent.xMinimum(), 0, 'f', 10 )
237  .arg( worldExtent.yMinimum(), 0, 'f', 10 )
238  .arg( worldExtent.xMaximum(), 0, 'f', 10 )
239  .arg( worldExtent.yMaximum(), 0, 'f', 10 ) );
240  identifyUrl.addQueryItem( "tolerance", "15" );
241  identifyUrl.addQueryItem( "layers", rlayerIds.join( "," ) );
242 
243  QNetworkRequest req( identifyUrl );
244  mRasterIdentifyReply = QgsNetworkAccessManager::instance()->get( req );
245  mRasterIdentifyReply->setProperty( "layerMap", rlayerMap );
246  mTimeoutTimer = new QTimer( mRasterIdentifyReply );
247  mTimeoutTimer->setSingleShot( true );
248  connect( mRasterIdentifyReply, SIGNAL( finished() ), this, SLOT( rasterIdentifyFinished() ) );
249  connect( mTimeoutTimer, SIGNAL( timeout() ), this, SLOT( rasterIdentifyFinished() ) );
250  mTimeoutTimer->start( 4000 );
251  }
252 }

286 void QgsMapIdentifyDialog::addVectorLayerResult( QgsVectorLayer *vLayer, const QgsFeature &feature )
287 {

289  {
290  QTreeWidgetItem* item = new QTreeWidgetItem( QStringList() << vLayer->name() );
291  item->setFirstColumnSpanned( true );
292  QFont font = item->font( 0 );
293  font.setBold( true );
294  item->setFont( 0, font );
295  mTreeWidget->invisibleRootItem()->addChild( item );
296  mLayerTreeItemMap.insert( vLayer->id(), item );
297  item->setExpanded( true );
298  }
299 
300  QString label = vLayer->displayField().isEmpty() ? QString::number( feature.id() ) : QString( "%1 [%2]" ).arg( feature.attribute( vLayer->displayField() ).toString() ).arg( feature.id() );
301  QTreeWidgetItem* item = new QTreeWidgetItem( QStringList() << label );
302  QgsAbstractGeometryV2* geomv2 = feature.geometry()->geometry()->clone();
303  mGeometries.append( geomv2 );
304  item->setData( 0, sGeometryRole, mGeometries.size() - 1 );
305  item->setData( 0, sGeometryCrsRole, vLayer->crs().authid() );
306  mLayerTreeItemMap[vLayer->id()]->addChild( item );
307 
308  if ( vLayer->type() == QgsMapLayer::RedliningLayer )
309  {
310  QStringList attributes = feature.attribute( "attributes" ).toString().split( QRegExp( "&(?!amp;)" ) );
311  for ( const QString& attribute : attributes )
312  {
313  QStringList pair = attribute.split( "=" );
314  item->addChild( new QTreeWidgetItem( pair ) );
315  }
316  }
317  else
318  {
319  QgsAttributes attributes = feature.attributes();
320  for ( int i = 0, n = attributes.size(); i < n; ++i )
321  {
322  QString attribName = vLayer->attributeDisplayName( i );
323  QStringList pair = QStringList() << attribName << attributes.at( i ).toString();
324  item->addChild( new QTreeWidgetItem( pair ) );
325  }
326  }
327  item->setExpanded( true );
328 }

330 void QgsMapIdentifyDialog::rasterIdentifyFinished()
331 {
332  if ( !mRasterIdentifyReply )
333  {
334  return;
335  }
336 
337  QVariantList results;
338  if ( mRasterIdentifyReply->error() == QNetworkReply::NoError && mTimeoutTimer->isActive() )
339  {
340 #if QT_VERSION < QT_VERSION_CHECK(5, 0, 0)
341  results = QJson::Parser().parse( mRasterIdentifyReply->readAll() ).toMap()["results"].toList();
342 #else
343  results = QJsonDocument::fromJson( mRasterIdentifyReply->readAll() ).object().toVariantMap()["results"].toList();
344 #endif
345  }
346 
347  QMap<QString, QVariant> layerMap = mRasterIdentifyReply->property( "layerMap" ).toMap();
348 
349  mRasterIdentifyReply->deleteLater();
350  mRasterIdentifyReply = nullptr;
351  mTimeoutTimer = nullptr;
352 
353  // Add results to tree
354  for ( int i = 0, n = results.size(); i < n; ++i )
355  {
356  QVariantMap result = results[i].toMap();
357  QString layerId = result["layerId"].toString();
358  if ( !layerMap.contains( layerId ) )
359  {
360  continue;
361  }
362  QString qgisLayerId = layerMap[layerId].toString();
363  if ( !mLayerTreeItemMap.contains( layerId ) )
364  {
365  QgsMapLayer* layer = QgsMapLayerRegistry::instance()->mapLayer( qgisLayerId );
366  if ( !layer )
367  {
368  mLayerTreeItemMap.insert( layerId, 0 );
369  }
370  else
371  {
372  QTreeWidgetItem* item = new QTreeWidgetItem( QStringList() << layer->name() );
373  item->setFirstColumnSpanned( true );
374  QFont font = item->font( 0 );
375  font.setBold( true );
376  item->setFont( 0, font );
377  mLayerTreeItemMap.insert( layerId, item );
378  mTreeWidget->invisibleRootItem()->addChild( item );
379  item->setExpanded( true );
380  }
381  }
382  QTreeWidgetItem* parent = mLayerTreeItemMap[layerId];
383  if ( !parent )
384  {
385  continue;
386  }
387  QTreeWidgetItem* resultItem = new QTreeWidgetItem( QStringList() << "" );
388  QgsCoordinateReferenceSystem crs;
389  QgsAbstractGeometryV2* geometryV2 = QgsArcGisRestUtils::parseEsriGeoJSON( result["geometry"].toMap(), result["geometryType"].toString(), false, false, &crs );
390  mGeometries.append( geometryV2 );
391 
392  resultItem->setData( 0, sGeometryRole, mGeometries.size() - 1 );
393  resultItem->setData( 0, sGeometryCrsRole, crs.authid() );
394 
395  resultItem->setFirstColumnSpanned( true );
396  QString displayFieldName = result["displayFieldName"].toString();
397  QVariantMap attributes = result["attributes"].toMap();
398  foreach ( const QString& key, attributes.keys() )
399  {
400  QString value = attributes[key].toString();
401  QTreeWidgetItem* attrItem = new QTreeWidgetItem( QStringList() << key << value );
402  if ( key == displayFieldName )
403  resultItem->setText( 0, value );
404  resultItem->addChild( attrItem );
405  }
406  parent->addChild( resultItem );
407  resultItem->setExpanded( true );
408  }
409 }
