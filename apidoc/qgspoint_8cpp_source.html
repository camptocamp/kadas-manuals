170 QString QgsPoint::toDegreesMinutesSeconds( int thePrecision, const bool useSuffix, const bool padded ) const
171 {
172  //first, limit longitude to -360 to 360 degree range
173  double myWrappedX = fmod( m_x, 360.0 );
174  //next, wrap around longitudes > 180 or < -180 degrees, so that eg "190E" -> "170W"
175  if ( myWrappedX > 180.0 )
176  {
177  myWrappedX = myWrappedX - 360.0;
178  }
179  else if ( myWrappedX < -180.0 )
180  {
181  myWrappedX = myWrappedX + 360.0;
182  }
183 
184  //first, limit latitude to -180 to 180 degree range
185  double myWrappedY = fmod( m_y, 180.0 );
186  //next, wrap around latitudes > 90 or < -90 degrees, so that eg "110S" -> "70N"
187  if ( myWrappedY > 90.0 )
188  {
189  myWrappedY = myWrappedY - 180.0;
190  }
191  else if ( myWrappedY < -90.0 )
192  {
193  myWrappedY = myWrappedY + 180.0;
194  }
195 
196  int myDegreesX = int( qAbs( myWrappedX ) );
197  double myFloatMinutesX = double(( qAbs( myWrappedX ) - myDegreesX ) * 60 );
198  int myIntMinutesX = int( myFloatMinutesX );
199  double mySecondsX = double( myFloatMinutesX - myIntMinutesX ) * 60;
200 
201  int myDegreesY = int( qAbs( myWrappedY ) );
202  double myFloatMinutesY = double(( qAbs( myWrappedY ) - myDegreesY ) * 60 );
203  int myIntMinutesY = int( myFloatMinutesY );
204  double mySecondsY = double( myFloatMinutesY - myIntMinutesY ) * 60;
205 
206  //make sure rounding to specified precision doesn't create seconds >= 60
207  if ( qRound( mySecondsX * pow( 10.0, thePrecision ) ) >= 60 * pow( 10.0, thePrecision ) )
208  {
209  mySecondsX = qMax( mySecondsX - 60, 0.0 );
210  myIntMinutesX++;
211  if ( myIntMinutesX >= 60 )
212  {
213  myIntMinutesX -= 60;
214  myDegreesX++;
215  }
216  }
217  if ( qRound( mySecondsY * pow( 10.0, thePrecision ) ) >= 60 * pow( 10.0, thePrecision ) )
218  {
219  mySecondsY = qMax( mySecondsY - 60, 0.0 );
220  myIntMinutesY++;
221  if ( myIntMinutesY >= 60 )
222  {
223  myIntMinutesY -= 60;
224  myDegreesY++;
225  }
226  }
227 
228  QString myXHemisphere;
229  QString myYHemisphere;
230  QString myXSign;
231  QString myYSign;
232  if ( useSuffix )
233  {
234  myXHemisphere = myWrappedX < 0 ? QObject::tr( "W" ) : QObject::tr( "E" );
235  myYHemisphere = myWrappedY < 0 ? QObject::tr( "S" ) : QObject::tr( "N" );
236  }
237  else
238  {
239  if ( myWrappedX < 0 )
240  {
241  myXSign = QObject::tr( "-" );
242  }
243  if ( myWrappedY < 0 )
244  {
245  myYSign = QObject::tr( "-" );
246  }
247  }
248  //check if coordinate is all zeros for the specified precision, and if so,
249  //remove the sign and hemisphere strings
250  if ( myDegreesX == 0 && myIntMinutesX == 0 && qRound( mySecondsX * pow( 10.0, thePrecision ) ) == 0 )
251  {
252  myXSign = QString();
253  myXHemisphere = QString();
254  }
255  if ( myDegreesY == 0 && myIntMinutesY == 0 && qRound( mySecondsY * pow( 10.0, thePrecision ) ) == 0 )
256  {
257  myYSign = QString();
258  myYHemisphere = QString();
259  }
260  //also remove directional prefix from 180 degree longitudes
261  if ( myDegreesX == 180 && myIntMinutesX == 0 && qRound( mySecondsX * pow( 10.0, thePrecision ) ) == 0 )
262  {
263  myXHemisphere = QString();
264  }
265  //pad minutes with leading digits if required
266  QString myMinutesX = padded ? QString( "%1" ).arg( myIntMinutesX, 2, 10, QChar( '0' ) ) : QString::number( myIntMinutesX );
267  QString myMinutesY = padded ? QString( "%1" ).arg( myIntMinutesY, 2, 10, QChar( '0' ) ) : QString::number( myIntMinutesY );
268  //pad seconds with leading digits if required
269  int digits = 2 + ( thePrecision == 0 ? 0 : 1 + thePrecision ); //1 for decimal place if required
270  QString myStrSecondsX = padded ? QString( "%1" ).arg( mySecondsX, digits, 'f', thePrecision, QChar( '0' ) ) : QString::number( mySecondsX, 'f', thePrecision );
271  QString myStrSecondsY = padded ? QString( "%1" ).arg( mySecondsY, digits, 'f', thePrecision, QChar( '0' ) ) : QString::number( mySecondsY, 'f', thePrecision );
272 
273  QString rep = myXSign + QString::number( myDegreesX ) + QChar( 176 ) +
274  myMinutesX + QChar( 0x2032 ) +
275  myStrSecondsX + QChar( 0x2033 ) +
276  myXHemisphere + QString( ", " ) +
277  myYSign + QString::number( myDegreesY ) + QChar( 176 ) +
278  myMinutesY + QChar( 0x2032 ) +
279  myStrSecondsY + QChar( 0x2033 ) +
280  myYHemisphere;
281  return rep;
282 }

424 int QgsPoint::onSegment( const QgsPoint& a, const QgsPoint& b ) const
425 {
426  //algorithm from 'graphics GEMS', A. Paeth: 'A Fast 2D Point-on-line test'
427  if (
428  qAbs(( b.y() - a.y() ) *( m_x - a.x() ) - ( m_y - a.y() ) *( b.x() - a.x() ) )
429  >= qMax( qAbs( b.x() - a.x() ), qAbs( b.y() - a.y() ) )
430  )
431  {
432  return 0;
433  }
434  if (( b.x() < a.x() && a.x() < m_x ) || ( b.y() < a.y() && a.y() < m_y ) )
435  {
436  return 1;
437  }
438  if (( m_x < a.x() && a.x() < b.x() ) || ( m_y < a.y() && a.y() < b.y() ) )
439  {
440  return 1;
441  }
442  if (( a.x() < b.x() && b.x() < m_x ) || ( a.y() < b.y() && b.y() < m_y ) )
443  {
444  return 3;
445  }
446  if (( m_x < b.x() && b.x() < a.x() ) || ( m_y < b.y() && b.y() < a.y() ) )
447  {
448  return 3;
449  }
450 
451  return 2;
452 }

454 double QgsPoint::sqrDistToSegment( double x1, double y1, double x2, double y2, QgsPoint& minDistPoint, double epsilon ) const
455 {
456  double nx, ny; //normal vector
457 
458  nx = y2 - y1;
459  ny = -( x2 - x1 );
460 
461  double t;
462  t = ( m_x * ny - m_y * nx - x1 * ny + y1 * nx ) / (( x2 - x1 ) * ny - ( y2 - y1 ) * nx );
463 
464  if ( t < 0.0 )
465  {
466  minDistPoint.setX( x1 );
467  minDistPoint.setY( y1 );
468  }
469  else if ( t > 1.0 )
470  {
471  minDistPoint.setX( x2 );
472  minDistPoint.setY( y2 );
473  }
474  else
475  {
476  minDistPoint.setX( x1 + t *( x2 - x1 ) );
477  minDistPoint.setY( y1 + t *( y2 - y1 ) );
478  }
479 
480  double dist = sqrDist( minDistPoint );
481  //prevent rounding errors if the point is directly on the segment
482  if ( qgsDoubleNear( dist, 0.0, epsilon ) )
483  {
484  minDistPoint.setX( m_x );
485  minDistPoint.setY( m_y );
486  return 0.0;
487  }
488  return dist;
489 }
