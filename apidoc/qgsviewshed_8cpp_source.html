32 static inline double geoToPixelX( double gtrans[6], double x, double y )
33 {
34  return ( -gtrans[0] * gtrans[5] + gtrans[2] * gtrans[3] - gtrans[2] * y + gtrans[5] * x ) / ( gtrans[1] * gtrans[5] - gtrans[2] * gtrans[4] );
35 }
36 
37 static inline double geoToPixelY( double gtrans[6], double x, double y )
38 {
39  return ( -gtrans[0] * gtrans[4] + gtrans[1] * gtrans[3] - gtrans[1] * y + gtrans[4] * x ) / ( gtrans[2] * gtrans[4] - gtrans[1] * gtrans[5] );
40 }

42 static inline double pixelToGeoX( double gtrans[6], double px, double py )
43 {
44  return gtrans[0] + px * gtrans[1] + py * gtrans[2];
45 }

52 bool QgsViewshed::computeViewshed( const QString &inputFile, const QString &outputFile, const QString &outputFormat, QgsPoint observerPos, const QgsCoordinateReferenceSystem &observerPosCrs, double observerHeight, double targetHeight, bool heightRelToTerr, double radius, const QGis::UnitType distanceElevUnit, const QVector<QgsPoint> &filterRegion, bool displayVisible, int accuracyFactor, QProgressDialog *progress )
53 {
54  // Open input file
55  GDALDatasetH inputDataset = GDALOpen( inputFile.toLocal8Bit().data(), GA_ReadOnly );
56  if ( inputDataset == 0 )
57  {
58  QgsDebugMsg( "Failed to open input dataset" );
59  return false;
60  }
61 
62  // Transform positions and measurements to dataset CRS
63  QgsCoordinateReferenceSystem datasetCrs( QString( GDALGetProjectionRef( inputDataset ) ) );
64  if ( !datasetCrs.isValid() )
65  {
66  QgsDebugMsg( "Could not determine input dataset CRS" );
67  GDALClose( inputDataset );
68  return false;
69  }
70  QgsCoordinateTransform ct( observerPosCrs, datasetCrs );
71  observerPos = ct.transform( observerPos );
72  if ( datasetCrs.mapUnits() != distanceElevUnit )
73  {
74  observerHeight *= QGis::fromUnitToUnitFactor( distanceElevUnit, datasetCrs.mapUnits() );
75  targetHeight *= QGis::fromUnitToUnitFactor( distanceElevUnit, datasetCrs.mapUnits() );
76  radius *= QGis::fromUnitToUnitFactor( distanceElevUnit, datasetCrs.mapUnits() );
77  }
78 
79 
80  // Open input band
81  GDALRasterBandH inputBand = GDALGetRasterBand( inputDataset, 1 );
82  if ( inputBand == NULL )
83  {
84  GDALClose( inputDataset );
85  QgsDebugMsg( "Failed to open input dataset band 1" );
86  return false;
87  }
88  float noDataValue = GDALGetRasterNoDataValue( inputBand, NULL );
89 
90 
91  // Compute window of raster to read
92  double gtrans[6] = {};
93  if ( GDALGetGeoTransform( inputDataset, &gtrans[0] ) != CE_None )
94  {
95  QgsDebugMsg( "Failed to query input dataset geotransform" );
96  return false;
97  }
98  int terWidth = GDALGetRasterXSize( inputDataset );
99  int terHeight = GDALGetRasterYSize( inputDataset );
100 
101  int obs[2] =
102  {
103  qRound( geoToPixelX( gtrans, observerPos.x(), observerPos.y() ) ),
104  qRound( geoToPixelY( gtrans, observerPos.x(), observerPos.y() ) )
105  };
106  double earthRadius = 6370000;
107  if ( datasetCrs.mapUnits() != QGis::Meters )
108  {
109  earthRadius *= QGis::fromUnitToUnitFactor( QGis::Meters, datasetCrs.mapUnits() );
110  }
111 
112  QList<QgsPoint> cornerPoints = QList<QgsPoint>()
113  << QgsPoint( observerPos.x() - radius, observerPos.y() - radius )
114  << QgsPoint( observerPos.x() + radius, observerPos.y() - radius )
115  << QgsPoint( observerPos.x() + radius, observerPos.y() + radius )
116  << QgsPoint( observerPos.x() - radius, observerPos.y() + radius );
117  int colStart = std::numeric_limits<int>::max();
118  int rowStart = std::numeric_limits<int>::max();
119  int colEnd = -std::numeric_limits<int>::max();
120  int rowEnd = -std::numeric_limits<int>::max();
121  foreach ( const QgsPoint& p, cornerPoints )
122  {
123  double x = geoToPixelX( gtrans, p.x(), p.y() );
124  double y = geoToPixelY( gtrans, p.x(), p.y() );
125  colStart = qMin( colStart, qFloor( x ) );
126  colEnd = qMax( colEnd, qCeil( x ) );
127  rowStart = qMin( rowStart, qFloor( y ) );
128  rowEnd = qMax( rowEnd, qCeil( y ) );
129  }
130  colStart = qMax( 0, colStart );
131  colEnd = qMin( terWidth - 1, colEnd );
132  rowStart = qMax( 0, rowStart );
133  rowEnd = qMin( terHeight - 1, rowEnd );
134  int hmapWidth = colEnd - colStart + 1;
135  int hmapHeight = rowEnd - rowStart + 1;
136  QPolygon filterPoly;
137  for ( int i = 0, n = filterRegion.size(); i < n; ++i )
138  {
139  QgsPoint p = ct.transform( filterRegion[i] );
140  filterPoly.append( QPoint( qRound( geoToPixelX( gtrans, p.x(), p.y() ) ), qRound( geoToPixelY( gtrans, p.x(), p.y() ) ) ) );
141  }
142 
143  if ( obs[0] < colStart || obs[0] > colEnd || obs[1] < rowStart || obs[1] > rowEnd )
144  {
145  GDALClose( inputDataset );
146  QgsDebugMsg( "Observer pos is outside vieweshed area, reprojection distortion?" );
147  return false;
148  }
149 
150  int scaledHmapHeight = hmapHeight / accuracyFactor;
151  int scaledHmapWidth = hmapWidth / accuracyFactor;
152 
153  // Read input heightmap
154  // Allow at most 1GB allocated
155  if ( scaledHmapWidth * scaledHmapHeight * sizeof( float ) > 1073741824 )
156  {
157  GDALClose( inputDataset );
158  QgsDebugMsg( "Too much memory required" );
159  return false;
160  }
161  QVector<float> heightmap( scaledHmapWidth * scaledHmapHeight, noDataValue );
162  GDALRasterIOExtraArg rioargs;
163  INIT_RASTERIO_EXTRA_ARG( rioargs );
164  rioargs.eResampleAlg = GRIORA_Average;
165  CPLErr err = GDALRasterIOEx( inputBand, GF_Read, colStart, rowStart, hmapWidth, hmapHeight, heightmap.data(), scaledHmapWidth, scaledHmapHeight, GDT_Float32, 0, 0, &rioargs );
166  if ( err != CE_None )
167  {
168  GDALClose( inputDataset );
169  QgsDebugMsg( "Failed to fetch raster pixels" );
170  return false;
171  }
172 
173  // Adjust for reduced resolution
174  gtrans[1] *= accuracyFactor;
175  gtrans[2] *= accuracyFactor;
176  gtrans[4] *= accuracyFactor;
177  gtrans[5] *= accuracyFactor;
178  colStart /= accuracyFactor;
179  colEnd /= accuracyFactor;
180  rowStart /= accuracyFactor;
181  rowEnd /= accuracyFactor;
182  obs[0] /= accuracyFactor;
183  obs[1] /= accuracyFactor;
184  hmapWidth = scaledHmapWidth;
185  hmapHeight = scaledHmapHeight;
186  for ( int i = 0, n = filterPoly.size(); i < n; ++i )
187  {
188  filterPoly[i] = QPoint( filterPoly[i].x() / accuracyFactor, filterPoly[i].y() / accuracyFactor );
189  }
190 
191  // Prepare output
192  GDALDriverH outputDriver = GDALGetDriverByName( outputFormat.toLocal8Bit().data() );
193  if ( outputDriver == 0 )
194  {
195  GDALClose( inputDataset );
196  QgsDebugMsg( "Failed to get driver for output" );
197  return false;
198  }
199  if ( !CSLFetchBoolean( GDALGetMetadata( outputDriver, NULL ), GDAL_DCAP_CREATE, false ) )
200  {
201  GDALClose( inputDataset );
202  QgsDebugMsg( "Driver for output does not support creation" );
203  return false;
204  }
205  char **papszOptions = CSLSetNameValue( 0, "COMPRESS", "LZW" );
206  GDALDatasetH outputDataset = GDALCreate( outputDriver, outputFile.toLocal8Bit().data(), hmapWidth, hmapHeight, 1, GDT_Byte, papszOptions );
207  if ( outputDataset == NULL )
208  {
209  GDALClose( inputDataset );
210  QgsDebugMsg( "Failed to open output dataset" );
211  return false;
212  }
213 
214  double outgtrans[6];
215  std::memcpy( outgtrans, gtrans, sizeof( gtrans ) );
216 
217  // Shift for origin of window
218  outgtrans[0] += colStart * outgtrans[1] + rowStart * outgtrans[2];
219  outgtrans[3] += colStart * outgtrans[4] + rowStart * outgtrans[5];
220 
221  GDALSetGeoTransform( outputDataset, outgtrans );
222  GDALSetProjection( outputDataset, GDALGetProjectionRef( inputDataset ) );
223 
224  GDALRasterBandH outputBand = GDALGetRasterBand( outputDataset, 1 );
225  if ( outputBand == 0 )
226  {
227  GDALClose( inputDataset );
228  GDALClose( outputDataset );
229  QgsDebugMsg( "Failed to get output dataset band 1" );
230  return false;
231  }
232  GDALSetRasterNoDataValue( outputBand, 255 * !displayVisible );
233 
234  // Offset observer elevation by position at point
235  if ( heightRelToTerr )
236  observerHeight += heightmap[( obs[1] - rowStart ) * hmapWidth + ( obs[0] - colStart )];
237 
238 
239  // Compute viewshed
240  int roi = .5 * qMin( hmapWidth, hmapHeight );
241  if ( progress )
242  {
243  progress->setRange( 0, 8 * roi );
244  }
245  QVector<unsigned char> viewshed( hmapWidth * hmapHeight, 255 * !displayVisible );
246  for ( int radiusNumber = 0; radiusNumber < 8 * roi; ++radiusNumber )
247  {
248  if ( progress )
249  {
250  if ( progress->wasCanceled() )
251  {
252  QgsDebugMsg( "Canceled" );
253  GDALClose( inputDataset );
254  GDALClose( outputDataset );
255  return false;
256  }
257  progress->setValue( radiusNumber );
258  }
259  int target[2];
260  if ( radiusNumber <= roi )
261  {
262  target[0] = obs[0] + roi;
263  target[1] = obs[1] + radiusNumber;
264  }
265  else if ( radiusNumber <= 3 * roi )
266  {
267  target[0] = obs[0] + 2 * roi - radiusNumber;
268  target[1] = obs[1] + roi;
269  }
270  else if ( radiusNumber <= 5 * roi )
271  {
272  target[0] = obs[0] - roi;
273  target[1] = obs[1] + 4 * roi - radiusNumber;
274  }
275  else if ( radiusNumber <= 7 * roi )
276  {
277  target[0] = obs[0] + radiusNumber - 6 * roi;
278  target[1] = obs[1] - roi;
279  }
280  else if ( radiusNumber < 8 * roi )
281  {
282  target[0] = obs[0] + roi;
283  target[1] = obs[1] + radiusNumber - 8 * roi;
284  }
285  else
286  {
287  break; // All terrain points processed.
288  }
289 
290  // Line of sight from observer to target.
291  int delta[2] = {target[0] - obs[0], target[1] - obs[1]};
292  int inciny = qAbs( delta[0] ) < qAbs( delta[1] );
293 
294  // Step along coord (X or Y) that varies most from observer to target.
295  // That coord is inciny. Slope is how fast the other coord varies.
296  double slope = ( double ) delta[1 - inciny] / ( double ) delta[inciny];
297  int step = delta[inciny] > 0 ? 1 : -1;
298  double horizon_slope = -99999; // Slope (in vertical plane) to horizon so far.
299 
300  // i = 0 would be the observer, which is always visible.
301  for ( int i = step; true; i += step )
302  {
303  int p[2];
304  p[inciny] = obs[inciny] + i;
305 
306  if ( i * slope > 0 )
307  {
308  p[1 - inciny] = obs[1 - inciny] + int( qCeil( i * slope - 0.5 ) );
309  }
310  else
311  {
312  p[1 - inciny] = obs[1 - inciny] + int( qFloor( i * slope + 0.5 ) );
313  }
314 
315  if ( p[0] < colStart || p[0] > colEnd || p[1] < rowStart || p[1] > rowEnd )
316  {
317  break;
318  }
319 
320  //Is the point in the outside of the viewshed area?
321  double dx = qAbs( p[0] - obs[0] ), dy = qAbs( p[1] - obs[1] );
322  if ( !( dx <= roi && dy <= roi && dx * dx + dy * dy <= double( roi ) * double( roi ) ) )
323  {
324  break;
325  }
326  if ( !filterPoly.isEmpty() && !filterPoly.containsPoint( QPoint( p[0], p[1] ), Qt::OddEvenFill ) )
327  {
328  continue;
329  }
330 
331  int idx = ( p[1] - rowStart ) * hmapWidth + ( p[0] - colStart );
332  if ( idx >= heightmap.size() )
333  {
334  continue;
335  }
336  float pElev = heightmap[idx];
337  if ( pElev == noDataValue )
338  {
339  continue;
340  }
341 
342  // Earth curvature correction
343  double pGeoX = pixelToGeoX( gtrans, p[0], p[1] );
344  double pGeoY = pixelToGeoY( gtrans, p[0], p[1] );
345  double geoDistSqr = ( observerPos.x() - pGeoX ) * ( observerPos.x() - pGeoX ) + ( observerPos.y() - pGeoY ) * ( observerPos.y() - pGeoY );
346  // http://www.swisstopo.admin.ch/internet/swisstopo/de/home/topics/survey/faq/curvature.html
347  pElev -= 0.87 * geoDistSqr / ( 2 * earthRadius );
348 
349  // Update the slope if the current slope is greater than the old one
350  double s = double( pElev - observerHeight ) / double( qAbs( p[inciny] - obs[inciny] ) );
351  horizon_slope = qMax( horizon_slope, s );
352 
353  double horizon_alt = observerHeight + horizon_slope * qAbs( p[inciny] - obs[inciny] );
354  double tHeight = targetHeight;
355  if ( heightRelToTerr )
356  tHeight += pElev;
357  if ( tHeight >= horizon_alt )
358  {
359  viewshed[( p[1] - rowStart ) * hmapWidth + ( p[0] - colStart )] = 255;
360  }
361  else
362  {
363  viewshed[( p[1] - rowStart ) * hmapWidth + ( p[0] - colStart )] = 0;
364  }
365  }
366  }
367  // The observer is always visible from itself
368  viewshed[( obs[1] - rowStart ) * hmapWidth + ( obs[0] - colStart )] = 255;
369 
370 
371  // Write output
372  err = GDALRasterIO( outputBand, GF_Write, 0, 0, hmapWidth, hmapHeight, viewshed.data(), hmapWidth, hmapHeight, GDT_Byte, 0, 0 );
373  GDALClose( inputDataset );
374  GDALClose( outputDataset );
375  if ( err != CE_None )
376  {
377  QgsDebugMsg( "Failed to write to output dataset" );
378  return false;
379  }
380  return true;
381 }
