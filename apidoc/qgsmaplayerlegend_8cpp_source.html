74 static QList<int> _makeNodeOrder( QgsLayerTreeLayer* nodeLayer )
75 {
76  if ( !nodeLayer->layer() || !nodeLayer->layer()->legend() )
77  {
78  QgsDebugMsg( "Legend node order manipulation is invalid without existing legend" );
79  return QList<int>();
80  }
81 
82  int numNodes = _originalLegendNodeCount( nodeLayer );
83 
84  QList<int> order;
85  for ( int i = 0; i < numNodes; ++i )
86  order << i;
87  return order;

90 QList<int> QgsMapLayerLegendUtils::legendNodeOrder( QgsLayerTreeLayer* nodeLayer )
91 {
92  QString orderStr = nodeLayer->customProperty( "legend/node-order" ).toString();
93 
94  if ( orderStr.isEmpty() )
95  return _makeNodeOrder( nodeLayer );
96 
97  if ( orderStr == "empty" )
98  return QList<int>();
99 
100  int numNodes = _originalLegendNodeCount( nodeLayer );
101 
102  QList<int> lst;
103  foreach ( QString item, orderStr.split( "," ) )
104  {
105  bool ok;
106  int id = item.toInt( &ok );
107  if ( !ok || id < 0 || id >= numNodes )
108  return _makeNodeOrder( nodeLayer );
109 
110  lst << id;
111  }
112 
113  return lst;
114 }

137 void QgsMapLayerLegendUtils::applyLayerNodeProperties( QgsLayerTreeLayer* nodeLayer, QList<QgsLayerTreeModelLegendNode*>& nodes )
138 {
139  // handle user labels
140  int i = 0;
141  foreach ( QgsLayerTreeModelLegendNode* legendNode, nodes )
142  {
143  QString userLabel = QgsMapLayerLegendUtils::legendNodeUserLabel( nodeLayer, i++ );
144  if ( !userLabel.isNull() )
145  legendNode->setUserLabel( userLabel );
146  }
147 
148  // handle user order of nodes
149  if ( QgsMapLayerLegendUtils::hasLegendNodeOrder( nodeLayer ) )
150  {
151  QList<int> order = QgsMapLayerLegendUtils::legendNodeOrder( nodeLayer );
152 
153  QList<QgsLayerTreeModelLegendNode*> newOrder;
154  QSet<int> usedIndices;
155  foreach ( int idx, order )
156  {
157  if ( usedIndices.contains( idx ) )
158  {
159  QgsDebugMsg( "invalid node order. ignoring." );
160  return;
161  }
162 
163  newOrder << nodes[idx];
164  usedIndices << idx;
165  }
166 
167  // delete unused nodes
168  for ( int i = 0; i < nodes.count(); ++i )
169  {
170  if ( !usedIndices.contains( i ) )
171  delete nodes[i];
172  }
173 
174  nodes = newOrder;
175  }
176 
177 }

188 QList<QgsLayerTreeModelLegendNode*> QgsDefaultVectorLayerLegend::createLayerTreeModelLegendNodes( QgsLayerTreeLayer* nodeLayer )
189 {
190  QList<QgsLayerTreeModelLegendNode*> nodes;
191 
192  QgsFeatureRendererV2* r = mLayer->rendererV2();
193  if ( !r )
194  return nodes;
195 
196  if ( nodeLayer->customProperty( "showFeatureCount", 0 ).toBool() )
197  mLayer->countSymbolFeatures();
198 
199  QSettings settings;
200  if ( settings.value( "/Qgis/showLegendClassifiers", false ).toBool() && !r->legendClassificationAttribute().isEmpty() )
201  {
202  nodes.append( new QgsSimpleLegendNode( nodeLayer, r->legendClassificationAttribute() ) );
203  }
204 
205  QgsLegendSymbolListV2 legendSymbolList = r->legendSymbolItemsV2();
206 
207  //check if the layer is a label layer only
208  bool labelLayer = true;
209  QgsLegendSymbolListV2::const_iterator it = legendSymbolList.constBegin();
210  for ( ; it != legendSymbolList.constEnd(); ++it )
211  {
212  if ( !legendSymbolIsEmpty( *it ) )
213  {
214  labelLayer = false;
215  }
216  }
217 
218  if ( mLayer->customProperty( "labeling/enabled", "false" ).toString().compare( "false", Qt::CaseInsensitive ) == 0 )
219  {
220  labelLayer = false;
221  }
222 
223  if ( labelLayer )
224  {
225  nodes.append( new QgsVectorLabelLegendNode( nodeLayer ) );
226  }
227  else
228  {
229  for ( int i = 0; i < legendSymbolList.size(); ++i )
230  {
231  nodes.append( new QgsSymbolV2LegendNode( nodeLayer, legendSymbolList.at( i ) ) );
232  }
233  }
234 
235  if ( nodes.count() == 1 && nodes[0]->data( Qt::EditRole ).toString().isEmpty() )
236  nodes[0]->setEmbeddedInParent( true );
237 
238  return nodes;
239 }

241 bool QgsDefaultVectorLayerLegend::legendSymbolIsEmpty( const QgsLegendSymbolItemV2& symbolItem )
242 {
243  //symbol has dedicated legend symbol
244  if ( symbolItem.legendSymbol() )
245  {
246  return false;
247  }
248 
249  QgsSymbolV2* symbol = symbolItem.symbol();
250  if ( !symbol )
251  {
252  return true;
253  }
254 
255  if ( symbol->symbolLayerCount() > 1 )
256  {
257  return false;
258  }
259 
260  if ( symbol->type() == QgsSymbolV2::Marker )
261  {
262  QgsSimpleMarkerSymbolLayerV2* simpleMarker = dynamic_cast<QgsSimpleMarkerSymbolLayerV2*>( symbol->symbolLayer( 0 ) );
263  if ( !simpleMarker )
264  {
265  return false;
266  }
267 
268  if ( simpleMarker-> size() <= 0 || ( simpleMarker->fillColor().alpha() <= 0 && simpleMarker->outlineColor().alpha() <= 0 ) )
269  {
270  return true;
271  }
272  }
273  else if ( symbol->type() == QgsSymbolV2::Line )
274  {
275  QgsSimpleLineSymbolLayerV2* simpleLine = dynamic_cast<QgsSimpleLineSymbolLayerV2*>( symbol->symbolLayer( 0 ) );
276  if ( !simpleLine )
277  {
278  return false;
279  }
280 
281  if ( simpleLine->width() <= 0 || simpleLine->color().alpha() <= 0 )
282  {
283  return true;
284  }
285  }
286  else if ( symbol->type() == QgsSymbolV2::Fill )
287  {
288  QgsSimpleFillSymbolLayerV2* simpleFill = dynamic_cast<QgsSimpleFillSymbolLayerV2*>( symbol->symbolLayer( 0 ) );
289  if ( !simpleFill )
290  {
291  return false;
292  }
293 
294  if ( simpleFill->fillColor().alpha() <= 0 && simpleFill->outlineColor().alpha() <= 0 )
295  {
296  return true;
297  }
298  }
299 
300  return false;
301 }

312 QList<QgsLayerTreeModelLegendNode*> QgsDefaultRasterLayerLegend::createLayerTreeModelLegendNodes( QgsLayerTreeLayer* nodeLayer )
313 {
314  QList<QgsLayerTreeModelLegendNode*> nodes;
315 
316  // temporary solution for WMS. Ideally should be done with a delegate.
317  if ( mLayer->dataProvider()->supportsLegendGraphic() )
318  {
319  nodes << new QgsWMSLegendNode( nodeLayer );
320  }
321 
322  //for palletted and multiband raster symbolisation, we only show an icon instead of all the color entries
323  QgsRasterRenderer* rasterRenderer = mLayer->renderer();
324  if ( !rasterRenderer )
325  {
326  return nodes;
327  }
328 
329  if ( rasterRenderer->type() == "paletted" )
330  {
331  nodes << new QgsSimpleLegendNode( nodeLayer, QString(), QImage( QgsApplication::iconsPath() + "paletted_raster.png" ) );
332  return nodes;
333  }
334  else if ( rasterRenderer->type() == "multibandcolor" )
335  {
336  nodes << new QgsSimpleLegendNode( nodeLayer, QString(), QImage( QgsApplication::iconsPath() + "multibandcolor_raster.png" ) );
337  return nodes;
338  }
339 
340  QgsLegendColorList rasterItemList = mLayer->legendSymbologyItems();
341  if ( rasterItemList.count() == 0 )
342  return nodes;
343 
344  // Paletted raster may have many colors, for example UInt16 may have 65536 colors
345  // and it is very slow, so we limit max count
346  int count = 0;
347  int max_count = 1000;
348 
349  for ( QgsLegendColorList::const_iterator itemIt = rasterItemList.constBegin();
350  itemIt != rasterItemList.constEnd(); ++itemIt, ++count )
351  {
352  nodes << new QgsRasterSymbolLegendNode( nodeLayer, itemIt->second, itemIt->first );
353 
354  if ( count == max_count )
355  {
356  QString label = tr( "following %1 items\nnot displayed" ).arg( rasterItemList.size() - max_count );
357  nodes << new QgsSimpleLegendNode( nodeLayer, label );
358  break;
359  }
360  }
361 
362  return nodes;
363 }
