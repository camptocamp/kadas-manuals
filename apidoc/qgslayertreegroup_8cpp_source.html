99 void QgsLayerTreeGroup::insertChildNodes( int index, QList<QgsLayerTreeNode*> nodes )
100 {
101  QgsLayerTreeNode* meChild = 0;
102  if ( mMutuallyExclusive && mMutuallyExclusiveChildIndex >= 0 && mMutuallyExclusiveChildIndex < mChildren.count() )
103  meChild = mChildren[mMutuallyExclusiveChildIndex];
104 
105  // low-level insert
106  insertChildrenPrivate( index, nodes );
107 
108  if ( mMutuallyExclusive )
109  {
110  if ( meChild )
111  {
112  // the child could have change its index - or the new children may have been also set as visible
113  mMutuallyExclusiveChildIndex = mChildren.indexOf( meChild );
114  }
115  else if ( mChecked == Qt::Checked )
116  {
117  // we have not picked a child index yet, but we should pick one now
118  // ... so pick the first one from the newly added
119  if ( index == -1 )
120  index = mChildren.count() - nodes.count(); // get real insertion index
121  mMutuallyExclusiveChildIndex = index;
122  }
123  updateChildVisibilityMutuallyExclusive();
124  }
125 
126  updateVisibilityFromChildren();
127 }

251 QgsLayerTreeGroup* QgsLayerTreeGroup::readXML( QDomElement& element )
252 {
253  if ( element.tagName() != "layer-tree-group" )
254  return 0;
255 
256  QString name = element.attribute( "name" );
257  bool isExpanded = ( element.attribute( "expanded", "1" ) == "1" );
258  Qt::CheckState checked = QgsLayerTreeUtils::checkStateFromXml( element.attribute( "checked" ) );
259  bool isMutuallyExclusive = element.attribute( "mutually-exclusive", "0" ) == "1";
260  int mutuallyExclusiveChildIndex = element.attribute( "mutually-exclusive-child", "-1" ).toInt();
261 
262  QgsLayerTreeGroup* groupNode = new QgsLayerTreeGroup( name, checked );
263  groupNode->setExpanded( isExpanded );
264 
265  groupNode->readCommonXML( element );
266 
267  groupNode->readChildrenFromXML( element );
268 
269  groupNode->setTitle( element.attribute( "title" ) );
270  groupNode->setAbstract( element.attribute( "abstract" ) );
271  groupNode->setWMSPublishLegend( element.attribute( "wmsPublishLegend", "1" ) == "1" );
272  groupNode->setWMSPublishMetadata( element.attribute( "wmsPublishMetadata", "1" ) == "1" );
273  groupNode->setWMSCheckable( element.attribute( "wmsCheckable", "1" ) == "1" );
274  groupNode->setIsMutuallyExclusive( isMutuallyExclusive, mutuallyExclusiveChildIndex );
275  groupNode->setVisible( checked );
276  return groupNode;
277 }

393 void QgsLayerTreeGroup::setIsMutuallyExclusive( bool enabled, int initialChildIndex )
394 {
395  mMutuallyExclusive = enabled;
396  mMutuallyExclusiveChildIndex = initialChildIndex;
397 
398  if ( !enabled )
399  {
400  updateVisibilityFromChildren();
401  return;
402  }
403 
404  if ( mMutuallyExclusiveChildIndex < 0 || mMutuallyExclusiveChildIndex >= mChildren.count() )
405  {
406  // try to use first checked index
407  int index = 0;
408  Q_FOREACH ( QgsLayerTreeNode* child, mChildren )
409  {
410  if ( _nodeIsChecked( child ) )
411  {
412  mMutuallyExclusiveChildIndex = index;
413  break;
414  }
415  index++;
416  }
417  }
418 
419  updateChildVisibilityMutuallyExclusive();
420 }

443 void QgsLayerTreeGroup::nodeVisibilityChanged( QgsLayerTreeNode* node )
444 {
445  int childIndex = mChildren.indexOf( node );
446  if ( childIndex == -1 )
447  return; // not a direct child - ignore
448 
449  if ( mMutuallyExclusive )
450  {
451  if ( _nodeIsChecked( node ) )
452  mMutuallyExclusiveChildIndex = childIndex;
453 
454  // we need to update this node's check status in two cases:
455  // 1. it was unchecked and a child node got checked
456  // 2. it was checked and the only checked child got unchecked
457  updateVisibilityFromChildren();
458 
459  // we also need to make sure there is only one child node checked
460  updateChildVisibilityMutuallyExclusive();
461  }
462  else
463  {
464  updateVisibilityFromChildren();
465  }
466 }

468 void QgsLayerTreeGroup::updateVisibilityFromChildren()
469 {
470  if ( mChangingChildVisibility )
471  return;
472 
473  if ( mChildren.count() == 0 )
474  return;
475 
476  if ( mMutuallyExclusive )
477  {
478  // if in mutually exclusive mode, our check state depends only on the check state of the chosen child index
479 
480  if ( mMutuallyExclusiveChildIndex < 0 || mMutuallyExclusiveChildIndex >= mChildren.count() )
481  return;
482 
483  Qt::CheckState meChildState = _nodeIsChecked( mChildren[mMutuallyExclusiveChildIndex] ) ? Qt::Checked : Qt::Unchecked;
484 
485  setVisible( meChildState );
486  return;
487  }
488 
489  bool hasVisible = false, hasHidden = false;
490 
491  foreach ( QgsLayerTreeNode* child, mChildren )
492  {
493  if ( QgsLayerTree::isLayer( child ) )
494  {
495  bool layerVisible = QgsLayerTree::toLayer( child )->isVisible() == Qt::Checked;
496  if ( layerVisible ) hasVisible = true;
497  if ( !layerVisible ) hasHidden = true;
498  }
499  else if ( QgsLayerTree::isGroup( child ) )
500  {
501  Qt::CheckState state = QgsLayerTree::toGroup( child )->isVisible();
502  if ( state == Qt::Checked || state == Qt::PartiallyChecked ) hasVisible = true;
503  if ( state == Qt::Unchecked || state == Qt::PartiallyChecked ) hasHidden = true;
504  }
505  }
506 
507  Qt::CheckState newState;
508  if ( hasVisible && !hasHidden )
509  newState = Qt::Checked;
510  else if ( hasHidden && !hasVisible )
511  newState = Qt::Unchecked;
512  else
513  newState = Qt::PartiallyChecked;
514 
515  setVisible( newState );
516 }

518 void QgsLayerTreeGroup::updateChildVisibilityMutuallyExclusive()
519 {
520  if ( mChildren.isEmpty() )
521  return;
522 
523  mChangingChildVisibility = true; // guard against running again setVisible() triggered from children
524 
525  int index = 0;
526  Q_FOREACH ( QgsLayerTreeNode* child, mChildren )

534  }
