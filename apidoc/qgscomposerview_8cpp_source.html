113  case QgsComposerView::AddAttributeTable:

130 void QgsComposerView::mousePressEvent( QMouseEvent* e )
131 {
132  if ( mRubberBandItem || mRubberBandLineItem || mKeyPanning || mMousePanning || mToolPanning || mMovingItemContent )
133  {
134  //ignore clicks during certain operations
135  return;
136  }
137 
138  if ( composition()->selectionHandles()->isDragging() || composition()->selectionHandles()->isResizing() )
139  {
140  //ignore clicks while dragging/resizing items
141  return;
142  }
143 
144  QPointF scenePoint = mapToScene( e->pos() );
145  QPointF snappedScenePoint = composition()->snapPointToGrid( scenePoint );
146  mMousePressStartPos = e->pos();
147 
148  if ( e->button() == Qt::RightButton )
149  {
150  //ignore right clicks for now
151  //TODO - show context menu
152  return;
153  }
154  else if ( e->button() == Qt::MidButton )
155  {
156  //pan composer with middle button
157  mMousePanning = true;
158  mMouseLastXY = e->pos();
159  //lock cursor to closed hand cursor
160  composition()->setPreventCursorChange( true );
161  viewport()->setCursor( Qt::ClosedHandCursor );
162  return;
163  }
164 
165  switch ( mCurrentTool )
166  {
167  //select/deselect items and pass mouse event further
168  case Select:
169  {
170  //check if we are clicking on a selection handle
171  if ( composition()->selectionHandles()->isVisible() )
172  {
173  //selection handles are being shown, get mouse action for current cursor position
174  QgsComposerMouseHandles::MouseAction mouseAction = composition()->selectionHandles()->mouseActionForScenePos( scenePoint );
175 
176  if ( mouseAction != QgsComposerMouseHandles::MoveItem && mouseAction != QgsComposerMouseHandles::NoAction && mouseAction != QgsComposerMouseHandles::SelectItem )
177  {
178  //mouse is over a resize handle, so propagate event onward
179  QGraphicsView::mousePressEvent( e );
180  return;
181  }
182  }
183 
184  QgsComposerItem* selectedItem = 0;
185  QgsComposerItem* previousSelectedItem = 0;
186 
187  if ( e->modifiers() & Qt::ControlModifier )
188  {
189  //CTRL modifier, so we are trying to select the next item below the current one
190  //first, find currently selected item
191  QList<QgsComposerItem*> selectedItems = composition()->selectedComposerItems();
192  if ( selectedItems.size() > 0 )
193  {
194  previousSelectedItem = selectedItems.at( 0 );
195  }
196  }
197 
198  if ( previousSelectedItem )
199  {
200  //select highest item just below previously selected item at position of event
201  selectedItem = composition()->composerItemAt( scenePoint, previousSelectedItem, true );
202 
203  //if we didn't find a lower item we'll use the top-most as fall-back
204  //this duplicates mapinfo/illustrator/etc behaviour where ctrl-clicks are "cyclic"
205  if ( !selectedItem )
206  {
207  selectedItem = composition()->composerItemAt( scenePoint, true );
208  }
209  }
210  else
211  {
212  //select topmost item at position of event
213  selectedItem = composition()->composerItemAt( scenePoint, true );
214  }
215 
216  if ( !selectedItem )
217  {
218  //not clicking over an item, so start marquee selection
219  startMarqueeSelect( scenePoint );
220  break;
221  }
222 
223  if (( !selectedItem->selected() ) && //keep selection if an already selected item pressed
224  !( e->modifiers() & Qt::ShiftModifier ) ) //keep selection if shift key pressed
225  {
226  composition()->setAllUnselected();
227  }
228 
229  if (( e->modifiers() & Qt::ShiftModifier ) && ( selectedItem->selected() ) )
230  {
231  //SHIFT-clicking a selected item deselects it
232  selectedItem->setSelected( false );
233 
234  //Check if we have any remaining selected items, and if so, update the item panel
235  QList<QgsComposerItem*> selectedItems = composition()->selectedComposerItems();
236  if ( selectedItems.size() > 0 )
237  {
238  emit selectedItemChanged( selectedItems.at( 0 ) );
239  }
240  }
241  else
242  {
243  selectedItem->setSelected( true );
244  QGraphicsView::mousePressEvent( e );
245  emit selectedItemChanged( selectedItem );
246  }
247  break;
248  }
249 
250  case Zoom:
251  {
252  if ( !( e->modifiers() & Qt::ShiftModifier ) )
253  {
254  //zoom in action
255  startMarqueeZoom( scenePoint );
256  }
257  else
258  {
259  //zoom out action, so zoom out and recenter on clicked point
260  double scaleFactor = 2;
261  //get current visible part of scene
262  QRect viewportRect( 0, 0, viewport()->width(), viewport()->height() );
263  QgsRectangle visibleRect = QgsRectangle( mapToScene( viewportRect ).boundingRect() );
264 
265  //transform the mouse pos to scene coordinates
266  QPointF scenePoint = mapToScene( e->pos() );
267 
268  visibleRect.scale( scaleFactor, scenePoint.x(), scenePoint.y() );
269  QRectF boundsRect = visibleRect.toRectF();
270 
271  //zoom view to fit desired bounds
272  fitInView( boundsRect, Qt::KeepAspectRatio );
273  }
274  break;
275  }
276 
277  case Pan:
278  {
279  //pan action
280  mToolPanning = true;
281  mMouseLastXY = e->pos();
282  viewport()->setCursor( Qt::ClosedHandCursor );
283  break;
284  }
285 
286  case MoveItemContent:
287  {
288  //get a list of items at clicked position
289  QList<QGraphicsItem *> itemsAtCursorPos = items( e->pos() );
290  if ( itemsAtCursorPos.size() == 0 )
291  {
292  //no items at clicked position
293  return;
294  }
295 
296  //find highest non-locked QgsComposerItem at clicked position
297  //(other graphics items may be higher, eg selection handles)
298  QList<QGraphicsItem*>::iterator itemIter = itemsAtCursorPos.begin();
299  for ( ; itemIter != itemsAtCursorPos.end(); ++itemIter )
300  {
301  QgsComposerItem* item = dynamic_cast<QgsComposerItem *>(( *itemIter ) );
302  if ( item && !item->positionLock() )
303  {
304  //we've found the highest QgsComposerItem
305  mMoveContentStartPos = scenePoint;
306  mMoveContentItem = item;
307  mMovingItemContent = true;
308  break;
309  }
310  }
311 
312  //no QgsComposerItem at clicked position
313  return;
314  }
315 
316  //create rubber band for adding line items
317  case AddArrow:
318  {
319  mRubberBandStartPos = QPointF( snappedScenePoint.x(), snappedScenePoint.y() );
320  mRubberBandLineItem = new QGraphicsLineItem( snappedScenePoint.x(), snappedScenePoint.y(), snappedScenePoint.x(), snappedScenePoint.y() );
321  mRubberBandLineItem->setPen( QPen( QBrush( QColor( 227, 22, 22, 200 ) ), 0 ) );
322  mRubberBandLineItem->setZValue( 1000 );
323  scene()->addItem( mRubberBandLineItem );
324  scene()->update();
325  break;
326  }
327 
328  //create rubber band for adding rectangular items
329  case AddMap:
330  case AddRectangle:
331  case AddTriangle:
332  case AddEllipse:
333  case AddHtml:
334  case AddPicture:
335  case AddLabel:
336  case AddLegend:
337  case AddTable:
338  case AddAttributeTable:
339  {
340  QTransform t;
341  mRubberBandItem = new QGraphicsRectItem( 0, 0, 0, 0 );
342  mRubberBandItem->setBrush( Qt::NoBrush );
343  mRubberBandItem->setPen( QPen( QBrush( QColor( 227, 22, 22, 200 ) ), 0 ) );
344  mRubberBandStartPos = QPointF( snappedScenePoint.x(), snappedScenePoint.y() );
345  t.translate( snappedScenePoint.x(), snappedScenePoint.y() );
346  mRubberBandItem->setTransform( t );
347  mRubberBandItem->setZValue( 1000 );
348  scene()->addItem( mRubberBandItem );
349  scene()->update();
350  }
351  break;
352 
353  case AddScalebar:
354  {
355  QgsComposerScaleBar* newScaleBar = new QgsComposerScaleBar( composition() );
356  newScaleBar->setSceneRect( QRectF( snappedScenePoint.x(), snappedScenePoint.y(), 20, 20 ) );
357  composition()->addComposerItem( newScaleBar );
358  QList<const QgsComposerMap*> mapItemList = composition()->composerMapItems();
359  if ( mapItemList.size() > 0 )
360  {
361  newScaleBar->setComposerMap( mapItemList.at( 0 ) );
362  }
363  newScaleBar->applyDefaultSize(); //4 segments, 1/5 of composer map width
364 
365  composition()->setAllUnselected();
366  newScaleBar->setSelected( true );
367  emit selectedItemChanged( newScaleBar );
368 
369  emit actionFinished();
370  composition()->pushAddRemoveCommand( newScaleBar, tr( "Scale bar added" ) );
371  }
372  break;
373 
374  default:
375  break;
376  }
377 }

503 void QgsComposerView::endMarqueeSelect( QMouseEvent* e )
504 {
505  mMarqueeSelect = false;
506 
507  bool subtractingSelection = false;
508  if ( e->modifiers() & Qt::ShiftModifier )
509  {
510  //shift modifer means adding to selection, nothing required here
511  }
512  else if ( e->modifiers() & Qt::ControlModifier )
513  {
514  //control modifier means subtract from current selection
515  subtractingSelection = true;
516  }
517  else
518  {
519  //not adding to or removing from selection, so clear current selection
520  composition()->setAllUnselected();
521  }
522 
523  if ( !mRubberBandItem || ( mRubberBandItem->rect().width() < 0.1 && mRubberBandItem->rect().height() < 0.1 ) )
524  {
525  //just a click, do nothing
526  removeRubberBand();
527  return;
528  }
529 
530  QRectF boundsRect = QRectF( mRubberBandItem->transform().dx(), mRubberBandItem->transform().dy(),
531  mRubberBandItem->rect().width(), mRubberBandItem->rect().height() );
532 
533  //determine item selection mode, default to intersection
534  Qt::ItemSelectionMode selectionMode = Qt::IntersectsItemShape;
535  if ( e->modifiers() & Qt::AltModifier )
536  {
537  //alt modifier switches to contains selection mode
538  selectionMode = Qt::ContainsItemShape;
539  }
540 
541  //find all items in rubber band
542  QList<QGraphicsItem *> itemList = composition()->items( boundsRect, selectionMode );
543  QList<QGraphicsItem *>::iterator itemIt = itemList.begin();
544  for ( ; itemIt != itemList.end(); ++itemIt )
545  {
546  QgsComposerItem* mypItem = dynamic_cast<QgsComposerItem *>( *itemIt );
547  QgsPaperItem* paperItem = dynamic_cast<QgsPaperItem*>( *itemIt );
548  if ( mypItem && !paperItem )
549  {
550  if ( !mypItem->positionLock() )
551  {
552  if ( subtractingSelection )
553  {
554  mypItem->setSelected( false );
555  }
556  else
557  {
558  mypItem->setSelected( true );
559  }
560  }
561  }
562  }
563  removeRubberBand();
564 
565  //update item panel
566  QList<QgsComposerItem*> selectedItemList = composition()->selectedComposerItems();
567  if ( selectedItemList.size() > 0 )
568  {
569  emit selectedItemChanged( selectedItemList[0] );
570  }
571 }

592 
593  QRectF boundsRect;
594 
595  if ( !mRubberBandItem || ( mRubberBandItem->rect().width() < 0.1 && mRubberBandItem->rect().height() < 0.1 ) )
596  {
597  //just a click, so zoom to clicked point and recenter

599  //get current visible part of scene
600  QRect viewportRect( 0, 0, viewport()->width(), viewport()->height() );

620

630 void QgsComposerView::mouseReleaseEvent( QMouseEvent* e )
631 {
632  if ( e->button() != Qt::LeftButton &&
633  ( composition()->selectionHandles()->isDragging() || composition()->selectionHandles()->isResizing() ) )
634  {
635  //ignore clicks while dragging/resizing items
636  return;
637  }
638 
639  QPoint mousePressStopPoint = e->pos();
640  int diffX = mousePressStopPoint.x() - mMousePressStartPos.x();
641  int diffY = mousePressStopPoint.y() - mMousePressStartPos.y();
642 
643  //was this just a click? or a click and drag?
644  bool clickOnly = false;
645  if ( qAbs( diffX ) < 2 && qAbs( diffY ) < 2 )
646  {
647  clickOnly = true;
648  }
649 
650  QPointF scenePoint = mapToScene( e->pos() );
651 
652  if ( mMousePanning || mToolPanning )
653  {
654  mMousePanning = false;
655  mToolPanning = false;
656 
657  if ( clickOnly && e->button() == Qt::MidButton )
658  {
659  //middle mouse button click = recenter on point
660 
661  //get current visible part of scene
662  QRect viewportRect( 0, 0, viewport()->width(), viewport()->height() );
663  QgsRectangle visibleRect = QgsRectangle( mapToScene( viewportRect ).boundingRect() );
664  visibleRect.scale( 1, scenePoint.x(), scenePoint.y() );
665  QRectF boundsRect = visibleRect.toRectF();
666 
667  //zoom view to fit desired bounds
668  fitInView( boundsRect, Qt::KeepAspectRatio );
669  }
670 
671  //set new cursor
672  if ( mCurrentTool != Pan )
673  {
674  //allow composer items to change cursor
675  composition()->setPreventCursorChange( false );
676  }
677  viewport()->setCursor( defaultCursorForTool( mCurrentTool ) );
678  }
679 
680  //for every other tool, ignore clicks of non-left button
681  if ( e->button() != Qt::LeftButton )
682  {
683  return;
684  }
685 
686  if ( mMarqueeSelect )
687  {
688  endMarqueeSelect( e );
689  return;
690  }
691 
692  switch ( mCurrentTool )
693  {
694  case Select:
695  {
696  QGraphicsView::mouseReleaseEvent( e );
697  break;
698  }
699 
700  case Zoom:
701  {
702  if ( mMarqueeZoom )
703  {
704  endMarqueeZoom( e );
705  }
706  break;
707  }
708 
709  case MoveItemContent:
710  {
711  if ( mMoveContentItem )
712  {
713  //update map preview if composer map
714  QgsComposerMap* composerMap = dynamic_cast<QgsComposerMap *>( mMoveContentItem );
715  if ( composerMap )
716  {
717  composerMap->setOffset( 0, 0 );
718  }

1551  //holding ctrl while wheel zooming results in a finer zoom
1552  zoomFactor = 1.0 + ( zoomFactor - 1.0 ) / 10.0;
1553  }
1554 
1555  //caculate zoom scale factor
1556  bool zoomIn = event->delta() > 0;
1557  double scaleFactor = ( zoomIn ? 1 / zoomFactor : zoomFactor );
1558 
1559  //get current visible part of scene
1560  QRect viewportRect( 0, 0, viewport()->width(), viewport()->height() );
1561  QgsRectangle visibleRect = QgsRectangle( mapToScene( viewportRect ).boundingRect() );

1565 
1566  //adjust view center according to wheel action setting

1579  scenePoint.y() + (( oldCenter.y() - scenePoint.y() ) * scaleFactor ) );

1582  }
1583

1585  break;
1586  }
1587 
1588  //zoom composition
1589  if ( zoomIn )
1590  {
1591  scale( zoomFactor, zoomFactor );
1592  }
1593  else
1594  {
1595  scale( 1 / zoomFactor, 1 / zoomFactor );
1596  }
