118 void QgsMessageBar::popItem( QgsMessageBarItem *item )
119 {
120  Q_ASSERT( item );
121 
122  if ( item != mCurrentItem && !mItems.contains( item ) )
123  return;
124 
125  if ( item == mCurrentItem )
126  {
127  if ( mCurrentItem )
128  {
129  QWidget *widget = dynamic_cast<QWidget*>( mCurrentItem );
130  mLayout->removeWidget( widget );
131  mCurrentItem->hide();
132  disconnect( mCurrentItem, SIGNAL( styleChanged( QString ) ), this, SLOT( setStyleSheet( QString ) ) );
133  delete mCurrentItem;
134  mCurrentItem = 0;
135  }
136 
137  if ( !mItems.isEmpty() )
138  {
139  showItem( mItems.first() );
140  }
141  else
142  {
143  hide();
144  }
145  }
146  else
147  {
148  mItems.removeOne( item );
149  }
150 
151  emit widgetRemoved( item );
152 }

154 bool QgsMessageBar::popWidget( QgsMessageBarItem *item )
155 {
156  if ( !item || !mCurrentItem )
157  return false;
158 
159  if ( item == mCurrentItem )
160  {
161  popItem( mCurrentItem );
162  return true;
163  }
164 
165  foreach ( QgsMessageBarItem *existingItem, mItems )
166  {
167  if ( existingItem == item )
168  {
169  mItems.removeOne( existingItem );
170  delete existingItem;
171  return true;
172  }
173  }
174 
175  return false;
176 }

225 void QgsMessageBar::showItem( QgsMessageBarItem *item )
226 {
227  Q_ASSERT( item );
228 
229  if ( mCurrentItem != 0 )
230  disconnect( mCurrentItem, SIGNAL( styleChanged( QString ) ), this, SLOT( setStyleSheet( QString ) ) );
231 
232  if ( item == mCurrentItem )
233  return;
234 
235  if ( mItems.contains( item ) )
236  mItems.removeOne( item );
237 
238  if ( mCurrentItem )
239  {
240  mItems.prepend( mCurrentItem );
241  mLayout->removeWidget( mCurrentItem );
242  mCurrentItem->hide();
243  }
244 
245  mCurrentItem = item;
246  mLayout->addWidget( item, 0, 1, 1, 1 );
247  mCurrentItem->show();
248 
249  if ( item->duration() > 0 )
250  {
251  mCountProgress->setRange( 0, item->duration() );
252  mCountProgress->setValue( item->duration() );
253  mCountProgress->setVisible( true );
254  mCountdownTimer->start();
255  }
256 
257  connect( mCurrentItem, SIGNAL( styleChanged( QString ) ), this, SLOT( setStyleSheet( QString ) ) );
258  setStyleSheet( item->getStyleSheet() );
259  show();
260 
261  emit widgetAdded( item );
262 }
