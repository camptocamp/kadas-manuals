135 void QgsLayerTreeView::modelRowsInserted( QModelIndex index, int start, int end )
136 {
137  QgsLayerTreeNode* parentNode = layerTreeModel()->index2node( index );
138  if ( !parentNode )
139  return;
140 
141  if ( QgsLayerTree::isLayer( parentNode ) )
142  {
143  // if ShowLegendAsTree flag is enabled in model, we may need to expand some legend nodes
144  QStringList expandedNodeKeys = parentNode->customProperty( "expandedLegendNodes" ).toStringList();
145  if ( expandedNodeKeys.isEmpty() )
146  return;
147 
148  foreach ( QgsLayerTreeModelLegendNode* legendNode, layerTreeModel()->layerLegendNodes( QgsLayerTree::toLayer( parentNode ) ) )
149  {
150  QString ruleKey = legendNode->data( QgsLayerTreeModelLegendNode::RuleKeyRole ).toString();
151  if ( expandedNodeKeys.contains( ruleKey ) )
152  setExpanded( layerTreeModel()->legendNode2index( legendNode ), true );
153  }
154  return;
155  }
156 
157  QList<QgsLayerTreeNode*> children = parentNode->children();
158  for ( int i = start; i <= end; ++i )
159  {
160  updateExpandedStateFromNode( children[i] );
161  }
162 
163  // make sure we still have correct current layer
164  onCurrentChanged();
165 }

173 void QgsLayerTreeView::updateExpandedStateToNode( QModelIndex index )
174 {
175  if ( QgsLayerTreeNode* node = layerTreeModel()->index2node( index ) )
176  {
177  node->setExpanded( isExpanded( index ) );
178  }
179  else if ( QgsLayerTreeModelLegendNode* node = layerTreeModel()->index2legendNode( index ) )
180  {
181  QString ruleKey = node->data( QgsLayerTreeModelLegendNode::RuleKeyRole ).toString();
182  QStringList lst = node->layerNode()->customProperty( "expandedLegendNodes" ).toStringList();
183  bool expanded = isExpanded( index );
184  bool isInList = lst.contains( ruleKey );
185  if ( expanded && !isInList )
186  {
187  lst.append( ruleKey );
188  node->layerNode()->setCustomProperty( "expandedLegendNodes", lst );
189  }
190  else if ( !expanded && isInList )
191  {
192  lst.removeAll( ruleKey );
193  node->layerNode()->setCustomProperty( "expandedLegendNodes", lst );
194  }
195  }
196 }

198 void QgsLayerTreeView::onCurrentChanged()
199 {
200  QgsMapLayer* layerCurrent = layerForIndex( currentIndex() );
201  QString layerCurrentID = layerCurrent ? layerCurrent->id() : QString();
202  if ( mCurrentLayerID == layerCurrentID )
203  return;
204 
205  // update the current index in model (the item will be underlined)
206  QModelIndex nodeLayerIndex;
207  if ( layerCurrent )
208  {
209  QgsLayerTreeLayer* nodeLayer = layerTreeModel()->rootGroup()->findLayer( layerCurrentID );
210  if ( nodeLayer )
211  nodeLayerIndex = layerTreeModel()->node2index( nodeLayer );
212  }
213  layerTreeModel()->setCurrentIndex( nodeLayerIndex );
214 
215  mCurrentLayerID = layerCurrentID;
216  emit currentLayerChanged( layerCurrent );
217 }

418 void QgsLayerTreeViewMenuProvider::addCustomLayerActions( QMenu* menu, QgsMapLayer* layer )
419 {
420  if ( !layer )
421  return;
422 
423  // add custom layer actions - should this go at end?
424  QList< LegendLayerAction > lyrActions = legendLayerActions( layer->type() );
425 
426  if ( ! lyrActions.isEmpty() )
427  {
428  menu->addSeparator();
429  QList<QMenu*> theMenus;
430  for ( int i = 0; i < lyrActions.count(); i++ )
431  {
432  if ( lyrActions[i].allLayers || lyrActions[i].layers.contains( layer ) )
433  {
434  if ( lyrActions[i].menu.isEmpty() )
435  {
436  menu->addAction( lyrActions[i].action );
437  }
438  else
439  {
440  // find or create menu for given menu name
441  // adapted from QgisApp::getPluginMenu( QString menuName )
442  QString menuName = lyrActions[i].menu;
443 #ifdef Q_OS_MAC
444  // Mac doesn't have '&' keyboard shortcuts.
445  menuName.remove( QChar( '&' ) );
446 #endif
447  QAction* before = 0;
448  QMenu* newMenu = 0;
449  QString dst = menuName;
450  dst.remove( QChar( '&' ) );
451  foreach ( QMenu* menu, theMenus )
452  {
453  QString src = menu->title();
454  src.remove( QChar( '&' ) );
455  int comp = dst.localeAwareCompare( src );
456  if ( comp < 0 )
457  {
458  // Add item before this one
459  before = menu->menuAction();
460  break;
461  }
462  else if ( comp == 0 )
463  {
464  // Plugin menu item already exists
465  newMenu = menu;
466  break;
467  }
468  }
469  if ( ! newMenu )
470  {
471  // It doesn't exist, so create
472  newMenu = new QMenu( menuName );
473  theMenus.append( newMenu );
474  // Where to put it? - we worked that out above...
475  menu->insertMenu( before, newMenu );
476  }
477  // QMenu* menu = getMenu( lyrActions[i].menu, &theBeforeSep, &theAfterSep, &theMenu );
478  newMenu->addAction( lyrActions[i].action );
479  }
480  }
481  }
482  menu->addSeparator();
483  }
