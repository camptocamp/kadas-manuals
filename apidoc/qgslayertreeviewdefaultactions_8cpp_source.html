108 QAction* QgsLayerTreeViewDefaultActions::actionTransparency( QObject* parent )
109 {
110  QgsMapLayer* layer = mView->currentLayer();
111  if ( !layer )
112  return 0;
113 
114  int curValue = 0;
115  if ( layer->type() == QgsMapLayer::VectorLayer || layer->type() == QgsMapLayer::RedliningLayer )
116  {
117  curValue = static_cast<QgsVectorLayer*>( layer )->layerTransparency();
118  }
119  else if ( layer->type() == QgsMapLayer::PluginLayer )
120  {
121  curValue = static_cast<QgsPluginLayer*>( layer )->layerTransparency();
122  }
123  else if ( layer->type() == QgsMapLayer::RasterLayer )
124  {
125  curValue = 100 - static_cast<QgsRasterLayer*>( layer )->renderer()->opacity() * 100;
126  }
127 
128  QWidget* transpWidget = new QWidget();
129  QHBoxLayout* transpLayout = new QHBoxLayout( transpWidget );
130 
131  QLabel* transpLabel = new QLabel( tr( "Transparency:" ) );
132  transpLabel->setSizePolicy( QSizePolicy::Preferred, QSizePolicy::Fixed );
133  transpLayout->addWidget( transpLabel );
134 
135  QSlider* transpSlider = new QSlider( Qt::Horizontal );
136  transpSlider->setRange( 0, 100 );
137  transpSlider->setValue( curValue );
138  transpSlider->setSizePolicy( QSizePolicy::Expanding, QSizePolicy::Fixed );
139  transpSlider->setTracking( false );
140  connect( transpSlider, SIGNAL( valueChanged( int ) ), this, SLOT( setLayerTransparency() ) );
141  transpLayout->addWidget( transpSlider );
142 
143  QWidgetAction* transpAction = new QWidgetAction( parent );
144  transpAction->setDefaultWidget( transpWidget );
145  return transpAction;

308 void QgsLayerTreeViewDefaultActions::setLayerTransparency()
309 {
310  QSlider* slider = qobject_cast<QSlider*>( QObject::sender() );
311  int value = slider->value();
312  QgsMapLayer* layer = mView->currentLayer();
313  if ( !layer )
314  return;
315 
316  if ( layer->type() == QgsMapLayer::VectorLayer || layer->type() == QgsMapLayer::RedliningLayer )
317  {
318  static_cast<QgsVectorLayer*>( layer )->setLayerTransparency( value );
319  mView->refreshLayerSymbology( layer->id() );
320  layer->triggerRepaint();
321  }
322  else if ( layer->type() == QgsMapLayer::PluginLayer )
323  {
324  static_cast<QgsPluginLayer*>( layer )->setLayerTransparency( value );
325  mView->refreshLayerSymbology( layer->id() );
326  layer->triggerRepaint();
327  }
328  else if ( layer->type() == QgsMapLayer::RasterLayer )
329  {
330  static_cast<QgsRasterLayer*>( layer )->renderer()->setOpacity( 1. - value / 100. );
331  mView->refreshLayerSymbology( layer->id() );
332  layer->triggerRepaint();
333  }
334 }

363 void QgsLayerTreeViewDefaultActions::zoomToLayers( QgsMapCanvas* canvas, const QList<QgsMapLayer*>& layers )
364 {
365  QgsRectangle extent;
366  extent.setMinimal();
367 
368  for ( int i = 0; i < layers.size(); ++i )
369  {
370  QgsMapLayer* layer = layers.at( i );
371  QgsRectangle layerExtent = layer->extent();
372 
373  QgsVectorLayer *vLayer = qobject_cast<QgsVectorLayer*>( layer );
374  if ( vLayer )
375  {
376  if ( vLayer->geometryType() == QGis::NoGeometry )
377  continue;
378 
379  if ( layerExtent.isEmpty() )
380  {
381  vLayer->updateExtents();
382  layerExtent = vLayer->extent();
383  }
384  }
385 
386  if ( layerExtent.isNull() )
387  continue;
388 
389  //transform extent if otf-projection is on
390  if ( canvas->hasCrsTransformEnabled() )
391  layerExtent = canvas->mapSettings().layerExtentToOutputExtent( layer, layerExtent );
392 
393  extent.combineExtentWith( &layerExtent );
394  }
395 
396  if ( extent.isNull() )
397  return;
398 
399  // Increase bounding box with 5%, so that layer is a bit inside the borders
400  extent.scale( 1.05 );
401 
402  //zoom to bounding box
403  canvas->setExtent( extent );
404  canvas->refresh();
405 }

435 void QgsLayerTreeViewDefaultActions::groupSelected()
436 {
437  QList<QgsLayerTreeNode*> nodes = mView->selectedNodes( true );
438  if ( nodes.count() < 2 || ! QgsLayerTree::isGroup( nodes[0]->parent() ) )
439  return;
440 
441  QgsLayerTreeGroup* parentGroup = QgsLayerTree::toGroup( nodes[0]->parent() );
442  int insertIdx = parentGroup->children().indexOf( nodes[0] );
443 
444  QgsLayerTreeGroup* newGroup = new QgsLayerTreeGroup( uniqueGroupName( parentGroup ) );
445  foreach ( QgsLayerTreeNode* node, nodes )
446  newGroup->addChildNode( node->clone() );
447 
448  parentGroup->insertChildNode( insertIdx, newGroup );
449 
450  foreach ( QgsLayerTreeNode* node, nodes )
451  {
452  QgsLayerTreeGroup* group = qobject_cast<QgsLayerTreeGroup*>( node->parent() );
453  if ( group )
454  group->removeChildNode( node );
455  }
456 
457  mView->setCurrentIndex( mView->layerTreeModel()->node2index( newGroup ) );
458 }
