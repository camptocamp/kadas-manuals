47 QgsMimeDataUtils::Uri::Uri( const QString& encData )
48 {
49  QgsDebugMsg( "encData: " + encData );
50  QStringList decoded = decode( encData );
51  if ( decoded.size() < 4 )
52  return;
53 
54  layerType = decoded[0];
55  providerKey = decoded[1];
56  name = decoded[2];
57  uri = decoded[3];
58 
59  if ( decoded.size() >= 6 )
60  {
61  supportedCrs = decode( decoded[4] );
62  supportedFormats = decode( decoded[5] );
63  }
64  if ( decoded.size() >= 7 )
65  layerInfoUrl = decoded[6];
66 
67  QgsDebugMsg( QString( "type:%1 key:%2 name:%3 uri:%4 supportedCRS:%5 supportedFormats:%6 metadataUrl:%7" )
68  .arg( layerType ).arg( providerKey ).arg( name ).arg( uri )
69  .arg( supportedCrs.join( ", " ) )
70  .arg( supportedFormats.join( ", " ) )
71  .arg( layerInfoUrl ) );
72 }

117 QString QgsMimeDataUtils::encode( const QStringList& items )
118 {
119  QString encoded;
120  foreach ( const QString& item, items )
121  {
122  QString str = item;
123  str.replace( "\\", "\\\\" );
124  str.replace( ":", "\\:" );

126  }
127  return encoded.left( encoded.length() - 1 );

130 QStringList QgsMimeDataUtils::decode( const QString& encoded )
131 {
132  QStringList items;
133  QString item;
134  bool inEscape = false;
135  foreach ( const QChar& c, encoded )
136  {
137  if ( c == '\\' && !inEscape )
138  {
139  inEscape = true;
140  continue;
141  }
142  if ( c == ':' && !inEscape )
143  {
144  items.append( item );
145  item = "";
146  }
147  else
148  {
149  item += c;
150  }
151  inEscape = false;
152  }
153  items.append( item );
154  return items;
155 }
