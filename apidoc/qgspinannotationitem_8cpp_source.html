43  Q_DECLARE_TR_FUNCTIONS( RemarksEdit )
44  public:
45  RemarksEdit( QWidget* parent = 0 ) : QTextBrowser( parent )

52  void contextMenuEvent( QContextMenuEvent *e ) override
53  {
54  QString anchor = anchorAt( e->pos() );
55  if ( !anchor.isEmpty() )
56  {
57  QMenu menu;
58  QAction* openAction = menu.addAction( tr( "Open link..." ) );
59  QAction* copyAction = menu.addAction( tr( "Copy link location" ) );
60  QAction* clickedAction = menu.exec( e->globalPos() );
61  if ( clickedAction == openAction )
62  {
63  QDesktopServices::openUrl( QUrl::fromUserInput( anchor ) );
64  }
65  else if ( clickedAction == copyAction )
66  {
67  QApplication::clipboard()->setText( anchor );
68  }
69  e->accept();
70  }
71  else
72  {
73  QTextBrowser::contextMenuEvent( e );
74  }
75  }
76  void keyPressEvent( QKeyEvent* e ) override
77  {
78  if ( e->key() == Qt::Key_Control )
79  {
80  setReadOnly( true );
81  e->accept();

83  }
84  setReadOnly( false );
85  QTextBrowser::keyPressEvent( e );
86  }

88  {

117 void QgsPinAnnotationItem::updateToolTip()
118 {
119  QString posStr = QgsCoordinateFormat::instance()->getDisplayString( mGeoPos, mGeoPosCrs );
120  if ( posStr.isEmpty() )
121  {
122  posStr = QString( "%1 (%2)" ).arg( mGeoPos.toString() ).arg( mGeoPosCrs.authid() );
123  }
124  QString toolTipText = QString( "<b>Position:</b> %1<br /><b>Height:</b> %2 %3<br /><b>Name:</b> %4<br /><b>Remarks:</b><br />%5" )
125  .arg( posStr )
126  .arg( QgsCoordinateFormat::instance()->getHeightAtPos( mGeoPos, mGeoPosCrs ) )
127  .arg( QgsCoordinateFormat::instance()->getHeightDisplayUnit() == QGis::Feet ? "ft" : "m" )
128  .arg( mName )
129  .arg( mRemarks );
130  setToolTip( toolTipText );
131 }

194 void QgsPinAnnotationItem::convertToWaypoint()
195 {
196  QgsRedliningLayer* gpsLayer = QgsMapLayerRegistry::instance()->getOrCreateGpsRoutesLayer();
197  QgsPoint wgsPos = QgsCoordinateTransformCache::instance()->transform( mapGeoPosCrs().authid(), "EPSG:4326" )->transform( mapGeoPos() );
198  QString name = getName();
199  QString flags( "shape=point,symbol=circle,r=0,bold=1" );
200  gpsLayer->addShape( new QgsGeometry( new QgsPointV2( wgsPos ) ), Qt::yellow, Qt::yellow, 2, Qt::SolidLine, Qt::SolidPattern, flags, QString(), name );
201  deleteLater();
202  gpsLayer->triggerRepaint();
203 }
