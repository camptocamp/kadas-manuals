41 void QgsMultiMapManager::addMapWidget()
42 {
43  int highestNumber = 1;
44  foreach ( const QPointer<QgsMapWidget>& mapWidget, mMapWidgets )
45  {
46  if ( mapWidget && mapWidget->getNumber() >= highestNumber )
47  {
48  highestNumber = mapWidget->getNumber() + 1;
49  }
50  }
51 
52  QgsMapWidget* mapWidget = new QgsMapWidget( highestNumber, tr( "View #%1" ).arg( highestNumber ), mMasterCanvas );
53  connect( mapWidget, SIGNAL( destroyed( QObject* ) ), this, SLOT( mapWidgetDestroyed( QObject* ) ) );
54 
55  // Determine whether to add the new dock widget in the right or bottom area
56  int nRight = 0, nBottom = 0;
57  double rightAreaWidth = 0;
58  double bottomAreaHeight = 0;
59  foreach ( const QPointer<QgsMapWidget>& mapWidget, mMapWidgets )
60  {
61  Qt::DockWidgetArea area = mMainWindow->dockWidgetArea( mapWidget );
62  if ( area == Qt::RightDockWidgetArea )
63  {
64  ++nRight;
65  rightAreaWidth = mapWidget->width();
66  }
67  else if ( area == Qt::BottomDockWidgetArea )
68  {
69  ++nBottom;
70  bottomAreaHeight = mapWidget->height();
71  }
72  }
73  QSize initialSize;
74  Qt::DockWidgetArea addArea;
75  if ( nBottom >= nRight - 1 )
76  {
77  addArea = Qt::RightDockWidgetArea;
78  initialSize.setHeight(( mMasterCanvas->height() + bottomAreaHeight ) / ( nRight + 1 ) );
79  initialSize.setWidth( nRight > 0 ? rightAreaWidth : mMasterCanvas->width() / 2 );
80  }
81  else
82  {
83  addArea = Qt::BottomDockWidgetArea;
84  initialSize.setHeight( nBottom > 0 ? bottomAreaHeight : mMasterCanvas->height() / 2 );
85  initialSize.setWidth( mMasterCanvas->width() / ( nBottom + 1 ) );
86  }
87 
88  // Set initial layers
89  QStringList initialLayers;
90  foreach ( QgsLayerTreeLayer* layerTreeLayer, QgsProject::instance()->layerTreeRoot()->findLayers() )
91  {
92  initialLayers.append( layerTreeLayer->layer()->id() );
93  }
94  mapWidget->setFixedSize( initialSize );
95  mMainWindow->addDockWidget( addArea, mapWidget );
96  mapWidget->resize( initialSize );
97  mMapWidgets.append( QPointer<QgsMapWidget>( mapWidget ) );
98 }

114 void QgsMultiMapManager::writeProjectSettings( QDomDocument& doc )
115 {
116  QDomNodeList nl = doc.elementsByTagName( "qgis" );
117  if ( nl.count() < 1 || nl.at( 0 ).toElement().isNull() )
118  {
119  return;
120  }
121  QDomElement qgisElem = nl.at( 0 ).toElement();
122 
123  QDomElement mapViewsElem = doc.createElement( "MapViews" );
124  foreach ( QgsMapWidget* mapWidget, mMapWidgets )
125  {
126  QByteArray ba;
127  QDataStream ds( &ba, QIODevice::WriteOnly );
128 
129  QDomElement mapWidgetItemElem = doc.createElement( "MapView" );
130  mapWidgetItemElem.setAttribute( "width", mapWidget->width() );
131  mapWidgetItemElem.setAttribute( "height", mapWidget->height() );
132  mapWidgetItemElem.setAttribute( "floating", mapWidget->isFloating() );
133  mapWidgetItemElem.setAttribute( "islocked", mapWidget->getLocked() );
134  mapWidgetItemElem.setAttribute( "area", mMainWindow->dockWidgetArea( mapWidget ) );
135  mapWidgetItemElem.setAttribute( "title", mapWidget->windowTitle() );
136  mapWidgetItemElem.setAttribute( "number", mapWidget->getNumber() );
137  ds << mapWidget->getLayers();
138  mapWidgetItemElem.setAttribute( "layers", QString( ba.toBase64() ) );
139  ba.clear();
140  ds << mapWidget->getMapExtent().toRectF();
141  mapWidgetItemElem.setAttribute( "extent", QString( ba.toBase64() ) );
142  mapViewsElem.appendChild( mapWidgetItemElem );
143  }
144  qgisElem.appendChild( mapViewsElem );
145 }

147 void QgsMultiMapManager::readProjectSettings( const QDomDocument& doc )
148 {
149  clearMapWidgets();
150 
151  QDomNodeList nl = doc.elementsByTagName( "MapViews" );
152  if ( nl.count() < 1 || nl.at( 0 ).toElement().isNull() )
153  {
154  return;
155  }
156 
157  QDomElement mapViewsElem = nl.at( 0 ).toElement();
158  QDomNodeList nodes = mapViewsElem.childNodes();
159  for ( int iNode = 0, nNodes = nodes.size(); iNode < nNodes; ++iNode )
160  {
161  QDomElement mapWidgetItemElem = nodes.at( iNode ).toElement();
162  if ( mapWidgetItemElem.nodeName() == "MapView" )
163  {
164  QDomNamedNodeMap attributes = mapWidgetItemElem.attributes();
165  QByteArray ba;
166  QDataStream ds( &ba, QIODevice::ReadOnly );
167  QgsMapWidget* mapWidget = new QgsMapWidget(
168  attributes.namedItem( "number" ).nodeValue().toInt(),
169  attributes.namedItem( "title" ).nodeValue(),
170  mMasterCanvas );
171  mapWidget->setAttribute( Qt::WA_DeleteOnClose );
172  ba = QByteArray::fromBase64( attributes.namedItem( "layers" ).nodeValue().toAscii() );
173  QStringList layersList;
174  ds >> layersList;
175  mapWidget->setInitialLayers( layersList );
176  connect( mapWidget, SIGNAL( destroyed( QObject* ) ), this, SLOT( mapWidgetDestroyed( QObject* ) ) );
177  // Compiler bug?! If I pass it directly, value is always false
178  bool islocked = attributes.namedItem( "islocked" ).nodeValue().toInt();
179  mapWidget->setLocked( islocked );
180  mapWidget->setFloating( attributes.namedItem( "floating" ).nodeValue().toInt() );
181  mapWidget->setFixedWidth( attributes.namedItem( "width" ).nodeValue().toInt() );
182  mapWidget->setFixedHeight( attributes.namedItem( "height" ).nodeValue().toInt() );
183  ba = QByteArray::fromBase64( attributes.namedItem( "extent" ).nodeValue().toAscii() );
184  QRectF extent;
185  ds >> extent;
186  mapWidget->setMapExtent( QgsRectangle( extent ) );
187  mMainWindow->addDockWidget( static_cast<Qt::DockWidgetArea>( attributes.namedItem( "area" ).nodeValue().toInt() ), mapWidget );
188  mMapWidgets.append( mapWidget );
189  }
190  }
