86 bool QgsMapRenderer::setExtent( const QgsRectangle& extent )
87 {
88  //remember the previous extent
89  mLastExtent = mExtent;
90 
91  // Don't allow zooms where the current extent is so small that it
92  // can't be accurately represented using a double (which is what
93  // currentExtent uses). Excluding 0 avoids a divide by zero and an
94  // infinite loop when rendering to a new canvas. Excluding extents
95  // greater than 1 avoids doing unnecessary calculations.
96 
97  // The scheme is to compare the width against the mean x coordinate
98  // (and height against mean y coordinate) and only allow zooms where
99  // the ratio indicates that there is more than about 12 significant
100  // figures (there are about 16 significant figures in a double).
101 
102  if ( extent.width() > 0 &&
103  extent.height() > 0 &&
104  extent.width() < 1 &&
105  extent.height() < 1 )
106  {
107  // Use abs() on the extent to avoid the case where the extent is
108  // symmetrical about 0.
109  double xMean = ( qAbs( extent.xMinimum() ) + qAbs( extent.xMaximum() ) ) * 0.5;
110  double yMean = ( qAbs( extent.yMinimum() ) + qAbs( extent.yMaximum() ) ) * 0.5;
111 
112  double xRange = extent.width() / xMean;
113  double yRange = extent.height() / yMean;
114 
115  static const double minProportion = 1e-12;
116  if ( xRange < minProportion || yRange < minProportion )
117  return false;
118  }
119 
120  mExtent = extent;
121  if ( !extent.isEmpty() )
122  adjustExtentToSize();
123 
124  emit extentsChanged();
125  return true;
126 }

171 void QgsMapRenderer::adjustExtentToSize()
172 {
173  double myHeight = mSize.height();
174  double myWidth = mSize.width();
175 
176  QgsMapToPixel newCoordXForm;
177 
178  if ( !myWidth || !myHeight )
179  {
180  mScale = 1.0;
181  newCoordXForm.setParameters( 1, 0, 0, 0 );
182  return;
183  }
184 
185  // calculate the translation and scaling parameters
186  // mapUnitsPerPixel = map units per pixel
187  double mapUnitsPerPixelY = mExtent.height() / myHeight;
188  double mapUnitsPerPixelX = mExtent.width() / myWidth;
189  mMapUnitsPerPixel = mapUnitsPerPixelY > mapUnitsPerPixelX ? mapUnitsPerPixelY : mapUnitsPerPixelX;
190 
191  // calculate the actual extent of the mapCanvas
192  double dxmin, dxmax, dymin, dymax, whitespace;
193 
194  if ( mapUnitsPerPixelY > mapUnitsPerPixelX )
195  {
196  dymin = mExtent.yMinimum();
197  dymax = mExtent.yMaximum();
198  whitespace = (( myWidth * mMapUnitsPerPixel ) - mExtent.width() ) * 0.5;
199  dxmin = mExtent.xMinimum() - whitespace;
200  dxmax = mExtent.xMaximum() + whitespace;
201  }
202  else
203  {
204  dxmin = mExtent.xMinimum();
205  dxmax = mExtent.xMaximum();
206  whitespace = (( myHeight * mMapUnitsPerPixel ) - mExtent.height() ) * 0.5;
207  dymin = mExtent.yMinimum() - whitespace;
208  dymax = mExtent.yMaximum() + whitespace;
209  }
210 
211  QgsDebugMsg( QString( "Map units per pixel (x,y) : %1, %2" ).arg( qgsDoubleToString( mapUnitsPerPixelX ) ).arg( qgsDoubleToString( mapUnitsPerPixelY ) ) );
212  QgsDebugMsg( QString( "Pixmap dimensions (x,y) : %1, %2" ).arg( qgsDoubleToString( myWidth ) ).arg( qgsDoubleToString( myHeight ) ) );
213  QgsDebugMsg( QString( "Extent dimensions (x,y) : %1, %2" ).arg( qgsDoubleToString( mExtent.width() ) ).arg( qgsDoubleToString( mExtent.height() ) ) );
214  QgsDebugMsg( mExtent.toString() );
215 
216  // update extent
217  mExtent.setXMinimum( dxmin );
218  mExtent.setXMaximum( dxmax );
219  mExtent.setYMinimum( dymin );
220  mExtent.setYMaximum( dymax );
221 
222  QgsDebugMsg( QString( "Adjusted map units per pixel (x,y) : %1, %2" ).arg( qgsDoubleToString( mExtent.width() / myWidth ) ).arg( qgsDoubleToString( mExtent.height() / myHeight ) ) );
223 
224  QgsDebugMsg( QString( "Recalced pixmap dimensions (x,y) : %1, %2" ).arg( qgsDoubleToString( mExtent.width() / mMapUnitsPerPixel ) ).arg( qgsDoubleToString( mExtent.height() / mMapUnitsPerPixel ) ) );
225 
226  // update the scale
227  updateScale();
228 
229  QgsDebugMsg( QString( "Scale (assuming meters as map units) = 1:%1" ).arg( qgsDoubleToString( mScale ) ) );
230 
231  newCoordXForm.setParameters( mMapUnitsPerPixel, dxmin, dymin, myHeight );
232  mRenderContext.setMapToPixel( newCoordXForm );
233  mRenderContext.setExtent( mExtent );
234 }

237 void QgsMapRenderer::render( QPainter* painter, double* forceWidthScale, bool logRenderTime )
238 {
239  //Lock render method for concurrent threads (e.g. from globe)
240  QMutexLocker renderLock( &mRenderMutex );
241 
242  QgsDebugMsg( "========== Rendering ==========" );
243 
244  if ( mExtent.isEmpty() )
245  {
246  QgsDebugMsg( "empty extent... not rendering" );
247  return;
248  }
249 
250  if ( mSize.width() == 1 && mSize.height() == 1 )
251  {
252  QgsDebugMsg( "size 1x1... not rendering" );
253  return;
254  }
255 
256  QPaintDevice* thePaintDevice = painter->device();
257  if ( !thePaintDevice )
258  {
259  return;
260  }
261 
262  // wait
263  if ( mDrawing )
264  {
265  QgsDebugMsg( "already rendering" );
266  QCoreApplication::processEvents();
267  }
268 
269  if ( mDrawing )
270  {
271  QgsDebugMsg( "still rendering - skipping" );
272  return;
273  }
274 
275  mDrawing = true;
276 
277  const QgsCoordinateTransform *ct;
278 
279  if ( mOverview )
280  mRenderContext.setDrawEditingInformation( !mOverview );
281 
282  mRenderContext.setPainter( painter );
283  mRenderContext.setCoordinateTransform( 0 );
284  //this flag is only for stopping during the current rendering progress,
285  //so must be false at every new render operation
286  mRenderContext.setRenderingStopped( false );
287 
288  // set selection color
289  QgsProject* prj = QgsProject::instance();
290  int myRed = prj->readNumEntry( "Gui", "/SelectionColorRedPart", 255 );
291  int myGreen = prj->readNumEntry( "Gui", "/SelectionColorGreenPart", 255 );
292  int myBlue = prj->readNumEntry( "Gui", "/SelectionColorBluePart", 0 );
293  int myAlpha = prj->readNumEntry( "Gui", "/SelectionColorAlphaPart", 255 );
294  mRenderContext.setSelectionColor( QColor( myRed, myGreen, myBlue, myAlpha ) );
295 
296  //calculate scale factor
297  //use the specified dpi and not those from the paint device
298  //because sometimes QPainter units are in a local coord sys (e.g. in case of QGraphicsScene)
299  double sceneDpi = mScaleCalculator->dpi();
300  double scaleFactor = 1.0;
301  if ( mOutputUnits == QgsMapRenderer::Millimeters )
302  {
303  if ( forceWidthScale )
304  {
305  scaleFactor = *forceWidthScale;
306  }
307  else
308  {
309  scaleFactor = sceneDpi / 25.4;
310  }
311  }
312  double rasterScaleFactor = ( thePaintDevice->logicalDpiX() + thePaintDevice->logicalDpiY() ) / 2.0 / sceneDpi;
313  if ( mRenderContext.rasterScaleFactor() != rasterScaleFactor )
314  {
315  mRenderContext.setRasterScaleFactor( rasterScaleFactor );
316  }
317  if ( mRenderContext.scaleFactor() != scaleFactor )
318  {
319  mRenderContext.setScaleFactor( scaleFactor );
320  }
321  if ( mRenderContext.rendererScale() != mScale )
322  {
323  //add map scale to render context
324  mRenderContext.setRendererScale( mScale );
325  }
326  if ( mLastExtent != mExtent )
327  {
328  mLastExtent = mExtent;
329  }
330 
331  mRenderContext.setLabelingEngine( mLabelingEngine );
332  if ( mLabelingEngine )
333  mLabelingEngine->init( mapSettings() );
334 
335  // render all layers in the stack, starting at the base
336  QListIterator<QString> li( mLayerSet );
337  li.toBack();
338 
339  QgsRectangle r1, r2;
340 
341  while ( li.hasPrevious() )
342  {
343  if ( mRenderContext.renderingStopped() )
344  {
345  break;
346  }
347 
348  // Store the painter in case we need to swap it out for the
349  // cache painter
350  QPainter * mypContextPainter = mRenderContext.painter();
351  // Flattened image for drawing when a blending mode is set
352  QImage * mypFlattenedImage = 0;
353 
354  QString layerId = li.previous();
355 
356  QgsDebugMsg( "Rendering at layer item " + layerId );
357 
358  // This call is supposed to cause the progress bar to
359  // advance. However, it seems that updating the progress bar is
360  // incompatible with having a QPainter active (the one that is
361  // passed into this function), as Qt produces a number of errors
362  // when try to do so. I'm (Gavin) not sure how to fix this, but
363  // added these comments and debug statement to help others...
364  QgsDebugMsg( "If there is a QPaintEngine error here, it is caused by an emit call" );
365 
366  //emit drawingProgress(myRenderCounter++, mLayerSet.size());
367  QgsMapLayer *ml = QgsMapLayerRegistry::instance()->mapLayer( layerId );
368 
369  if ( !ml )
370  {
371  QgsDebugMsg( "Layer not found in registry!" );
372  continue;
373  }
374 
375  QgsDebugMsg( QString( "layer %1: minscale:%2 maxscale:%3 scaledepvis:%4 extent:%5 blendmode:%6" )
376  .arg( ml->name() )
377  .arg( ml->minimumScale() )
378  .arg( ml->maximumScale() )
379  .arg( ml->hasScaleBasedVisibility() )
380  .arg( ml->extent().toString() )
381  .arg( ml->blendMode() )
382  );
383 
384  if ( mRenderContext.useAdvancedEffects() )
385  {
386  // Set the QPainter composition mode so that this layer is rendered using
387  // the desired blending mode
388  mypContextPainter->setCompositionMode( ml->blendMode() );
389  }
390 
391  if ( !ml->hasScaleBasedVisibility() || ( ml->minimumScale() <= mScale && mScale < ml->maximumScale() ) || mOverview )
392  {
393  connect( ml, SIGNAL( drawingProgress( int, int ) ), this, SLOT( onDrawingProgress( int, int ) ) );
394 
395  //
396  // Now do the call to the layer that actually does
397  // the rendering work!
398  //
399 
400  bool split = false;
401 
402  if ( hasCrsTransformEnabled() )
403  {
404  r1 = mExtent;
405  split = splitLayersExtent( ml, r1, r2 );
406  ct = transformation( ml );
407  mRenderContext.setExtent( r1 );
408  QgsDebugMsg( " extent 1: " + r1.toString() );
409  QgsDebugMsg( " extent 2: " + r2.toString() );
410  if ( !r1.isFinite() || !r2.isFinite() ) //there was a problem transforming the extent. Skip the layer
411  {
412  continue;
413  }
414  }
415  else
416  {
417  ct = NULL;
418  }
419 
420  mRenderContext.setCoordinateTransform( ct );
421 
422  QTime t;
423  if ( logRenderTime )
424  {
425  t.start();
426  }
427 
428  //decide if we have to scale the raster
429  //this is necessary in case QGraphicsScene is used
430  bool scaleRaster = false;
431  QgsMapToPixel rasterMapToPixel;
432  QgsMapToPixel bk_mapToPixel;
433 
434  if ( ml->type() == QgsMapLayer::RasterLayer && qAbs( rasterScaleFactor - 1.0 ) > 0.000001 )
435  {
436  scaleRaster = true;
437  }
438 
439  QSettings mySettings;
440 
441  // If we are drawing with an alternative blending mode then we need to render to a separate image
442  // before compositing this on the map. This effectively flattens the layer and prevents
443  // blending occuring between objects on the layer
444  // (this is not required for raster layers or when layer caching is enabled, since that has the same effect)
445  bool flattenedLayer = false;
446  if (( mRenderContext.useAdvancedEffects() ) && ( ml->type() == QgsMapLayer::VectorLayer || ml->type() == QgsMapLayer::RedliningLayer ) )
447  {
448  QgsVectorLayer* vl = qobject_cast<QgsVectorLayer *>( ml );
449  if ((( vl->blendMode() != QPainter::CompositionMode_SourceOver )
450  || ( vl->featureBlendMode() != QPainter::CompositionMode_SourceOver )
451  || ( vl->layerTransparency() != 0 ) ) )
452  {
453  flattenedLayer = true;
454  mypFlattenedImage = new QImage( mRenderContext.painter()->device()->width(),
455  mRenderContext.painter()->device()->height(), QImage::Format_ARGB32 );
456  if ( mypFlattenedImage->isNull() )
457  {
458  QgsDebugMsg( "insufficient memory for image " + QString::number( mRenderContext.painter()->device()->width() ) + "x" + QString::number( mRenderContext.painter()->device()->height() ) );
459  emit drawError( ml );
460  painter->end(); // drawError is not caught by anyone, so we end painting to notify caller
461  return;
462  }
463  mypFlattenedImage->fill( 0 );
464  QPainter * mypPainter = new QPainter( mypFlattenedImage );
465  if ( mySettings.value( "/Qgis/enable_anti_aliasing", true ).toBool() )
466  {
467  mypPainter->setRenderHint( QPainter::Antialiasing );
468  }
469  mypPainter->scale( rasterScaleFactor, rasterScaleFactor );
470  mRenderContext.setPainter( mypPainter );
471  }
472  }
473 
474  // Per feature blending mode
475  if (( mRenderContext.useAdvancedEffects() ) && ( ml->type() == QgsMapLayer::VectorLayer || ml->type() == QgsMapLayer::RedliningLayer ) )
476  {
477  QgsVectorLayer* vl = qobject_cast<QgsVectorLayer *>( ml );
478  if ( vl->featureBlendMode() != QPainter::CompositionMode_SourceOver )
479  {
480  // set the painter to the feature blend mode, so that features drawn
481  // on this layer will interact and blend with each other
482  mRenderContext.painter()->setCompositionMode( vl->featureBlendMode() );
483  }
484  }
485 
486  if ( scaleRaster )
487  {
488  bk_mapToPixel = mRenderContext.mapToPixel();
489  rasterMapToPixel = mRenderContext.mapToPixel();
490  rasterMapToPixel.setMapUnitsPerPixel( mRenderContext.mapToPixel().mapUnitsPerPixel() / rasterScaleFactor );
491  rasterMapToPixel.setYMaximum( mSize.height() * rasterScaleFactor );
492  mRenderContext.setMapToPixel( rasterMapToPixel );
493  mRenderContext.painter()->save();
494  mRenderContext.painter()->scale( 1.0 / rasterScaleFactor, 1.0 / rasterScaleFactor );
495  }
496 
497  if ( !ml->draw( mRenderContext ) )
498  {
499  emit drawError( ml );
500  }
501  else
502  {
503  QgsDebugMsg( "Layer rendered without issues" );
504  }
505 
506  if ( split )
507  {
508  mRenderContext.setExtent( r2 );
509  if ( !ml->draw( mRenderContext ) )
510  {
511  emit drawError( ml );
512  }
513  }
514 
515  if ( scaleRaster )
516  {
517  mRenderContext.setMapToPixel( bk_mapToPixel );
518  mRenderContext.painter()->restore();
519  }
520 
521  //apply layer transparency for vector and plugin layers
522  if (( mRenderContext.useAdvancedEffects() ) && ( ml->type() == QgsMapLayer::VectorLayer || ml->type() == QgsMapLayer::RedliningLayer || ml->type() == QgsMapLayer::PluginLayer ) )
523  {
524  int transparency = 0;
525  if ( qobject_cast<QgsVectorLayer *>( ml ) )
526  {
527  transparency = static_cast<QgsVectorLayer*>( ml )->layerTransparency();
528  }
529  else if ( qobject_cast<QgsPluginLayer*>( ml ) )
530  {
531  transparency = static_cast<QgsPluginLayer*>( ml )->layerTransparency();
532  }
533  if ( transparency != 0 )
534  {
535  // a layer transparency has been set, so update the alpha for the flattened layer
536  // by combining it with the layer transparency
537  QColor transparentFillColor = QColor( 0, 0, 0, 255 - ( 255 * transparency / 100 ) );
538  // use destination in composition mode to merge source's alpha with destination
539  mRenderContext.painter()->setCompositionMode( QPainter::CompositionMode_DestinationIn );
540  mRenderContext.painter()->fillRect( 0, 0, mRenderContext.painter()->device()->width(),
541  mRenderContext.painter()->device()->height(), transparentFillColor );
542  }
543  }
544 
545  if ( flattenedLayer )
546  {
547  // If we flattened this layer for alternate blend modes, composite it now
548  delete mRenderContext.painter();
549  mRenderContext.setPainter( mypContextPainter );
550  mypContextPainter->save();
551  mypContextPainter->scale( 1.0 / rasterScaleFactor, 1.0 / rasterScaleFactor );
552  mypContextPainter->drawImage( 0, 0, *( mypFlattenedImage ) );
553  mypContextPainter->restore();
554  delete mypFlattenedImage;
555  mypFlattenedImage = 0;
556  }
557 
558  disconnect( ml, SIGNAL( drawingProgress( int, int ) ), this, SLOT( onDrawingProgress( int, int ) ) );
559 
560  if ( logRenderTime )
561  {
562  QgsMessageLog::logMessage( "Layer " + ml->name() + " rendered in " + QString::number( t.elapsed() ) + " ms", "Rendering", QgsMessageLog::INFO );
563  }
564  }
565  else // layer not visible due to scale
566  {
567  QgsDebugMsg( "Layer not rendered because it is not within the defined "
568  "visibility scale range" );
569  }
570  } // while (li.hasPrevious())
571 
572  QgsDebugMsg( "Done rendering map layers" );
573 
574  // Reset the composition mode before rendering the labels
575  mRenderContext.painter()->setCompositionMode( QPainter::CompositionMode_SourceOver );
576 
577  if ( !mOverview )
578  {
579  // render all labels for vector layers in the stack, starting at the base
580  li.toBack();
581  while ( li.hasPrevious() )
582  {
583  if ( mRenderContext.renderingStopped() )
584  {
585  break;
586  }
587 
588  QString layerId = li.previous();
589 
590  // TODO: emit drawingProgress((myRenderCounter++),zOrder.size());
591  QgsMapLayer *ml = QgsMapLayerRegistry::instance()->mapLayer( layerId );
592 
593  if ( ml && ( ml->type() != QgsMapLayer::RasterLayer ) )
594  {
595  // only make labels if the layer is visible
596  // after scale dep viewing settings are checked
597  if ( !ml->hasScaleBasedVisibility() || ( ml->minimumScale() < mScale && mScale < ml->maximumScale() ) )
598  {
599  bool split = false;
600 
601  if ( hasCrsTransformEnabled() )
602  {
603  QgsRectangle r1 = mExtent;
604  split = splitLayersExtent( ml, r1, r2 );
605  ct = transformation( ml );
606  mRenderContext.setExtent( r1 );
607  }
608  else
609  {
610  ct = NULL;
611  }
612 
613  mRenderContext.setCoordinateTransform( ct );
614 
615  ml->drawLabels( mRenderContext );
616  if ( split )
617  {
618  mRenderContext.setExtent( r2 );
619  ml->drawLabels( mRenderContext );
620  }
621  }
622  }
623  }
624  } // if (!mOverview)
625 
626  // make sure progress bar arrives at 100%!
627  emit drawingProgress( 1, 1 );
628 
629  if ( mLabelingEngine )
630  {
631  // set correct extent
632  mRenderContext.setExtent( mExtent );
633  mRenderContext.setCoordinateTransform( NULL );
634 
635  mLabelingEngine->drawLabeling( mRenderContext );
636  mLabelingEngine->exit();
637  }
638  mDrawing = false;
639 }

732 bool QgsMapRenderer::splitLayersExtent( QgsMapLayer* layer, QgsRectangle& extent, QgsRectangle& r2 )
733 {
734  bool split = false;
735 
736  if ( hasCrsTransformEnabled() )
737  {
738  try
739  {
740 #ifdef QGISDEBUG
741  // QgsLogger::debug<QgsRectangle>("Getting extent of canvas in layers CS. Canvas is ", extent, __FILE__, __FUNCTION__, __LINE__);
742 #endif
743  // Split the extent into two if the source CRS is
744  // geographic and the extent crosses the split in
745  // geographic coordinates (usually +/- 180 degrees,
746  // and is assumed to be so here), and draw each
747  // extent separately.
748  static const double splitCoord = 180.0;
749 
750  const QgsCoordinateTransform *transform = transformation( layer );
751  if ( layer->crs().geographicFlag() )
752  {
753  // Note: ll = lower left point
754  // and ur = upper right point
755 
756  QgsPoint ll( extent.xMinimum(), extent.yMinimum() );
757  QgsPoint ur( extent.xMaximum(), extent.yMaximum() );
758 
759  if ( transform )
760  {
761  ll = transform->transform( ll.x(), ll.y(),
762  QgsCoordinateTransform::ReverseTransform );
763  ur = transform->transform( ur.x(), ur.y(),
764  QgsCoordinateTransform::ReverseTransform );
765  extent = transform->transformBoundingBox( extent, QgsCoordinateTransform::ReverseTransform );
766  }
767 
768  if ( ll.x() > ur.x() )
769  {
770  r2 = extent;
771  extent.setXMinimum( splitCoord );
772  r2.setXMaximum( splitCoord );
773  split = true;
774  }
775  }
776  else // can't cross 180
777  {
778  if ( transform )
779  {
780  extent = transform->transformBoundingBox( extent, QgsCoordinateTransform::ReverseTransform );
781  }
782  }
783  }
784  catch ( QgsCsException &cse )
785  {
786  Q_UNUSED( cse );
787  QgsDebugMsg( "Transform error caught" );
788  extent = QgsRectangle( -std::numeric_limits<double>::max(), -std::numeric_limits<double>::max(), std::numeric_limits<double>::max(), std::numeric_limits<double>::max() );
789  r2 = QgsRectangle( -std::numeric_limits<double>::max(), -std::numeric_limits<double>::max(), std::numeric_limits<double>::max(), std::numeric_limits<double>::max() );
790  }
791  }
792  return split;

939 void QgsMapRenderer::updateFullExtent()
940 {
941  QgsDebugMsg( "called." );
942  QgsMapLayerRegistry* registry = QgsMapLayerRegistry::instance();
943 
944  // reset the map canvas extent since the extent may now be smaller
945  // We can't use a constructor since QgsRectangle normalizes the rectangle upon construction
946  mFullExtent.setMinimal();
947 
948  // iterate through the map layers and test each layers extent
949  // against the current min and max values
950  QStringList::iterator it = mLayerSet.begin();
951  QgsDebugMsg( QString( "Layer count: %1" ).arg( mLayerSet.count() ) );
952  while ( it != mLayerSet.end() )
953  {
954  QgsMapLayer * lyr = registry->mapLayer( *it );
955  if ( lyr == NULL )
956  {
957  QgsDebugMsg( QString( "WARNING: layer '%1' not found in map layer registry!" ).arg( *it ) );
958  }
959  else
960  {
961  QgsDebugMsg( "Updating extent using " + lyr->name() );
962  QgsDebugMsg( "Input extent: " + lyr->extent().toString() );
963 
964  if ( lyr->extent().isNull() )
965  {
966  ++it;
967  continue;
968  }
969 
970  // Layer extents are stored in the coordinate system (CS) of the
971  // layer. The extent must be projected to the canvas CS
972  QgsRectangle extent = layerExtentToOutputExtent( lyr, lyr->extent() );
973 
974  QgsDebugMsg( "Output extent: " + extent.toString() );
975  mFullExtent.unionRect( extent );
976 
977  }
978  ++it;
979  }
980 
981  if ( mFullExtent.width() == 0.0 || mFullExtent.height() == 0.0 )
982  {
983  // If all of the features are at the one point, buffer the
984  // rectangle a bit. If they are all at zero, do something a bit
985  // more crude.
986 
987  if ( mFullExtent.xMinimum() == 0.0 && mFullExtent.xMaximum() == 0.0 &&
988  mFullExtent.yMinimum() == 0.0 && mFullExtent.yMaximum() == 0.0 )
989  {
990  mFullExtent.set( -1.0, -1.0, 1.0, 1.0 );
991  }
992  else
993  {
994  const double padFactor = 1e-8;
995  double widthPad = mFullExtent.xMinimum() * padFactor;
996  double heightPad = mFullExtent.yMinimum() * padFactor;
997  double xmin = mFullExtent.xMinimum() - widthPad;
998  double xmax = mFullExtent.xMaximum() + widthPad;
999  double ymin = mFullExtent.yMinimum() - heightPad;
1000  double ymax = mFullExtent.yMaximum() + heightPad;
1001  mFullExtent.set( xmin, ymin, xmax, ymax );
1002  }
1003  }
1004 
1005  QgsDebugMsg( "Full extent: " + mFullExtent.toString() );
1006 }
