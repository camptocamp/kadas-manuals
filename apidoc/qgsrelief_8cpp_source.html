86 int QgsRelief::processRaster( QProgressDialog* p )
87 {
88  //open input file
89  int xSize, ySize;
90  GDALDatasetH inputDataset = openInputFile( xSize, ySize );
91  if ( inputDataset == NULL )
92  {
93  return 1; //opening of input file failed
94  }
95 
96  //output driver
97  GDALDriverH outputDriver = openOutputDriver();
98  if ( outputDriver == 0 )
99  {
100  return 2;
101  }
102 
103  GDALDatasetH outputDataset = openOutputFile( inputDataset, outputDriver );
104  if ( outputDataset == NULL )
105  {
106  return 3; //create operation on output file failed
107  }
108 
109  //initialize dependency filters with cell sizes
110  mHillshadeFilter285->setCellSizeX( mCellSizeX );
111  mHillshadeFilter285->setCellSizeY( mCellSizeY );
112  mHillshadeFilter285->setZFactor( mZFactor );
113  mHillshadeFilter300->setCellSizeX( mCellSizeX );
114  mHillshadeFilter300->setCellSizeY( mCellSizeY );
115  mHillshadeFilter300->setZFactor( mZFactor );
116  mHillshadeFilter315->setCellSizeX( mCellSizeX );
117  mHillshadeFilter315->setCellSizeY( mCellSizeY );
118  mHillshadeFilter315->setZFactor( mZFactor );
119  mSlopeFilter->setCellSizeX( mCellSizeX );
120  mSlopeFilter->setCellSizeY( mCellSizeY );
121  mSlopeFilter->setZFactor( mZFactor );
122  mAspectFilter->setCellSizeX( mCellSizeX );
123  mAspectFilter->setCellSizeY( mCellSizeY );
124  mAspectFilter->setZFactor( mZFactor );
125 
126  //open first raster band for reading (operation is only for single band raster)
127  GDALRasterBandH rasterBand = GDALGetRasterBand( inputDataset, 1 );
128  if ( rasterBand == NULL )
129  {
130  GDALClose( inputDataset );
131  GDALClose( outputDataset );
132  return 4;
133  }
134  mInputNodataValue = GDALGetRasterNoDataValue( rasterBand, NULL );
135  mSlopeFilter->setInputNodataValue( mInputNodataValue );
136  mAspectFilter->setInputNodataValue( mInputNodataValue );
137  mHillshadeFilter285->setInputNodataValue( mInputNodataValue );
138  mHillshadeFilter300->setInputNodataValue( mInputNodataValue );
139  mHillshadeFilter315->setInputNodataValue( mInputNodataValue );
140 
141  GDALRasterBandH outputRedBand = GDALGetRasterBand( outputDataset, 1 );
142  GDALRasterBandH outputGreenBand = GDALGetRasterBand( outputDataset, 2 );
143  GDALRasterBandH outputBlueBand = GDALGetRasterBand( outputDataset, 3 );
144 
145  if ( outputRedBand == NULL || outputGreenBand == NULL || outputBlueBand == NULL )
146  {
147  GDALClose( inputDataset );
148  GDALClose( outputDataset );
149  return 5;
150  }
151  //try to set -9999 as nodata value
152  GDALSetRasterNoDataValue( outputRedBand, -9999 );
153  GDALSetRasterNoDataValue( outputGreenBand, -9999 );
154  GDALSetRasterNoDataValue( outputBlueBand, -9999 );
155  mOutputNodataValue = GDALGetRasterNoDataValue( outputRedBand, NULL );
156  mSlopeFilter->setOutputNodataValue( mOutputNodataValue );
157  mAspectFilter->setOutputNodataValue( mOutputNodataValue );
158  mHillshadeFilter285->setOutputNodataValue( mOutputNodataValue );
159  mHillshadeFilter300->setOutputNodataValue( mOutputNodataValue );
160  mHillshadeFilter315->setOutputNodataValue( mOutputNodataValue );
161 
162  if ( ySize < 3 ) //we require at least three rows (should be true for most datasets)
163  {
164  GDALClose( inputDataset );
165  GDALClose( outputDataset );
166  return 6;
167  }
168 
169  //keep only three scanlines in memory at a time
170  float* scanLine1 = ( float * ) CPLMalloc( sizeof( float ) * xSize );
171  float* scanLine2 = ( float * ) CPLMalloc( sizeof( float ) * xSize );
172  float* scanLine3 = ( float * ) CPLMalloc( sizeof( float ) * xSize );
173 
174  unsigned char* resultRedLine = ( unsigned char * ) CPLMalloc( sizeof( unsigned char ) * xSize );
175  unsigned char* resultGreenLine = ( unsigned char * ) CPLMalloc( sizeof( unsigned char ) * xSize );
176  unsigned char* resultBlueLine = ( unsigned char * ) CPLMalloc( sizeof( unsigned char ) * xSize );
177 
178  if ( p )
179  {
180  p->setMaximum( ySize );
181  }
182 
183  bool resultOk;
184 
185  //values outside the layer extent (if the 3x3 window is on the border) are sent to the processing method as (input) nodata values
186  for ( int i = 0; i < ySize; ++i )
187  {
188  if ( p )
189  {
190  p->setValue( i );
191  }
192 
193  if ( p && p->wasCanceled() )
194  {
195  break;
196  }
197 
198  if ( i == 0 )
199  {
200  //fill scanline 1 with (input) nodata for the values above the first row and feed scanline2 with the first row
201  for ( int a = 0; a < xSize; ++a )
202  {
203  scanLine1[a] = mInputNodataValue;
204  }
205  CPLErr err = GDALRasterIO( rasterBand, GF_Read, 0, 0, xSize, 1, scanLine2, xSize, 1, GDT_Float32, 0, 0 );
206  Q_UNUSED( err );
207  }
208  else
209  {
210  //normally fetch only scanLine3 and release scanline 1 if we move forward one row
211  CPLFree( scanLine1 );
212  scanLine1 = scanLine2;
213  scanLine2 = scanLine3;
214  scanLine3 = ( float * ) CPLMalloc( sizeof( float ) * xSize );
215  }
216 
217  if ( i == ySize - 1 ) //fill the row below the bottom with nodata values
218  {
219  for ( int a = 0; a < xSize; ++a )
220  {
221  scanLine3[a] = mInputNodataValue;
222  }
223  }
224  else
225  {
226  CPLErr err = GDALRasterIO( rasterBand, GF_Read, 0, i + 1, xSize, 1, scanLine3, xSize, 1, GDT_Float32, 0, 0 );
227  Q_UNUSED( err );
228  }
229 
230  for ( int j = 0; j < xSize; ++j )
231  {
232  if ( j == 0 )
233  {
234  resultOk = processNineCellWindow( &mInputNodataValue, &scanLine1[j], &scanLine1[j+1], &mInputNodataValue, &scanLine2[j], \
235  &scanLine2[j+1], &mInputNodataValue, &scanLine3[j], &scanLine3[j+1], \
236  &resultRedLine[j], &resultGreenLine[j], &resultBlueLine[j] );
237  }
238  else if ( j == xSize - 1 )
239  {
240  resultOk = processNineCellWindow( &scanLine1[j-1], &scanLine1[j], &mInputNodataValue, &scanLine2[j-1], &scanLine2[j], \
241  &mInputNodataValue, &scanLine3[j-1], &scanLine3[j], &mInputNodataValue, \
242  &resultRedLine[j], &resultGreenLine[j], &resultBlueLine[j] );
243  }
244  else
245  {
246  resultOk = processNineCellWindow( &scanLine1[j-1], &scanLine1[j], &scanLine1[j+1], &scanLine2[j-1], &scanLine2[j], \
247  &scanLine2[j+1], &scanLine3[j-1], &scanLine3[j], &scanLine3[j+1], \
248  &resultRedLine[j], &resultGreenLine[j], &resultBlueLine[j] );
249  }
250 
251  if ( !resultOk )
252  {
253  resultRedLine[j] = mOutputNodataValue;
254  resultGreenLine[j] = mOutputNodataValue;
255  resultBlueLine[j] = mOutputNodataValue;
256  }
257  }
258 
259  CPLErr err = GDALRasterIO( outputRedBand, GF_Write, 0, i, xSize, 1, resultRedLine, xSize, 1, GDT_Byte, 0, 0 );
260  err = GDALRasterIO( outputGreenBand, GF_Write, 0, i, xSize, 1, resultGreenLine, xSize, 1, GDT_Byte, 0, 0 );
261  err = GDALRasterIO( outputBlueBand, GF_Write, 0, i, xSize, 1, resultBlueLine, xSize, 1, GDT_Byte, 0, 0 );
262  Q_UNUSED( err );
263  }
264 
265  if ( p )
266  {
267  p->setValue( ySize );
268  }
269 
270  CPLFree( resultRedLine );
271  CPLFree( resultBlueLine );
272  CPLFree( resultGreenLine );
273  CPLFree( scanLine1 );
274  CPLFree( scanLine2 );
275  CPLFree( scanLine3 );
276 
277  GDALClose( inputDataset );
278 
279  if ( p && p->wasCanceled() )
280  {
281  //delete the dataset without closing (because it is faster)
282  GDALDeleteDataset( outputDriver, TO8F( mOutputFile ) );
283  return 7;
284  }
285  GDALClose( outputDataset );
286 
287  return 0;
288 }

290 bool QgsRelief::processNineCellWindow( float* x1, float* x2, float* x3, float* x4, float* x5, float* x6, float* x7, float* x8, float* x9,
291  unsigned char* red, unsigned char* green, unsigned char* blue )
292 {
293  //1. component: color and hillshade from 300 degrees
294  int r = 0;
295  int g = 0;
296  int b = 0;
297 
298  float hillShadeValue300 = mHillshadeFilter300->processNineCellWindow( x1, x2, x3, x4, x5, x6, x7, x8, x9 );
299  if ( hillShadeValue300 != mOutputNodataValue )
300  {
301  if ( !setElevationColor( *x5, &r, &g, &b ) )
302  {
303  r = hillShadeValue300;
304  g = hillShadeValue300;
305  b = hillShadeValue300;
306  }
307  else
308  {
309  r = r / 2.0 + hillShadeValue300 / 2.0;
310  g = g / 2.0 + hillShadeValue300 / 2.0;
311  b = b / 2.0 + hillShadeValue300 / 2.0;
312  }
313  }
314 
315  //2. component: hillshade and slope
316  float hillShadeValue315 = mHillshadeFilter315->processNineCellWindow( x1, x2, x3, x4, x5, x6, x7, x8, x9 );
317  float slope = mSlopeFilter->processNineCellWindow( x1, x2, x3, x4, x5, x6, x7, x8, x9 );
318  if ( hillShadeValue315 != mOutputNodataValue && slope != mOutputNodataValue )
319  {
320  int r2, g2, b2;
321  if ( slope > 15 )
322  {
323  r2 = 0 / 2.0 + hillShadeValue315 / 2.0;
324  g2 = 0 / 2.0 + hillShadeValue315 / 2.0;
325  b2 = 0 / 2.0 + hillShadeValue315 / 2.0;
326  }
327  else if ( slope >= 1 )
328  {
329  int slopeValue = 255 - ( slope / 15.0 * 255.0 );
330  r2 = slopeValue / 2.0 + hillShadeValue315 / 2.0;
331  g2 = slopeValue / 2.0 + hillShadeValue315 / 2.0;
332  b2 = slopeValue / 2.0 + hillShadeValue315 / 2.0;
333  }
334  else
335  {
336  r2 = hillShadeValue315; g2 = hillShadeValue315; b2 = hillShadeValue315;
337  }
338 
339  //combine with r,g,b with 70 percentage coverage
340  r = r * 0.7 + r2 * 0.3;
341  g = g * 0.7 + g2 * 0.3;
342  b = b * 0.7 + b2 * 0.3;
343  }
344 
345  //3. combine yellow aspect with 10% transparency, illumination from 285 degrees
346  float hillShadeValue285 = mHillshadeFilter285->processNineCellWindow( x1, x2, x3, x4, x5, x6, x7, x8, x9 );
347  float aspect = mAspectFilter->processNineCellWindow( x1, x2, x3, x4, x5, x6, x7, x8, x9 );
348  if ( hillShadeValue285 != mOutputNodataValue && aspect != mOutputNodataValue )
349  {
350  double angle_diff = qAbs( 285 - aspect );
351  if ( angle_diff > 180 )
352  {
353  angle_diff -= 180;
354  }
355 
356  int r3, g3, b3;
357  if ( angle_diff < 90 )
358  {
359  int aspectVal = ( 1 - cos( angle_diff * M_PI / 180 ) ) * 255;
360  r3 = 0.5 * 255 + hillShadeValue315 * 0.5;
361  g3 = 0.5 * 255 + hillShadeValue315 * 0.5;
362  b3 = 0.5 * aspectVal + hillShadeValue315 * 0.5;
363  }
364  else //white
365  {
366  r3 = 0.5 * 255 + hillShadeValue315 * 0.5;
367  g3 = 0.5 * 255 + hillShadeValue315 * 0.5;
368  b3 = 0.5 * 255 + hillShadeValue315 * 0.5;
369  }
370 
371  r = r3 * 0.1 + r * 0.9;
372  g = g3 * 0.1 + g * 0.9;
373  b = b3 * 0.1 + b * 0.9;
374  }
375 
376  *red = ( unsigned char )r;
377  *green = ( unsigned char )g;
378  *blue = ( unsigned char )b;
379  return true;
380 }

490 //this function is mainly there for debugging
491 bool QgsRelief::exportFrequencyDistributionToCsv( const QString& file )
492 {
493  int nCellsX, nCellsY;
494  GDALDatasetH inputDataset = openInputFile( nCellsX, nCellsY );
495  if ( inputDataset == NULL )
496  {
497  return false;
498  }
499 
500  //open first raster band for reading (elevation raster is always single band)
501  GDALRasterBandH elevationBand = GDALGetRasterBand( inputDataset, 1 );
502  if ( elevationBand == NULL )
503  {
504  GDALClose( inputDataset );
505  return false;
506  }
507 
508  //1. get minimum and maximum of elevation raster -> 252 elevation classes
509  int minOk, maxOk;
510  double minMax[2];
511  minMax[0] = GDALGetRasterMinimum( elevationBand, &minOk );
512  minMax[1] = GDALGetRasterMaximum( elevationBand, &maxOk );
513 
514  if ( !minOk || !maxOk )
515  {
516  GDALComputeRasterMinMax( elevationBand, TRUE, minMax );
517  }
518 
519  //2. go through raster cells and get frequency of classes
520 
521  //store elevation frequency in 256 elevation classes
522  double frequency[252];
523  double frequencyClassRange = ( minMax[1] - minMax[0] ) / 252.0;
524  //initialize to zero
525  for ( int i = 0; i < 252; ++i )
526  {
527  frequency[i] = 0;
528  }
529 
530  float* scanLine = ( float * ) CPLMalloc( sizeof( float ) * nCellsX );
531  int elevationClass = -1;
532 
533  for ( int i = 0; i < nCellsY; ++i )
534  {
535  CPLErr err = GDALRasterIO( elevationBand, GF_Read, 0, i, nCellsX, 1,
536  scanLine, nCellsX, 1, GDT_Float32,
537  0, 0 );
538  Q_UNUSED( err );
539  for ( int j = 0; j < nCellsX; ++j )
540  {
541  elevationClass = frequencyClassForElevation( scanLine[j], minMax[0], frequencyClassRange );
542  if ( elevationClass >= 0 )
543  {
544  frequency[elevationClass] += 1.0;
545  }
546  }
547  }
548 
549  CPLFree( scanLine );
550 
551  //log10 transformation for all frequency values
552  for ( int i = 0; i < 252; ++i )
553  {
554  frequency[i] = log10( frequency[i] );
555  }
556 
557  //write out frequency values to csv file for debugging
558  QFile outFile( file );
559  if ( !outFile.open( QIODevice::WriteOnly ) )
560  {
561  return false;
562  }
563 
564  QTextStream outstream( &outFile );
565  for ( int i = 0; i < 252; ++i )
566  {
567  outstream << QString::number( i ) + "," + QString::number( frequency[i] ) << endl;
568  }
569  outFile.close();
570  return true;
571 }

692 void QgsRelief::optimiseClassBreaks( QList<int>& breaks, double* frequencies )
693 {
694  int nClasses = breaks.size() - 1;
695  double* a = new double[nClasses]; //slopes
696  double* b = new double[nClasses]; //y-offsets
697 
698  for ( int i = 0; i < nClasses; ++i )
699  {
700  //get all the values between the class breaks into input
701  QList< QPair < int, double > > regressionInput;
702  for ( int j = breaks.at( i ); j < breaks.at( i + 1 ); ++j )
703  {
704  regressionInput.push_back( qMakePair( j, frequencies[j] ) );
705  }
706 
707  double aParam, bParam;
708  if ( regressionInput.size() > 0 && calculateRegression( regressionInput, aParam, bParam ) )
709  {
710  a[i] = aParam;
711  b[i] = bParam;
712  }
713  else
714  {
715  a[i] = 0;
716  b[i] = 0; //better default value
717  }
718  }
719 
720  QList<int> classesToRemove;
721 
722  //shift class boundaries or eliminate classes which fall together
723  for ( int i = 1; i < nClasses ; ++i )
724  {
725  if ( breaks[i] == breaks[ i - 1 ] )
726  {
727  continue;
728  }
729 
730  if ( qgsDoubleNear( a[i - 1 ], a[i] ) )
731  {
732  continue;
733  }
734  else
735  {
736  int newX = ( b[i - 1] - b[ i ] ) / ( a[ i ] - a[ i - 1 ] );
737 
738  if ( newX <= breaks[i - 1] )
739  {
740  newX = breaks[i - 1];
741  // classesToRemove.push_back( i );//remove this class later as it falls together with the preceding one
742  }
743  else if ( i < nClasses - 1 && newX >= breaks[i + 1] )
744  {
745  newX = breaks[i + 1];
746  // classesToRemove.push_back( i );//remove this class later as it falls together with the next one
747  }
748 
749  breaks[i] = newX;
750  }
751  }
752 
753  for ( int i = classesToRemove.size() - 1; i >= 0; --i )
754  {
755  breaks.removeAt( classesToRemove.at( i ) );
756  }
757 
758  delete[] a;
759  delete[] b;
760 }

767 bool QgsRelief::calculateRegression( const QList< QPair < int, double > >& input, double& a, double& b )
768 {
769  double xMean, yMean;
770  double xSum = 0;
771  double ySum = 0;
772  QList< QPair < int, double > >::const_iterator inputIt = input.constBegin();
773  for ( ; inputIt != input.constEnd(); ++inputIt )
774  {
775  xSum += inputIt->first;
776  ySum += inputIt->second;
777  }
778  xMean = xSum / input.size();
779  yMean = ySum / input.size();
780 
781  double sumCounter = 0;
782  double sumDenominator = 0;
783  inputIt = input.constBegin();
784  for ( ; inputIt != input.constEnd(); ++inputIt )
785  {
786  sumCounter += (( inputIt->first - xMean ) * ( inputIt->second - yMean ) );
787  sumDenominator += (( inputIt->first - xMean ) * ( inputIt->first - xMean ) );
788  }
789 
790  a = sumCounter / sumDenominator;
791  b = yMean - a * xMean;
792 
793  return true;
794 }
