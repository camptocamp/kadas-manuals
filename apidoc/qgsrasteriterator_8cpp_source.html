52 bool QgsRasterIterator::readNextRasterPart( int bandNumber,
53  int& nCols, int& nRows,
54  QgsRasterBlock **block,
55  int& topLeftCol, int& topLeftRow )
56 {
57  QgsDebugMsg( "Entered" );
58  *block = 0;
59  //get partinfo
60  QMap<int, RasterPartInfo>::iterator partIt = mRasterPartInfos.find( bandNumber );
61  if ( partIt == mRasterPartInfos.end() )
62  {
63  return false;
64  }
65 
66  RasterPartInfo& pInfo = partIt.value();
67 
68  // If we started with zero cols or zero rows, just return (avoids divide by zero below)
69  if ( 0 == pInfo.nCols || 0 == pInfo.nRows )
70  {
71  return false;
72  }
73 
74  //remove last data block
75  delete pInfo.prj;
76  pInfo.prj = 0;
77 
78  //already at end
79  if ( pInfo.currentCol == pInfo.nCols && pInfo.currentRow == pInfo.nRows )
80  {
81  return false;
82  }
83 
84  //read data block
85  nCols = qMin( mMaximumTileWidth, pInfo.nCols - pInfo.currentCol );
86  nRows = qMin( mMaximumTileHeight, pInfo.nRows - pInfo.currentRow );
87  QgsDebugMsg( QString( "nCols = %1 nRows = %2" ).arg( nCols ).arg( nRows ) );
88 
89  //get subrectangle
90  QgsRectangle viewPortExtent = mExtent;
91  double xmin = viewPortExtent.xMinimum() + pInfo.currentCol / ( double )pInfo.nCols * viewPortExtent.width();
92  double xmax = viewPortExtent.xMinimum() + ( pInfo.currentCol + nCols ) / ( double )pInfo.nCols * viewPortExtent.width();
93  double ymin = viewPortExtent.yMaximum() - ( pInfo.currentRow + nRows ) / ( double )pInfo.nRows * viewPortExtent.height();
94  double ymax = viewPortExtent.yMaximum() - pInfo.currentRow / ( double )pInfo.nRows * viewPortExtent.height();
95  QgsRectangle blockRect( xmin, ymin, xmax, ymax );
96 
97  *block = mInput->block2( bandNumber, blockRect, nCols, nRows, mFeedback );
98  topLeftCol = pInfo.currentCol;
99  topLeftRow = pInfo.currentRow;
100 
101  pInfo.currentCol += nCols;
102  if ( pInfo.currentCol == pInfo.nCols && pInfo.currentRow + nRows == pInfo.nRows ) //end of raster
103  {
104  pInfo.currentRow = pInfo.nRows;
105  }
106  else if ( pInfo.currentCol == pInfo.nCols ) //start new row
107  {
108  pInfo.currentCol = 0;
109  pInfo.currentRow += nRows;
110  }
111 
112  return true;
113 }
