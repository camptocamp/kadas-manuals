49 void QgsLayerTreeRegistryBridge::layersAdded( QList<QgsMapLayer*> layers )
50 {
51  if ( !mEnabled )
52  return;
53 
54  QList<QgsLayerTreeNode*> nodes;
55  int maxPriority = 0;
56  foreach ( QgsMapLayer* layer, layers )
57  {
58  QgsLayerTreeLayer* nodeLayer = new QgsLayerTreeLayer( layer );
59  nodeLayer->setVisible( mNewLayersVisible ? Qt::Checked : Qt::Unchecked );
60 
61  nodes << nodeLayer;
62 
63  // check whether the layer is marked as embedded
64  QString projectFile = QgsProject::instance()->layerIsEmbedded( nodeLayer->layerId() );
65  if ( !projectFile.isEmpty() )
66  {
67  nodeLayer->setCustomProperty( "embedded", 1 );
68  nodeLayer->setCustomProperty( "embedded_project", projectFile );
69  }
70  maxPriority = std::max( maxPriority, layer->priority() );
71  }
72 
73  int addMode = QSettings().value( "/Qgis/layerLegendAddMode", 0 ).toInt();
74  int ins = addMode == 0 ? 0 : mInsertionPointIndex;
75  QgsLayerTreeGroup* group = addMode == 0 ? mRoot : mInsertionPointGroup;
76 
77  // Don't add layers into mutually exclusive groups or disabled groups
78  while ( group != mRoot && ( group->isMutuallyExclusive() || group->isVisible() != Qt::Checked ) )
79  {
80  QgsLayerTreeNode* parent = group->parent();
81  if ( !QgsLayerTree::isGroup( parent ) )
82  {
83  break;
84  }
85  group = QgsLayerTree::toGroup( parent );
86  }
87 
88  // Modify insertion point to the first possible slot below layers with higher priority
89  QList<QgsLayerTreeNode*> childNodes = group->children();
90  for ( int i = 0, n = childNodes.size(); i < n; ++i )
91  {
92  if ( childNodes[i]->nodeType() == QgsLayerTreeNode::NodeLayer )
93  {
94  QgsLayerTreeLayer* layerNode = static_cast<QgsLayerTreeLayer*>( childNodes[i] );
95  if ( layerNode->layer()->priority() > maxPriority )
96  {
97  ins = i + 1;
98  }
99  }
100  }
101 
102  // add new layers to the right place
103  group->insertChildNodes( ins, nodes );
104 
105  // tell other components that layers have been added - this signal is used in QGIS to auto-select the first layer
106  emit addedLayersToLayerTree( layers );
107 }

109 void QgsLayerTreeRegistryBridge::layersWillBeRemoved( QStringList layerIds )
110 {
111  QgsDebugMsg( QString( "%1 layers will be removed, enabled:%2" ).arg( layerIds.count() ).arg( mEnabled ) );
112 
113  if ( !mEnabled )
114  return;
115 
116  // when we start removing child nodes, the bridge would try to remove those layers from
117  // the registry _again_ in groupRemovedChildren() - this prevents it
118  mRegistryRemovingLayers = true;
119 
120  foreach ( QString layerId, layerIds )
121  {
122  QgsLayerTreeLayer* nodeLayer = mRoot->findLayer( layerId );
123  if ( nodeLayer )
124  qobject_cast<QgsLayerTreeGroup*>( nodeLayer->parent() )->removeChildNode( nodeLayer );
125  }
126 
127  mRegistryRemovingLayers = false;
128 }

131 static void _collectLayerIdsInGroup( QgsLayerTreeGroup* group, int indexFrom, int indexTo, QStringList& lst )
132 {
133  for ( int i = indexFrom; i <= indexTo; ++i )
134  {
135  QgsLayerTreeNode* child = group->children()[i];
136  if ( QgsLayerTree::isLayer( child ) )
137  {
138  lst << QgsLayerTree::toLayer( child )->layerId();
139  }
140  else if ( QgsLayerTree::isGroup( child ) )
141  {
142  _collectLayerIdsInGroup( QgsLayerTree::toGroup( child ), 0, child->children().count() - 1, lst );
143  }
144  }
145 }

147 void QgsLayerTreeRegistryBridge::groupWillRemoveChildren( QgsLayerTreeNode* node, int indexFrom, int indexTo )
148 {
149  if ( mRegistryRemovingLayers )
150  return; // do not try to remove those layers again
151 
152  Q_ASSERT( mLayerIdsForRemoval.isEmpty() );
153 
154  Q_ASSERT( QgsLayerTree::isGroup( node ) );
155  QgsLayerTreeGroup* group = QgsLayerTree::toGroup( node );
156 
157  _collectLayerIdsInGroup( group, indexFrom, indexTo, mLayerIdsForRemoval );
158 }

160 void QgsLayerTreeRegistryBridge::groupRemovedChildren()
161 {
162  if ( mRegistryRemovingLayers )
163  return; // do not try to remove those layers again
164 
165  // remove only those that really do not exist in the tree
166  // (ignores layers that were dragged'n'dropped: 1. drop new 2. remove old)
167  QStringList toRemove;
168  foreach ( QString layerId, mLayerIdsForRemoval )
169  if ( !mRoot->findLayer( layerId ) )
170  toRemove << layerId;
171  mLayerIdsForRemoval.clear();
172 
173  QgsDebugMsg( QString( "%1 layers will be removed" ).arg( toRemove.count() ) );
174 
175  // delay the removal of layers from the registry. There may be other slots connected to map layer registry's signals
176  // that might disrupt the execution flow - e.g. a processEvents() call may force update of layer tree view with
177  // semi-broken tree model
178  QMetaObject::invokeMethod( this, "removeLayersFromRegistry", Qt::QueuedConnection, Q_ARG( QStringList, toRemove ) );
179 }
