505 bool QgsRuleBasedRendererV2::Rule::renderFeature( QgsRuleBasedRendererV2::FeatureToRender& featToRender, QgsRenderContext& context, QgsRuleBasedRendererV2::RenderQueue& renderQueue )
506 {
507  if ( !isFilterOK( featToRender.feat ) )
508  return false;
509 
510  bool rendered = false;
511 
512  // create job for this feature and this symbol, add to list of jobs
513  if ( mSymbol )
514  {
515  // add job to the queue: each symbol's zLevel must be added
516  foreach ( int normZLevel, mSymbolNormZLevels )
517  {
518  //QgsDebugMsg(QString("add job at level %1").arg(normZLevel));
519  renderQueue[normZLevel].jobs.append( new RenderJob( featToRender, mSymbol ) );
520  }
521  rendered = true;
522  }
523 
524  bool willrendersomething = false;
525 
526  // process children
527  for ( QList<Rule*>::iterator it = mActiveChildren.begin(); it != mActiveChildren.end(); ++it )
528  {
529  Rule* rule = *it;
530  if ( rule->isElse() )
531  {
532  // Don't process else rules yet
533  continue;
534  }
535  willrendersomething |= rule->renderFeature( featToRender, context, renderQueue );
536  rendered |= willrendersomething;
537  }
538 
539  // If none of the rules passed then we jump into the else rules and process them.
540  if ( !willrendersomething )
541  {
542  foreach ( Rule* rule, mElseRules )
543  {
544  rendered |= rule->renderFeature( featToRender, context, renderQueue );
545  }
546  }
547 
548  return rendered;
549 }
