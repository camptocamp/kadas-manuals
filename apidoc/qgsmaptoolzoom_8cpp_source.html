48 void QgsMapToolZoom::canvasMoveEvent( QMouseEvent * e )
49 {
50  if ( !( e->buttons() & Qt::LeftButton ) )
51  return;
52 
53  if ( !mDragging )

81 void QgsMapToolZoom::canvasReleaseEvent( QMouseEvent * e )
82 {
83  if ( e->button() != Qt::LeftButton )
84  return;
85 
86  // We are not really dragging in this case. This is sometimes caused by
87  // a pen based computer reporting a press, move, and release, all the
88  // one point.
89  if ( mDragging && ( mZoomRect.topLeft() == mZoomRect.bottomRight() ) )
90  {
91  mDragging = false;
92  delete mRubberBand;
93  mRubberBand = 0;
94  }
95 
96  if ( mDragging )
97  {
98  mDragging = false;
99  delete mRubberBand;
100  mRubberBand = 0;
101 
102  // store the rectangle
103  mZoomRect.setRight( e->pos().x() );
104  mZoomRect.setBottom( e->pos().y() );
105 
106  //account for bottom right -> top left dragging
107  mZoomRect = mZoomRect.normalized();
108 
109  // set center and zoom
110  QSize zoomRectSize = mZoomRect.size();
111  const QgsMapSettings& mapSettings = mCanvas->mapSettings();
112  QSize canvasSize = mapSettings.outputSize();
113  double sfx = ( double )zoomRectSize.width() / canvasSize.width();
114  double sfy = ( double )zoomRectSize.height() / canvasSize.height();
115  double sf = qMax( sfx, sfy );
116 
117  const QgsMapToPixel* m2p = mCanvas->getCoordinateTransform();
118  QgsPoint c = m2p->toMapCoordinates( mZoomRect.center() );
119 
120  mCanvas->setCenter( c );
121  mCanvas->zoomByFactor( mZoomOut ? 1.0 / sf : sf );
122  }
123  else // not dragging
124  {
125  // change to zoom in/out by the default multiple
126  mCanvas->zoomWithCenter( e->x(), e->y(), !mZoomOut );
127  }
128 }
