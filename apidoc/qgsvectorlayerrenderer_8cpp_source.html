107 bool QgsVectorLayerRenderer::render()
108 {
109  if ( mGeometryType == QGis::NoGeometry || mGeometryType == QGis::UnknownGeometry )
110  return true;
111 
112  if ( !mRendererV2 )
113  {
114  mErrors.append( QObject::tr( "No renderer for drawing." ) );
115  return false;
116  }
117 
118  // Per feature blending mode
119  if ( mContext.useAdvancedEffects() && mFeatureBlendMode != QPainter::CompositionMode_SourceOver )
120  {
121  // set the painter to the feature blend mode, so that features drawn
122  // on this layer will interact and blend with each other
123  mContext.painter()->setCompositionMode( mFeatureBlendMode );
124  }
125 
126  mRendererV2->startRender( mContext, mFields );
127 
128  QgsRectangle requestExtent = mContext.extent();
129  mRendererV2->modifyRequestExtent( requestExtent, mContext );
130 
131  QgsFeatureRequest featureRequest = QgsFeatureRequest()
132  .setFilterRect( requestExtent )
133  .setSubsetOfAttributes( mAttrNames, mFields );
134 
135  // enable the simplification of the geometries (Using the current map2pixel context) before send it to renderer engine.
136  if ( mSimplifyGeometry )
137  {
138  double map2pixelTol = mSimplifyMethod.threshold();
139  bool validTransform = true;
140 
141  const QgsMapToPixel& mtp = mContext.mapToPixel();
142  map2pixelTol *= mtp.mapUnitsPerPixel();
143  const QgsCoordinateTransform* ct = mContext.coordinateTransform();
144 
145  // resize the tolerance using the change of size of an 1-BBOX from the source CoordinateSystem to the target CoordinateSystem
146  if ( ct && !(( QgsCoordinateTransform* )ct )->isShortCircuited() )
147  {
148  try
149  {
150  QgsPoint center = mContext.extent().center();
151  double rectSize = ct->sourceCrs().geographicFlag() ? 0.0008983 /* ~100/(40075014/360=111319.4833) */ : 100;
152 
153  QgsRectangle sourceRect = QgsRectangle( center.x(), center.y(), center.x() + rectSize, center.y() + rectSize );
154  QgsRectangle targetRect = ct->transform( sourceRect );
155 
156  QgsDebugMsg( QString( "Simplify - SourceTransformRect=%1" ).arg( sourceRect.toString( 16 ) ) );
157  QgsDebugMsg( QString( "Simplify - TargetTransformRect=%1" ).arg( targetRect.toString( 16 ) ) );
158 
159  if ( !sourceRect.isEmpty() && sourceRect.isFinite() && !targetRect.isEmpty() && targetRect.isFinite() )
160  {
161  QgsPoint minimumSrcPoint( sourceRect.xMinimum(), sourceRect.yMinimum() );
162  QgsPoint maximumSrcPoint( sourceRect.xMaximum(), sourceRect.yMaximum() );
163  QgsPoint minimumDstPoint( targetRect.xMinimum(), targetRect.yMinimum() );
164  QgsPoint maximumDstPoint( targetRect.xMaximum(), targetRect.yMaximum() );
165 
166  double sourceHypothenuse = sqrt( minimumSrcPoint.sqrDist( maximumSrcPoint ) );
167  double targetHypothenuse = sqrt( minimumDstPoint.sqrDist( maximumDstPoint ) );
168 
169  QgsDebugMsg( QString( "Simplify - SourceHypothenuse=%1" ).arg( sourceHypothenuse ) );
170  QgsDebugMsg( QString( "Simplify - TargetHypothenuse=%1" ).arg( targetHypothenuse ) );
171 
172  if ( targetHypothenuse != 0 )
173  map2pixelTol *= ( sourceHypothenuse / targetHypothenuse );
174  }
175  }
176  catch ( QgsCsException &cse )
177  {
178  QgsMessageLog::logMessage( QObject::tr( "Simplify transform error caught: %1" ).arg( cse.what() ), QObject::tr( "CRS" ) );
179  validTransform = false;
180  }
181  }
182 
183  if ( validTransform )
184  {
185  QgsSimplifyMethod simplifyMethod;
186  simplifyMethod.setMethodType( QgsSimplifyMethod::OptimizeForRendering );
187  simplifyMethod.setTolerance( map2pixelTol );
188  simplifyMethod.setForceLocalOptimization( mSimplifyMethod.forceLocalOptimization() );
189 
190  featureRequest.setSimplifyMethod( simplifyMethod );
191 
192  QgsVectorSimplifyMethod vectorMethod = mSimplifyMethod;
193  mContext.setVectorSimplifyMethod( vectorMethod );
194  }
195  else
196  {
197  QgsVectorSimplifyMethod vectorMethod;
198  vectorMethod.setSimplifyHints( QgsVectorSimplifyMethod::NoSimplification );
199  mContext.setVectorSimplifyMethod( vectorMethod );
200  }
201  }
202  else
203  {
204  QgsVectorSimplifyMethod vectorMethod;
205  vectorMethod.setSimplifyHints( QgsVectorSimplifyMethod::NoSimplification );
206  mContext.setVectorSimplifyMethod( vectorMethod );
207  }
208 
209  QgsVectorLayerFeatureSource source( mLayer );
210  QgsFeatureIterator fit = source.getFeatures( featureRequest );
211 
212  if (( mRendererV2->capabilities() & QgsFeatureRendererV2::SymbolLevels ) && mRendererV2->usingSymbolLevels() )
213  drawRendererV2Levels( fit );
214  else
215  drawRendererV2( fit );
216 
217  //apply layer transparency for vector layers
218  if ( mContext.useAdvancedEffects() && mLayerTransparency != 0 )
219  {
220  // a layer transparency has been set, so update the alpha for the flattened layer
221  // by combining it with the layer transparency
222  QColor transparentFillColor = QColor( 0, 0, 0, 255 - ( 255 * mLayerTransparency / 100 ) );
223  // use destination in composition mode to merge source's alpha with destination
224  mContext.painter()->setCompositionMode( QPainter::CompositionMode_DestinationIn );
225  mContext.painter()->fillRect( 0, 0, mContext.painter()->device()->width(),
226  mContext.painter()->device()->height(), transparentFillColor );
227  }
228 
229  return true;
230 }

245 void QgsVectorLayerRenderer::drawRendererV2( QgsFeatureIterator& fit )
246 {
247  QgsFeature fet;
248  while ( fit.nextFeature( fet ) )
249  {
250  try
251  {
252  if ( !fet.geometry() )
253  continue; // skip features without geometry
254 
255  if ( mContext.renderingStopped() )
256  {
257  QgsDebugMsg( QString( "Drawing of vector layer %1 cancelled." ).arg( layerID() ) );
258  break;
259  }
260 
261  bool sel = mContext.showSelection() && mSelectedFeatureIds.contains( fet.id() );
262  bool drawMarker = ( mDrawVertexMarkers && mContext.drawEditingInformation() && ( !mVertexMarkerOnlyForSelection || sel ) );
263 
264  if ( mCache )
265  {
266  // Cache this for the use of (e.g.) modifying the feature's uncommitted geometry.
267  mCache->cacheGeometry( fet.id(), *fet.geometry() );
268  }
269 
270  // render feature
271  bool rendered = mRendererV2->renderFeature( fet, mContext, -1, sel, drawMarker );
272 
273  // labeling - register feature
274  Q_UNUSED( rendered );
275  if ( rendered && mContext.labelingEngine() )
276  {
277  if ( mLabeling )
278  {
279  mContext.labelingEngine()->registerFeature( mLayerID, fet, mContext );
280  }
281  if ( mDiagrams )
282  {
283  mContext.labelingEngine()->registerDiagramFeature( mLayerID, fet, mContext );
284  }
285  }
286  }
287  catch ( const QgsCsException &cse )
288  {
289  Q_UNUSED( cse );
290  QgsDebugMsg( QString( "Failed to transform a point while drawing a feature with ID '%1'. Ignoring this feature. %2" )
291  .arg( fet.id() ).arg( cse.what() ) );
292  }
293  }
294 
295  stopRendererV2( NULL );
296 }

298 void QgsVectorLayerRenderer::drawRendererV2Levels( QgsFeatureIterator& fit )
299 {
300  QHash< QgsSymbolV2*, QList<QgsFeature> > features; // key = symbol, value = array of features
301 
302  QgsSingleSymbolRendererV2* selRenderer = NULL;
303  if ( !mSelectedFeatureIds.isEmpty() )
304  {
305  selRenderer = new QgsSingleSymbolRendererV2( QgsSymbolV2::defaultSymbol( mGeometryType ) );
306  selRenderer->symbol()->setColor( mContext.selectionColor() );
307  selRenderer->setVertexMarkerAppearance( mVertexMarkerStyle, mVertexMarkerSize );
308  selRenderer->startRender( mContext, mFields );
309  }
310 
311  // 1. fetch features
312  QgsFeature fet;
313  while ( fit.nextFeature( fet ) )
314  {
315  if ( !fet.geometry() )
316  continue; // skip features without geometry
317 
318  if ( mContext.renderingStopped() )
319  {
320  qDebug( "rendering stop!" );
321  stopRendererV2( selRenderer );
322  return;
323  }
324 
325  QgsSymbolV2* sym = mRendererV2->symbolForFeature( fet );
326  if ( !sym )
327  {
328  continue;
329  }
330 
331  if ( !features.contains( sym ) )
332  {
333  features.insert( sym, QList<QgsFeature>() );
334  }
335  features[sym].append( fet );
336 
337  if ( mCache )
338  {
339  // Cache this for the use of (e.g.) modifying the feature's uncommitted geometry.
340  mCache->cacheGeometry( fet.id(), *fet.geometry() );
341  }
342 
343  if ( mContext.labelingEngine() )
344  {
345  if ( mLabeling )
346  {
347  mContext.labelingEngine()->registerFeature( mLayerID, fet, mContext );
348  }
349  if ( mDiagrams )
350  {
351  mContext.labelingEngine()->registerDiagramFeature( mLayerID, fet, mContext );
352  }
353  }
354  }
355 
356  // find out the order
357  QgsSymbolV2LevelOrder levels;
358  QgsSymbolV2List symbols = mRendererV2->symbols();
359  for ( int i = 0; i < symbols.count(); i++ )
360  {
361  QgsSymbolV2* sym = symbols[i];
362  for ( int j = 0; j < sym->symbolLayerCount(); j++ )
363  {
364  int level = sym->symbolLayer( j )->renderingPass();
365  if ( level < 0 || level >= 1000 ) // ignore invalid levels
366  continue;
367  QgsSymbolV2LevelItem item( sym, j );
368  while ( level >= levels.count() ) // append new empty levels
369  levels.append( QgsSymbolV2Level() );
370  levels[level].append( item );
371  }
372  }
373 
374  // 2. draw features in correct order
375  for ( int l = 0; l < levels.count(); l++ )
376  {
377  QgsSymbolV2Level& level = levels[l];
378  for ( int i = 0; i < level.count(); i++ )
379  {
380  QgsSymbolV2LevelItem& item = level[i];
381  if ( !features.contains( item.symbol() ) )
382  {
383  QgsDebugMsg( "level item's symbol not found!" );
384  continue;
385  }
386  int layer = item.layer();
387  QList<QgsFeature>& lst = features[item.symbol()];
388  QList<QgsFeature>::iterator fit;
389  for ( fit = lst.begin(); fit != lst.end(); ++fit )
390  {
391  if ( mContext.renderingStopped() )
392  {
393  stopRendererV2( selRenderer );
394  return;
395  }
396 
397  bool sel = mSelectedFeatureIds.contains( fit->id() );
398  // maybe vertex markers should be drawn only during the last pass...
399  bool drawMarker = ( mDrawVertexMarkers && mContext.drawEditingInformation() && ( !mVertexMarkerOnlyForSelection || sel ) );
400 
401  try
402  {
403  mRendererV2->renderFeature( *fit, mContext, layer, sel, drawMarker );
404  }
405  catch ( const QgsCsException &cse )
406  {
407  Q_UNUSED( cse );
408  QgsDebugMsg( QString( "Failed to transform a point while drawing a feature with ID '%1'. Ignoring this feature. %2" )
409  .arg( fet.id() ).arg( cse.what() ) );
410  }
411  }
412  }
413  }
414 
415  stopRendererV2( selRenderer );
416 }

472 void QgsVectorLayerRenderer::prepareDiagrams( QgsVectorLayer* layer, QStringList& attributeNames )
473 {

484 
485  mContext.labelingEngine()->addDiagramLayer( layer, diagSettings ); // will make internal copy of diagSettings + initialize it
486 
487  //add attributes needed by the diagram renderer

493  QStringList columns = expression->referencedColumns();
494  QStringList::const_iterator columnsIterator = columns.constBegin();
495  for ( ; columnsIterator != columns.constEnd(); ++columnsIterator )
496  {
497  if ( !attributeNames.contains( *columnsIterator ) )
498  attributeNames << *columnsIterator;
499  }
500  }
501

503  if ( linearlyInterpolatedDiagramRenderer != NULL )
