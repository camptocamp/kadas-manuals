307 //

310 inline bool qgsDoubleNear( double a, double b, double epsilon = 4 * std::numeric_limits<double>::epsilon() )
311 {
312  const double diff = a - b;
313  return diff > -epsilon && diff <= epsilon;
314 }

316 //
317 // compare two doubles using specified number of significant digits
318 //
319 inline bool qgsDoubleNearSig( double a, double b, int significantDigits = 10 )
320 {
321  // The most simple would be to print numbers as %.xe and compare as strings
322  // but that is probably too costly
323  // Then the fastest would be to set some bits directly, but little/big endian
324  // has to be considered (maybe TODO)
325  // Is there a better way?
326  int aexp, bexp;
327  double ar = frexp( a, &aexp );
328  double br = frexp( b, &bexp );
329 
330  return aexp == bexp &&
331  qRound( ar * pow( 10.0, significantDigits ) ) == qRound( br * pow( 10.0, significantDigits ) );
332 }
