305 void QgsRubberBand::addGeometry( QgsGeometry* geom, QgsVectorLayer* layer )
306 {
307  if ( !geom )
308  {
309  return;
310  }
311 
312  //maprender object of canvas
313  const QgsMapSettings& ms = mMapCanvas->mapSettings();
314 
315  int idx = mPoints.size();
316 
317  switch ( QgsWKBTypes::flatType( geom->geometry()->wkbType() ) )
318  {
319 
320  case QgsWKBTypes::Point:
321  {
322  QgsPoint pt;
323  if ( layer )
324  {
325  pt = ms.layerToMapCoordinates( layer, geom->asPoint() );
326  }
327  else
328  {
329  pt = geom->asPoint();
330  }
331  addPoint( pt, false, idx );
332  removeLastPoint( idx, false );
333  }
334  break;
335 
336  case QgsWKBTypes::MultiPoint:
337  {
338  QgsMultiPoint mpt = geom->asMultiPoint();
339  for ( int i = 0; i < mpt.size(); ++i, ++idx )
340  {
341  QgsPoint pt = mpt[i];
342  if ( layer )
343  {
344  addPoint( ms.layerToMapCoordinates( layer, pt ), false, idx );
345  removeLastPoint( idx, false );
346  }
347  else
348  {
349  addPoint( pt, false, idx );
350  removeLastPoint( idx, false );
351  }
352  }
353  }
354  break;
355 
356  case QgsWKBTypes::LineString:
357  case QgsWKBTypes::CircularString:
358  case QgsWKBTypes::CompoundCurve:
359  {
360  QgsPolyline line = geom->asPolyline();
361  for ( int i = 0; i < line.count(); i++ )
362  {
363  if ( layer )
364  {
365  addPoint( ms.layerToMapCoordinates( layer, line[i] ), false, idx );
366  }
367  else
368  {
369  addPoint( line[i], false, idx );
370  }
371  }
372  }
373  break;
374 
375  case QgsWKBTypes::MultiLineString:
376  case QgsWKBTypes::MultiCurve:
377  {
378 
379  QgsMultiPolyline mline = geom->asMultiPolyline();
380  for ( int i = 0; i < mline.size(); ++i, ++idx )
381  {
382  QgsPolyline line = mline[i];
383 
384  if ( line.size() == 0 )
385  {
386  --idx;
387  }
388 
389  for ( int j = 0; j < line.size(); ++j )
390  {
391  if ( layer )
392  {
393  addPoint( ms.layerToMapCoordinates( layer, line[j] ), false, idx );
394  }
395  else
396  {
397  addPoint( line[j], false, idx );
398  }
399  }
400  }
401  }
402  break;
403 
404  case QgsWKBTypes::Polygon:
405  case QgsWKBTypes::CurvePolygon:
406  {
407  QgsPolygon poly = geom->asPolygon();
408  QgsPolyline line = poly[0];
409  for ( int i = 0; i < line.count(); i++ )
410  {
411  if ( layer )
412  {
413  addPoint( ms.layerToMapCoordinates( layer, line[i] ), false, idx );
414  }
415  else
416  {
417  addPoint( line[i], false, idx );
418  }
419  }
420  }
421  break;
422 
423  case QgsWKBTypes::MultiPolygon:
424  {
425 
426  QgsMultiPolygon multipoly = geom->asMultiPolygon();
427  for ( int i = 0; i < multipoly.size(); ++i, ++idx )
428  {
429  QgsPolygon poly = multipoly[i];
430  QgsPolyline line = poly[0];
431  for ( int j = 0; j < line.count(); ++j )
432  {
433  if ( layer )
434  {
435  addPoint( ms.layerToMapCoordinates( layer, line[j] ), false, idx );
436  }
437  else
438  {
439  addPoint( line[j], false, idx );
440  }
441  }
442  }
443  }
444  break;
445 
446  case QGis::WKBUnknown:
447  default:
448  return;
449  }
450 
451  setVisible( true );
452  updateRect();
453  update();
454 }

479 void QgsRubberBand::paint( QPainter* p )
480 {
481  if ( mPoints.size() > 0 )
482  {
483  p->setBrush( mBrush );
484  p->setPen( mPen );
485 
486  Q_FOREACH ( const QList<QgsPoint>& line, mPoints )
487  {
488  QVector<QPointF> pts;
489  Q_FOREACH ( const QgsPoint& pt, line )
490  {
491  const QPointF cur = toCanvasCoordinates( QgsPoint( pt.x() + mTranslationOffsetX, pt.y() + mTranslationOffsetY ) ) - pos();
492  if ( pts.empty() || std::abs( pts.back().x() - cur.x() ) > 1 || std::abs( pts.back().y() - cur.y() ) > 1 )
493  pts.append( cur );
494  }
495 
496  switch ( mGeometryType )
497  {
498  case QGis::Polygon:
499  {
500  p->drawPolygon( pts );
501  }
502  break;
503 
504  case QGis::Point:
505  {
506  Q_FOREACH ( const QPointF& pt, pts )
507  {
508  double x = pt.x();
509  double y = pt.y();
510 
511  qreal s = ( mIconSize - 1 ) / 2.0;
512 
513  switch ( mIconType )
514  {
515  case ICON_NONE:
516  break;
517 
518  case ICON_CROSS:
519  p->drawLine( QLineF( x - s, y, x + s, y ) );
520  p->drawLine( QLineF( x, y - s, x, y + s ) );
521  break;
522 
523  case ICON_X:
524  p->drawLine( QLineF( x - s, y - s, x + s, y + s ) );
525  p->drawLine( QLineF( x - s, y + s, x + s, y - s ) );
526  break;
527 
528  case ICON_BOX:
529  p->drawLine( QLineF( x - s, y - s, x + s, y - s ) );
530  p->drawLine( QLineF( x + s, y - s, x + s, y + s ) );
531  p->drawLine( QLineF( x + s, y + s, x - s, y + s ) );
532  p->drawLine( QLineF( x - s, y + s, x - s, y - s ) );
533  break;
534 
535  case ICON_FULL_BOX:
536  p->drawRect( x - s, y - s, mIconSize, mIconSize );
537  break;
538 
539  case ICON_CIRCLE:
540  p->drawEllipse( x - s, y - s, mIconSize, mIconSize );
541  break;
542 
543  case ICON_SVG:
544  QRectF viewBox = mSvgRenderer->viewBoxF();
545  QRectF r( mSvgOffset.x(), mSvgOffset.y(), viewBox.width(), viewBox.height() );
546  p->save();
547  p->translate( pt );
548  mSvgRenderer->render( p, r );
549  p->restore();
550  break;
551  }
552  }
553  }
554  break;
555 
556  case QGis::Line:
557  default:
558  {
559  p->drawPolyline( pts );
560  }
561  break;
562  }
563  }
564  }
565 }

567 void QgsRubberBand::updateRect()
568 {
569  if ( mPoints.empty() )
570  {
571  setRect( QgsRectangle() );
572  setVisible( false );
573  return;
574  }
575 
576  qreal mupp = mMapCanvas->getCoordinateTransform()->mapUnitsPerPixel();
577 
578  double iconSize = ( mIconSize + 1 ) / 2.;
579  if ( mSvgRenderer )
580  {
581  QRectF viewBox = mSvgRenderer->viewBoxF();
582  iconSize = qMax( qAbs( mSvgOffset.x() ) + .5 * viewBox.width(), qAbs( mSvgOffset.y() ) + .5 * viewBox.height() );
583  }
584  qreal w = qMax( iconSize, ( mPen.width() * qSqrt( 2 ) ) / 2. + mPen.width() ) * mupp;
585 
586  QgsRectangle newRect;
587  for ( int i = 0; i < mPoints.size(); ++i )
588  {
589  foreach ( const QgsPoint& point, mPoints[i] )
590  {
591  QgsPoint p( point.x() + mTranslationOffsetX, point.y() + mTranslationOffsetY );
592  QgsRectangle pRect( p.x() - w, p.y() - w, p.x() + w, p.y() + w );
593 
594  if ( newRect.isEmpty() )
595  {
596  // Get rectangle of the first point
597  newRect = pRect;
598  }
599  else
600  {
601  newRect.combineExtentWith( &pRect );
602  }
603  }
604  }
605  setRect( newRect );
606 }

642 QgsPoint QgsRubberBand::partMidpoint( int geometryIndex ) const
643 {
644  int n = mPoints.size() <= geometryIndex ? 0 : mPoints[geometryIndex].size();
645  if ( n == 0 )
646  {
647  return QgsPoint( 0, 0 );
648  }
649 
650  if ( mGeometryType == QGis::Point )
651  {
652  return mPoints[geometryIndex][0];
653  }
654  else if ( mGeometryType == QGis::Line )
655  {
656  if ( n < 2 )
657  {
658  return mPoints[geometryIndex][0];
659  }
660  double totLen = 0;
661  QList<double> cumLength;
662  cumLength.append( 0 );
663  for ( int i = 0; i < n - 1; ++i )
664  {
665  double len = qSqrt( mPoints[geometryIndex][i].sqrDist( mPoints[geometryIndex][i + 1] ) );
666  cumLength.append( cumLength.back() + len );
667  totLen += len;
668  }
669  for ( int i = 0; i < n - 1; ++i )
670  {
671  if ( cumLength[i + 1] > .5 * totLen )
672  {
673  double k = ( .5 * totLen - cumLength[i] ) / ( cumLength[i + 1] - cumLength[i] );
674  return mPoints[geometryIndex][i] + ( mPoints[geometryIndex][i + 1] - mPoints[geometryIndex][i] ) * k;
675  }
676  }
677  return mPoints[geometryIndex][0];
678  }
679  else if ( mGeometryType == QGis::Polygon )
680  {
681  if ( n < 2 )
682  {
683  return mPoints[geometryIndex].front();
684  }
685 
686  // Reduce numerical instability by doing all the computations relative to first point
687  QgsPoint center = mPoints[geometryIndex][0];
688  QList<QgsPoint> points;
689  for ( int i = 0; i < n; ++i )
690  {
691  points.append( QgsPoint( mPoints[geometryIndex][i].x() - center.x(), mPoints[geometryIndex][i].y() - center.y() ) );
692  }
693 
694  double A = 0.;
695  double Cx = 0.;
696  double Cy = 0.;
697  int i = n - 1, j = 0;
698  for ( ; j < n; i = j++ )
699  {
700  const QgsPoint& vi = points[i];
701  const QgsPoint& vj = points[j];
702  double d = vi.x() * vj.y() - vj.x() * vi.y();
703  A += d;
704  Cx += ( vi.x() + vj.x() ) * d;
705  Cy += ( vi.y() + vj.y() ) * d;
706  }
707 
708  if ( qAbs( A ) < 1E-12 )
709  return mPoints[geometryIndex].front();
710  else
711  return QgsPoint( center.x() + Cx / ( 3. * A ), center.y() + Cy / ( 3. * A ) );
712  }
713  return QgsPoint( 0, 0 );
714 }

750 QgsGeometry *QgsRubberBand::asGeometry()
751 {
752  QgsGeometry *geom = NULL;
753 
754  switch ( mGeometryType )
755  {
756  case QGis::Polygon:
757  {
758  QgsPolygon polygon;
759  QList< QList<QgsPoint> >::const_iterator it = mPoints.constBegin();
760  for ( ; it != mPoints.constEnd(); ++it )
761  {
762  polygon.append( getPolyline( *it ) );
763  }
764  geom = QgsGeometry::fromPolygon( polygon );
765  break;
766  }
767 
768  case QGis::Point:
769  {
770  QgsMultiPoint multiPoint;
771 
772  QList< QList<QgsPoint> >::const_iterator it = mPoints.constBegin();
773  for ( ; it != mPoints.constEnd(); ++it )
774  {
775  multiPoint += getPolyline( *it );
776  }
777  geom = QgsGeometry::fromMultiPoint( multiPoint );
778  break;
779  }
780 
781  case QGis::Line:
782  default:
783  {
784  if ( mPoints.size() > 0 )
785  {
786  if ( mPoints.size() > 1 )
787  {
788  QgsMultiPolyline multiPolyline;
789  QList< QList<QgsPoint> >::const_iterator it = mPoints.constBegin();
790  for ( ; it != mPoints.constEnd(); ++it )
791  {
792  multiPolyline.append( getPolyline( *it ) );
793  }
794  geom = QgsGeometry::fromMultiPolyline( multiPolyline );
795  }
796  else
797  {
798  geom = QgsGeometry::fromPolyline( getPolyline( mPoints[0] ) );
799  }
