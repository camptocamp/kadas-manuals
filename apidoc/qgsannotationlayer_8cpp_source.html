72 QgsRectangle QgsAnnotationLayer::extent()
73 {
74  QgsRectangle extent;
75  bool empty = true;
76  foreach ( QGraphicsItem* item, mCanvas->items() )
77  {
78  QgsAnnotationItem* annotationItem = dynamic_cast<QgsAnnotationItem*>( item );
79  if ( annotationItem && mItemIds.contains( annotationItem->id() ) )
80  {
81  const QgsCoordinateTransform* t = QgsCoordinateTransformCache::instance()->transform( annotationItem->mapGeoPosCrs().authid(), crs().authid() );
82  QgsPoint p = t->transform( annotationItem->mapGeoPos() );
83  if ( empty )
84  {
85  extent = QgsRectangle( p, p );
86  empty = false;
87  }
88  else
89  {
90  extent.include( p );
91  }
92  }
93  }
94  return extent;
95 }

155 bool QgsAnnotationLayer::readXml( const QDomNode& layer_node )
156 {
157  mTransparency = layer_node.toElement().attribute( "transparency" ).toInt();
158  mSymbolScale = layer_node.toElement().attribute( "scale", "1." ).toDouble();
159  QDomElement annotationLayerEl = layer_node.firstChildElement( "AnnotationLayer" );
160  if ( !annotationLayerEl.isNull() )
161  {
162  mItemType = annotationLayerEl.attribute( "itemType" );
163  QDomNodeList annotationItemEls = annotationLayerEl.elementsByTagName( "AnnotationItem" );
164  for ( int i = 0, n = annotationItemEls.count(); i < n; ++i )
165  {
166  QDomElement annotationItemEl = annotationItemEls.at( i ).toElement();
167  mItemIds.insert( annotationItemEl.attribute( "id" ) );
168  }
169  }
170  return true;
171 }

174 {

180  QDomElement annotationLayerEl = document.createElement( "AnnotationLayer" );
181  annotationLayerEl.setAttribute( "itemType", mItemType );
182  layerElement.appendChild( annotationLayerEl );
183 
184  // Only write ids of items which actually still exists
185  foreach ( QGraphicsItem* item, mCanvas->items() )
186  {
187  QgsAnnotationItem* annotationItem = dynamic_cast<QgsAnnotationItem*>( item );
188  if ( annotationItem && mItemIds.contains( annotationItem->id() ) )
189  {
190  QDomElement annotationItemEl = document.createElement( "AnnotationItem" );
191  annotationItemEl.setAttribute( "id", annotationItem->id() );
192  annotationLayerEl.appendChild( annotationItemEl );
193  }
194  }
195  return true;
