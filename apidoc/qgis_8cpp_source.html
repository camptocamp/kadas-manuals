151 double QGis::fromUnitToUnitFactor( QGis::UnitType fromUnit, QGis::UnitType toUnit )
152 {
153 #define DEGREE_TO_METER 111319.49079327358
154 #define FEET_TO_METER 0.3048
155 #define NMILE_TO_METER 1852.0
156 
157  // Unify degree units
158  if ( fromUnit == QGis::DecimalDegrees || fromUnit == QGis::DegreesMinutesSeconds || fromUnit == QGis::DegreesDecimalMinutes )
159  fromUnit = QGis::Degrees;
160  if ( toUnit == QGis::DecimalDegrees || toUnit == QGis::DegreesMinutesSeconds || toUnit == QGis::DegreesDecimalMinutes )
161  toUnit = QGis::Degrees;
162 
163  // Calculate the conversion factor between the specified units
164  if ( fromUnit != toUnit && fromUnit != QGis::UnknownUnit && toUnit != QGis::UnknownUnit )
165  {
166  switch ( fromUnit )
167  {
168  case QGis::Meters:
169  {
170  if ( toUnit == QGis::Feet ) return 1.0 / FEET_TO_METER;
171  if ( toUnit == QGis::Degrees ) return 1.0 / DEGREE_TO_METER;
172  if ( toUnit == QGis::NauticalMiles ) return 1.0 / NMILE_TO_METER;
173  break;
174  }
175  case QGis::Feet:
176  {
177  if ( toUnit == QGis::Meters ) return FEET_TO_METER;
178  if ( toUnit == QGis::Degrees ) return FEET_TO_METER / DEGREE_TO_METER;
179  if ( toUnit == QGis::NauticalMiles ) return FEET_TO_METER / NMILE_TO_METER;
180  break;
181  }
182  case QGis::Degrees:
183  {
184  if ( toUnit == QGis::Meters ) return DEGREE_TO_METER;
185  if ( toUnit == QGis::Feet ) return DEGREE_TO_METER / FEET_TO_METER;
186  if ( toUnit == QGis::NauticalMiles ) return DEGREE_TO_METER / NMILE_TO_METER;
187  break;
188  }
189  case QGis::NauticalMiles:
190  {
191  if ( toUnit == QGis::Meters ) return NMILE_TO_METER;
192  if ( toUnit == QGis::Feet ) return NMILE_TO_METER / FEET_TO_METER;
193  if ( toUnit == QGis::Degrees ) return NMILE_TO_METER / DEGREE_TO_METER;
194  break;
195  }
196  case QGis::UnknownUnit:
197  break;
198  }
199  }
200  return 1.0;
201 }

203 bool QGis::addFileToZip( QuaZip* zip, QString filePath, QString zipFileName )
204 {
205  if ( !zip )
206  {
207  return false;
208  }
209 
210  QFileInfo fi( filePath );
211  QuaZipFile zipLocalFile( zip );
212  if ( !zipLocalFile.open( QIODevice::WriteOnly, QuaZipNewInfo( zipFileName ) ) )
213  {
214  return false;
215  }
216 
217  //copy from local file to zip local file
218  QFile localFile( filePath );
219  if ( !localFile.open( QIODevice::ReadOnly ) )
220  {
221  return false;
222  }
223 
224  QByteArray buffer;
225  while ( !localFile.atEnd() )
226  {
227  buffer = localFile.read( 4096 );
228  if ( !buffer.isEmpty() )
229  {
230  zipLocalFile.write( buffer );
231  }
232  }
233  return true;
234 }

271 bool qgsVariantLessThan( const QVariant& lhs, const QVariant& rhs )
272 {
273  switch ( lhs.type() )
274  {
275  case QVariant::Int:
276  return lhs.toInt() < rhs.toInt();
277  case QVariant::UInt:
278  return lhs.toUInt() < rhs.toUInt();
279  case QVariant::LongLong:
280  return lhs.toLongLong() < rhs.toLongLong();
281  case QVariant::ULongLong:
282  return lhs.toULongLong() < rhs.toULongLong();
283  case QVariant::Double:
284  return lhs.toDouble() < rhs.toDouble();
285  case QVariant::Char:
286  return lhs.toChar() < rhs.toChar();
287  case QVariant::Date:
288  return lhs.toDate() < rhs.toDate();
289  case QVariant::Time:
290  return lhs.toTime() < rhs.toTime();
291  case QVariant::DateTime:
292  return lhs.toDateTime() < rhs.toDateTime();
293  default:
294  return QString::localeAwareCompare( lhs.toString(), rhs.toString() ) < 0;
295  }
296 }

303 QString qgsInsertLinkAnchors( const QString& text )
304 {
305  QString value = text;
306  // http://alanstorm.com/url_regex_explained
307  static QRegExp urlRegEx( "(\\b(([\\w-]+://?|www[.])[^\\s()<>]+(?:\\([\\w\\d]+\\)|([^!\"#$%&'()*+,\\-./:;<=>?@[\\\\\\]^_`{|}~\\s]|/))))" );
308  static QRegExp protoRegEx( "^(?:f|ht)tps?://" );
309  static QRegExp emailRegEx( "([\\w._%+-]+@[\\w.-]+\\.[A-Za-z]+)" );
310 
311  int offset = 0;
312  while ( urlRegEx.indexIn( value, offset ) != -1 )
313  {
314  QString url = urlRegEx.cap( 1 );
315  QString protoUrl = url;
316  if ( protoRegEx.indexIn( protoUrl ) == -1 )
317  {
318  protoUrl.prepend( "http://" );
319  }
320  QString anchor = QString( "<a href=\"%1\">%2</a>" ).arg( protoUrl.toHtmlEscaped() ).arg( url.toHtmlEscaped() );
321  value.replace( urlRegEx.pos( 1 ), url.length(), anchor );
322  offset = urlRegEx.pos( 1 ) + anchor.length();
323  }
324  offset = 0;
325  while ( emailRegEx.indexIn( value, offset ) != -1 )
326  {
327  QString email = emailRegEx.cap( 1 );
328  QString anchor = QString( "<a href=\"mailto:%1\">%1</a>" ).arg( email.toHtmlEscaped() );
329  value.replace( emailRegEx.pos( 1 ), email.length(), anchor );
330  offset = emailRegEx.pos( 1 ) + anchor.length();
331  }
332  return value;
333 }
