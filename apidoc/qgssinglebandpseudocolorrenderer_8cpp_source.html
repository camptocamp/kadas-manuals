109 QgsRasterBlock* QgsSingleBandPseudoColorRenderer::block2( int bandNo, QgsRectangle const & extent, int width, int height, QgsRasterBlockFeedback* feedback )
110 {
111  Q_UNUSED( bandNo );
112 
113  QgsRasterBlock *outputBlock = new QgsRasterBlock();
114  if ( !mInput || !mShader )
115  {
116  return outputBlock;
117  }
118 
119 
120  QgsRasterBlock *inputBlock = mInput->block2( mBand, extent, width, height, feedback );
121  if ( !inputBlock || inputBlock->isEmpty() )
122  {
123  QgsDebugMsg( "No raster data!" );
124  delete inputBlock;
125  return outputBlock;
126  }
127 
128  //rendering is faster without considering user-defined transparency
129  bool hasTransparency = usesTransparency();
130 
131  QgsRasterBlock *alphaBlock = 0;
132  if ( mAlphaBand > 0 && mAlphaBand != mBand )
133  {
134  alphaBlock = mInput->block2( mAlphaBand, extent, width, height, feedback );
135  if ( !alphaBlock || alphaBlock->isEmpty() )
136  {
137  delete inputBlock;
138  delete alphaBlock;
139  return outputBlock;
140  }
141  }
142  else if ( mAlphaBand == mBand )
143  {
144  alphaBlock = inputBlock;
145  }
146 
147  if ( !outputBlock->reset( QGis::ARGB32_Premultiplied, width, height ) )
148  {
149  delete inputBlock;
150  delete alphaBlock;
151  return outputBlock;
152  }
153 
154  QRgb myDefaultColor = NODATA_COLOR;
155 
156  for ( qgssize i = 0; i < ( qgssize )width*height; i++ )
157  {
158  if ( inputBlock->isNoData( i ) )
159  {
160  outputBlock->setColor( i, myDefaultColor );
161  continue;
162  }
163  double val = inputBlock->value( i );
164  int red, green, blue, alpha;
165  if ( !mShader->shade( val, &red, &green, &blue, &alpha ) )
166  {
167  outputBlock->setColor( i, myDefaultColor );
168  continue;
169  }
170 
171  if ( alpha < 255 )
172  {
173  // Working with premultiplied colors, so multiply values by alpha
174  red *= ( alpha / 255.0 );
175  blue *= ( alpha / 255.0 );
176  green *= ( alpha / 255.0 );
177  }
178 
179  if ( !hasTransparency )
180  {
181  outputBlock->setColor( i, qRgba( red, green, blue, alpha ) );
182  }
183  else
184  {
185  //opacity
186  double currentOpacity = mOpacity;
187  if ( mRasterTransparency )
188  {
189  currentOpacity = mRasterTransparency->alphaValue( val, mOpacity * 255 ) / 255.0;
190  }
191  if ( mAlphaBand > 0 )
192  {
193  currentOpacity *= alphaBlock->value( i ) / 255.0;
194  }
195 
196  outputBlock->setColor( i, qRgba( currentOpacity * red, currentOpacity * green, currentOpacity * blue, currentOpacity * alpha ) );
197  }
198  }
199 
200  delete inputBlock;
201  if ( mAlphaBand > 0 && mBand != mAlphaBand )
202  {
203  delete alphaBlock;
204  }
205 
206  return outputBlock;
207 }

210 {

213  return;

215

217  _writeXML( doc, rasterRendererElem );

226
