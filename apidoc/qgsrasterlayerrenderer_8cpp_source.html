24 QgsRasterLayerRenderer::QgsRasterLayerRenderer( QgsRasterLayer* layer, QgsRenderContext& rendererContext )
25  : QgsMapLayerRenderer( layer->id() )
26  , mRasterViewPort( 0 )
27  , mPipe( 0 )
28  , mFeedback( nullptr )
29 {
30 
31  mPainter = rendererContext.painter();
32  const QgsMapToPixel& theQgsMapToPixel = rendererContext.mapToPixel();
33  mMapToPixel = &theQgsMapToPixel;
34  mRenderContext = &rendererContext;
35  mFeedback = new Feedback( this );
36 
37  QgsMapToPixel mapToPixel = theQgsMapToPixel;
38  if ( mapToPixel.mapRotation() )
39  {
40  // unset rotation for the sake of local computations.
41  // Rotation will be handled by QPainter later
42  // TODO: provide a method of QgsMapToPixel to fetch map center
43  // in geographical units
44  QgsPoint center = mapToPixel.toMapCoordinates(
45  mapToPixel.mapWidth() / 2.0,
46  mapToPixel.mapHeight() / 2.0
47  );
48  mapToPixel.setMapRotation( 0, center.x(), center.y() );
49  }
50 
51  QgsRectangle myProjectedViewExtent;
52  QgsRectangle myProjectedLayerExtent;
53 
54  if ( rendererContext.coordinateTransform() )
55  {
56  QgsDebugMsg( "coordinateTransform set -> project extents." );
57  try
58  {
59  myProjectedViewExtent = rendererContext.coordinateTransform()->transformBoundingBox( rendererContext.extent() );
60  }
61  catch ( QgsCsException &cs )
62  {
63  QgsMessageLog::logMessage( QObject::tr( "Could not reproject view extent: %1" ).arg( cs.what() ), QObject::tr( "Raster" ) );
64  myProjectedViewExtent.setMinimal();
65  }
66 
67  try
68  {
69  myProjectedLayerExtent = rendererContext.coordinateTransform()->transformBoundingBox( layer->extent() );
70  }
71  catch ( QgsCsException &cs )
72  {
73  QgsMessageLog::logMessage( QObject::tr( "Could not reproject layer extent: %1" ).arg( cs.what() ), QObject::tr( "Raster" ) );
74  myProjectedLayerExtent.setMinimal();
75  }
76  }
77  else
78  {
79  QgsDebugMsg( "coordinateTransform not set" );
80  myProjectedViewExtent = rendererContext.extent();
81  myProjectedLayerExtent = layer->extent();
82  }
83 
84  // clip raster extent to view extent
85  QgsRectangle myRasterExtent = myProjectedViewExtent.intersect( &myProjectedLayerExtent );
86  if ( myRasterExtent.isEmpty() )
87  {
88  QgsDebugMsg( "draw request outside view extent." );
89  // nothing to do
90  return;
91  }
92 
93  QgsDebugMsg( "theViewExtent is " + rendererContext.extent().toString() );
94  QgsDebugMsg( "myProjectedViewExtent is " + myProjectedViewExtent.toString() );
95  QgsDebugMsg( "myProjectedLayerExtent is " + myProjectedLayerExtent.toString() );
96  QgsDebugMsg( "myRasterExtent is " + myRasterExtent.toString() );
97 
98  //
99  // The first thing we do is set up the QgsRasterViewPort. This struct stores all the settings
100  // relating to the size (in pixels and coordinate system units) of the raster part that is
101  // in view in the map window. It also stores the origin.
102  //
103  //this is not a class level member because every time the user pans or zooms
104  //the contents of the rasterViewPort will change
105  mRasterViewPort = new QgsRasterViewPort();
106 
107  mRasterViewPort->mDrawnExtent = myRasterExtent;
108  if ( rendererContext.coordinateTransform() )
109  {
110  mRasterViewPort->mSrcCRS = layer->crs();
111  mRasterViewPort->mDestCRS = rendererContext.coordinateTransform()->destCRS();
112  mRasterViewPort->mSrcDatumTransform = rendererContext.coordinateTransform()->sourceDatumTransform();
113  mRasterViewPort->mDestDatumTransform = rendererContext.coordinateTransform()->destinationDatumTransform();
114  }
115  else
116  {
117  mRasterViewPort->mSrcCRS = QgsCoordinateReferenceSystem(); // will be invalid
118  mRasterViewPort->mDestCRS = QgsCoordinateReferenceSystem(); // will be invalid
119  mRasterViewPort->mSrcDatumTransform = -1;
120  mRasterViewPort->mDestDatumTransform = -1;
121  }
122 
123  // get dimensions of clipped raster image in device coordinate space (this is the size of the viewport)
124  mRasterViewPort->mTopLeftPoint = mapToPixel.transform( myRasterExtent.xMinimum(), myRasterExtent.yMaximum() );
125  mRasterViewPort->mBottomRightPoint = mapToPixel.transform( myRasterExtent.xMaximum(), myRasterExtent.yMinimum() );
126 
127  // align to output device grid, i.e. floor/ceil to integers
128  // TODO: this should only be done if paint device is raster - screen, image
129  // for other devices (pdf) it can have floating point origin
130  // we could use floating point for raster devices as well, but respecting the
131  // output device grid should make it more effective as the resampling is done in
132  // the provider anyway
133  mRasterViewPort->mTopLeftPoint.setX( floor( mRasterViewPort->mTopLeftPoint.x() ) );
134  mRasterViewPort->mTopLeftPoint.setY( floor( mRasterViewPort->mTopLeftPoint.y() ) );
135  mRasterViewPort->mBottomRightPoint.setX( ceil( mRasterViewPort->mBottomRightPoint.x() ) );
136  mRasterViewPort->mBottomRightPoint.setY( ceil( mRasterViewPort->mBottomRightPoint.y() ) );
137  // recalc myRasterExtent to aligned values
138  myRasterExtent.set(
139  mapToPixel.toMapCoordinatesF( mRasterViewPort->mTopLeftPoint.x(),
140  mRasterViewPort->mBottomRightPoint.y() ),
141  mapToPixel.toMapCoordinatesF( mRasterViewPort->mBottomRightPoint.x(),
142  mRasterViewPort->mTopLeftPoint.y() )
143  );
144 
145  //raster viewport top left / bottom right are already rounded to int
146  mRasterViewPort->mWidth = static_cast<int>( mRasterViewPort->mBottomRightPoint.x() - mRasterViewPort->mTopLeftPoint.x() );
147  mRasterViewPort->mHeight = static_cast<int>( mRasterViewPort->mBottomRightPoint.y() - mRasterViewPort->mTopLeftPoint.y() );
148 
149  //the drawable area can start to get very very large when you get down displaying 2x2 or smaller, this is becasue
150  //mapToPixel.mapUnitsPerPixel() is less then 1,
151  //so we will just get the pixel data and then render these special cases differently in paintImageToCanvas()
152 
153  QgsDebugMsgLevel( QString( "mapUnitsPerPixel = %1" ).arg( mapToPixel.mapUnitsPerPixel() ), 3 );
154  QgsDebugMsgLevel( QString( "mWidth = %1" ).arg( layer->width() ), 3 );
155  QgsDebugMsgLevel( QString( "mHeight = %1" ).arg( layer->height() ), 3 );
156  QgsDebugMsgLevel( QString( "myRasterExtent.xMinimum() = %1" ).arg( myRasterExtent.xMinimum() ), 3 );
157  QgsDebugMsgLevel( QString( "myRasterExtent.xMaximum() = %1" ).arg( myRasterExtent.xMaximum() ), 3 );
158  QgsDebugMsgLevel( QString( "myRasterExtent.yMinimum() = %1" ).arg( myRasterExtent.yMinimum() ), 3 );
159  QgsDebugMsgLevel( QString( "myRasterExtent.yMaximum() = %1" ).arg( myRasterExtent.yMaximum() ), 3 );
160 
161  QgsDebugMsgLevel( QString( "mTopLeftPoint.x() = %1" ).arg( mRasterViewPort->mTopLeftPoint.x() ), 3 );
162  QgsDebugMsgLevel( QString( "mBottomRightPoint.x() = %1" ).arg( mRasterViewPort->mBottomRightPoint.x() ), 3 );
163  QgsDebugMsgLevel( QString( "mTopLeftPoint.y() = %1" ).arg( mRasterViewPort->mTopLeftPoint.y() ), 3 );
164  QgsDebugMsgLevel( QString( "mBottomRightPoint.y() = %1" ).arg( mRasterViewPort->mBottomRightPoint.y() ), 3 );
165 
166  QgsDebugMsgLevel( QString( "mWidth = %1" ).arg( mRasterViewPort->mWidth ), 3 );
167  QgsDebugMsgLevel( QString( "mHeight = %1" ).arg( mRasterViewPort->mHeight ), 3 );
168 
169  // /\/\/\ - added to handle zoomed-in rasters
170 
171  // TODO R->mLastViewPort = *mRasterViewPort;
172 
173  // TODO: is it necessary? Probably WMS only?
174  layer->dataProvider()->setDpi( rendererContext.rasterScaleFactor() * 25.4 * rendererContext.scaleFactor() );
175 
176 
177  // copy the whole raster pipe!
178  mPipe = new QgsRasterPipe( *layer->pipe() );
179 }

188 bool QgsRasterLayerRenderer::render()
189 {
190  if ( !mRasterViewPort )
191  return true; // outside of layer extent - nothing to do
192 
193  //R->draw( mPainter, mRasterViewPort, &mMapToPixel );
194 
195  QTime time;
196  time.start();
197  //
198  //
199  // The goal here is to make as many decisions as possible early on (outside of the rendering loop)
200  // so that we can maximise performance of the rendering process. So now we check which drawing
201  // procedure to use :
202  //
203 
204  QgsRasterProjector *projector = mPipe->projector();
205 
206  // TODO add a method to interface to get provider and get provider
207  // params in QgsRasterProjector
208  if ( projector )
209  {
210  projector->setCRS( mRasterViewPort->mSrcCRS, mRasterViewPort->mDestCRS );
211  }
212 
213  // Drawer to pipe?
214  QgsRasterIterator iterator( mPipe->last() );
215  QgsRasterDrawer drawer( &iterator );
216 
217  drawer.draw( mPainter, mRasterViewPort, mMapToPixel, mRenderContext, mFeedback );
218 
219  QgsDebugMsg( QString( "total raster draw time (ms): %1" ).arg( time.elapsed(), 5 ) );
220 
221  return true;
222 }

239 void QgsRasterLayerRenderer::Feedback::onNewData()
240 {
241  if ( !renderPartialOutput() )
242  return; // we were not asked for partial renders and we may not have a temporary image for overwriting...
243 
244  // update only once upon a time
245  // (preview itself takes some time)
246  if ( mLastPreview.isValid() && mLastPreview.msecsTo( QTime::currentTime() ) < mMinimalPreviewInterval )
247  return;
248 
249  // TODO: update only the area that got new data
250 
251  QgsDebugMsg( QString( "new raster preview! %1" ).arg( mLastPreview.msecsTo( QTime::currentTime() ) ) );
252  QTime t;
253  t.start();
254  QgsRasterBlockFeedback feedback;
255  feedback.setPreviewOnly( true );
256  feedback.setRenderPartialOutput( true );
257  QgsRasterIterator iterator( mR->mPipe->last() );
258  QgsRasterDrawer drawer( &iterator );
259  drawer.draw( mR->mPainter, mR->mRasterViewPort, mR->mMapToPixel, 0, &feedback );
260  QgsDebugMsg( QString( "total raster preview time: %1 ms" ).arg( t.elapsed() ) );
261  mLastPreview = QTime::currentTime();
262 }
