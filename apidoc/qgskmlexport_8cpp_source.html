218 void QgsKMLExport::writeVectorLayerFeatures( QgsVectorLayer* vl, QTextStream& outStream, bool labelLayer, QgsKMLPalLabeling& labeling, QgsRenderContext& rc )
219 {
220  QgsFeatureRendererV2* renderer = vl->rendererV2();
221  if ( !renderer )
222  {
223  return;
224  }
225  renderer->startRender( rc, vl->pendingFields() );
226 
227  const QgsCoordinateTransform* ct = QgsCoordinateTransformCache::instance()->transform( vl->crs().authid(), "EPSG:4326" );
228 
229  QgsFeatureIterator it = vl->getFeatures();
230  QgsFeature f;
231  while ( it.nextFeature( f ) )
232  {
233  outStream << "<Placemark>" << "\n";
234  if ( f.geometry() && f.geometry()->geometry() )
235  {
236  // Name (from labeling expression, if possible)
237  if ( vl->customProperty( "labeling/enabled", false ).toBool() && !vl->customProperty( "labeling/fieldName" ).isNull() )
238  {
239  QString expr = QString( "\"%1\"" ).arg( vl->customProperty( "labeling/fieldName" ).toString() );
240  outStream << QString( "<name>%1</name>\n" ).arg( QgsExpression( expr ).evaluate( f ).toString() );
241  }
242  else
243  {
244  outStream << QString( "<name>Feature %1</name>\n" ).arg( f.id() );
245  }
246 
247  // Style
248  addStyle( outStream, f, *renderer, rc );
249 
250  // Attributes
251  outStream << QString( "<ExtendedData><SchemaData schemaUrl=\"#%1\">" ).arg( vl->name() ) << "\n";
252  const QgsFields& fields = *f.fields();
253  const QgsAttributes& attributes = f.attributes();
254  for ( int i = 0, n = attributes.size(); i < n; ++i )
255  {
256  outStream << QString( "<SimpleData name=\"%1\">%2</SimpleData>" ).arg( fields[i].name() ).arg( attributes[i].toString() ) << "\n";
257  }
258  outStream << QString( "</SchemaData></ExtendedData>" );
259 
260  // Segmentize immediately, since otherwise for instance circles are distorted if segmentation occurs after transformation
261  QgsAbstractGeometryV2* geom = f.geometry()->geometry()->segmentize();
262  if ( geom )
263  {
264  geom->transform( *ct ); //KML must be WGS84
265  outStream << geom->asKML( 6 );
266  }
267  delete geom;
268  }
269  outStream << "</Placemark>" << "\n";
270 
271  if ( labelLayer )
272  {
273  labeling.registerFeature( vl->id(), f, rc, vl->name() );
274  }
275  }
276  renderer->stopRender( rc );
277 }

279 void QgsKMLExport::writeTiles( QgsMapLayer* mapLayer, const QgsRectangle& layerExtent, double exportScale, QTextStream& outStream, int drawingOrder, QuaZip* quaZip, QProgressDialog* progress )
280 {
281  const int tileSize = 512;
282 
283  // Make extent square
284  QgsPoint center = layerExtent.center();
285  double extension = qMax( layerExtent.width(), layerExtent.height() ) * 0.5;
286  QgsRectangle renderExtent = QgsRectangle( center.x() - extension, center.y() - extension, center.x() + extension, center.y() + extension );
287 
288  // Compute pixels to match extent at scale
289  // px / dpi * 0.0254 * scale = meters
290  QImage image( tileSize, tileSize, QImage::Format_ARGB32 );
291  double meters = QgsScaleCalculator().calculateGeographicDistance( renderExtent );
292  int dpi = image.logicalDpiX();
293  int totPixels = meters / ( exportScale * 0.0254 ) * dpi;
294  double resolution = renderExtent.width() / totPixels;
295 
296  // Round up to next <tileSize> multiple, adding margin
297  totPixels = qCeil(( totPixels + 2 * mapLayer->margin() ) / double( tileSize ) ) * tileSize;
298  extension = totPixels * resolution * 0.5;
299  renderExtent = QgsRectangle( center.x() - extension, center.y() - extension, center.x() + extension, center.y() + extension );
300 
301  progress->setRange( 0, ( totPixels / tileSize ) * ( totPixels / tileSize ) );
302  QApplication::processEvents();
303 
304  // Render in <tileSize> blocks
305  int tileCounter = 0;
306  for ( int iy = 0; iy < totPixels; iy += tileSize )
307  {
308  for ( int ix = 0; ix < totPixels; ix += tileSize )
309  {
310 
311  progress->setValue( tileCounter );
312  QApplication::processEvents();
313  if ( progress->wasCanceled() )
314  {
315  return;
316  }
317 
318  QgsRectangle tileExtent( renderExtent.xMinimum() + ix * resolution, renderExtent.yMinimum() + iy * resolution,
319  renderExtent.xMinimum() + ( ix + tileSize ) * resolution, renderExtent.yMinimum() + ( iy + tileSize ) * resolution );
320  if ( renderTile( image, tileExtent, mapLayer ) )
321  {
322  QString filename = QString( "%1_%2.png" ).arg( mapLayer->id() ).arg( tileCounter++ );
323  QuaZipFile outputFile( quaZip );
324  if ( outputFile.open( QIODevice::WriteOnly, QuaZipNewInfo( filename ) ) && image.save( &outputFile, "PNG" ) )
325  writeGroundOverlay( outStream, QString( "Tile %1" ).arg( tileExtent.toString( 3 ) ), filename, tileExtent, drawingOrder );
326  }
327  }
328  }
329 }

409  bool rendered = false;
410  if ( layerRenderer )

418 void QgsKMLExport::addStyle( QTextStream& outStream, QgsFeature& f, QgsFeatureRendererV2& r, QgsRenderContext& rc )
419 {
420  // Take first symbollayer
421  QgsSymbolV2List symbolList = r.symbolsForFeature( f );
422  if ( symbolList.isEmpty() )
423  {
424  return;
425  }
426 
427  QgsSymbolV2* s = symbolList.first();
428  if ( !s || s->symbolLayerCount() < 1 )
429  {
430  return;
431  }
432  QgsExpression* expr = 0;
433 
434  outStream << "<Style>";
435  if ( s->type() == QgsSymbolV2::Line )
436  {
437  double width = 1;
438  expr = s->symbolLayer( 0 )->expression( "width" );
439  if ( expr )
440  {
441  width = expr->evaluate( f ).toDouble( ) * QgsSymbolLayerV2Utils::lineWidthScaleFactor( rc, QgsSymbolV2::MM );
442  }
443  else if ( dynamic_cast<QgsLineSymbolLayerV2*>( s->symbolLayer( 0 ) ) )
444  {
445  QgsLineSymbolLayerV2* lineSymbolLayer = static_cast<QgsLineSymbolLayerV2*>( s->symbolLayer( 0 ) );
446  width = lineSymbolLayer->width() * QgsSymbolLayerV2Utils::lineWidthScaleFactor( rc, lineSymbolLayer->widthUnit() );
447  }
448 
449  QColor c = s->symbolLayer( 0 )->color();
450  expr = s->symbolLayer( 0 )->expression( "color" );
451  if ( expr )
452  {
453  c = QgsSymbolLayerV2Utils::decodeColor( expr->evaluate( f ).toString() );
454  }
455 
456  outStream << QString( "<LineStyle><color>%1</color><width>%2</width></LineStyle>" ).arg( convertColor( c ) ).arg( QString::number( width ) );
457  }
458  else if ( s->type() == QgsSymbolV2::Fill )
459  {
460  double width = 1;
461  expr = s->symbolLayer( 0 )->expression( "width_border" );
462  if ( expr )
463  {
464  width = expr->evaluate( f ).toDouble( ) * QgsSymbolLayerV2Utils::lineWidthScaleFactor( rc, QgsSymbolV2::MM );
465  }

468 
469  expr = s->symbolLayer( 0 )->expression( "color_border" );
470  if ( expr )
471  {
472  outlineColor = QgsSymbolLayerV2Utils::decodeColor( expr->evaluate( f ).toString() );
473  }
474

476  expr = s->symbolLayer( 0 )->expression( "color" );
477  if ( expr )
478  {
479  fillColor = QgsSymbolLayerV2Utils::decodeColor( expr->evaluate( f ).toString() );
480  }
481 
482  outStream << QString( "<LineStyle><width>%1</width><color>%2</color></LineStyle><PolyStyle><fill>1</fill><color>%3</color></PolyStyle>" )
483  .arg( width ).arg( convertColor( outlineColor ) ).arg( convertColor( fillColor ) );
484  }
485  outStream << "</Style>\n";
