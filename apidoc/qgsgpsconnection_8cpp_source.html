93 bool QgsGPSConnection::gpsInfoValid( const QgsGPSInformation& info )
94 {
95  bool valid = false;
96  if ( info.status == 'V' || info.fixType == NMEA_FIX_BAD || info.quality == 0 ) // some sources say that 'V' indicates position fix, but is below acceptable quality
97  {
98  return false;
99  }
100  else if ( info.fixType == NMEA_FIX_2D )
101  {
102  valid = true;
103  }
104  else if ( info.status == 'A' || info.fixType == NMEA_FIX_3D || info.quality > 0 ) // good
105  {
106  valid = true;
107  }
108 
109  return valid;
110 }
