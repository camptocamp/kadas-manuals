108 void QgsMapCanvasGPSDisplay::updateGPSInformation( const QgsGPSInformation& info )
109 {
110  FixStatus fixStatus = NoData;
111 
112  // no fix if any of the three report bad; default values are invalid values and won't be changed if the corresponding NMEA msg is not received
113  if ( info.status == 'V' || info.fixType == NMEA_FIX_BAD || info.quality == 0 ) // some sources say that 'V' indicates position fix, but is below acceptable quality
114  {
115  fixStatus = NoFix;
116  }
117  else if ( info.fixType == NMEA_FIX_2D ) // 2D indication (from GGA)
118  {
119  fixStatus = Fix2D;
120  }
121  else if ( info.status == 'A' || info.fixType == NMEA_FIX_3D || info.quality > 0 ) // good
122  {
123  fixStatus = Fix3D;
124  }
125  if ( fixStatus != mCurFixStatus )
126  {
127  emit gpsFixStatusChanged( fixStatus );
128  }
129  mCurFixStatus = fixStatus;
130 
131  emit gpsInformationReceived( info ); //send signal for service who want to do further actions (e.g. satellite position display, digitising, ...)
132 
133  if ( !mCanvas || !QgsGPSConnection::gpsInfoValid( info ) )
134  {
135  return;
136  }
137 
138  QgsPoint position( info.longitude, info.latitude );
139  if ( mRecenterMap != Never && mLastGPSPosition != position )
140  {
141  //recenter map
142  QgsCoordinateReferenceSystem destCRS = mCanvas->mapSettings().destinationCrs();
143  QgsCoordinateTransform myTransform( mWgs84CRS, destCRS ); // use existing WGS84 CRS
144 
145  QgsPoint centerPoint = myTransform.transform( position );
146  QgsRectangle myRect( centerPoint, centerPoint );
147 
148  // testing if position is outside some proportion of the map extent
149  // this is a user setting - useful range: 5% to 100% (0.05 to 1.0)
150  QgsRectangle extentLimit( mCanvas->extent() );
151  extentLimit.scale( mSpinMapExtentMultiplier * 0.01 );
152 
153  if ( mRecenterMap == Always || !extentLimit.contains( centerPoint ) )
154  {
155  mCanvas->setExtent( myRect );
156  mCanvas->refresh();
157  }
158  mLastGPSPosition = position;
159  }
160 
161  if ( mShowMarker )
162  {
163  if ( ! mMarker )
164  {
165  mMarker = new QgsGpsMarker( mCanvas );
166  }
167  mMarker->setSize( mMarkerSize );
168  mMarker->setCenter( position );
169  }
170 }
