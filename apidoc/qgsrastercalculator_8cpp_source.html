55 int QgsRasterCalculator::processCalculation( QProgressDialog* p )
56 {
57  //prepare search string / tree
58  QString errorString;
59  QgsRasterCalcNode* calcNode = QgsRasterCalcNode::parseRasterCalcString( mFormulaString, errorString );
60  if ( !calcNode )
61  {
62  //error
63  return 4;
64  }
65 
66  double targetGeoTransform[6];
67  outputGeoTransform( targetGeoTransform );
68 
69  //open all input rasters for reading
70  QMap< QString, GDALRasterBandH > mInputRasterBands; //raster references and corresponding scanline data
71  QMap< QString, QgsRasterMatrix* > inputScanLineData; //stores raster references and corresponding scanline data
72  QVector< GDALDatasetH > mInputDatasets; //raster references and corresponding dataset
73 
74  QVector<QgsRasterCalculatorEntry>::const_iterator it = mRasterEntries.constBegin();
75  for ( ; it != mRasterEntries.constEnd(); ++it )
76  {
77  if ( !it->raster ) // no raster layer in entry
78  {
79  return 2;
80  }
81  GDALDatasetH inputDataset = GDALOpen( TO8F( it->raster->source() ), GA_ReadOnly );
82  if ( !inputDataset )
83  {
84  return 2;
85  }
86 
87  //check if the input dataset is south up or rotated. If yes, use GDALAutoCreateWarpedVRT to create a north up raster
88  double inputGeoTransform[6];
89  if ( GDALGetGeoTransform( inputDataset, inputGeoTransform ) == CE_None
90  && ( inputGeoTransform[1] < 0.0
91  || inputGeoTransform[2] != 0.0
92  || inputGeoTransform[4] != 0.0
93  || inputGeoTransform[5] > 0.0 ) )
94  {
95  GDALDatasetH vDataset = GDALAutoCreateWarpedVRT( inputDataset, NULL, NULL, GRA_NearestNeighbour, 0.2, NULL );
96  mInputDatasets.push_back( vDataset );
97  mInputDatasets.push_back( inputDataset );
98  inputDataset = vDataset;
99  }
100  else
101  {
102  mInputDatasets.push_back( inputDataset );
103  }
104 
105  GDALRasterBandH inputRasterBand = GDALGetRasterBand( inputDataset, it->bandNumber );
106  if ( inputRasterBand == NULL )
107  {
108  return 2;
109  }
110 
111  int nodataSuccess;
112  double nodataValue = GDALGetRasterNoDataValue( inputRasterBand, &nodataSuccess );
113 
114  mInputRasterBands.insert( it->ref, inputRasterBand );
115  inputScanLineData.insert( it->ref, new QgsRasterMatrix( mNumOutputColumns, 1, new float[mNumOutputColumns], nodataValue ) );
116  }
117 
118  //open output dataset for writing
119  GDALDriverH outputDriver = openOutputDriver();
120  if ( outputDriver == NULL )
121  {
122  return 1;
123  }
124  GDALDatasetH outputDataset = openOutputFile( outputDriver );
125 
126  //copy the projection info from the first input raster
127  if ( mRasterEntries.size() > 0 )
128  {
129  QgsRasterLayer* rl = mRasterEntries.at( 0 ).raster;
130  if ( rl )
131  {
132  char* crsWKT = 0;
133  OGRSpatialReferenceH ogrSRS = OSRNewSpatialReference( NULL );
134  if ( OSRSetFromUserInput( ogrSRS, rl->crs().authid().toUtf8().constData() ) == OGRERR_NONE )
135  {
136  OSRExportToWkt( ogrSRS, &crsWKT );
137  GDALSetProjection( outputDataset, crsWKT );
138  }
139  else
140  {
141  GDALSetProjection( outputDataset, TO8( rl->crs().toWkt() ) );
142  }
143  OSRDestroySpatialReference( ogrSRS );
144  CPLFree( crsWKT );
145  }
146  }
147 
148 
149  GDALRasterBandH outputRasterBand = GDALGetRasterBand( outputDataset, 1 );
150 
151  float outputNodataValue = -std::numeric_limits<float>::max();
152  GDALSetRasterNoDataValue( outputRasterBand, outputNodataValue );
153 
154  float* resultScanLine = ( float * ) CPLMalloc( sizeof( float ) * mNumOutputColumns );
155 
156  if ( p )
157  {
158  p->setMaximum( mNumOutputRows );
159  }
160 
161  QgsRasterMatrix resultMatrix;
162 
163  //read / write line by line
164  for ( int i = 0; i < mNumOutputRows; ++i )
165  {
166  if ( p )
167  {
168  p->setValue( i );
169  }
170 
171  if ( p && p->wasCanceled() )
172  {
173  break;
174  }
175 
176  //fill buffers
177  QMap< QString, QgsRasterMatrix* >::iterator bufferIt = inputScanLineData.begin();
178  for ( ; bufferIt != inputScanLineData.end(); ++bufferIt )
179  {
180  double sourceTransformation[6];
181  GDALRasterBandH sourceRasterBand = mInputRasterBands[bufferIt.key()];
182  if ( GDALGetGeoTransform( GDALGetBandDataset( sourceRasterBand ), sourceTransformation ) != CE_None )
183  {
184  qWarning( "GDALGetGeoTransform failed!" );
185  }
186 
187  //the function readRasterPart calls GDALRasterIO (and ev. does some conversion if raster transformations are not the same)
188  readRasterPart( targetGeoTransform, 0, i, mNumOutputColumns, 1, sourceTransformation, sourceRasterBand, bufferIt.value()->data() );
189  }
190 
191  if ( calcNode->calculate( inputScanLineData, resultMatrix ) )
192  {
193  bool resultIsNumber = resultMatrix.isNumber();
194  float* calcData;
195 
196  if ( resultIsNumber ) //scalar result. Insert number for every pixel
197  {
198  calcData = new float[mNumOutputColumns];
199  for ( int j = 0; j < mNumOutputColumns; ++j )
200  {
201  calcData[j] = resultMatrix.number();
202  }
203  }
204  else //result is real matrix
205  {
206  calcData = resultMatrix.data();
207  }
208 
209  //replace all matrix nodata values with output nodatas
210  for ( int j = 0; j < mNumOutputColumns; ++j )
211  {
212  if ( calcData[j] == resultMatrix.nodataValue() )
213  {
214  calcData[j] = outputNodataValue;
215  }
216  }
217 
218  //write scanline to the dataset
219  if ( GDALRasterIO( outputRasterBand, GF_Write, 0, i, mNumOutputColumns, 1, calcData, mNumOutputColumns, 1, GDT_Float32, 0, 0 ) != CE_None )
220  {
221  qWarning( "RasterIO error!" );
222  }
223 
224  if ( resultIsNumber )
225  {
226  delete[] calcData;
227  }
228  }
229 
230  }
231 
232  if ( p )
233  {
234  p->setValue( mNumOutputRows );
235  }
236 
237  //close datasets and release memory
238  delete calcNode;
239  QMap< QString, QgsRasterMatrix* >::iterator bufferIt = inputScanLineData.begin();
240  for ( ; bufferIt != inputScanLineData.end(); ++bufferIt )
241  {
242  delete bufferIt.value();
243  }
244  inputScanLineData.clear();
245 
246  QVector< GDALDatasetH >::iterator datasetIt = mInputDatasets.begin();
247  for ( ; datasetIt != mInputDatasets.end(); ++ datasetIt )
248  {
249  GDALClose( *datasetIt );
250  }
251 
252  if ( p && p->wasCanceled() )
253  {
254  //delete the dataset without closing (because it is faster)
255  GDALDeleteDataset( outputDriver, TO8F( mOutputFile ) );
256  return 3;
257  }
258  GDALClose( outputDataset );
259  CPLFree( resultScanLine );
260  return 0;
261 }

308 void QgsRasterCalculator::readRasterPart( double* targetGeotransform, int xOffset, int yOffset, int nCols, int nRows, double* sourceTransform, GDALRasterBandH sourceBand, float* rasterBuffer )
309 {
310  //If dataset transform is the same as the requested transform, do a normal GDAL raster io
311  if ( transformationsEqual( targetGeotransform, sourceTransform ) )
312  {
313  CPLErr err = GDALRasterIO( sourceBand, GF_Read, xOffset, yOffset, nCols, nRows, rasterBuffer, nCols, nRows, GDT_Float32, 0, 0 );
314  Q_UNUSED( err );
315  return;
316  }
317 
318  int sourceBandXSize = GDALGetRasterBandXSize( sourceBand );
319  int sourceBandYSize = GDALGetRasterBandYSize( sourceBand );
320 
321  //pixel calculation needed because of different raster position / resolution
322  int nodataSuccess;
323  double nodataValue = GDALGetRasterNoDataValue( sourceBand, &nodataSuccess );
324  QgsRectangle targetRect( targetGeotransform[0] + targetGeotransform[1] * xOffset, targetGeotransform[3] + yOffset * targetGeotransform[5] + nRows * targetGeotransform[5]
325  , targetGeotransform[0] + targetGeotransform[1] * xOffset + targetGeotransform[1] * nCols, targetGeotransform[3] + yOffset * targetGeotransform[5] );
326  QgsRectangle sourceRect( sourceTransform[0], sourceTransform[3] + GDALGetRasterBandYSize( sourceBand ) * sourceTransform[5],
327  sourceTransform[0] + GDALGetRasterBandXSize( sourceBand )* sourceTransform[1], sourceTransform[3] );
328  QgsRectangle intersection = targetRect.intersect( &sourceRect );
329 
330  //no intersection, fill all the pixels with nodata values
331  if ( intersection.isEmpty() )
332  {
333  int nPixels = nCols * nRows;
334  for ( int i = 0; i < nPixels; ++i )
335  {
336  rasterBuffer[i] = nodataValue;
337  }
338  return;
339  }
340 
341  //do raster io in source resolution
342  int sourcePixelOffsetXMin = floor(( intersection.xMinimum() - sourceTransform[0] ) / sourceTransform[1] );
343  int sourcePixelOffsetXMax = ceil(( intersection.xMaximum() - sourceTransform[0] ) / sourceTransform[1] );
344  if ( sourcePixelOffsetXMax > sourceBandXSize )
345  {
346  sourcePixelOffsetXMax = sourceBandXSize;
347  }
348  int nSourcePixelsX = sourcePixelOffsetXMax - sourcePixelOffsetXMin;
349 
350  int sourcePixelOffsetYMax = floor(( intersection.yMaximum() - sourceTransform[3] ) / sourceTransform[5] );
351  int sourcePixelOffsetYMin = ceil(( intersection.yMinimum() - sourceTransform[3] ) / sourceTransform[5] );
352  if ( sourcePixelOffsetYMin > sourceBandYSize )
353  {
354  sourcePixelOffsetYMin = sourceBandYSize;
355  }
356  int nSourcePixelsY = sourcePixelOffsetYMin - sourcePixelOffsetYMax;
357  float* sourceRaster = ( float * ) CPLMalloc( sizeof( float ) * nSourcePixelsX * nSourcePixelsY );
358  double sourceRasterXMin = sourceRect.xMinimum() + sourcePixelOffsetXMin * sourceTransform[1];
359  double sourceRasterYMax = sourceRect.yMaximum() + sourcePixelOffsetYMax * sourceTransform[5];
360  if ( GDALRasterIO( sourceBand, GF_Read, sourcePixelOffsetXMin, sourcePixelOffsetYMax, nSourcePixelsX, nSourcePixelsY,
361  sourceRaster, nSourcePixelsX, nSourcePixelsY, GDT_Float32, 0, 0 ) != CE_None )
362  {
363  //IO error, fill array with nodata values
364  CPLFree( sourceRaster );
365  int npixels = nRows * nCols;
366  for ( int i = 0; i < npixels; ++i )
367  {
368  rasterBuffer[i] = nodataValue;
369  }
370  return;
371  }
372 
373 
374  double targetPixelX;
375  double targetPixelXMin = targetGeotransform[0] + targetGeotransform[1] * xOffset + targetGeotransform[1] / 2.0;
376  double targetPixelY = targetGeotransform[3] + targetGeotransform[5] * yOffset + targetGeotransform[5] / 2.0; //coordinates of current target pixel
377  int sourceIndexX, sourceIndexY; //current raster index in source pixels
378  double sx, sy;
379  for ( int i = 0; i < nRows; ++i )
380  {
381  targetPixelX = targetPixelXMin;
382  for ( int j = 0; j < nCols; ++j )
383  {
384  sx = ( targetPixelX - sourceRasterXMin ) / sourceTransform[1];
385  sourceIndexX = sx > 0 ? sx : floor( sx );
386  sy = ( targetPixelY - sourceRasterYMax ) / sourceTransform[5];
387  sourceIndexY = sy > 0 ? sy : floor( sy );
388  if ( sourceIndexX >= 0 && sourceIndexX < nSourcePixelsX
389  && sourceIndexY >= 0 && sourceIndexY < nSourcePixelsY )
390  {
391  rasterBuffer[j + i*nRows] = sourceRaster[ sourceIndexX + nSourcePixelsX * sourceIndexY ];
392  }
393  else
394  {
395  rasterBuffer[j + i*j] = nodataValue;
396  }
397  targetPixelX += targetGeotransform[1];
398  }
399  targetPixelY += targetGeotransform[5];
400  }
401 
402  CPLFree( sourceRaster );
403  return;
404 }
