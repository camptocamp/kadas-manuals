165  int CostCalculator::finalizeCandidatesCosts( Feats* feat, int max_p, RTree <PointSet*, double, 2, double> *obstacles, double bbx[4], double bby[4] )
166  {
167  // If candidates list is smaller than expected
168  if ( max_p > feat->nblp )
169  max_p = feat->nblp;
170  //
171  // sort candidates list, best label to worst
172  sort(( void** ) feat->lPos, feat->nblp, LabelPosition::costGrow );
173 
174  // try to exclude all conflitual labels (good ones have cost < 1 by pruning)
175  double discrim = 0.0;
176  int stop;
177  do
178  {
179  discrim += 1.0;
180  for ( stop = 0; stop < feat->nblp && feat->lPos[stop]->getCost() < discrim; stop++ )
181  ;
182  }
183  while ( stop == 0 && discrim < feat->lPos[feat->nblp-1]->getCost() + 2.0 );
184 
185  if ( discrim > 1.5 )
186  {
187  int k;
188  for ( k = 0; k < stop; k++ )
189  feat->lPos[k]->setCost( 0.0021 );
190  }
191 
192  if ( max_p > stop )
193  max_p = stop;
194 
195 #ifdef _DEBUG_FULL_
196  std::cout << "Nblabel kept for feat " << feat->feature->getUID() << "/" << feat->feature->getLayer()->getName() << ": " << max_p << "/" << feat->nblp << std::endl;
197 #endif
198 
199  // Sets costs for candidates of polygon
200 
201  if ( feat->feature->getGeosType() == GEOS_POLYGON )
202  {
203  int arrangement = feat->feature->getLayer()->getArrangement();
204  if ( arrangement == P_FREE || arrangement == P_HORIZ )
205  setPolygonCandidatesCost( stop, ( LabelPosition** ) feat->lPos, max_p, obstacles, bbx, bby );
206  }
207 
208  // add size penalty (small lines/polygons get higher cost)
209  feat->feature->addSizePenalty( max_p, feat->lPos, bbx, bby );
210 
211  return max_p;
212  }
