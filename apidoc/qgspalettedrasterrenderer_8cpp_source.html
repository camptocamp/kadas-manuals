59 QgsRasterRenderer* QgsPalettedRasterRenderer::create( const QDomElement& elem, QgsRasterInterface* input )
60 {
61  if ( elem.isNull() )
62  {
63  return 0;
64  }
65 
66  int bandNumber = elem.attribute( "band", "-1" ).toInt();
67  int nColors = 0;
68  QRgb* colors = 0;
69  QVector<QString> labels;
70 
71  QDomElement paletteElem = elem.firstChildElement( "colorPalette" );
72  if ( !paletteElem.isNull() )
73  {
74  QDomNodeList paletteEntries = paletteElem.elementsByTagName( "paletteEntry" );
75 
76  QDomElement entryElem;
77  int value;
78  nColors = 0;
79 
80  // We cannot believe that data are correct, check first max value
81  for ( int i = 0; i < paletteEntries.size(); ++i )
82  {
83  entryElem = paletteEntries.at( i ).toElement();
84  // Could be written as doubles (with .0000) in old project files
85  value = ( int )entryElem.attribute( "value", "0" ).toDouble();
86  if ( value >= nColors && value <= 10000 ) nColors = value + 1;
87  }
88  QgsDebugMsg( QString( "nColors = %1" ).arg( nColors ) );
89 
90  colors = new QRgb[ nColors ];
91 
92  for ( int i = 0; i < nColors; ++i )
93  {
94  entryElem = paletteEntries.at( i ).toElement();
95  value = ( int )entryElem.attribute( "value", "0" ).toDouble();
96  QgsDebugMsg( entryElem.attribute( "color", "#000000" ) );
97  if ( value >= 0 && value < nColors )
98  {
99  colors[value] = QColor( entryElem.attribute( "color", "#000000" ) ).rgba();
100  QString label = entryElem.attribute( "label" );
101  if ( !label.isEmpty() )
102  {
103  if ( value >= labels.size() ) labels.resize( value + 1 );
104  labels[value] = label;
105  }
106  }
107  else
108  {
109  QgsDebugMsg( QString( "value %1 out of range" ).arg( value ) );
110  }
111  }
112  }
113  QgsPalettedRasterRenderer* r = new QgsPalettedRasterRenderer( input, bandNumber, colors, nColors, labels );
114  r->readXML( elem );
115  return r;
116 }

160 QgsRasterBlock * QgsPalettedRasterRenderer::block2( int bandNo, QgsRectangle const & extent, int width, int height, QgsRasterBlockFeedback* feedback )
161 {
162  QgsRasterBlock *outputBlock = new QgsRasterBlock();
163  if ( !mInput || mNColors == 0 )
164  {
165  return outputBlock;
166  }
167 
168  QgsRasterBlock *inputBlock = mInput->block2( bandNo, extent, width, height, feedback );
169 
170  if ( !inputBlock || inputBlock->isEmpty() )
171  {
172  QgsDebugMsg( "No raster data!" );
173  delete inputBlock;
174  return outputBlock;
175  }
176 
177  double currentOpacity = mOpacity;
178 
179  //rendering is faster without considering user-defined transparency
180  bool hasTransparency = usesTransparency();
181  QgsRasterBlock *alphaBlock = 0;
182 
183  if ( mAlphaBand > 0 && mAlphaBand != mBand )
184  {
185  alphaBlock = mInput->block2( mAlphaBand, extent, width, height, feedback );
186  if ( !alphaBlock || alphaBlock->isEmpty() )
187  {
188  delete inputBlock;
189  delete alphaBlock;
190  return outputBlock;
191  }
192  }
193  else if ( mAlphaBand == mBand )
194  {
195  alphaBlock = inputBlock;
196  }
197 
198  if ( !outputBlock->reset( QGis::ARGB32_Premultiplied, width, height ) )
199  {
200  delete inputBlock;
201  delete alphaBlock;
202  return outputBlock;
203  }
204 
205  QRgb myDefaultColor = NODATA_COLOR;
206 
207  //use direct data access instead of QgsRasterBlock::setValue
208  //because of performance
209  unsigned int* outputData = ( unsigned int* )( outputBlock->bits() );
210 
211  qgssize rasterSize = ( qgssize )width * height;
212  for ( qgssize i = 0; i < rasterSize; ++i )
213  {
214  if ( inputBlock->isNoData( i ) )
215  {
216  outputData[i] = myDefaultColor;
217  continue;
218  }
219  int val = ( int ) inputBlock->value( i );
220  if ( !hasTransparency )
221  {
222  outputData[i] = mColors[val];
223  }
224  else
225  {
226  currentOpacity = mOpacity;
227  if ( mRasterTransparency )
228  {
229  currentOpacity = mRasterTransparency->alphaValue( val, mOpacity * 255 ) / 255.0;
230  }
231  if ( mAlphaBand > 0 )
232  {
233  currentOpacity *= alphaBlock->value( i ) / 255.0;
234  }
235  QColor currentColor = QColor( mColors[val] );
236  outputData[i] = qRgba( currentOpacity * currentColor.red(), currentOpacity * currentColor.green(), currentOpacity * currentColor.blue(), currentOpacity * 255 );
237  }
238  }
239 
240  delete inputBlock;
241  if ( mAlphaBand > 0 && mBand != mAlphaBand )
242  {
243  delete alphaBlock;
244  }
245 
246  return outputBlock;
247 }

249 void QgsPalettedRasterRenderer::writeXML( QDomDocument& doc, QDomElement& parentElem ) const
250 {
251  if ( parentElem.isNull() )
252  {
253  return;
254  }
255 
256  QDomElement rasterRendererElem = doc.createElement( "rasterrenderer" );
257  _writeXML( doc, rasterRendererElem );
258 
259  rasterRendererElem.setAttribute( "band", mBand );
260  QDomElement colorPaletteElem = doc.createElement( "colorPalette" );
261  for ( int i = 0; i < mNColors; ++i )
262  {
263  QDomElement colorElem = doc.createElement( "paletteEntry" );
264  colorElem.setAttribute( "value", i );
265  colorElem.setAttribute( "color", QColor( mColors[i] ).name() );
266  if ( !label( i ).isEmpty() )
267  {
268  colorElem.setAttribute( "label", label( i ) );
269  }
270  colorPaletteElem.appendChild( colorElem );
271  }
272  rasterRendererElem.appendChild( colorPaletteElem );
273 
274  parentElem.appendChild( rasterRendererElem );
275 }
