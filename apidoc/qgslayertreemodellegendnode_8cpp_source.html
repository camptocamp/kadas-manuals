90 QSizeF QgsLayerTreeModelLegendNode::drawSymbolText( const QgsLegendSettings& settings, ItemContext* ctx, const QSizeF& symbolSize ) const
91 {
92  QSizeF labelSize( 0, 0 );
93 
94  QFont symbolLabelFont = settings.style( QgsComposerLegendStyle::SymbolLabel ).font();
95  double textHeight = settings.fontHeightCharacterMM( symbolLabelFont, QChar( '0' ) );
96  QStringList lines;
97 
98  //show html formatted string only in wms GetLegendGraphic, not in the composer
99  if ( settings.wmsLegend() )
100  {
101  lines = settings.splitStringForWrapping( data( Qt::UserRole ).toString() );
102  }
103  else
104  {
105  lines = settings.splitStringForWrapping( data( Qt::DisplayRole ).toString() );
106  }
107 
108  labelSize.rheight() = lines.count() * textHeight + ( lines.count() - 1 ) * settings.lineSpacing();
109 
110  double labelX = 0.0;
111  double labelY = 0.0;
112  if ( ctx )
113  {
114  ctx->painter->setPen( settings.fontColor() );
115 
116  labelX = ctx->point.x() + qMax(( double ) symbolSize.width(), ctx->labelXOffset );
117  labelY = ctx->point.y();
118 
119  // Vertical alignment of label with symbol
120  if ( labelSize.height() < symbolSize.height() )
121  labelY += symbolSize.height() / 2 + textHeight / 2; // label centered with symbol
122  else
123  labelY += textHeight; // label starts at top and runs under symbol
124  }
125 
126  //html mode
127  if ( settings.wmsLegend() )
128  {
129  double itemHeight = 0;
130  double itemWidth = 0;
131  for ( QStringList::Iterator itemPart = lines.begin(); itemPart != lines.end(); ++itemPart )
132  {
133  QList< QList< QPair< QFont, QString> > > htmlFragments = formatTextAsHtml( symbolLabelFont, *itemPart );
134  for ( int i = 0; i < htmlFragments.size(); ++i )
135  {
136  double maxTextHeight = 0;
137  double currentLineWidth = 0;
138  double currentLabelX = labelX;
139 
140  const QList< QPair< QFont, QString> >& currentLine = htmlFragments.at( i );
141  QList< QPair< QFont, QString> >::const_iterator currentLineIt = currentLine.constBegin();
142  for ( ; currentLineIt != currentLine.constEnd(); ++currentLineIt ) //for each line section
143  {
144  maxTextHeight = 0;
145  if ( ctx )
146  {
147  settings.drawText( ctx->painter, currentLabelX, labelY, currentLineIt->second, currentLineIt->first );
148  }
149  currentLabelX += settings.textWidthMillimeters( currentLineIt->first, currentLineIt->second );
150  currentLineWidth += settings.textWidthMillimeters( currentLineIt->first, currentLineIt->second );
151 
152  maxTextHeight = qMax( settings.fontHeightCharacterMM( symbolLabelFont, QChar( '0' ) ), maxTextHeight );
153  }
154  labelY += maxTextHeight;
155  itemHeight += maxTextHeight;
156  itemWidth = qMax( itemWidth, currentLineWidth );
157  }
158  }
159  //set label width / height
160  labelSize.rheight() = itemHeight;
161  labelSize.rwidth() = itemWidth;
162  }
163  else
164  {
165  for ( QStringList::Iterator itemPart = lines.begin(); itemPart != lines.end(); ++itemPart )
166  {
167  labelSize.rwidth() = qMax( settings.textWidthMillimeters( symbolLabelFont, *itemPart ), double( labelSize.width() ) );
168 
169  if ( ctx )
170  {
171  settings.drawText( ctx->painter, labelX, labelY, *itemPart, symbolLabelFont );
172  if ( itemPart != lines.end() )
173  {
174  labelY += settings.lineSpacing() + textHeight;
175  }
176  }
177  }
178  }
179 
180  return labelSize;
181 }

183 QList< QList< QPair< QFont, QString> > > QgsLayerTreeModelLegendNode::formatTextAsHtml( const QFont& baseFont, const QString& text )
184 {
185  QList< QList< QPair< QFont, QString> > > result;
186  QList< QPair< QFont, QString> > textLine;
187  QStringList boldSplit = text.split( QRegExp( "</{0,1}b{0,1}B{0,1}>" ), QString::KeepEmptyParts );
188  bool bold = false;
189  for ( int i = 0; i < boldSplit.size(); ++i )
190  {
191  QString boldSplitText = boldSplit.at( i );
192  if ( !boldSplitText.isEmpty() )
193  {
194  bool italic = false;
195  QStringList italicSplit = boldSplitText.split( QRegExp( "</{0,1}i{0,1}I{0,1}>" ), QString::KeepEmptyParts );
196  for ( int j = 0; j < italicSplit.size(); ++j )
197  {
198  QString italicSplitText = italicSplit.at( j );
199  if ( !italicSplitText.isEmpty() )
200  {
201  bool underline = false;
202  QStringList underlineSplit = italicSplitText.split( QRegExp( "</{0,1}u{0,1}U{0,1}>" ), QString::KeepEmptyParts );
203  for ( int k = 0; k < underlineSplit.size(); ++k )
204  {
205  QString underlineSplitText = underlineSplit.at( k );
206  if ( !underlineSplitText.isEmpty() )
207  {
208 
209  QFont font = baseFont;
210  if ( bold )
211  {
212  font.setBold( true );
213  }
214  if ( italic )
215  {
216  font.setItalic( true );
217  }
218  if ( underline )
219  {
220  font.setUnderline( true );
221  }
222 
223  QStringList lineSplit = underlineSplitText.split( "<br>", QString::KeepEmptyParts, Qt::CaseInsensitive );
224  for ( int j = 0; j < lineSplit.size(); ++j )
225  {
226  if ( j > 0 )
227  {
228  result.push_back( textLine );
229  textLine.clear();
230  }
231  if ( !lineSplit.at( j ).isEmpty() )
232  {
233  textLine.push_back( qMakePair( font, lineSplit.at( j ) ) );
234  }
235  }
236  }
237  underline = !underline;
238  }
239  }
240  italic = !italic;
241  }
242  }
243  bold = !bold;
244  }
245  result.push_back( textLine );
246 
247  return result;
248 }

277 QVariant QgsSymbolV2LegendNode::data( int role ) const
278 {
279  if ( role == Qt::UserRole ) //user role is html text if available or label text else
280  {
281  return mItem.html().isEmpty() ? mItem.label() : mItem.html();
282  }
283  if ( role == Qt::DisplayRole )
284  {
285  return mLabel;
286  }
287  else if ( role == Qt::EditRole )
288  {
289  return mUserLabel.isEmpty() ? mItem.label() : mUserLabel;
290  }
291  else if ( role == Qt::DecorationRole )
292  {
293  QSize iconSize( 16, 16 ); // TODO: configurable
294  const int indentSize = 20;
295  if ( mPixmap.isNull() )
296  {
297  QPixmap pix;
298  if ( mItem.symbol() )
299  {
300  double scale = 0.0;
301  double mupp = 0.0;
302  int dpi = 0;
303  if ( model() )
304  model()->legendMapViewData( &mupp, &dpi, &scale );
305  bool validData = mupp != 0 && dpi != 0 && scale != 0;
306 
307  // setup temporary render context
308  QgsRenderContext context;
309  context.setScaleFactor( dpi / 25.4 );
310  context.setRendererScale( scale );
311  context.setMapToPixel( QgsMapToPixel( mupp ) ); // hope it's ok to leave out other params
312 
313  pix = QgsSymbolLayerV2Utils::symbolPreviewPixmap( mItem.symbol(), iconSize, validData ? &context : 0 );
314  }
315  else
316  {
317  pix = QPixmap( iconSize );
318  pix.fill( Qt::transparent );
319  }
320 
321  if ( mItem.level() == 0 || ( model() && model()->testFlag( QgsLayerTreeModel::ShowLegendAsTree ) ) )
322  mPixmap = pix;
323  else
324  {
325  // ident the symbol icon to make it look like a tree structure
326  QPixmap pix2( pix.width() + mItem.level() * indentSize, pix.height() );
327  pix2.fill( Qt::transparent );
328  QPainter p( &pix2 );
329  p.drawPixmap( mItem.level() * indentSize, 0, pix );
330  p.end();
331  mPixmap = pix2;
332  }
333  }
334  return mPixmap;
335  }
336  else if ( role == Qt::CheckStateRole )
337  {
338  if ( !mItem.isCheckable() )
339  return QVariant();
340 
341  QgsVectorLayer* vlayer = qobject_cast<QgsVectorLayer*>( mLayerNode->layer() );
342  if ( !vlayer || !vlayer->rendererV2() )
343  return QVariant();
344 
345  return vlayer->rendererV2()->legendSymbolItemChecked( mItem.ruleKey() ) ? Qt::Checked : Qt::Unchecked;
346  }
347  else if ( role == RuleKeyRole )
348  {
349  return mItem.ruleKey();
350  }
351  else if ( role == SymbolV2LegacyRuleKeyRole )
352  {
353  return QVariant::fromValue<void*>( mItem.legacyRuleKey() );
354  }
355  else if ( role == ParentRuleKeyRole )
356  {
357  return mItem.parentRuleKey();
358  }
359 
360  return QVariant();
361 }

386 QSizeF QgsSymbolV2LegendNode::drawSymbol( const QgsLegendSettings& settings, ItemContext* ctx, double itemHeight ) const
387 {
388  QgsSymbolV2* s = mItem.legendSymbol() ? mItem.legendSymbol() : mItem.symbol();
389  if ( !s )
390  {
391  return QSizeF();
392  }
393 
394  // setup temporary render context
395  QgsRenderContext context;
396  context.setScaleFactor( settings.dpi() / 25.4 );
397  context.setRendererScale( settings.mapScale() );
398  context.setMapToPixel( QgsMapToPixel( 1 / ( settings.mmPerMapUnit() * context.scaleFactor() ) ) ); // hope it's ok to leave out other params
399  context.setForceVectorOutput( true );
400  context.setPainter( ctx ? ctx->painter : 0 );
401 
402  //Consider symbol size for point markers
403  double height = settings.symbolSize().height();
404  double width = settings.symbolSize().width();
405  double size = 0;
406  //Center small marker symbols
407  double widthOffset = 0;
408  double heightOffset = 0;
409 
410  if ( QgsMarkerSymbolV2* markerSymbol = dynamic_cast<QgsMarkerSymbolV2*>( s ) )
411  {
412  // allow marker symbol to occupy bigger area if necessary
413  size = markerSymbol->size() * QgsSymbolLayerV2Utils::lineWidthScaleFactor( context, s->outputUnit(), s->mapUnitScale() ) / context.scaleFactor();
414  height = size;
415  width = size;
416  if ( width < settings.symbolSize().width() )
417  {
418  widthOffset = ( settings.symbolSize().width() - width ) / 2.0;
419  }
420  if ( height < settings.symbolSize().height() )
421  {
422  heightOffset = ( settings.symbolSize().height() - height ) / 2.0;
423  }
424  }
425 
426  if ( ctx )
427  {
428  double currentXPosition = ctx->point.x();
429  double currentYCoord = ctx->point.y() + ( itemHeight - settings.symbolSize().height() ) / 2;
430  QPainter* p = ctx->painter;
431 
432  //setup painter scaling to dots so that raster symbology is drawn to scale
433  double dotsPerMM = context.scaleFactor();
434 
435  int opacity = 255;
436  if ( QgsVectorLayer* vectorLayer = dynamic_cast<QgsVectorLayer*>( layerNode()->layer() ) )
437  opacity = 255 - ( 255 * vectorLayer->layerTransparency() / 100 );
438 
439  p->save();
440  p->setRenderHint( QPainter::Antialiasing );
441  p->translate( currentXPosition + widthOffset, currentYCoord + heightOffset );
442  p->scale( 1.0 / dotsPerMM, 1.0 / dotsPerMM );
443  if ( opacity != 255 && settings.useAdvancedEffects() )
444  {
445  //semi transparent layer, so need to draw symbol to an image (to flatten it first)
446  //create image which is same size as legend rect, in case symbol bleeds outside its alloted space
447  QSize tempImageSize( width * dotsPerMM, height * dotsPerMM );
448  QImage tempImage = QImage( tempImageSize, QImage::Format_ARGB32 );
449  tempImage.fill( Qt::transparent );
450  QPainter imagePainter( &tempImage );
451  context.setPainter( &imagePainter );
452  s->drawPreviewIcon( &imagePainter, tempImageSize, &context );
453  context.setPainter( ctx->painter );
454  //reduce opacity of image
455  imagePainter.setCompositionMode( QPainter::CompositionMode_DestinationIn );
456  imagePainter.fillRect( tempImage.rect(), QColor( 0, 0, 0, opacity ) );
457  imagePainter.end();
458  //draw rendered symbol image
459  p->drawImage( 0, 0, tempImage );
460  }
461  else
462  {
463  s->drawPreviewIcon( p, QSize( width * dotsPerMM, height * dotsPerMM ), &context );
464  }
465  p->restore();
466  }
467 
468  return QSizeF( qMax( width + 2 * widthOffset, ( double ) settings.symbolSize().width() ),
469  qMax( height + 2 * heightOffset, ( double ) settings.symbolSize().height() ) );
470 }

552 {

554  double fontSizePoints = mFontSizeInMapUnits ? mFontSize * settings.mmPerMapUnit() / 0.376 : mFontSize;

573  }

580 void QgsVectorLabelLegendNode::updateSettings()
581 {
582  QgsLayerTreeLayer* treeLayer = layerNode();
583  if ( !treeLayer )
584  {
585  return;
586  }
587 
588  QgsVectorLayer* layer = dynamic_cast<QgsVectorLayer*>( treeLayer->layer() );
589  if ( !layer )
590  {
591  return;
592  }
593 
594  bool showFeatureCount = mLayerNode->customProperty( "showFeatureCount", 0 ).toBool();
595  mLayerLabel = layer->name();
596  if ( showFeatureCount && layer && layer->pendingFeatureCount() >= 0 )
597  {
598  mLayerLabel += QString( " [%1]" ).arg( layer->pendingFeatureCount() );
599  }
600 
601  //labeling enabled?
602  mLabelsEnabled = layer->customProperty( "labeling/enabled", "false" ).toString().compare( "true", Qt::CaseInsensitive ) == 0;
603  if ( !mLabelsEnabled )
604  {
605  return;
606  }
607 
608  //color
609  int alpha = layer->customProperty( "labeling/textColorA", "255" ).toInt();
610  int red = layer->customProperty( "labeling/textColorR", "0" ).toInt();
611  int green = layer->customProperty( "labeling/textColorG", "0" ).toInt();
612  int blue = layer->customProperty( "labeling/textColorB", "0" ).toInt();
613 
614  mFontColor = QColor( red, green, blue, alpha );
615 
616  mFontSize = layer->customProperty( "labeling/fontSize", "9" ).toDouble();
617  mFontSizeInMapUnits = layer->customProperty( "labeling/fontSizeInMapUnits", "false" ).toString().compare( "true", Qt::CaseInsensitive ) == 0;
618  QString fontFamily = layer->customProperty( "labeling/fontFamily", "" ).toString();
619  bool italic = layer->customProperty( "labeling/fontItalic", "false" ).toString().compare( "true", Qt::CaseInsensitive ) == 0;
620  bool bold = layer->customProperty( "labeling/fontBold", "false" ).toString().compare( "true", Qt::CaseInsensitive ) == 0;
621  bool underline = layer->customProperty( "labeling/fontUnderline", "false" ).toString().compare( "true", Qt::CaseInsensitive ) == 0;
622  bool strikeout = layer->customProperty( "labeling/fontStrikeout", "false" ).toString().compare( "true", Qt::CaseInsensitive ) == 0;
623 
624  mFont.setFamily( fontFamily );
625  mFont.setItalic( italic );
626  mFont.setBold( bold );
627  mFont.setUnderline( underline );
628  mFont.setStrikeOut( strikeout );
629 }
