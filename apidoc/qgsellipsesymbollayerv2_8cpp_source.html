256 void QgsEllipseSymbolLayerV2::renderPoint( const QPointF& point, QgsSymbolV2RenderContext& context )
257 {
258  QgsExpression* outlineWidthExpression = expression( "outline_width" );
259  QgsExpression* outlineStyleExpression = expression( "outline_style" );
260  QgsExpression* fillStyleExpression = expression( "fill_style" );
261  QgsExpression* fillColorExpression = expression( "fill_color" );
262  QgsExpression* outlineColorExpression = expression( "outline_color" );
263  QgsExpression* widthExpression = expression( "width" );
264  QgsExpression* heightExpression = expression( "height" );
265  QgsExpression* symbolNameExpression = expression( "symbol_name" );
266  QgsExpression* rotationExpression = expression( "rotation" );
267 
268  if ( outlineWidthExpression )
269  {
270  double width = outlineWidthExpression->evaluate( const_cast<QgsFeature*>( context.feature() ) ).toDouble();
271  width *= QgsSymbolLayerV2Utils::lineWidthScaleFactor( context.renderContext(), mOutlineWidthUnit, mOutlineWidthMapUnitScale );
272  mPen.setWidthF( width );
273  }
274  if ( outlineStyleExpression )
275  {
276  Qt::PenStyle style = QgsSymbolLayerV2Utils::decodePenStyle( outlineStyleExpression->evaluate( const_cast<QgsFeature*>( context.feature() ) ).toString() );
277  mPen.setStyle( style );
278  }
279  if ( fillStyleExpression )
280  {
281  Qt::BrushStyle style = QgsSymbolLayerV2Utils::decodeBrushStyle( fillStyleExpression->evaluate( const_cast<QgsFeature*>( context.feature() ) ).toString() );
282  mBrush.setStyle( style );
283  }
284  if ( fillColorExpression )
285  {
286  QString colorString = fillColorExpression->evaluate( const_cast<QgsFeature*>( context.feature() ) ).toString();
287  mBrush.setColor( QColor( QgsSymbolLayerV2Utils::decodeColor( colorString ) ) );
288  }
289  if ( outlineColorExpression )
290  {
291  QString colorString = outlineColorExpression->evaluate( const_cast<QgsFeature*>( context.feature() ) ).toString();
292  mPen.setColor( QColor( QgsSymbolLayerV2Utils::decodeColor( colorString ) ) );
293  }
294  double scaledWidth = mSymbolWidth;
295  double scaledHeight = mSymbolHeight;
296  if ( widthExpression || heightExpression || symbolNameExpression )
297  {
298  QString symbolName = mSymbolName;
299  if ( symbolNameExpression )
300  {
301  symbolName = symbolNameExpression->evaluate( const_cast<QgsFeature*>( context.feature() ) ).toString();
302  }
303  preparePath( symbolName, context, &scaledWidth, &scaledHeight, context.feature() );
304  }
305 
306  //offset
307  double offsetX = 0;
308  double offsetY = 0;
309  markerOffset( context, scaledWidth, scaledHeight, mSymbolWidthUnit, mSymbolHeightUnit, offsetX, offsetY, mSymbolWidthMapUnitScale, mSymbolHeightMapUnitScale );
310  QPointF off( offsetX, offsetY );
311 
312  QPainter* p = context.renderContext().painter();
313  if ( !p )
314  {
315  return;
316  }
317 
318  //priority for rotation: 1. data defined symbol level, 2. symbol layer rotation (mAngle)
319  double rotation = 0.0;
320  if ( rotationExpression )
321  {
322  rotation = rotationExpression->evaluate( const_cast<QgsFeature*>( context.feature() ) ).toDouble();
323  }
324  else if ( !qgsDoubleNear( mAngle, 0.0 ) )
325  {
326  rotation = mAngle;
327  }
328  if ( rotation )
329  off = _rotatedOffset( off, rotation );
330 
331  QMatrix transform;
332  transform.translate( point.x() + off.x(), point.y() + off.y() );
333  if ( !qgsDoubleNear( rotation, 0.0 ) )
334  {
335  transform.rotate( rotation );
336  }
337 
338  p->setPen( mPen );
339  p->setBrush( mBrush );
340  p->drawPath( transform.map( mPainterPath ) );
341 }

385 void QgsEllipseSymbolLayerV2::writeSldMarker( QDomDocument &doc, QDomElement &element, QgsStringMap props ) const
386 {
387  // <Graphic>
388  QDomElement graphicElem = doc.createElement( "se:Graphic" );
389  element.appendChild( graphicElem );
390 
391  QgsSymbolLayerV2Utils::wellKnownMarkerToSld( doc, graphicElem, mSymbolName, mFillColor, mOutlineColor, mOutlineStyle, mOutlineWidth, mSymbolWidth );
392 
393  // store w/h factor in a <VendorOption>
394  double widthHeightFactor = mSymbolWidth / mSymbolHeight;
395  QDomElement factorElem = QgsSymbolLayerV2Utils::createVendorOptionElement( doc, "widthHeightFactor", QString::number( widthHeightFactor ) );
396  graphicElem.appendChild( factorElem );
397 
398  // <Rotation>
399  const QgsExpression* rotationExpression = dataDefinedProperty( "rotation" );
400  QString angleFunc = props.value( "angle", "" );
401  if ( angleFunc.isEmpty() ) // symbol has no angle set
402  {
403 
404  if ( rotationExpression )
405  angleFunc = rotationExpression->expression();
406  else if ( !qgsDoubleNear( mAngle, 0.0 ) )
407  angleFunc = QString::number( mAngle );
408  }
409  else if ( rotationExpression )
410  {
411  // the symbol has an angle and the symbol layer have a rotation
412  // property set
413  angleFunc = QString( "%1 + %2" ).arg( angleFunc ).arg( rotationExpression->expression() );
414  }
415  else if ( !qgsDoubleNear( mAngle, 0.0 ) )
416  {
417  // both the symbol and the symbol layer have angle value set
418  bool ok;
419  double angle = angleFunc.toDouble( &ok );
420  if ( !ok )
421  {
422  // its a string (probably a property name or a function)
423  angleFunc = QString( "%1 + %2" ).arg( angleFunc ).arg( mAngle );
424  }
425  else if ( !qgsDoubleNear( angle + mAngle, 0.0 ) )
426  {
427  // it's a double value
428  angleFunc = QString::number( angle + mAngle );
429  }
430  }
431  QgsSymbolLayerV2Utils::createRotationElement( doc, graphicElem, angleFunc );
432 }

434 QgsSymbolLayerV2* QgsEllipseSymbolLayerV2::createFromSld( QDomElement &element )
435 {
436  QgsDebugMsg( "Entered." );
437 
438  QDomElement graphicElem = element.firstChildElement( "Graphic" );
439  if ( graphicElem.isNull() )
440  return NULL;
441 
442  QString name = "circle";
443  QColor fillColor, borderColor;
444  double borderWidth, size;
445  double widthHeightFactor = 1.0;
446  Qt::PenStyle borderStyle;
447 
448  QgsStringMap vendorOptions = QgsSymbolLayerV2Utils::getVendorOptionList( graphicElem );
449  for ( QgsStringMap::iterator it = vendorOptions.begin(); it != vendorOptions.end(); ++it )
450  {
451  if ( it.key() == "widthHeightFactor" )
452  {
453  bool ok;
454  double v = it.value().toDouble( &ok );
455  if ( ok && !qgsDoubleNear( v, 0.0 ) && v > 0 )
456  widthHeightFactor = v;
457  }
458  }
459 
460  if ( !QgsSymbolLayerV2Utils::wellKnownMarkerFromSld( graphicElem, name, fillColor, borderColor, borderStyle, borderWidth, size ) )
461  return NULL;
462 
463  double angle = 0.0;
464  QString angleFunc;
465  if ( QgsSymbolLayerV2Utils::rotationFromSldElement( graphicElem, angleFunc ) )
466  {
467  bool ok;
468  double d = angleFunc.toDouble( &ok );
469  if ( ok )
470  angle = d;
471  }
472 
473  QgsEllipseSymbolLayerV2 *m = new QgsEllipseSymbolLayerV2();
474  m->setSymbolName( name );
475  m->setFillColor( fillColor );
476  m->setOutlineColor( borderColor );
477  m->setOutlineStyle( borderStyle );
478  m->setOutlineWidth( borderWidth );
479  m->setSymbolWidth( size );
480  m->setSymbolHeight( size / widthHeightFactor );
481  m->setAngle( angle );
482  return m;
483 }

522 void QgsEllipseSymbolLayerV2::preparePath( const QString& symbolName, QgsSymbolV2RenderContext& context, double* scaledWidth, double* scaledHeight, const QgsFeature* f )
523 {
524  mPainterPath = QPainterPath();
525  const QgsRenderContext& ct = context.renderContext();
526 
527  double width = 0;
528 
529  QgsExpression* widthExpression = expression( "width" );
530  if ( widthExpression ) //1. priority: data defined setting on symbol layer level
531  {
532  width = widthExpression->evaluate( const_cast<QgsFeature*>( f ) ).toDouble();
533  }
534  else if ( context.renderHints() & QgsSymbolV2::DataDefinedSizeScale ) //2. priority: is data defined size on symbol level
535  {
536  width = mSize;
537  }
538  else //3. priority: global width setting
539  {
540  width = mSymbolWidth;
541  }
542  if ( scaledWidth )
543  {
544  *scaledWidth = width;
545  }
546  width *= QgsSymbolLayerV2Utils::lineWidthScaleFactor( ct, mSymbolWidthUnit, mSymbolHeightMapUnitScale );
547 
548  double height = 0;
549  QgsExpression* heightExpression = expression( "height" );
550  if ( heightExpression ) //1. priority: data defined setting on symbol layer level
551  {
552  height = heightExpression->evaluate( const_cast<QgsFeature*>( f ) ).toDouble();
553  }
554  else if ( context.renderHints() & QgsSymbolV2::DataDefinedSizeScale ) //2. priority: is data defined size on symbol level
555  {
556  height = mSize;
557  }
558  else //3. priority: global height setting
559  {
560  height = mSymbolHeight;
561  }
562  if ( scaledHeight )
563  {
564  *scaledHeight = height;
565  }
566  height *= QgsSymbolLayerV2Utils::lineWidthScaleFactor( ct, mSymbolHeightUnit, mSymbolHeightMapUnitScale );
567 
568  if ( symbolName == "circle" )
569  {
570  mPainterPath.addEllipse( QRectF( -width / 2.0, -height / 2.0, width, height ) );
571  }
572  else if ( symbolName == "rectangle" )
573  {
574  mPainterPath.addRect( QRectF( -width / 2.0, -height / 2.0, width, height ) );
575  }
576  else if ( symbolName == "cross" )
577  {
578  mPainterPath.moveTo( 0, -height / 2.0 );
579  mPainterPath.lineTo( 0, height / 2.0 );
580  mPainterPath.moveTo( -width / 2.0, 0 );
581  mPainterPath.lineTo( width / 2.0, 0 );
582  }
583  else if ( symbolName == "triangle" )

629 bool QgsEllipseSymbolLayerV2::writeDxf( QgsDxfExport& e, double mmMapUnitScaleFactor, const QString& layerName, const QgsSymbolV2RenderContext* context, const QgsFeature* f, const QPointF& shift ) const
630 {
631  //width
632  double symbolWidth = mSymbolWidth;
633  QgsExpression* widthExpression = expression( "width" );
634  if ( widthExpression ) //1. priority: data defined setting on symbol layer level
635  {
636  symbolWidth = widthExpression->evaluate( const_cast<QgsFeature*>( f ) ).toDouble();
637  }
638  else if ( context->renderHints() & QgsSymbolV2::DataDefinedSizeScale ) //2. priority: is data defined size on symbol level
639  {
640  symbolWidth = mSize;
641  }
642  if ( mSymbolWidthUnit == QgsSymbolV2::MM )
643  {
644  symbolWidth *= mmMapUnitScaleFactor;
645  }
646 
647  //height
648  double symbolHeight = mSymbolHeight;
649  QgsExpression* heightExpression = expression( "height" );
650  if ( heightExpression ) //1. priority: data defined setting on symbol layer level
651  {
652  symbolHeight = heightExpression->evaluate( const_cast<QgsFeature*>( f ) ).toDouble();
653  }
654  else if ( context->renderHints() & QgsSymbolV2::DataDefinedSizeScale ) //2. priority: is data defined size on symbol level
655  {
656  symbolHeight = mSize;
657  }
658  if ( mSymbolHeightUnit == QgsSymbolV2::MM )
659  {
660  symbolHeight *= mmMapUnitScaleFactor;
661  }
662 
663  //outline width
664  double outlineWidth = mOutlineWidth;
665  QgsExpression* outlineWidthExpression = expression( "outline_width" );
666  if ( outlineWidthExpression )
667  {
668  outlineWidth = outlineWidthExpression->evaluate( const_cast<QgsFeature*>( context->feature() ) ).toDouble();
669  }
670  if ( mOutlineWidthUnit == QgsSymbolV2::MM )
671  {
672  outlineWidth *= outlineWidth;
673  }
674 
675  //fill color
676  QColor fc = mFillColor;
677  QgsExpression* fillColorExpression = expression( "fill_color" );
678  if ( fillColorExpression )
679  {
680  fc = QColor( fillColorExpression->evaluate( const_cast<QgsFeature*>( context->feature() ) ).toString() );
681  }
682 
683  //outline color
684  QColor oc = mOutlineColor;
685  QgsExpression* outlineColorExpression = expression( "outline_color" );
686  if ( outlineColorExpression )
687  {
688  oc = QColor( outlineColorExpression->evaluate( const_cast<QgsFeature*>( context->feature() ) ).toString() );
689  }
690 
691  //symbol name
692  QString symbolName = mSymbolName;
693  QgsExpression* symbolNameExpression = expression( "symbol_name" );
694  if ( symbolNameExpression )
695  {
696  QgsExpression* symbolNameExpression = expression( "symbol_name" );
697  symbolName = symbolNameExpression->evaluate( const_cast<QgsFeature*>( context->feature() ) ).toString();
698  }
699 
700  //offset
701  double offsetX = 0;
702  double offsetY = 0;
703  markerOffset( *context, offsetX, offsetY );
704  QPointF off( offsetX, offsetY );
705 
706  //priority for rotation: 1. data defined symbol level, 2. symbol layer rotation (mAngle)
707  double rotation = 0.0;
708  QgsExpression* rotationExpression = expression( "rotation" );
709  if ( rotationExpression )
710  {
711  rotation = rotationExpression->evaluate( const_cast<QgsFeature*>( context->feature() ) ).toDouble();
712  }
713  else if ( !qgsDoubleNear( mAngle, 0.0 ) )
714  {
715  rotation = mAngle;
716  }
717  rotation = -rotation; //rotation in Qt is counterclockwise
718  if ( rotation )
719  off = _rotatedOffset( off, rotation );
720 
721  QTransform t;
722  t.translate( shift.x() + offsetX, shift.y() + offsetY );
723 
724  if ( rotation != 0 )
725  t.rotate( rotation );
726 
727  double halfWidth = symbolWidth / 2.0;
728  double halfHeight = symbolHeight / 2.0;
729 
730  if ( symbolName == "circle" )
731  {
732  if ( qgsDoubleNear( halfWidth, halfHeight ) )
733  {
734  QPointF pt( t.map( QPointF( 0, 0 ) ) );
735  e.writeFilledCircle( layerName, oc, pt, halfWidth );
736  }
737  else
738  {
739  QgsPolyline line;
740  double stepsize = 2 * M_PI / 40;
741  for ( int i = 0; i < 39; ++i )
742  {
743  double angle = stepsize * i;
744  double x = halfWidth * cos( angle );
745  double y = halfHeight * sin( angle );
746  QPointF pt( t.map( QPointF( x, y ) ) );
747  line.push_back( pt );
748  }
749  //close ellipse with first point
750  line.push_back( line.at( 0 ) );
751  if ( mBrush.style() != Qt::NoBrush )
752  e.writePolygon( QgsPolygon() << line, layerName, "SOLID", fc );
753  if ( mPen.style() != Qt::NoPen )
754  e.writePolyline( line, layerName, "CONTINUOUS", oc, outlineWidth );
755  }
756  }
757  else if ( symbolName == "rectangle" )
758  {
759  QgsPolygon p( 1 );
760  p[0].resize( 5 );
761  p[0][0] = t.map( QPointF( -halfWidth, -halfHeight ) );
762  p[0][1] = t.map( QPointF( halfWidth, -halfHeight ) );
763  p[0][2] = t.map( QPointF( halfWidth, halfHeight ) );
764  p[0][3] = t.map( QPointF( -halfWidth, halfHeight ) );
765  p[0][4] = p[0][0];
766  if ( mBrush.style() != Qt::NoBrush )
767  e.writePolygon( p, layerName, "SOLID", fc );
768  if ( mPen.style() != Qt::NoPen )
769  e.writePolyline( p[0], layerName, "CONTINUOUS", oc, outlineWidth );
770  return true;
771  }
772  else if ( symbolName == "cross" && mPen.style() != Qt::NoPen )
773  {
774  QgsPolyline line( 2 );
775  line[0] = t.map( QPointF( -halfWidth, 0 ) );
776  line[1] = t.map( QPointF( halfWidth, 0 ) );
777  e.writePolyline( line, layerName, "CONTINUOUS", oc, outlineWidth );
778 
779  line[0] = t.map( QPointF( 0, halfHeight ) );
780  line[1] = t.map( QPointF( 0, -halfHeight ) );
781  e.writePolyline( line, layerName, "CONTINUOUS", oc, outlineWidth );
782 
783  return true;
784  }
785  else if ( symbolName == "triangle" )
786  {
787  QgsPolygon p( 1 );
788  p[0].resize( 4 );
789  p[0][0] = QPointF( t.map( QPointF( -halfWidth, -halfHeight ) ) );
790  p[0][1] = QPointF( t.map( QPointF( halfWidth, -halfHeight ) ) );
791  p[0][2] = QPointF( t.map( QPointF( 0, halfHeight ) ) );
792  p[0][3] = p[0][0];
793  if ( mBrush.style() != Qt::NoBrush )
794  e.writePolygon( p, layerName, "SOLID", fc );
795  if ( mPen.style() != Qt::NoPen )
796  e.writePolyline( p[0], layerName, "CONTINUOUS", oc, outlineWidth );
797  return true;
798  }
799 
800  return false; //soon...
801 }
