52 const QgsCoordinateTransform* QgsCoordinateTransformCache::transform( const QString& srcAuthId, const QString& destAuthId, int srcDatumTransform, int destDatumTransform )
53 {
54  QList< QgsCoordinateTransform* > values =
55  mTransforms.values( qMakePair( srcAuthId, destAuthId ) );
56 
57  QList< QgsCoordinateTransform* >::const_iterator valIt = values.constBegin();
58  for ( ; valIt != values.constEnd(); ++valIt )
59  {
60  if ( *valIt &&
61  ( *valIt )->sourceDatumTransform() == srcDatumTransform &&
62  ( *valIt )->destinationDatumTransform() == destDatumTransform )
63  {
64  return *valIt;
65  }
66  }
67 
68  //not found, insert new value
69  const QgsCoordinateReferenceSystem& srcCrs = QgsCRSCache::instance()->crsByAuthId( srcAuthId );
70  const QgsCoordinateReferenceSystem& destCrs = QgsCRSCache::instance()->crsByAuthId( destAuthId );
71  QgsCoordinateTransform* ct = new QgsCoordinateTransform( srcCrs, destCrs );
72  ct->setSourceDatumTransform( srcDatumTransform );
73  ct->setDestinationDatumTransform( destDatumTransform );
74  ct->initialise();
75  mTransforms.insertMulti( qMakePair( srcAuthId, destAuthId ), ct );
76  return ct;
77 }

80 {

83  QList< QPair< QString, QString > > updateList;

232 const QgsEllipsoidCache::Params& QgsEllipsoidCache::getParams( const QString& ellipsoid )
233 {
234  QMap<QString, Params>::iterator it = mParams.find( ellipsoid );
235  if ( it != mParams.end() )
236  {
237  return it.value();
238  }
239 
240  Params params;
241  //
242  // SQLITE3 stuff - get parameters for selected ellipsoid
243  //
244  sqlite3 *myDatabase;
245  const char *myTail;
246  sqlite3_stmt *myPreparedStatement;
247  int myResult;
248 
249  // Continue with PROJ.4 list of ellipsoids.
250 
251  //check the db is available
252  myResult = sqlite3_open_v2( QgsApplication::srsDbFilePath().toUtf8().data(), &myDatabase, SQLITE_OPEN_READONLY, NULL );
253  if ( myResult )
254  {
255  QgsMessageLog::logMessage( QObject::tr( "Can't open database: %1" ).arg( sqlite3_errmsg( myDatabase ) ) );
256  // XXX This will likely never happen since on open, sqlite creates the
257  // database if it does not exist.
258  return mParams.insert( ellipsoid, params ).value();
259  }
260  // Set up the query to retrieve the projection information needed to populate the ELLIPSOID list
261  QString mySql = "select radius, parameter2 from tbl_ellipsoid where acronym='" + ellipsoid + "'";
262  myResult = sqlite3_prepare( myDatabase, mySql.toUtf8(), mySql.toUtf8().length(), &myPreparedStatement, &myTail );
263  // XXX Need to free memory from the error msg if one is set
264  if ( myResult == SQLITE_OK )
265  {
266  if ( sqlite3_step( myPreparedStatement ) == SQLITE_ROW )
267  {
268  params.radius = QString(( char * )sqlite3_column_text( myPreparedStatement, 0 ) );
269  params.parameter2 = QString(( char * )sqlite3_column_text( myPreparedStatement, 1 ) );
270  }
271  }
272  // close the sqlite3 statement
273  sqlite3_finalize( myPreparedStatement );
274  sqlite3_close( myDatabase );
275 
276  return mParams.insert( ellipsoid, params ).value();
277 }
