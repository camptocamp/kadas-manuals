256 void QgsVectorLayer::setDisplayField( QString fldName )
257 {
258  if ( !hasGeometryType() )
259  return;
260 
261  // If fldName is provided, use it as the display field, otherwise
262  // determine the field index for the feature column of the identify
263  // dialog. We look for fields containing "name" first and second for
264  // fields containing "id". If neither are found, the first field
265  // is used as the node.
266  QString idxName = "";
267  QString idxId = "";
268 
269  if ( !fldName.isEmpty() )
270  {
271  mDisplayField = fldName;
272  }
273  else
274  {
275  const QgsFields &fields = pendingFields();
276  int fieldsSize = fields.size();
277 
278  for ( int idx = 0; idx < fields.count(); ++idx )
279  {
280  QString fldName = fields[idx].name();
281  QgsDebugMsg( "Checking field " + fldName + " of " + QString::number( fieldsSize ) + " total" );
282 
283  // Check the fields and keep the first one that matches.
284  // We assume that the user has organized the data with the
285  // more "interesting" field names first. As such, name should
286  // be selected before oldname, othername, etc.
287  if ( fldName.indexOf( "name", 0, Qt::CaseInsensitive ) > -1 )
288  {
289  if ( idxName.isEmpty() )
290  {
291  idxName = fldName;
292  }
293  }
294  if ( fldName.indexOf( "descrip", 0, Qt::CaseInsensitive ) > -1 )
295  {
296  if ( idxName.isEmpty() )
297  {
298  idxName = fldName;
299  }
300  }
301  if ( fldName.indexOf( "id", 0, Qt::CaseInsensitive ) > -1 )
302  {
303  if ( idxId.isEmpty() )
304  {
305  idxId = fldName;
306  }
307  }
308  }
309 
310  //if there were no fields in the dbf just return - otherwise qgis segfaults!
311  if ( fieldsSize == 0 )
312  return;
313 
314  if ( idxName.length() > 0 )
315  {
316  mDisplayField = idxName;
317  }
318  else
319  {
320  if ( idxId.length() > 0 )
321  {
322  mDisplayField = idxId;
323  }
324  else
325  {
326  mDisplayField = fields[0].name();
327  }
328  }
329 
330  }
331 }

343  if ( mRendererV2 && mLabelOn && mLabel &&
344  ( !mLabel->scaleBasedVisibility() ||
345  ( mLabel->minScale() <= rendererContext.rendererScale() &&
346  rendererContext.rendererScale() <= mLabel->maxScale() ) ) )
347  {
348  QgsAttributeList attributes;
349  foreach ( QString attrName, mRendererV2->usedAttributes() )
350  {
351  int attrNum = fieldNameIndex( attrName );
352  attributes.append( attrNum );
353  }
354  // make sure the renderer is ready for classification ("symbolForFeature")
355  mRendererV2->startRender( rendererContext, pendingFields() );

359 
360  QgsDebugMsg( "Selecting features based on view extent" );
361 
362  int featureCount = 0;
363

365  {
366  // select the records in the extent. The provider sets a spatial filter
367  // and sets up the selection set for retrieval

369  .setFilterRect( rendererContext.extent() )

374  {
375  if ( mRendererV2->willRenderFeature( fet ) )

628 QgsRectangle QgsVectorLayer::boundingBoxOfSelected()
629 {
630  if ( mSelectedFeatureIds.size() == 0 ) //no selected features
631  {
632  return QgsRectangle( 0, 0, 0, 0 );
633  }
634 
635  QgsRectangle r, retval;
636  retval.setMinimal();
637 
638  QgsFeature fet;
639  if ( mDataProvider->capabilities() & QgsVectorDataProvider::SelectAtId )
640  {
641  foreach ( QgsFeatureId fid, mSelectedFeatureIds )
642  {
643  if ( getFeatures( QgsFeatureRequest()
644  .setFilterFid( fid )
645  .setSubsetOfAttributes( QgsAttributeList() ) )
646  .nextFeature( fet ) &&
647  fet.geometry() )
648  {
649  r = fet.geometry()->boundingBox();
650  retval.combineExtentWith( &r );
651  }
652  }
653  }
654  else
655  {
656  QgsFeatureIterator fit = getFeatures( QgsFeatureRequest()
657  .setSubsetOfAttributes( QgsAttributeList() ) );
658 
659  while ( fit.nextFeature( fet ) )
660  {
661  if ( mSelectedFeatureIds.contains( fet.id() ) )
662  {
663  if ( fet.geometry() )
664  {
665  r = fet.geometry()->boundingBox();
666  retval.combineExtentWith( &r );
667  }
668  }
669  }
670  }
671 
672  if ( retval.width() == 0.0 || retval.height() == 0.0 )
673  {
674  // If all of the features are at the one point, buffer the
675  // rectangle a bit. If they are all at zero, do something a bit
676  // more crude.
677 
678  if ( retval.xMinimum() == 0.0 && retval.xMaximum() == 0.0 &&
679  retval.yMinimum() == 0.0 && retval.yMaximum() == 0.0 )
680  {
681  retval.set( -1.0, -1.0, 1.0, 1.0 );
682  }
683  }
684 
685  return retval;
686 }

705 bool QgsVectorLayer::countSymbolFeatures( bool showProgress )
706 {
707  if ( mSymbolFeatureCounted ) return true;
708  mSymbolFeatureCountMap.clear();
709 
710  if ( !mDataProvider )
711  {
712  QgsDebugMsg( "invoked with null mDataProvider" );
713  return false;
714  }
715  if ( !mRendererV2 )
716  {
717  QgsDebugMsg( "invoked with null mRendererV2" );
718  return false;
719  }
720 
721  QgsLegendSymbolList symbolList = mRendererV2->legendSymbolItems();
722  QgsLegendSymbolList::const_iterator symbolIt = symbolList.constBegin();
723 
724  for ( ; symbolIt != symbolList.constEnd(); ++symbolIt )
725  {
726  mSymbolFeatureCountMap.insert( symbolIt->second, 0 );
727  }
728 
729  long nFeatures = pendingFeatureCount();
730  QProgressDialog progressDialog( tr( "Updating feature count for layer %1" ).arg( name() ), tr( "Abort" ), 0, nFeatures );
731  progressDialog.setWindowModality( Qt::WindowModal );
732  int featuresCounted = 0;
733 
734  QgsFeatureIterator fit = getFeatures( QgsFeatureRequest().setFlags( QgsFeatureRequest::NoGeometry ) );
735 
736  // Renderer (rule based) may depend on context scale, with scale is ignored if 0
737  QgsRenderContext renderContext;
738  renderContext.setRendererScale( 0 );
739  mRendererV2->startRender( renderContext, pendingFields() );
740 
741  QgsFeature f;
742  while ( fit.nextFeature( f ) )
743  {
744  QgsSymbolV2List featureSymbolList = mRendererV2->originalSymbolsForFeature( f );
745  for ( QgsSymbolV2List::iterator symbolIt = featureSymbolList.begin(); symbolIt != featureSymbolList.end(); ++symbolIt )
746  {
747  mSymbolFeatureCountMap[*symbolIt] += 1;
748  }
749  ++featuresCounted;
750 
751  if ( showProgress )
752  {
753  if ( featuresCounted % 50 == 0 )
754  {
755  if ( featuresCounted > nFeatures ) //sometimes the feature count is not correct
756  {
757  progressDialog.setMaximum( 0 );
