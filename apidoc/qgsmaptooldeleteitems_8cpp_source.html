66 void QgsMapToolDeleteItems::deleteItems( const QgsRectangle &filterRect, const QgsCoordinateReferenceSystem &filterRectCrs )
67 {
68  // Search annotation items
69  QList<QgsAnnotationItem*> delAnnotationItems;
70  foreach ( QGraphicsItem* item, canvas()->scene()->items() )
71  {
72  QgsAnnotationItem* aitem = dynamic_cast<QgsAnnotationItem*>( item );
73  if ( aitem && aitem->mapPositionFixed() )
74  {
75  QgsPoint p = QgsCoordinateTransformCache::instance()->transform( aitem->mapGeoPosCrs().authid(), filterRectCrs.authid() )->transform( aitem->mapGeoPos() );
76  if ( filterRect.contains( p ) )
77  {
78  delAnnotationItems.append( aitem );
79  }
80  }
81  }
82 
83  // Search redlining and plugin layers
84  QgsRenderContext renderContext = QgsRenderContext::fromMapSettings( canvas()->mapSettings() );
85  QMap<QgsRedliningLayer*, QgsFeatureIds> delRedliningItems;
86  QMap<QgsPluginLayer*, QVariantList> delPluginItems;
87  foreach ( QgsMapLayer* layer, canvas()->layers() )
88  {
89  if ( layer->type() == QgsMapLayer::RedliningLayer )
90  {
91  QgsRedliningLayer* rlayer = static_cast<QgsRedliningLayer*>( layer );
92 
93  if ( rlayer->hasScaleBasedVisibility() &&
94  ( rlayer->minimumScale() > canvas()->mapSettings().scale() ||
95  rlayer->maximumScale() <= canvas()->mapSettings().scale() ) )
96  {
97  continue;
98  }
99 
100 
101  QgsFeatureRendererV2* renderer = rlayer->rendererV2();
102  bool filteredRendering = false;
103  if ( renderer && renderer->capabilities() & QgsFeatureRendererV2::ScaleDependent )
104  {
105  // setup scale for scale dependent visibility (rule based)
106  renderer->startRender( renderContext, rlayer->pendingFields() );
107  filteredRendering = renderer->capabilities() & QgsFeatureRendererV2::Filter;
108  }
109 
110  QgsRectangle layerFilterRect = QgsCoordinateTransformCache::instance()->transform( filterRectCrs.authid(), rlayer->crs().authid() )->transform( filterRect );
111  QgsFeatureIterator fit = rlayer->getFeatures( QgsFeatureRequest( layerFilterRect ).setFlags( QgsFeatureRequest::ExactIntersect ) );
112  QgsFeature feature;
113  while ( fit.nextFeature( feature ) )
114  {
115  if ( filteredRendering && !renderer->willRenderFeature( feature ) )
116  {
117  continue;
118  }
119  delRedliningItems[rlayer].insert( feature.id() );
120  }
121  if ( renderer && renderer->capabilities() & QgsFeatureRendererV2::ScaleDependent )
122  {
123  renderer->stopRender( renderContext );
124  }
125  }
126  else if ( layer->type() == QgsMapLayer::PluginLayer )
127  {
128  QgsPluginLayer* player = static_cast<QgsPluginLayer*>( layer );
129  QgsRectangle layerFilterRect = QgsCoordinateTransformCache::instance()->transform( filterRectCrs.authid(), player->crs().authid() )->transform( filterRect );
130  QVariantList items = player->getItems( layerFilterRect );
131  if ( !items.isEmpty() )
132  {
133  delPluginItems[player] = items;
134  }
135  }
136  }
137 
138  if ( !delAnnotationItems.isEmpty() || !delRedliningItems.isEmpty() || !delPluginItems.isEmpty() )
139  {
140  QCheckBox* checkBoxAnnotationItems = 0;
141  QMap<QgsRedliningLayer*, QCheckBox*> checkBoxRedliningItems;
142  QMap<QgsPluginLayer*, QCheckBox*> checkBoxPluginItems;
143  QDialog confirmDialog;
144  confirmDialog.setWindowTitle( tr( "Delete items" ) );
145  confirmDialog.setLayout( new QVBoxLayout() );
146  confirmDialog.layout()->addWidget( new QLabel( tr( "Do you want to delete the following items?" ) ) );
147  if ( !delAnnotationItems.isEmpty() )
148  {
149  checkBoxAnnotationItems = new QCheckBox( tr( "%1 annotation item(s)" ).arg( delAnnotationItems.size() ) );
150  checkBoxAnnotationItems->setChecked( true );
151  confirmDialog.layout()->addWidget( checkBoxAnnotationItems );
152  }
153  foreach ( QgsRedliningLayer* layer, delRedliningItems.keys() )
154  {
155  QCheckBox* checkBox = new QCheckBox( tr( "%1 items(s) from layer %2" ).arg( delRedliningItems[layer].size() ).arg( layer->name() ) );
156  checkBox->setChecked( true );
157  checkBoxRedliningItems[layer] = checkBox;
158  confirmDialog.layout()->addWidget( checkBox );
159  }
160  foreach ( QgsPluginLayer* layer, delPluginItems.keys() )
161  {
162  QCheckBox* checkBox = new QCheckBox( tr( "%1 items(s) from layer %2" ).arg( delPluginItems[layer].size() ).arg( layer->name() ) );
163  checkBox->setChecked( true );
164  checkBoxPluginItems[layer] = checkBox;
165  confirmDialog.layout()->addWidget( checkBox );
166  }
167  confirmDialog.layout()->addItem( new QSpacerItem( 1, 1, QSizePolicy::Minimum, QSizePolicy::Expanding ) );
168  QDialogButtonBox* bbox = new QDialogButtonBox( QDialogButtonBox::Ok | QDialogButtonBox::Cancel, Qt::Horizontal );
169  connect( bbox, SIGNAL( accepted() ), &confirmDialog, SLOT( accept() ) );
170  connect( bbox, SIGNAL( rejected() ), &confirmDialog, SLOT( reject() ) );
171  confirmDialog.layout()->addWidget( bbox );
172  if ( confirmDialog.exec() == QDialog::Accepted )
173  {
174  if ( checkBoxAnnotationItems && checkBoxAnnotationItems->isChecked() )
175  {
176  qDeleteAll( delAnnotationItems );
177  }
178  foreach ( QgsRedliningLayer* layer, checkBoxRedliningItems.keys() )
179  {
180  if ( checkBoxRedliningItems[layer]->isChecked() )
181  {
182  layer->dataProvider()->deleteFeatures( delRedliningItems[layer] );
183  }
184  layer->triggerRepaint();
185  }
186  foreach ( QgsPluginLayer* layer, checkBoxPluginItems.keys() )
187  {
188  if ( checkBoxPluginItems[layer]->isChecked() )
189  {
190  layer->deleteItems( delPluginItems[layer] );
191  }
192  layer->triggerRepaint();
193  }
194  }
195  }
196 }
