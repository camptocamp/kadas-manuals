64 void QgsComposerLegend::paint( QPainter* painter, const QStyleOptionGraphicsItem* itemStyle, QWidget* pWidget )
65 {
66  Q_UNUSED( itemStyle );
67  Q_UNUSED( pWidget );
68 
69  if ( !painter )
70  return;
71 
72  if ( !shouldDrawItem() )
73  {
74  return;
75  }
76 
77  int dpi = painter->device()->logicalDpiX();
78  double dotsPerMM = dpi / 25.4;
79 
80  if ( mComposition )
81  {
82  mSettings.setUseAdvancedEffects( mComposition->useAdvancedEffects() );
83  mSettings.setDpi( dpi );
84  }
85  if ( mComposerMap )
86  {
87  mSettings.setMmPerMapUnit( mComposerMap->mapUnitsToMM() );
88 
89  // use a temporary QgsMapSettings to find out real map scale
90  QgsMapSettings ms = mComposerMap->composition()->mapSettings();
91  ms.setOutputSize( QSizeF( mComposerMap->rect().width() * dotsPerMM, mComposerMap->rect().height() * dotsPerMM ).toSize() );
92  ms.setExtent( *mComposerMap->currentMapExtent() );
93  ms.setOutputDpi( dpi );
94  mSettings.setMapScale( ms.scale() );
95  }
96 
97  drawBackground( painter );
98  painter->save();
99  //antialiasing on
100  painter->setRenderHint( QPainter::Antialiasing, true );
101  painter->setPen( QPen( QColor( 0, 0, 0 ) ) );
102 
103  QgsLegendRenderer legendRenderer( mLegendModel2, mSettings );
104  legendRenderer.setLegendSize( rect().size() );
105 
106  //adjust box if width or height is too small
107  QSizeF size = legendRenderer.minimumSize();
108  if ( size.height() > rect().height() || size.width() > rect().width() )
109  {
110  //need to resize box
111  QRectF targetRect = QRectF( pos().x(), pos().y(), rect().width(), rect().height() );
112  if ( size.height() > targetRect.height() )
113  targetRect.setHeight( size.height() );
114  if ( size.width() > rect().width() )
115  targetRect.setWidth( size.width() );
116 
117  //set new rect, respecting position mode and data defined size/position
118  setSceneRect( evalItemRect( targetRect, true ) );
119  }
120 
121  legendRenderer.drawLegend( painter );
122 
123  painter->restore();
124 
125  //draw frame and selection boxes if necessary
126  drawFrame( painter );
127  if ( isSelected() )
128  {
129  drawSelectionBoxes( painter );
130  }
131 }

309 static void _readOldLegendGroup( QDomElement& elem, QgsLayerTreeGroup* parentGroup )
310 {
311  QDomElement itemElem = elem.firstChildElement();
312 
313  while ( !itemElem.isNull() )
314  {
315 
316  if ( itemElem.tagName() == "LayerItem" )
317  {
318  QString layerId = itemElem.attribute( "layerId" );
319  if ( QgsMapLayer* layer = QgsMapLayerRegistry::instance()->mapLayer( layerId ) )
320  {
321  QgsLayerTreeLayer* nodeLayer = parentGroup->addLayer( layer );
322  QString userText = itemElem.attribute( "userText" );
323  if ( !userText.isEmpty() )
324  nodeLayer->setCustomProperty( "legend/title-label", userText );
325  QString style = itemElem.attribute( "style" );
326  if ( !style.isEmpty() )
327  nodeLayer->setCustomProperty( "legend/title-style", style );
328  QString showFeatureCount = itemElem.attribute( "showFeatureCount" );
329  if ( showFeatureCount.toInt() )
330  nodeLayer->setCustomProperty( "showFeatureCount", 1 );
331 
332  // support for individual legend items (user text, order) not implemented yet
333  }
334  }
335  else if ( itemElem.tagName() == "GroupItem" )
336  {
337  QgsLayerTreeGroup* nodeGroup = parentGroup->addGroup( itemElem.attribute( "userText" ) );
338  QString style = itemElem.attribute( "style" );
339  if ( !style.isEmpty() )
340  nodeGroup->setCustomProperty( "legend/title-style", style );
341 
342  _readOldLegendGroup( itemElem, nodeGroup );
343  }
344 
345  itemElem = itemElem.nextSiblingElement();
346  }
347 }

349 bool QgsComposerLegend::readXML( const QDomElement& itemElem, const QDomDocument& doc )
350 {
351  if ( itemElem.isNull() )
352  {
353  return false;
354  }
355 
356  //read general properties
357  mSettings.setTitle( itemElem.attribute( "title" ) );
358  if ( !itemElem.attribute( "titleAlignment" ).isEmpty() )
359  {
360  mSettings.setTitleAlignment(( Qt::AlignmentFlag )itemElem.attribute( "titleAlignment" ).toInt() );
361  }
362  int colCount = itemElem.attribute( "columnCount", "1" ).toInt();
363  if ( colCount < 1 ) colCount = 1;
364  mSettings.setColumnCount( colCount );
365  mSettings.setSplitLayer( itemElem.attribute( "splitLayer", "0" ).toInt() == 1 );
366  mSettings.setEqualColumnWidth( itemElem.attribute( "equalColumnWidth", "0" ).toInt() == 1 );
367 
368  QDomNodeList stylesNodeList = itemElem.elementsByTagName( "styles" );
369  if ( stylesNodeList.size() > 0 )
370  {
371  QDomNode stylesNode = stylesNodeList.at( 0 );
372  for ( int i = 0; i < stylesNode.childNodes().size(); i++ )
373  {
374  QDomElement styleElem = stylesNode.childNodes().at( i ).toElement();
375  QgsComposerLegendStyle style;
376  style.readXML( styleElem, doc );
377  QString name = styleElem.attribute( "name" );
378  QgsComposerLegendStyle::Style s;
379  if ( name == "title" ) s = QgsComposerLegendStyle::Title;
380  else if ( name == "group" ) s = QgsComposerLegendStyle::Group;
381  else if ( name == "subgroup" ) s = QgsComposerLegendStyle::Subgroup;
382  else if ( name == "symbol" ) s = QgsComposerLegendStyle::Symbol;
383  else if ( name == "symbolLabel" ) s = QgsComposerLegendStyle::SymbolLabel;
384  else continue;
385  setStyle( s, style );
386  }
387  }
388 
389  //font color
390  QColor fontClr;
391  fontClr.setNamedColor( itemElem.attribute( "fontColor", "#000000" ) );
392  mSettings.setFontColor( fontClr );
393 
394  //spaces
395  mSettings.setBoxSpace( itemElem.attribute( "boxSpace", "2.0" ).toDouble() );
396  mSettings.setColumnSpace( itemElem.attribute( "columnSpace", "2.0" ).toDouble() );
397 
398  mSettings.setSymbolSize( QSizeF( itemElem.attribute( "symbolWidth", "7.0" ).toDouble(), itemElem.attribute( "symbolHeight", "14.0" ).toDouble() ) );
399 
400  mSettings.setWrapChar( itemElem.attribute( "wrapChar" ) );
401 
402  //composer map
403  mLegendFilterByMap = itemElem.attribute( "legendFilterByMap", "0" ).toInt();
404  if ( !itemElem.attribute( "map" ).isEmpty() )
405  {
406  setComposerMap( mComposition->getComposerMapById( itemElem.attribute( "map" ).toInt() ) );
407  }
408 
409  QDomElement oldLegendModelElem = itemElem.firstChildElement( "Model" );
410  if ( !oldLegendModelElem.isNull() )
411  {
412  // QGIS <= 2.4
413  QgsLayerTreeGroup* nodeRoot = new QgsLayerTreeGroup();
414  _readOldLegendGroup( oldLegendModelElem, nodeRoot );
415  setCustomLayerTree( nodeRoot );
416  }
417  else
418  {
419  // QGIS >= 2.6
420  QDomElement layerTreeElem = itemElem.firstChildElement( "layer-tree-group" );
421  setCustomLayerTree( QgsLayerTreeGroup::readXML( layerTreeElem ) );
422  }
423 
424  //restore general composer item properties
425  QDomNodeList composerItemList = itemElem.elementsByTagName( "ComposerItem" );
426  if ( composerItemList.size() > 0 )
427  {
428  QDomElement composerItemElem = composerItemList.at( 0 ).toElement();
429  _readXML( composerItemElem, doc );
430  }
431 
432  // < 2.0 projects backward compatibility >>>>>
433  //title font
434  QString titleFontString = itemElem.attribute( "titleFont" );
435  if ( !titleFontString.isEmpty() )
436  {
437  rstyle( QgsComposerLegendStyle::Title ).rfont().fromString( titleFontString );
438  }
439  //group font
440  QString groupFontString = itemElem.attribute( "groupFont" );
441  if ( !groupFontString.isEmpty() )
442  {
443  rstyle( QgsComposerLegendStyle::Group ).rfont().fromString( groupFontString );
444  }
445 
446  //layer font
447  QString layerFontString = itemElem.attribute( "layerFont" );
448  if ( !layerFontString.isEmpty() )
449  {
450  rstyle( QgsComposerLegendStyle::Subgroup ).rfont().fromString( layerFontString );
451  }
452  //item font
453  QString itemFontString = itemElem.attribute( "itemFont" );
454  if ( !itemFontString.isEmpty() )
455  {
456  rstyle( QgsComposerLegendStyle::SymbolLabel ).rfont().fromString( itemFontString );
457  }
458 
459  if ( !itemElem.attribute( "groupSpace" ).isEmpty() )
460  {
461  rstyle( QgsComposerLegendStyle::Group ).setMargin( QgsComposerLegendStyle::Top, itemElem.attribute( "groupSpace", "3.0" ).toDouble() );
462  }
463  if ( !itemElem.attribute( "layerSpace" ).isEmpty() )
464  {
465  rstyle( QgsComposerLegendStyle::Subgroup ).setMargin( QgsComposerLegendStyle::Top, itemElem.attribute( "layerSpace", "3.0" ).toDouble() );
466  }
467  if ( !itemElem.attribute( "symbolSpace" ).isEmpty() )
468  {
469  rstyle( QgsComposerLegendStyle::Symbol ).setMargin( QgsComposerLegendStyle::Top, itemElem.attribute( "symbolSpace", "2.0" ).toDouble() );
470  rstyle( QgsComposerLegendStyle::SymbolLabel ).setMargin( QgsComposerLegendStyle::Top, itemElem.attribute( "symbolSpace", "2.0" ).toDouble() );
471  }
472  // <<<<<<< < 2.0 projects backward compatibility
473 
474  emit itemChanged();
475  return true;
476 }

527 void QgsComposerLegend::updateFilterByMap()
528 {
529  if ( isRemoved() )
530  return;
531 
532  if ( mComposerMap && mLegendFilterByMap )
533  {
534  int dpi = mComposition->printResolution();
535 
536  QgsRectangle requestRectangle;
537  mComposerMap->requestedExtent( requestRectangle );
538 
539  QSizeF theSize( requestRectangle.width(), requestRectangle.height() );
540  theSize *= mComposerMap->mapUnitsToMM() * dpi / 25.4;
541 
542  QgsMapSettings ms = mComposerMap->mapSettings( requestRectangle, theSize, dpi );
543 
544  mLegendModel2->setLegendFilterByMap( &ms );
545  }
546  else
547  mLegendModel2->setLegendFilterByMap( 0 );
548

564 QVariant QgsLegendModelV2::data( const QModelIndex& index, int role ) const
565 {
566  // handle custom layer node labels
567  if ( QgsLayerTreeNode* node = index2node( index ) )
568  {
569  if ( QgsLayerTree::isLayer( node ) && ( role == Qt::DisplayRole || role == Qt::EditRole ) && !node->customProperty( "legend/title-label" ).isNull() )
570  {
571  QgsLayerTreeLayer* nodeLayer = QgsLayerTree::toLayer( node );
572  QString name = node->customProperty( "legend/title-label" ).toString();
573  if ( nodeLayer->customProperty( "showFeatureCount", 0 ).toInt() && role == Qt::DisplayRole )
574  {
575  QgsVectorLayer* vlayer = qobject_cast<QgsVectorLayer*>( nodeLayer->layer() );
576  if ( vlayer && vlayer->pendingFeatureCount() >= 0 )
577  name += QString( " [%1]" ).arg( vlayer->pendingFeatureCount() );
578  }
579  return name;
580  }
581  }
582 
583  return QgsLayerTreeModel::data( index, role );
