35 int QgsDxfPaintDevice::metric( PaintDeviceMetric metric ) const
36 {
37  switch ( metric )

41  case QPaintDevice::PdmHeight:
42  return mDrawingSize.height();

62 double QgsDxfPaintDevice::widthScaleFactor() const
63 {
64  if ( !mDrawingSize.isValid() || mRectangle.isEmpty() )
65  {
66  return 1.0;
67  }
68 
69  double widthFactor = mRectangle.width() / mDrawingSize.width();
70  double heightFactor = mRectangle.height() / mDrawingSize.height();
71  return ( widthFactor + heightFactor ) / 2.0;
72 }

74 QPointF QgsDxfPaintDevice::dxfCoordinates( const QPointF& pt ) const
75 {
76  if ( !mDrawingSize.isValid() || mRectangle.isEmpty() )
77  {
78  return QPointF( pt.x(), pt.y() );
79  }
80 
81  double x = mRectangle.left() + pt.x() * ( mRectangle.width() / mDrawingSize.width() );
82  double y = mRectangle.bottom() - pt.y() * ( mRectangle.height() / mDrawingSize.height() );
83  return QPointF( x, y );
84 }
