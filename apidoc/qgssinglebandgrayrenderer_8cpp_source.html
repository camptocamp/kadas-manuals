49 QgsRasterRenderer* QgsSingleBandGrayRenderer::create( const QDomElement& elem, QgsRasterInterface* input )
50 {
51  if ( elem.isNull() )
52  {
53  return 0;
54  }
55 
56  int grayBand = elem.attribute( "grayBand", "-1" ).toInt();
57  QgsSingleBandGrayRenderer* r = new QgsSingleBandGrayRenderer( input, grayBand );
58  r->readXML( elem );
59 
60  if ( elem.attribute( "gradient" ) == "WhiteToBlack" )
61  {
62  r->setGradient( WhiteToBlack ); // BlackToWhite is default
63  }
64 
65  QDomElement contrastEnhancementElem = elem.firstChildElement( "contrastEnhancement" );
66  if ( !contrastEnhancementElem.isNull() )
67  {
68  QgsContrastEnhancement* ce = new QgsContrastEnhancement(( QGis::DataType )(
69  input->dataType( grayBand ) ) );
70  ce->readXML( contrastEnhancementElem );
71  r->setContrastEnhancement( ce );
72  }
73  return r;

87 QgsRasterBlock* QgsSingleBandGrayRenderer::block2( int bandNo, QgsRectangle const & extent, int width, int height, QgsRasterBlockFeedback* feedback )
88 {
89  Q_UNUSED( bandNo );
90  QgsDebugMsg( QString( "width = %1 height = %2" ).arg( width ).arg( height ) );
91 
92  QgsRasterBlock *outputBlock = new QgsRasterBlock();
93  if ( !mInput )
94  {
95  return outputBlock;
96  }
97 
98  QgsRasterBlock *inputBlock = mInput->block2( mGrayBand, extent, width, height, feedback );
99  if ( !inputBlock || inputBlock->isEmpty() )
100  {
101  QgsDebugMsg( "No raster data!" );
102  delete inputBlock;
103  return outputBlock;
104  }
105 
106  QgsRasterBlock *alphaBlock = 0;
107 
108  if ( mAlphaBand > 0 && mGrayBand != mAlphaBand )
109  {
110  alphaBlock = mInput->block2( mAlphaBand, extent, width, height, feedback );
111  if ( !alphaBlock || alphaBlock->isEmpty() )
112  {
113  // TODO: better to render without alpha
114  delete inputBlock;
115  delete alphaBlock;
116  return outputBlock;
117  }
118  }
119  else if ( mAlphaBand > 0 )
120  {
121  alphaBlock = inputBlock;
122  }
123 
124  if ( !outputBlock->reset( QGis::ARGB32_Premultiplied, width, height ) )
125  {
126  delete inputBlock;
127  delete alphaBlock;
128  return outputBlock;
129  }
130 
131  QRgb myDefaultColor = NODATA_COLOR;
132  for ( qgssize i = 0; i < ( qgssize )width*height; i++ )
133  {
134  if ( inputBlock->isNoData( i ) )
135  {
136  outputBlock->setColor( i, myDefaultColor );
137  continue;
138  }
139  double grayVal = inputBlock->value( i );
140 
141  double currentAlpha = mOpacity;
142  if ( mRasterTransparency )
143  {
144  currentAlpha = mRasterTransparency->alphaValue( grayVal, mOpacity * 255 ) / 255.0;
145  }
146  if ( mAlphaBand > 0 )
147  {
148  currentAlpha *= alphaBlock->value( i ) / 255.0;
149  }
150 
151  if ( mContrastEnhancement )
152  {
153  if ( !mContrastEnhancement->isValueInDisplayableRange( grayVal ) )
154  {
155  outputBlock->setColor( i, myDefaultColor );
156  continue;
157  }
158  grayVal = mContrastEnhancement->enhanceContrast( grayVal );
159  }
160 
161  if ( mGradient == WhiteToBlack )
162  {
163  grayVal = 255 - grayVal;
164  }
165 
166  if ( qgsDoubleNear( currentAlpha, 1.0 ) )
167  {
168  outputBlock->setColor( i, qRgba( grayVal, grayVal, grayVal, 255 ) );
169  }
170  else
171  {
172  outputBlock->setColor( i, qRgba( currentAlpha * grayVal, currentAlpha * grayVal, currentAlpha * grayVal, currentAlpha * 255 ) );
173  }
174  }
175 
176  delete inputBlock;
177  if ( mAlphaBand > 0 && mGrayBand != mAlphaBand )
178  {
179  delete alphaBlock;
180  }
181 
182  return outputBlock;
183 }

185 void QgsSingleBandGrayRenderer::writeXML( QDomDocument& doc, QDomElement& parentElem ) const
186 {
187  if ( parentElem.isNull() )
188  {
189  return;
190  }
191 
192  QDomElement rasterRendererElem = doc.createElement( "rasterrenderer" );
193  _writeXML( doc, rasterRendererElem );
194 
195  rasterRendererElem.setAttribute( "grayBand", mGrayBand );
196 
197  QString gradient;
198  if ( mGradient == BlackToWhite )
199  {
200  gradient = "BlackToWhite";
201  }
202  else
203  {
204  gradient = "WhiteToBlack";
205  }
206  rasterRendererElem.setAttribute( "gradient", gradient );
207 
208  if ( mContrastEnhancement )
209  {
210  QDomElement contrastElem = doc.createElement( "contrastEnhancement" );
211  mContrastEnhancement->writeXML( doc, contrastElem );
212  rasterRendererElem.appendChild( contrastElem );
213  }
214  parentElem.appendChild( rasterRendererElem );
215 }

217 void QgsSingleBandGrayRenderer::legendSymbologyItems( QList< QPair< QString, QColor > >& symbolItems ) const
218 {
219  if ( mContrastEnhancement && mContrastEnhancement->contrastEnhancementAlgorithm() != QgsContrastEnhancement::NoEnhancement )
220  {
221  QColor minColor = ( mGradient == BlackToWhite ) ? Qt::black : Qt::white;
222  QColor maxColor = ( mGradient == BlackToWhite ) ? Qt::white : Qt::black;
223  symbolItems.push_back( qMakePair( QString::number( mContrastEnhancement->minimumValue() ), minColor ) );
224  symbolItems.push_back( qMakePair( QString::number( mContrastEnhancement->maximumValue() ), maxColor ) );
