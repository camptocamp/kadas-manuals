245 QRectF QgsComposition::compositionBounds() const
246 {
247  //start with an empty rectangle
248  QRectF bounds = QRectF( 0, 0, 0, 0 );
249 
250  //add all QgsComposerItems and QgsPaperItems which are in the composition
251  QList<QGraphicsItem *> itemList = items();
252  QList<QGraphicsItem *>::iterator itemIt = itemList.begin();
253  for ( ; itemIt != itemList.end(); ++itemIt )
254  {
255  const QgsComposerItem* composerItem = dynamic_cast<const QgsComposerItem *>( *itemIt );
256  const QgsPaperItem* paperItem = dynamic_cast<const QgsPaperItem*>( *itemIt );
257  if (( composerItem || paperItem ) )
258  {
259  //expand bounds with current item's bounds
260  bounds = bounds.united(( *itemIt )->sceneBoundingRect() );
261  }
262  }
263 
264  //finally, expand bounds out by 5% page size to give a bit of a margin
265  bounds.adjust( -mPageWidth * 0.05, -mPageWidth * 0.05, mPageWidth * 0.05, mPageWidth * 0.05 );
266

270 void QgsComposition::setPaperSize( const double width, const double height )
271 {
272  if ( width == mPageWidth && height == mPageHeight )
273  {
274  return;
275  }
276 
277  //update item positions
278  if ( mPages.size() > 1 )
279  {
280  QList<QGraphicsItem *> itemList = items();
281  QList<QGraphicsItem *>::iterator itemIt = itemList.begin();
282  for ( ; itemIt != itemList.end(); ++itemIt )
283  {
284  QgsComposerItem* composerItem = dynamic_cast<QgsComposerItem *>( *itemIt );
285  if ( composerItem )
286  {
287  composerItem->updatePagePos( width, height );
288  }
289  }
290  }
291  //update guide positions and size
292  QList< QGraphicsLineItem* >* guides = snapLines();
293  QList< QGraphicsLineItem* >::iterator guideIt = guides->begin();
294  double totalHeight = ( height + spaceBetweenPages() ) * ( numPages() - 1 ) + height;
295  for ( ; guideIt != guides->end(); ++guideIt )
296  {
297  QLineF line = ( *guideIt )->line();
298  if ( line.dx() == 0 )
299  {
300  //vertical line, change height of line
301  ( *guideIt )->setLine( line.x1(), 0, line.x1(), totalHeight );
302  }
303  else
304  {
305  //horizontal line
306  //move to new vertical position and change width of line
307  QPointF curPagePos = positionOnPage( line.p1() );
308  int curPage = pageNumberForPoint( line.p1() ) - 1;
309  double newY = curPage * ( height + spaceBetweenPages() ) + curPagePos.y();
310  ( *guideIt )->setLine( 0, newY, width, newY );
311  }
312  }
313 
314  mPageWidth = width;
315  mPageHeight = height;
316  double currentY = 0;
317  for ( int i = 0; i < mPages.size(); ++i )
318  {
319  mPages.at( i )->setSceneRect( QRectF( 0, currentY, width, height ) );
320  currentY += ( height + mSpaceBetweenPages );
321  }
322  QgsProject::instance()->dirty( true );
323  updateBounds();
324  emit paperSizeChanged();
325 }

327 void QgsComposition::setNumPages( const int pages )
328 {
329  int currentPages = numPages();
330  int desiredPages = pages;
331 
332  //data defined num pages set?
333  QVariant exprVal;
334  if ( dataDefinedEvaluate( QgsComposerObject::NumPages, exprVal, &mDataDefinedProperties ) )
335  {
336  bool ok = false;
337  int pagesD = exprVal.toInt( &ok );
338  QgsDebugMsg( QString( "exprVal NumPages:%1" ).arg( pagesD ) );
339  if ( ok )
340  {
341  desiredPages = pagesD;
342  }
343  }
344 
345  int diff = desiredPages - currentPages;
346  if ( diff >= 0 )
347  {
348  for ( int i = 0; i < diff; ++i )
349  {
350  addPaperItem();
351  }
352  }
353  else
354  {
355  diff = -diff;
356  for ( int i = 0; i < diff; ++i )
357  {
358  delete mPages.last();
359  mPages.removeLast();
360  }
361  }
362 
363  //update vertical guide height
364  QList< QGraphicsLineItem* >* guides = snapLines();
365  QList< QGraphicsLineItem* >::iterator guideIt = guides->begin();
366  double totalHeight = ( mPageHeight + spaceBetweenPages() ) * ( pages - 1 ) + mPageHeight;
367  for ( ; guideIt != guides->end(); ++guideIt )
368  {
369  QLineF line = ( *guideIt )->line();
370  if ( line.dx() == 0 )
371  {
372  //vertical line, change height of line
373  ( *guideIt )->setLine( line.x1(), 0, line.x1(), totalHeight );
374  }
375  }
376 
377  //update the corresponding variable
378  QgsExpression::setSpecialColumn( "$numpages", QVariant(( int )numPages() ) );
379 
380  QgsProject::instance()->dirty( true );
381  updateBounds();
382 
383  emit nPagesChanged();
384 }

387 {
388  //get all items on page
389  QList<QgsComposerItem*> items;
390  //composerItemsOnPage uses 0-based page numbering
391  composerItemsOnPage( items, page - 1 );
392 
393  //loop through and check for non-paper items
394  QList<QgsComposerItem*>::const_iterator itemIt = items.constBegin();
395  for ( ; itemIt != items.constEnd(); ++itemIt )
396  {
397  //is item a paper item?
398  QgsPaperItem* paper = dynamic_cast<QgsPaperItem*>( *itemIt );
399  if ( !paper )

402  return false;
403  }
404  }
405  //no non-paper items
406  return true;

451 QPointF QgsComposition::positionOnPage( const QPointF & position ) const
452 {
453  double y;
454  if ( position.y() > ( mPages.size() - 1 ) * ( paperHeight() + spaceBetweenPages() ) )
455  {
456  //y coordinate is greater then the end of the last page, so return distance between
457  //top of last page and y coordinate
458  y = position.y() - ( mPages.size() - 1 ) * ( paperHeight() + spaceBetweenPages() );
459  }
460  else
461  {
462  //y coordinate is less then the end of the last page
463  y = fmod( position.y(), ( paperHeight() + spaceBetweenPages() ) );
464  }
465  return QPointF( position.x(), y );
466 }

486 QgsComposerItem* QgsComposition::composerItemAt( const QPointF & position, const QgsComposerItem* belowItem, const bool ignoreLocked ) const
487 {
488  //get a list of items which intersect the specified position, in descending z order
489  QList<QGraphicsItem*> itemList;
490  itemList = items( position, Qt::IntersectsItemShape, Qt::DescendingOrder );
491  QList<QGraphicsItem *>::iterator itemIt = itemList.begin();
492 
493  bool foundBelowItem = false;
494  for ( ; itemIt != itemList.end(); ++itemIt )
495  {
496  QgsComposerItem* composerItem = dynamic_cast<QgsComposerItem *>( *itemIt );
497  QgsPaperItem* paperItem = dynamic_cast<QgsPaperItem*>( *itemIt );
498  if ( composerItem && !paperItem )
499  {
500  // If we are not checking for a an item below a specified item, or if we've
501  // already found that item, then we've found our target
502  if (( ! belowItem || foundBelowItem ) && ( !ignoreLocked || !composerItem->positionLock() ) )
503  {
504  return composerItem;
505  }
506  else
507  {
508  if ( composerItem == belowItem )
509  {
510  //Target item is next in list
511  foundBelowItem = true;
512  }
513  }
514  }
515  }
516  return 0;
517 }
