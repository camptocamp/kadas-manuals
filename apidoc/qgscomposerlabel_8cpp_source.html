89 void QgsComposerLabel::paint( QPainter* painter, const QStyleOptionGraphicsItem* itemStyle, QWidget* pWidget )
90 {
91  Q_UNUSED( itemStyle );
92  Q_UNUSED( pWidget );
93  if ( !painter )
94  {
95  return;
96  }
97  if ( !shouldDrawItem() )
98  {
99  return;
100  }
101 
102  drawBackground( painter );
103  painter->save();
104 
105  //antialiasing on
106  painter->setRenderHint( QPainter::Antialiasing, true );
107 
108  double penWidth = hasFrame() ? ( pen().widthF() / 2.0 ) : 0;
109  double xPenAdjust = mMarginX < 0 ? -penWidth : penWidth;
110  double yPenAdjust = mMarginY < 0 ? -penWidth : penWidth;
111  QRectF painterRect( xPenAdjust + mMarginX, yPenAdjust + mMarginY, rect().width() - 2 * xPenAdjust - 2 * mMarginX, rect().height() - 2 * yPenAdjust - 2 * mMarginY );
112 
113  QString textToDraw = displayText();
114 
115  if ( mHtmlState )
116  {
117  painter->scale( 1.0 / mHtmlUnitsToMM / 10.0, 1.0 / mHtmlUnitsToMM / 10.0 );
118  QWebPage *webPage = new QWebPage();
119  webPage->setNetworkAccessManager( QgsNetworkAccessManager::instance() );
120 
121  //Setup event loop and timeout for rendering html
122  QEventLoop loop;
123  QTimer timeoutTimer;
124  timeoutTimer.setSingleShot( true );
125 
126  //This makes the background transparent. Found on http://blog.qt.digia.com/blog/2009/06/30/transparent-qwebview-or-qwebpage/
127  QPalette palette = webPage->palette();
128  palette.setBrush( QPalette::Base, Qt::transparent );
129  webPage->setPalette( palette );
130  //webPage->setAttribute(Qt::WA_OpaquePaintEvent, false); //this does not compile, why ?
131 
132  webPage->setViewportSize( QSize( painterRect.width() * mHtmlUnitsToMM * 10.0, painterRect.height() * mHtmlUnitsToMM * 10.0 ) );
133  webPage->mainFrame()->setZoomFactor( 10.0 );
134  webPage->mainFrame()->setScrollBarPolicy( Qt::Horizontal, Qt::ScrollBarAlwaysOff );
135  webPage->mainFrame()->setScrollBarPolicy( Qt::Vertical, Qt::ScrollBarAlwaysOff );
136 
137  // QGIS segfaults when rendering web page while in composer if html
138  // contains images. So if we are not printing the composition, then
139  // disable image loading
140  if ( mComposition->plotStyle() != QgsComposition::Print &&
141  mComposition->plotStyle() != QgsComposition::Postscript )
142  {
143  webPage->settings()->setAttribute( QWebSettings::AutoLoadImages, false );
144  }
145 
146  //Connect timeout and webpage loadFinished signals to loop
147  connect( &timeoutTimer, SIGNAL( timeout() ), &loop, SLOT( quit() ) );
148  connect( webPage, SIGNAL( loadFinished( bool ) ), &loop, SLOT( quit() ) );
149 
150  //mHtmlLoaded tracks whether the QWebPage has completed loading
151  //its html contents, set it initially to false. The loadingHtmlFinished slot will
152  //set this to true after html is loaded.
153  mHtmlLoaded = false;
154  connect( webPage, SIGNAL( loadFinished( bool ) ), SLOT( loadingHtmlFinished( bool ) ) );
155 
156  webPage->mainFrame()->setHtml( textToDraw );
157 
158  //For very basic html labels with no external assets, the html load will already be
159  //complete before we even get a chance to start the QEventLoop. Make sure we check
160  //this before starting the loop
161  if ( !mHtmlLoaded )
162  {
163  // Start a 20 second timeout in case html loading will never complete
164  timeoutTimer.start( 20000 );
165  // Pause until html is loaded
166  loop.exec();
167  }
168  webPage->mainFrame()->render( painter );//DELETE WEBPAGE ?
169  }
170  else
171  {
172  painter->setFont( mFont );
173  //debug
174  //painter->setPen( QColor( Qt::red ) );
175  //painter->drawRect( painterRect );
176  QgsComposerUtils::drawText( painter, painterRect, textToDraw, mFont, mFontColor, mHAlignment, mVAlignment, Qt::TextWordWrap );
177  }
178 
179  painter->restore();
180 
181  drawFrame( painter );
182  if ( isSelected() )
183  {
184  drawSelectionBoxes( painter );
185  }
186 }

290 void QgsComposerLabel::replaceDateText( QString& text ) const
291 {
292  QString constant = "$CURRENT_DATE";
293  int currentDatePos = text.indexOf( constant );
294  if ( currentDatePos != -1 )
295  {
296  //check if there is a bracket just after $CURRENT_DATE
297  QString formatText;
298  int openingBracketPos = text.indexOf( "(", currentDatePos );
299  int closingBracketPos = text.indexOf( ")", openingBracketPos + 1 );
300  if ( openingBracketPos != -1 &&
301  closingBracketPos != -1 &&
302  ( closingBracketPos - openingBracketPos ) > 1 &&
303  openingBracketPos == currentDatePos + constant.size() )
304  {
305  formatText = text.mid( openingBracketPos + 1, closingBracketPos - openingBracketPos - 1 );
306  text.replace( currentDatePos, closingBracketPos - currentDatePos + 1, QDate::currentDate().toString( formatText ) );
307  }
308  else //no bracket
309  {
310  text.replace( "$CURRENT_DATE", QDate::currentDate().toString() );
311  }
312  }
313 }

339 void QgsComposerLabel::adjustSizeToText()
340 {
341  QString text = displayText();
342  double maxTextWidth = 0;
343 
344  QStringList multiLineSplit = text.split( "\n" );
345  QStringList::const_iterator mlSplitIt = multiLineSplit.constBegin();
346  for ( ; mlSplitIt != multiLineSplit.constEnd(); ++mlSplitIt )
347  {
348  double currentTextWidth = QgsComposerUtils::textWidthMM( mFont, *mlSplitIt );
349  if ( currentTextWidth > maxTextWidth )
350  {
351  maxTextWidth = currentTextWidth;
352  }
353  }
354  double fontHeight = QgsComposerUtils::fontHeightMM( mFont );
355 
356  double penWidth = hasFrame() ? ( pen().widthF() / 2.0 ) : 0;
357 
358  double width = maxTextWidth + 2 * mMarginX + 2 * penWidth + 1;
359  double height = multiLineSplit.size() * ( fontHeight + 2 * mMarginY + 2 * penWidth );
360 
361  //keep alignment point constant
362  double xShift = 0;
363  double yShift = 0;
364  itemShiftAdjustSize( width, height, xShift, yShift );
365 
366  //update rect for data defined size and position
367  QRectF evaluatedRect = evalItemRect( QRectF( pos().x() + xShift, pos().y() + yShift, width, height ) );
368  setSceneRect( evaluatedRect );
369 }

412 bool QgsComposerLabel::readXML( const QDomElement& itemElem, const QDomDocument& doc )
413 {
414  QString alignment;
415 
416  if ( itemElem.isNull() )
417  {
418  return false;
419  }
420 
421  //restore label specific properties
422 
423  //text
424  mText = itemElem.attribute( "labelText" );
425 
426  //html state
427  mHtmlState = itemElem.attribute( "htmlState" ).toInt();
428 
429  //margin
430  bool marginXOk = false;
431  bool marginYOk = false;
432  mMarginX = itemElem.attribute( "marginX" ).toDouble( &marginXOk );
433  mMarginY = itemElem.attribute( "marginY" ).toDouble( &marginYOk );
434  if ( !marginXOk || !marginYOk )
435  {
436  //upgrade old projects where margins where stored in a single attribute
437  double margin = itemElem.attribute( "margin", "1.0" ).toDouble();
438  mMarginX = margin;
439  mMarginY = margin;
440  }
441 
442  //Horizontal alignment
443  mHAlignment = ( Qt::AlignmentFlag )( itemElem.attribute( "halign" ).toInt() );
444 
445  //Vertical alignment
446  mVAlignment = ( Qt::AlignmentFlag )( itemElem.attribute( "valign" ).toInt() );
447 
448  mMaxLength = itemElem.attribute( "maxLength", "-1" ).toInt();
449 
450  //font
451  QDomNodeList labelFontList = itemElem.elementsByTagName( "LabelFont" );
452  if ( labelFontList.size() > 0 )
453  {
454  QDomElement labelFontElem = labelFontList.at( 0 ).toElement();
455  mFont.fromString( labelFontElem.attribute( "description" ) );
456  }
457 
458  //font color
459  QDomNodeList fontColorList = itemElem.elementsByTagName( "FontColor" );
460  if ( fontColorList.size() > 0 )
461  {
462  QDomElement fontColorElem = fontColorList.at( 0 ).toElement();
463  int red = fontColorElem.attribute( "red", "0" ).toInt();
464  int green = fontColorElem.attribute( "green", "0" ).toInt();
465  int blue = fontColorElem.attribute( "blue", "0" ).toInt();
466  mFontColor = QColor( red, green, blue );
467  }
468  else
469  {
470  mFontColor = QColor( 0, 0, 0 );
471  }
472 
473  //restore general composer item properties
474  QDomNodeList composerItemList = itemElem.elementsByTagName( "ComposerItem" );
475  if ( composerItemList.size() > 0 )
476  {
477  QDomElement composerItemElem = composerItemList.at( 0 ).toElement();
478 
479  //rotation
480  if ( composerItemElem.attribute( "rotation", "0" ).toDouble() != 0 )
481  {
482  //check for old (pre 2.1) rotation attribute
483  setItemRotation( composerItemElem.attribute( "rotation", "0" ).toDouble() );
484  }
485 
486  _readXML( composerItemElem, doc );
487  }
488  emit itemChanged();
489  return true;
490 }

550 void QgsComposerLabel::itemShiftAdjustSize( double newWidth, double newHeight, double& xShift, double& yShift ) const
551 {
552  //keep alignment point constant
553  double currentWidth = rect().width();
554  double currentHeight = rect().height();
555  xShift = 0;
556  yShift = 0;
557 
558  if ( mItemRotation >= 0 && mItemRotation < 90 )
559  {
560  if ( mHAlignment == Qt::AlignHCenter )
561  {
562  xShift = - ( newWidth - currentWidth ) / 2.0;
563  }
564  else if ( mHAlignment == Qt::AlignRight )
565  {
566  xShift = - ( newWidth - currentWidth );
567  }
568  if ( mVAlignment == Qt::AlignVCenter )
569  {
570  yShift = -( newHeight - currentHeight ) / 2.0;
571  }
572  else if ( mVAlignment == Qt::AlignBottom )
573  {
574  yShift = - ( newHeight - currentHeight );
575  }
576  }
577  if ( mItemRotation >= 90 && mItemRotation < 180 )
578  {
579  if ( mHAlignment == Qt::AlignHCenter )
580  {
581  yShift = -( newHeight - currentHeight ) / 2.0;
582  }
583  else if ( mHAlignment == Qt::AlignRight )
584  {
585  yShift = -( newHeight - currentHeight );
586  }
587  if ( mVAlignment == Qt::AlignTop )
588  {
589  xShift = -( newWidth - currentWidth );
590  }
591  else if ( mVAlignment == Qt::AlignVCenter )
592  {
593  xShift = -( newWidth - currentWidth / 2.0 );
594  }
595  }
596  else if ( mItemRotation >= 180 && mItemRotation < 270 )
597  {
598  if ( mHAlignment == Qt::AlignHCenter )
599  {
600  xShift = -( newWidth - currentWidth ) / 2.0;
601  }
602  else if ( mHAlignment == Qt::AlignLeft )
603  {
604  xShift = -( newWidth - currentWidth );
605  }
606  if ( mVAlignment == Qt::AlignVCenter )
607  {
608  yShift = ( newHeight - currentHeight ) / 2.0;
609  }
610  else if ( mVAlignment == Qt::AlignTop )
611  {
612  yShift = ( newHeight - currentHeight );
613  }
614  }
615  else if ( mItemRotation >= 270 && mItemRotation < 360 )
616  {
617  if ( mHAlignment == Qt::AlignHCenter )
618  {
619  yShift = -( newHeight - currentHeight ) / 2.0;
620  }
621  else if ( mHAlignment == Qt::AlignLeft )
622  {
623  yShift = -( newHeight - currentHeight );
624  }
625  if ( mVAlignment == Qt::AlignBottom )
626  {
627  xShift = -( newWidth - currentWidth );
628  }
629  else if ( mVAlignment == Qt::AlignVCenter )
630  {
631  xShift = -( newWidth - currentWidth / 2.0 );
632  }
633  }
