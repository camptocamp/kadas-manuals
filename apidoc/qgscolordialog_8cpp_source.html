426  QString filePath = QFileDialog::getOpenFileName( this, tr( "Select palette file" ), lastDir, "GPL (*.gpl);;All files (*.*)" );
427  activateWindow();
428  if ( filePath.isEmpty() )
429  {
430  return;
431  }
432 
433  //check if file exists
434  QFileInfo fileInfo( filePath );
435  if ( !fileInfo.exists() || !fileInfo.isReadable() )
436  {
437  QMessageBox::critical( 0, tr( "Invalid file" ), tr( "Error, file does not exist or is not readable" ) );
438  return;
439  }
440 
441  s.setValue( "/UI/lastGplPaletteDir", fileInfo.absolutePath() );
442  QFile file( filePath );

452  }

465 
466  QgsColorSchemeRegistry::instance()->addColorScheme( importedScheme );
467

473 void QgsColorDialogV2::removePalette()
474 {
475  //get current scheme
476  QList<QgsColorScheme *> schemeList = QgsColorSchemeRegistry::instance()->schemes( QgsColorScheme::ShowInColorDialog );
477  int prevIndex = mSchemeComboBox->currentIndex();
478  if ( prevIndex >= schemeList.length() )
479  {
480  return;
481  }
482 
483  //make user scheme is a user removable scheme
484  QgsUserColorScheme* userScheme = dynamic_cast<QgsUserColorScheme*>( schemeList.at( prevIndex ) );
485  if ( !userScheme )
486  {
487  return;
488  }
489 
490  if ( QMessageBox::question( this, tr( "Remove Color Palette" ),
491  QString( tr( "Are you sure you want to remove %1?" ) ).arg( userScheme->schemeName() ),
492  QMessageBox::Yes | QMessageBox::No, QMessageBox::No ) != QMessageBox::Yes )
493  {
494  //user cancelled
495  return;
496  }
497 
498  //remove palette and associated gpl file
499  if ( !userScheme->erase() )
500  {
501  //something went wrong
502  return;
503  }
504 
505  //remove scheme from registry
506  QgsColorSchemeRegistry::instance()->removeColorScheme( userScheme );
507  refreshSchemeComboBox();
508  prevIndex = qMax( qMin( prevIndex, mSchemeComboBox->count() - 1 ), 0 );
509  mSchemeComboBox->setCurrentIndex( prevIndex );
510 }

512 void QgsColorDialogV2::newPalette()
513 {
514  bool ok = false;
515  QString name = QInputDialog::getText( this, tr( "Create New Palette" ), tr( "Enter a name for the new palette:" ),
516  QLineEdit::Normal, tr( "New palette" ), &ok );
517 
518  if ( !ok || name.isEmpty() )
519  {
520  //user cancelled
521  return;
522  }
523 
524  //generate file name for new palette
525  QDir palettePath( gplFilePath() );
526  QRegExp badChars( "[,^@={}\\[\\]~!?:&*\"|#%<>$\"'();`' /\\\\]" );
527  QString filename = name.simplified().toLower().replace( badChars, QString( "_" ) );
528  if ( filename.isEmpty() )
529  {
530  filename = tr( "new_palette" );
531  }
532  QFileInfo destFileInfo( palettePath.filePath( filename + ".gpl" ) );
533  int fileNumber = 1;
534  while ( destFileInfo.exists() )
535  {
536  //try to generate a unique file name
537  destFileInfo = QFileInfo( palettePath.filePath( filename + QString( "%1.gpl" ).arg( fileNumber ) ) );
538  fileNumber++;
539  }
540 
541  QgsUserColorScheme* newScheme = new QgsUserColorScheme( destFileInfo.fileName() );
542  newScheme->setName( name );
543 
544  QgsColorSchemeRegistry::instance()->addColorScheme( newScheme );
545 
546  //refresh combobox and set new scheme as active
547  refreshSchemeComboBox();
548  mSchemeComboBox->setCurrentIndex( mSchemeComboBox->count() - 1 );
549 }

831 QColor QgsColorDialogV2::averageColor( const QImage &image ) const
832 {
833  QRgb tmpRgb;
834  int colorCount = 0;
835  int sumRed = 0;
836  int sumBlue = 0;
837  int sumGreen = 0;
838  //scan through image and sum rgb components
839  for ( int heightIndex = 0; heightIndex < image.height(); ++heightIndex )
840  {
841  QRgb* scanLine = ( QRgb* )image.constScanLine( heightIndex );
842  for ( int widthIndex = 0; widthIndex < image.width(); ++widthIndex )
843  {
844  tmpRgb = scanLine[widthIndex];
845  sumRed += qRed( tmpRgb );
846  sumBlue += qBlue( tmpRgb );
847  sumGreen += qGreen( tmpRgb );
848  colorCount++;
849  }
850  }
851  //calculate average components as floats
852  double avgRed = ( double )sumRed / ( 255.0 * colorCount );
853  double avgGreen = ( double )sumGreen / ( 255.0 * colorCount );
854  double avgBlue = ( double )sumBlue / ( 255.0 * colorCount );
855 
856  //create a new color representing the average
857  return QColor::fromRgbF( avgRed, avgGreen, avgBlue );
858 }
