37 void QgsRasterDrawer::draw( QPainter* p, QgsRasterViewPort* viewPort, const QgsMapToPixel* theQgsMapToPixel, const QgsRenderContext* ctx, QgsRasterBlockFeedback* feedback )
38 {
39  QgsDebugMsg( "Entered" );
40  if ( !p || !mIterator || !viewPort || !theQgsMapToPixel )
41  {
42  return;
43  }
44 
45  int width = viewPort->mWidth;
46  int height = viewPort->mHeight;
47  QgsMapToPixel mapToPixel( *theQgsMapToPixel );
48  double scaleFactor = 1.0;
49 
50  if ( ctx && ctx->customRenderFlags().split( ";" ).contains( "composer" ) && isWMTSLayer( mIterator->input() ) ) //wmts case, comply with standard pixel size of 0.28mm from specification
51  {
52  QImage image( 1, 1, QImage::Format_ARGB32 );
53  scaleFactor = double( image.logicalDpiX() ) / p->device()->logicalDpiX();
54  QgsDebugMsg( QString( "Scale factor: %1 (device DPI: %2)" ).arg( scaleFactor ).arg( p->device()->logicalDpiX() ) );
55  width = width * scaleFactor;
56  height = height * scaleFactor;
57  mapToPixel.setParameters( ctx->extent().width() / width, ctx->extent().xMinimum(), ctx->extent().yMinimum(), height );
58  }
59 
60  // last pipe filter has only 1 band
61  int bandNumber = 1;
62  mIterator->startRasterRead( bandNumber, width, height, viewPort->mDrawnExtent, feedback );
63 
64  //number of cols/rows in output pixels
65  int nCols = 0;
66  int nRows = 0;
67  //shift to top left point for the raster part
68  int topLeftCol = 0;
69  int topLeftRow = 0;
70 
71  // We know that the output data type of last pipe filter is QImage data
72 
73  QgsRasterBlock *block;
74 
75  // readNextRasterPart calcs and resets nCols, nRows, topLeftCol, topLeftRow
76  while ( mIterator->readNextRasterPart( bandNumber, nCols, nRows,
77  &block, topLeftCol, topLeftRow ) )
78  {
79  if ( !block )
80  {
81  QgsDebugMsg( "Cannot get block" );
82  continue;
83  }
84 
85  QImage img = block->image();
86 
87  // Because of bug in Acrobat Reader we must use "white" transparent color instead
88  // of "black" for PDF. See #9101.
89  QPrinter *printer = dynamic_cast<QPrinter *>( p->device() );
90  if ( printer && printer->outputFormat() == QPrinter::PdfFormat )
91  {
92  QgsDebugMsg( "PdfFormat" );
93 
94  img = img.convertToFormat( QImage::Format_ARGB32 );
95  QRgb transparentBlack = qRgba( 0, 0, 0, 0 );
96  QRgb transparentWhite = qRgba( 255, 255, 255, 0 );
97  for ( int x = 0; x < img.width(); x++ )
98  {
99  for ( int y = 0; y < img.height(); y++ )
100  {
101  if ( img.pixel( x, y ) == transparentBlack )
102  {
103  img.setPixel( x, y, transparentWhite );
104  }
105  }
106  }
107  }
108 
109  if ( feedback && feedback->renderPartialOutput() )
110  {
111  // there could have been partial preview written before
112  // so overwrite anything with the resulting image.
113  // (we are guaranteed to have a temporary image for this layer, see QgsMapRendererJob::needTemporaryImage)
114  p->setCompositionMode( QPainter::CompositionMode_Source );
115  }
116 
117  drawImage( p, viewPort, img, topLeftCol, topLeftRow, scaleFactor, &mapToPixel );
118  delete block;
119 
120  if ( feedback && feedback->renderPartialOutput() )
121  {
122  p->setCompositionMode( QPainter::CompositionMode_SourceOver ); //go back to the default composition mode
123  }
124 
125  if ( ctx && ctx->renderingStopped() ) { break; }
126 
127  // ok this does not matter much anyway as the tile size quite big so most of the time
128  // there would be just one tile for the whole display area, but it won't hurt...
129  if ( feedback && feedback->isCanceled() )
130  {
131  break;
132  }
133  }
134 }

136 void QgsRasterDrawer::drawImage( QPainter* p, QgsRasterViewPort* viewPort, const QImage& img, int topLeftCol, int topLeftRow, double scaleFactor, const QgsMapToPixel* theQgsMapToPixel ) const
137 {
138  if ( !p || !viewPort )
139  {
140  return;
141  }
142 
143  //top left position in device coords
144  QPoint tlPoint = QPoint( viewPort->mTopLeftPoint.x() + topLeftCol, viewPort->mTopLeftPoint.y() + topLeftRow );
145  p->save();
146  p->setRenderHint( QPainter::Antialiasing, false );
147 
148  // Blending problem was reported with PDF output if background color has alpha < 255
149  // in #7766, it seems to be a bug in Qt, setting a brush with alpha 255 is a workaround
150  // which should not harm anything
151  p->setBrush( QBrush( QColor( Qt::white ), Qt::NoBrush ) );
152 
153  if ( theQgsMapToPixel )
154  {
155  int w = theQgsMapToPixel->mapWidth();
156  int h = theQgsMapToPixel->mapHeight();
157 
158  double rotation = theQgsMapToPixel->mapRotation();
159  if ( rotation )
160  {
161  // both viewPort and image sizes are dependent on scale
162  double cx = w / 2.0;
163  double cy = h / 2.0;
164  p->translate( cx, cy );
165  p->rotate( rotation );
166  p->translate( -cx, -cy );
167  }
168  }
169 
170  if ( qgsDoubleNear( scaleFactor, 1.0 ) )
171  {
172  p->drawImage( tlPoint, img );
173  }
174  else
175  {
176  QRect source( 0, 0, img.width(), img.height() );
177  QRect target( tlPoint.x(), tlPoint.y(), img.width() / scaleFactor, img.height() / scaleFactor );
178  p->drawImage( target, img, source );
179  }
180 
181 #if 0
182  // For debugging:
183  QRectF br = QRectF( tlPoint, img.size() );
184  QPointF c = br.center();
185  double rad = std::max( br.width(), br.height() ) / 10;
186  p->drawRoundedRect( br, rad, rad );
187  p->drawLine( QLineF( br.x(), br.y(), br.x() + br.width(), br.y() + br.height() ) );
188  p->drawLine( QLineF( br.x() + br.width(), br.y(), br.x(), br.y() + br.height() ) );
189 
190  double nw = br.width() * 0.5; double nh = br.height() * 0.5;
191  br = QRectF( c - QPointF( nw / 2, nh / 2 ), QSize( nw, nh ) );
192  p->drawRoundedRect( br, rad, rad );
193 
194  nw = br.width() * 0.5; nh = br.height() * 0.5;
195  br = QRectF( c - QPointF( nw / 2, nh / 2 ), QSize( nw, nh ) );
196  p->drawRoundedRect( br, rad, rad );
197 #endif
198 
199  p->restore();
200 }

202 bool QgsRasterDrawer::isWMTSLayer( const QgsRasterInterface* iface )
203 {
204  if ( iface && iface->srcInput() )
205  {
206  const QgsRasterDataProvider* provider = dynamic_cast<const QgsRasterDataProvider*>( iface->srcInput() );
207  QVariant resolutions = provider->property( "resolutions" );
208  if ( provider->name() == "wms" && !resolutions.isNull() )
209  {
210  return true;
211  }
212  }
213  return false;
214 }
