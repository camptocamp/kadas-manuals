91 void QgsColorButtonV2::showColorDialog()
92 {
93  QColor newColor;
94  QSettings settings;
95 
96  //using native color dialogs?
97  bool useNative = settings.value( "/Qgis/native_color_dialogs", false ).toBool();
98 
99  if ( useNative )
100  {
101  // use native o/s dialogs
102  if ( mAcceptLiveUpdates && settings.value( "/Qgis/live_color_dialogs", false ).toBool() )
103  {
104  newColor = QgsColorDialog::getLiveColor(
105  color(), this, SLOT( setValidColor( const QColor& ) ),
106  this->parentWidget(), mColorDialogTitle, mAllowAlpha ? QColorDialog::ShowAlphaChannel : ( QColorDialog::ColorDialogOption )0 );
107  }
108  else
109  {
110  newColor = QColorDialog::getColor( color(), this->parentWidget(), mColorDialogTitle, mAllowAlpha ? QColorDialog::ShowAlphaChannel : ( QColorDialog::ColorDialogOption )0 );
111  }
112  }
113  else
114  {
115  //use QGIS style color dialogs
116  if ( mAcceptLiveUpdates && settings.value( "/Qgis/live_color_dialogs", false ).toBool() )
117  {
118  newColor = QgsColorDialogV2::getLiveColor(
119  color(), this, SLOT( setValidColor( const QColor& ) ),
120  this->parentWidget(), mColorDialogTitle, mAllowAlpha );
121  }
122  else
123  {
124  QgsColorDialogV2 dialog( this, 0, color() );
125  dialog.setTitle( mColorDialogTitle );
126  dialog.setAllowAlpha( mAllowAlpha );
127 
128  if ( dialog.exec() )
129  {
130  newColor = dialog.color();
131  }
132  }
133  }
134 
135  if ( newColor.isValid() )

184 bool QgsColorButtonV2::colorFromMimeData( const QMimeData * mimeData, QColor& resultColor )
185 {
186  bool hasAlpha = false;
187  QColor mimeColor = QgsSymbolLayerV2Utils::colorFromMimeData( mimeData, hasAlpha );
188 
189  if ( mimeColor.isValid() )
190  {
191  if ( !mAllowAlpha )
192  {
193  //remove alpha channel
194  mimeColor.setAlpha( 255 );
195  }
196  else if ( !hasAlpha )
197  {
198  //mime color has no explicit alpha component, so keep existing alpha
199  mimeColor.setAlpha( mColor.alpha() );
200  }
201  resultColor = mimeColor;
202  return true;
203  }
204 
205  //could not get color from mime data
206  return false;
207 }

209 void QgsColorButtonV2::mouseMoveEvent( QMouseEvent *e )
210 {
211  if ( mPickingColor )
212  {
213  //currently in color picker mode
214  if ( e->buttons() & Qt::LeftButton )
215  {
216  //if left button depressed, sample color under cursor and temporarily update button color
217  //to give feedback to user
218  QPixmap snappedPixmap = QPixmap::grabWindow( QApplication::desktop()->winId(), e->globalPos().x(), e->globalPos().y(), 1, 1 );
219  QImage snappedImage = snappedPixmap.toImage();
220  QColor hoverColor = snappedImage.pixel( 0, 0 );
221  setButtonBackground( hoverColor );
222  }
223  e->accept();
224  return;
225  }
226 
227  //handle dragging colors from button
228 
229  if ( !( e->buttons() & Qt::LeftButton ) || !mColor.isValid() )
230  {
231  //left button not depressed or no color set, so not a drag
232  QToolButton::mouseMoveEvent( e );
233  return;
234  }
235 
236  if (( e->pos() - mDragStartPosition ).manhattanLength() < QApplication::startDragDistance() )
237  {
238  //mouse not moved, so not a drag
239  QToolButton::mouseMoveEvent( e );
240  return;
241  }
242 
243  //user is dragging color
244  QDrag *drag = new QDrag( this );
245  drag->setMimeData( QgsSymbolLayerV2Utils::colorToMimeData( mColor ) );
246  drag->setPixmap( QgsColorWidget::createDragIcon( mColor ) );
247  drag->exec( Qt::CopyAction );
248  setDown( false );
249 }

383 void QgsColorButtonV2::prepareMenu()
384 {
385  //we need to tear down and rebuild this menu every time it is shown. Otherwise the space allocated to any
386  //QgsColorSwatchGridAction is not recalculated by Qt and the swatch grid may not be the correct size
387  //for the number of colors shown in the grid. Note that we MUST refresh color swatch grids every time this
388  //menu is opened, otherwise color schemes like the recent color scheme grid are meaningless
389  mMenu->clear();
390 
391  //show default color option if set
392  if ( mDefaultColor.isValid() )
393  {
394  QAction* defaultColorAction = new QAction( tr( "Default color" ), this );
395  defaultColorAction->setIcon( createMenuIcon( mDefaultColor ) );
396  mMenu->addAction( defaultColorAction );
397  connect( defaultColorAction, SIGNAL( triggered() ), this, SLOT( setToDefaultColor() ) );
398  }
399 
400  if ( mShowNoColorOption && mAllowAlpha )
401  {
402  QAction* noColorAction = new QAction( mNoColorString, this );
403  noColorAction->setIcon( createMenuIcon( Qt::transparent, false ) );
404  mMenu->addAction( noColorAction );
405  connect( noColorAction, SIGNAL( triggered() ), this, SLOT( setToNoColor() ) );
406  }
407 
408  if ( mColorSchemeRegistry )
409  {
410  //get schemes with ShowInColorButtonMenu flag set
411  QList< QgsColorScheme* > schemeList = mColorSchemeRegistry->schemes( QgsColorScheme::ShowInColorButtonMenu );
412  QList< QgsColorScheme* >::iterator it = schemeList.begin();
413  for ( ; it != schemeList.end(); ++it )
414  {
415  QgsColorSwatchGridAction* colorAction = new QgsColorSwatchGridAction( *it, mMenu, mContext, this );
416  colorAction->setBaseColor( mColor );
417  mMenu->addAction( colorAction );
418  connect( colorAction, SIGNAL( colorChanged( const QColor& ) ), this, SLOT( setValidColor( const QColor& ) ) );
419  connect( colorAction, SIGNAL( colorChanged( const QColor& ) ), this, SLOT( addRecentColor( const QColor& ) ) );
420  }
421  }
422 
423  mMenu->addSeparator();
424 
425  QAction* copyColorAction = new QAction( tr( "Copy color" ), this );
426  mMenu->addAction( copyColorAction );
427  connect( copyColorAction, SIGNAL( triggered() ), this, SLOT( copyColor() ) );
428 
429  QAction* pasteColorAction = new QAction( tr( "Paste color" ), this );
430  //enable or disable paste action based on current clipboard contents. We always show the paste
431  //action, even if it's disabled, to give hint to the user that pasting colors is possible
432  QColor clipColor;
433  if ( colorFromMimeData( QApplication::clipboard()->mimeData(), clipColor ) )
434  {
435  pasteColorAction->setIcon( createMenuIcon( clipColor ) );
436  }
437  else
438  {
439  pasteColorAction->setEnabled( false );
440  }
441  mMenu->addAction( pasteColorAction );
442  connect( pasteColorAction, SIGNAL( triggered() ), this, SLOT( pasteColor() ) );
443 
444 #ifndef Q_OS_MAC
445  //disabled for OSX, as it is impossible to grab the mouse under OSX
446  //see note for QWidget::grabMouse() re OSX Cocoa
447  //http://qt-project.org/doc/qt-4.8/qwidget.html#grabMouse
448  QAction* pickColorAction = new QAction( tr( "Pick color" ), this );
449  mMenu->addAction( pickColorAction );
450  connect( pickColorAction, SIGNAL( triggered() ), this, SLOT( activatePicker() ) );
451 #endif
452  QAction* chooseColorAction = new QAction( tr( "Choose color..." ), this );
453  mMenu->addAction( chooseColorAction );
454  connect( chooseColorAction, SIGNAL( triggered() ), this, SLOT( showColorDialog() ) );
455 }

511 void QgsColorButtonV2::addRecentColor( const QColor& color )
512 {
513  if ( !color.isValid() )
514  {
515  return;
516  }
517 
518  //strip alpha from color
519  QColor opaqueColor = color;
520  opaqueColor.setAlpha( 255 );
521 
522  QSettings settings;
523  QList< QVariant > recentColorVariants = settings.value( QString( "/colors/recent" ) ).toList();
524 
525  //remove colors by name
526  for ( int colorIdx = recentColorVariants.length() - 1; colorIdx >= 0; --colorIdx )
527  {
528  if (( recentColorVariants.at( colorIdx ).value<QColor>() ).name() == opaqueColor.name() )
529  {
530  recentColorVariants.removeAt( colorIdx );
531  }
532  }
533 
534  //add color
535  QVariant colorVariant = QVariant( opaqueColor );
536  recentColorVariants.prepend( colorVariant );
537 
538  //trim to 20 colors
539  while ( recentColorVariants.count() > 20 )
540  {
541  recentColorVariants.pop_back();
542  }
543 
544  settings.setValue( QString( "/colors/recent" ), recentColorVariants );
545 }

547 void QgsColorButtonV2::setButtonBackground( const QColor &color )
548 {
549  QColor backgroundColor = color;
550 
551  if ( !color.isValid() )
552  {
553  backgroundColor = mColor;
554  }
555 
556  QSize currentIconSize;
557  //icon size is button size with a small margin
558  if ( menu() )
559  {
560  if ( !mIconSize.isValid() )
561  {
562  //calculate size of push button part of widget (ie, without the menu dropdown button part)
563  QStyleOptionToolButton opt;
564  initStyleOption( &opt );
565  QRect buttonSize = QApplication::style()->subControlRect( QStyle::CC_ToolButton, &opt, QStyle::SC_ToolButton,
566  this );
567  //make sure height of icon looks good under different platforms
568  mIconSize = QSize( buttonSize.width() - 10, height() - 12 );
569  }
570  currentIconSize = mIconSize;
571  }
572  else
573  {
574  //no menu
575  currentIconSize = QSize( width() - 10, height() - 12 );
576  }
577 
578  if ( !currentIconSize.isValid() || currentIconSize.width() <= 0 || currentIconSize.height() <= 0 )
579  {
580  return;
581  }
582 
583  //create an icon pixmap
584  QPixmap pixmap( currentIconSize );
585  pixmap.fill( Qt::transparent );
586 
587  if ( backgroundColor.isValid() )
588  {
589  QRect rect( 0, 0, currentIconSize.width(), currentIconSize.height() );
590  QPainter p;
591  p.begin( &pixmap );
592  p.setRenderHint( QPainter::Antialiasing );
593  p.setPen( Qt::NoPen );
594  if ( mAllowAlpha && backgroundColor.alpha() < 255 )
595  {
596  //start with checkboard pattern
597  QBrush checkBrush = QBrush( transparentBackground() );
598  p.setBrush( checkBrush );
599  p.drawRoundedRect( rect, 3, 3 );
600  }
601 
602  //draw semi-transparent color on top
603  p.setBrush( backgroundColor );
604  p.drawRoundedRect( rect, 3, 3 );
605  p.end();
606  }
607 
608  setIconSize( currentIconSize );
609  setIcon( pixmap );
610 }
