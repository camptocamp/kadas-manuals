86 void QgsMapSettings::updateDerived()
87 {
88  QgsRectangle extent = mExtent;
89 
90  if ( extent.isEmpty() || !extent.isFinite() )
91  {
92  mValid = false;
93  return;
94  }
95 
96  // Don't allow zooms where the current extent is so small that it
97  // can't be accurately represented using a double (which is what
98  // currentExtent uses). Excluding 0 avoids a divide by zero and an
99  // infinite loop when rendering to a new canvas. Excluding extents
100  // greater than 1 avoids doing unnecessary calculations.
101 
102  // The scheme is to compare the width against the mean x coordinate
103  // (and height against mean y coordinate) and only allow zooms where
104  // the ratio indicates that there is more than about 12 significant
105  // figures (there are about 16 significant figures in a double).
106 
107  if ( extent.width() > 0 &&
108  extent.height() > 0 &&
109  extent.width() < 1 &&
110  extent.height() < 1 )
111  {
112  // Use abs() on the extent to avoid the case where the extent is
113  // symmetrical about 0.
114  double xMean = ( qAbs( extent.xMinimum() ) + qAbs( extent.xMaximum() ) ) * 0.5;
115  double yMean = ( qAbs( extent.yMinimum() ) + qAbs( extent.yMaximum() ) ) * 0.5;
116 
117  double xRange = extent.width() / xMean;
118  double yRange = extent.height() / yMean;
119 
120  static const double minProportion = 1e-12;
121  if ( xRange < minProportion || yRange < minProportion )
122  {
123  mValid = false;
124  return;
125  }
126  }
127 
128  double myHeight = mSize.height();
129  double myWidth = mSize.width();
130 
131  if ( !myWidth || !myHeight )
132  {
133  mValid = false;
134  return;
135  }
136 
137  // calculate the translation and scaling parameters
138  double mapUnitsPerPixelY = mExtent.height() / myHeight;
139  double mapUnitsPerPixelX = mExtent.width() / myWidth;
140  mMapUnitsPerPixel = mapUnitsPerPixelY > mapUnitsPerPixelX ? mapUnitsPerPixelY : mapUnitsPerPixelX;
141 
142  // calculate the actual extent of the mapCanvas
143  double dxmin = mExtent.xMinimum(), dxmax = mExtent.xMaximum(),
144  dymin = mExtent.yMinimum(), dymax = mExtent.yMaximum(), whitespace;
145 
146  if ( mapUnitsPerPixelY > mapUnitsPerPixelX )
147  {
148  whitespace = (( myWidth * mMapUnitsPerPixel ) - mExtent.width() ) * 0.5;
149  dxmin -= whitespace;
150  dxmax += whitespace;
151  }
152  else
153  {
154  whitespace = (( myHeight * mMapUnitsPerPixel ) - mExtent.height() ) * 0.5;
155  dymin -= whitespace;
156  dymax += whitespace;
157  }
158 
159  mVisibleExtent.set( dxmin, dymin, dxmax, dymax );
160 
161  // update the scale
162  mScale = mScaleCalculator.calculate( mVisibleExtent, mSize.width() );
163 
164  mMapToPixel.setParameters( mapUnitsPerPixel(),
165  visibleExtent().center().x(),
166  visibleExtent().center().y(),
167  outputSize().width(),
168  outputSize().height(),
169  mRotation );
170 
171 #if 1 // set visible extent taking rotation in consideration
172  if ( mRotation )
173  {
174  QgsPoint p1 = mMapToPixel.toMapCoordinates( QPoint( 0, 0 ) );
175  QgsPoint p2 = mMapToPixel.toMapCoordinates( QPoint( 0, myHeight ) );
176  QgsPoint p3 = mMapToPixel.toMapCoordinates( QPoint( myWidth, 0 ) );
177  QgsPoint p4 = mMapToPixel.toMapCoordinates( QPoint( myWidth, myHeight ) );
178  dxmin = std::min( p1.x(), std::min( p2.x(), std::min( p3.x(), p4.x() ) ) );
179  dymin = std::min( p1.y(), std::min( p2.y(), std::min( p3.y(), p4.y() ) ) );
180  dxmax = std::max( p1.x(), std::max( p2.x(), std::max( p3.x(), p4.x() ) ) );
181  dymax = std::max( p1.y(), std::max( p2.y(), std::max( p3.y(), p4.y() ) ) );
182  mVisibleExtent.set( dxmin, dymin, dxmax, dymax );
183  }
184 #endif
185 
186  QgsDebugMsg( QString( "Map units per pixel (x,y) : %1, %2" ).arg( qgsDoubleToString( mapUnitsPerPixelX ) ).arg( qgsDoubleToString( mapUnitsPerPixelY ) ) );
187  QgsDebugMsg( QString( "Pixmap dimensions (x,y) : %1, %2" ).arg( qgsDoubleToString( mSize.width() ) ).arg( qgsDoubleToString( mSize.height() ) ) );
188  QgsDebugMsg( QString( "Extent dimensions (x,y) : %1, %2" ).arg( qgsDoubleToString( mExtent.width() ) ).arg( qgsDoubleToString( mExtent.height() ) ) );
189  QgsDebugMsg( mExtent.toString() );
190  QgsDebugMsg( QString( "Adjusted map units per pixel (x,y) : %1, %2" ).arg( qgsDoubleToString( mVisibleExtent.width() / myWidth ) ).arg( qgsDoubleToString( mVisibleExtent.height() / myHeight ) ) );
191  QgsDebugMsg( QString( "Recalced pixmap dimensions (x,y) : %1, %2" ).arg( qgsDoubleToString( mVisibleExtent.width() / mMapUnitsPerPixel ) ).arg( qgsDoubleToString( mVisibleExtent.height() / mMapUnitsPerPixel ) ) );
192  QgsDebugMsg( QString( "Scale (assuming meters as map units) = 1:%1" ).arg( qgsDoubleToString( mScale ) ) );
193  QgsDebugMsg( QString( "Rotation: %1 degrees" ).arg( mRotation ) );
194 
195  mValid = true;
196 }

346 QgsRectangle QgsMapSettings::computeExtentForScale( const QgsPoint& point, double scale, const QgsCoordinateReferenceSystem& sourceCrs ) const
347 {
348  QgsPoint center = QgsCoordinateTransformCache::instance()->transform( sourceCrs.authid(), destinationCrs().authid() )->transform( point );
349 
350  // Output width in inches
351  double outWIn = outputSize().width() / double( outputDpi() );
352 
353  // Desired visible width (honouring scale)
354  double scaledWIn = outWIn * scale;
355 
356  // Conversion factor to inches
357  double conversionFactor = 0;
358  switch ( mapUnits() )
359  {
360  case QGis::Meters:
361  conversionFactor = 39.3700787;
362  break;
363  case QGis::Feet:
364  conversionFactor = 12.0;
365  break;
366  case QGis::NauticalMiles:
367  conversionFactor = 72913.4;
368  break;
369 
370  default:
371  case QGis::Degrees:
372  // Start with an 1x1 extent around the center
373  QgsRectangle ext( center.x() - 0.5, center.y() - 0.5, center.x() + 0.5, center.y() + 0.5 );
374  // Get scale at extent, and then scale extent to the desired scale
375  double testScale = mScaleCalculator.calculate( ext, outputSize().width() );
376  ext.scale( scale / testScale );
377  return ext;
378  break;
379  }
380 
381  double delta = 0.5 * scaledWIn / conversionFactor;
382  return QgsRectangle( center.x() - delta, center.y() - delta, center.x() + delta, center.y() + delta );
383 }

390 double QgsMapSettings::layerToMapUnits( QgsMapLayer *theLayer, const QgsRectangle& referenceExtent ) const
391 {
392  QgsRectangle extent = referenceExtent.isEmpty() ? theLayer->extent() : referenceExtent;
393  QgsPoint l1( extent.xMinimum(), extent.yMinimum() );
394  QgsPoint l2( extent.xMaximum(), extent.yMaximum() );
395  double distLayerUnits = std::sqrt( l1.sqrDist( l2 ) );
396  QgsPoint m1 = layerToMapCoordinates( theLayer, l1 );
397  QgsPoint m2 = layerToMapCoordinates( theLayer, l2 );
398  double distMapUnits = std::sqrt( m1.sqrDist( m2 ) );
399  return distMapUnits / distLayerUnits;
400 }

539 QgsRectangle QgsMapSettings::fullExtent() const
540 {
541  QgsDebugMsg( "called." );
542  QgsMapLayerRegistry* registry = QgsMapLayerRegistry::instance();
543 
544  // reset the map canvas extent since the extent may now be smaller
545  // We can't use a constructor since QgsRectangle normalizes the rectangle upon construction
546  QgsRectangle fullExtent;
547  fullExtent.setMinimal();
548 
549  // iterate through the map layers and test each layers extent
550  // against the current min and max values
551  QStringList::const_iterator it = mLayers.begin();
552  QgsDebugMsg( QString( "Layer count: %1" ).arg( mLayers.count() ) );
553  while ( it != mLayers.end() )
554  {
555  QgsMapLayer * lyr = registry->mapLayer( *it );
556  if ( lyr == NULL )
557  {
558  QgsDebugMsg( QString( "WARNING: layer '%1' not found in map layer registry!" ).arg( *it ) );
559  }
560  else
561  {
562  QgsDebugMsg( "Updating extent using " + lyr->name() );
563  QgsDebugMsg( "Input extent: " + lyr->extent().toString() );
564 
565  if ( lyr->extent().isNull() )
566  {
567  ++it;
568  continue;
569  }
570 
571  // Layer extents are stored in the coordinate system (CS) of the
572  // layer. The extent must be projected to the canvas CS
573  QgsRectangle extent = layerExtentToOutputExtent( lyr, lyr->extent() );
574 
575  QgsDebugMsg( "Output extent: " + extent.toString() );
576  fullExtent.unionRect( extent );
577 
578  }
579  ++it;
580  }
581 
582  if ( fullExtent.width() == 0.0 || fullExtent.height() == 0.0 )
583  {
584  // If all of the features are at the one point, buffer the
585  // rectangle a bit. If they are all at zero, do something a bit
586  // more crude.
587 
588  if ( fullExtent.xMinimum() == 0.0 && fullExtent.xMaximum() == 0.0 &&
589  fullExtent.yMinimum() == 0.0 && fullExtent.yMaximum() == 0.0 )
590  {
591  fullExtent.set( -1.0, -1.0, 1.0, 1.0 );
592  }
593  else
594  {
595  const double padFactor = 1e-8;
596  double widthPad = fullExtent.xMinimum() * padFactor;
597  double heightPad = fullExtent.yMinimum() * padFactor;
598  double xmin = fullExtent.xMinimum() - widthPad;
599  double xmax = fullExtent.xMaximum() + widthPad;
600  double ymin = fullExtent.yMinimum() - heightPad;
601  double ymax = fullExtent.yMaximum() + heightPad;
602  fullExtent.set( xmin, ymin, xmax, ymax );
603  }
604  }
605 
606  QgsDebugMsg( "Full extent: " + fullExtent.toString() );
607  return fullExtent;
608 }
