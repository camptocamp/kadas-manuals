177 void QgsAnnotationItem::updateBalloon()
178 {
179  if ( mFlags & ItemHasNoFrame )
180  {
181  return;
182  }
183  //first test if the point is in the frame. In that case we don't need a balloon.
184  if ( !mMapPositionFixed ||
185  ( mOffsetFromReferencePoint.x() < 0 && ( mOffsetFromReferencePoint.x() + mFrameSize.width() ) > 0
186  && mOffsetFromReferencePoint.y() < 0 && ( mOffsetFromReferencePoint.y() + mFrameSize.height() ) > 0 ) )
187  {
188  mBalloonSegment = -1;
189  return;
190  }
191 
192  //edge list
193  QList<QLineF> segmentList;
194  segmentList << segment( 0 ); segmentList << segment( 1 ); segmentList << segment( 2 ); segmentList << segment( 3 );
195 
196  //find closest edge / closest edge point
197  double minEdgeDist = std::numeric_limits<double>::max();
198  int minEdgeIndex = -1;
199  QLineF minEdge;
200  QgsPoint minEdgePoint;
201  QgsPoint origin( 0, 0 );
202 
203  for ( int i = 0; i < 4; ++i )
204  {
205  QLineF currentSegment = segmentList.at( i );
206  QgsPoint currentMinDistPoint;
207  double currentMinDist = origin.sqrDistToSegment( currentSegment.x1(), currentSegment.y1(), currentSegment.x2(), currentSegment.y2(), currentMinDistPoint );
208  if ( currentMinDist < minEdgeDist )
209  {
210  minEdgeIndex = i;
211  minEdgePoint = currentMinDistPoint;
212  minEdgeDist = currentMinDist;
213  minEdge = currentSegment;
214  }
215  }
216 
217  if ( minEdgeIndex < 0 )
218  {
219  return;
220  }
221 
222  //make that configurable for the item
223  double segmentPointWidth = 10;
224 
225  mBalloonSegment = minEdgeIndex;
226  QPointF minEdgeEnd = minEdge.p2();
227  mBalloonSegmentPoint1 = QPointF( minEdgePoint.x(), minEdgePoint.y() );
228  if ( sqrt( minEdgePoint.sqrDist( minEdgeEnd.x(), minEdgeEnd.y() ) ) < segmentPointWidth )
229  {
230  mBalloonSegmentPoint1 = pointOnLineWithDistance( minEdge.p2(), minEdge.p1(), segmentPointWidth );
231  }
232 
233  mBalloonSegmentPoint2 = pointOnLineWithDistance( mBalloonSegmentPoint1, minEdge.p2(), 10 );
234 }

350 QPointF QgsAnnotationItem::pointOnLineWithDistance( const QPointF& startPoint, const QPointF& directionPoint, double distance ) const
351 {
352  double dx = directionPoint.x() - startPoint.x();
353  double dy = directionPoint.y() - startPoint.y();
354  double length = sqrt( dx * dx + dy * dy );
355  double scaleFactor = distance / length;
356  return QPointF( startPoint.x() + dx * scaleFactor, startPoint.y() + dy * scaleFactor );

359 int QgsAnnotationItem::moveActionForPosition( const QPointF& pos ) const
360 {
361  QPointF itemPos = mapFromScene( pos );
362 
363  int cursorSensitivity = 7. / scale();
364 
365  // Check whether to move anchor point
366  if (( mFlags & ItemMapPositionLocked ) == 0 && qAbs( itemPos.x() ) < cursorSensitivity && qAbs( itemPos.y() ) < cursorSensitivity )
367  {
368  return MoveMapPosition;
369  }
370 
371  if ( mFlags & ItemRotatable )
372  {
373  QPointF itemPos = mapFromScene( pos );
374  double alpha = mAngle / 180. * M_PI;
375  QPointF rotPos = mOffsetFromReferencePoint +
376  QPointF( 0.5 * mFrameSize.width() + ( 0.5 * mFrameSize.width() - 4. / scale() ) * qCos( alpha ),
377  0.5 * mFrameSize.height() + ( 0.5 * mFrameSize.height() - 4. / scale() ) * qSin( alpha ) );
378  if ( qAbs( rotPos.x() - itemPos.x() ) < cursorSensitivity && qAbs( rotPos.y() - itemPos.y() ) < cursorSensitivity )
379  {
380  return RotateItem;
381  }
382  }
383 
384  if (( mFlags & ItemIsNotResizeable ) == 0 &&
385  QRectF( mOffsetFromReferencePoint.x() - cursorSensitivity,
386  mOffsetFromReferencePoint.y() - cursorSensitivity,
387  mFrameSize.width() + cursorSensitivity,
388  mFrameSize.height() + cursorSensitivity ).contains( itemPos )
389  )
390  {
391 
392  bool left, right, up, down;
393  left = qAbs( itemPos.x() - mOffsetFromReferencePoint.x() ) < cursorSensitivity;
394  right = qAbs( itemPos.x() - ( mOffsetFromReferencePoint.x() + mFrameSize.width() ) ) < cursorSensitivity;
395  up = qAbs( itemPos.y() - mOffsetFromReferencePoint.y() ) < cursorSensitivity;
396  down = qAbs( itemPos.y() - ( mOffsetFromReferencePoint.y() + mFrameSize.height() ) ) < cursorSensitivity;
397 
398  if ( left && up )
399  {
400  return ResizeFrameLeftUp;
401  }
402  else if ( right && up )
403  {
404  return ResizeFrameRightUp;
405  }
406  else if ( left && down )
407  {
408  return ResizeFrameLeftDown;
409  }
410  else if ( right && down )
411  {
412  return ResizeFrameRightDown;
413  }
414  if ( left )
415  {
416  return ResizeFrameLeft;
417  }
418  if ( right )
419  {
420  return ResizeFrameRight;
421  }
422  if ( up )
423  {
424  return ResizeFrameUp;
425  }
426  if ( down )
427  {
428  return ResizeFrameDown;
429  }
430  }
431 
432  // If inside frame, move frame if frame is visible, else move map position (unless position is fixed)
433  if ( itemPos.x() >= mOffsetFromReferencePoint.x() && itemPos.x() <= ( mOffsetFromReferencePoint.x() + mFrameSize.width() ) &&
434  itemPos.y() >= mOffsetFromReferencePoint.y() && itemPos.y() <= ( mOffsetFromReferencePoint.y() + mFrameSize.height() ) )
435  {
436  if (( mFlags & ItemHasNoFrame ) == 0 )
437  {
438  return MoveFramePosition;
439  }
440  else if (( mFlags & ItemMapPositionLocked ) == 0 )
441  {
442  return MoveMapPosition;
443  }
444  }
445  return NoAction;
446 }

448 void QgsAnnotationItem::handleMoveAction( int moveAction, const QPointF &newPos, const QPointF &oldPos )
449 {
450  if ( moveAction == QgsAnnotationItem::MoveMapPosition )
451  {
452  QgsPoint newMapPos = mMapCanvas->getCoordinateTransform()->toMapCoordinates( newPos.toPoint() );
453  QgsPoint oldMapPos = mMapCanvas->getCoordinateTransform()->toMapCoordinates( oldPos.toPoint() );
454  setMapPosition( mMapPosition + ( newMapPos - oldMapPos ) );
455  update();
456  }
457  else if ( moveAction == QgsAnnotationItem::MoveFramePosition )
458  {
459  if ( mMapPositionFixed )
460  {
461  setOffsetFromReferencePoint( mOffsetFromReferencePoint + ( newPos - oldPos ) );
462  }
463  else
464  {
465  QPointF newCanvasPos = pos() + ( newPos - oldPos );
466  setMapPosition( mMapCanvas->getCoordinateTransform()->toMapCoordinates( newCanvasPos.toPoint() ) );
467  }
468  update();
469  }
470  else if ( moveAction == QgsAnnotationItem::RotateItem )
471  {
472  QPointF center = mOffsetFromReferencePoint + QPointF( 0.5 * mFrameSize.width(), 0.5 * mFrameSize.height() );
473  QPointF delta = mapFromScene( newPos ) - center;
474  mAngle = qAtan2( delta.y(), delta.x() ) / M_PI * 180.;
475  if ( qAbs( mAngle - qRound( mAngle / 90. ) * 90. ) < 5 )
476  {
477  // If less than 5 deg from quarter, snap to quarter
478  mAngle = qRound( mAngle / 90. ) * 90.;
479  }
480  notifyItemUpdated();
481  update();
482  }
483  else if ( moveAction != QgsAnnotationItem::NoAction )
484  {
485  //handle the frame resize actions
486  double xmin = mOffsetFromReferencePoint.x();
487  double ymin = mOffsetFromReferencePoint.y();
488  double xmax = xmin + mFrameSize.width();
489  double ymax = ymin + mFrameSize.height();
490  double dx = newPos.x() - oldPos.x();
491  double dy = newPos.y() - oldPos.y();
492 
493  if ( moveAction == QgsAnnotationItem::ResizeFrameRight ||
494  moveAction == QgsAnnotationItem::ResizeFrameRightDown ||
495  moveAction == QgsAnnotationItem::ResizeFrameRightUp )
496  {
497  xmax += dx;
498  }
499  if ( moveAction == QgsAnnotationItem::ResizeFrameLeft ||
500  moveAction == QgsAnnotationItem::ResizeFrameLeftDown ||
501  moveAction == QgsAnnotationItem::ResizeFrameLeftUp )
502  {
503  xmin += dx;
504  }
505  if ( moveAction == QgsAnnotationItem::ResizeFrameUp ||
506  moveAction == QgsAnnotationItem::ResizeFrameLeftUp ||
507  moveAction == QgsAnnotationItem::ResizeFrameRightUp )
508  {
509  ymin += dy;
510  }
511  if ( moveAction == QgsAnnotationItem::ResizeFrameDown ||
512  moveAction == QgsAnnotationItem::ResizeFrameLeftDown ||
513  moveAction == QgsAnnotationItem::ResizeFrameRightDown )
514  {
515  ymax += dy;
516  }
517 
518  //switch min / max if necessary
519  double tmp;
520  if ( xmax < xmin )
521  {
522  tmp = xmax; xmax = xmin; xmin = tmp;
523  }
524  if ( ymax < ymin )
525  {
526  tmp = ymax; ymax = ymin; ymin = tmp;
527  }
528 
529  // Preserve aspect ratio
530  if ( mFlags & ItemKeepsAspectRatio )
531  {
532  double ratio = mFrameSize.height() / mFrameSize.width();
533  double dx = qAbs( mFrameSize.width() - ( xmax - xmin ) );
534  double dy = qAbs( mFrameSize.height() - ( ymax - ymin ) );
535  if ( dx > dy )
536  {
537  double ymid = mOffsetFromReferencePoint.y() + 0.5 * mFrameSize.height();
538  ymin = ymid - 0.5 * ( xmax - xmin ) * ratio;
539  ymax = ymid + 0.5 * ( xmax - xmin ) * ratio;
540  }
541  else
542  {
543  double xmid = mOffsetFromReferencePoint.x() + 0.5 * mFrameSize.width();
544  xmin = xmid - 0.5 * ( ymax - ymin ) / ratio;
545  xmax = xmid + 0.5 * ( ymax - ymin ) / ratio;
546  }
547  }
548 
549  setOffsetFromReferencePoint( QPointF( xmin, ymin ) );
550  setFrameSize( QSizeF( xmax - xmin, ymax - ymin ) );
551  update();
552  }
553 }
