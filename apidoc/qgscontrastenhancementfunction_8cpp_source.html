37 int QgsContrastEnhancementFunction::enhance( double theValue )
38 {
39  if ( mQgsRasterDataType == QGis::Byte )
40  {
41  return static_cast<int>( theValue );
42  }
43  else
44  {
45  return static_cast<int>(((( theValue - QgsContrastEnhancement::minimumValuePossible( mQgsRasterDataType ) ) / ( QgsContrastEnhancement::maximumValuePossible( mQgsRasterDataType ) - QgsContrastEnhancement::minimumValuePossible( mQgsRasterDataType ) ) )*255.0 ) );
46  }
47 }
