106 static QgsPointLocator::Match _findClosestSegmentIntersection( const QgsPoint& pt, const QgsPointLocator::MatchList& segments )
107 {
108  if ( segments.isEmpty() )
109  return QgsPointLocator::Match();
110 
111  QSet<QgsPoint> endpoints;
112 
113  // make a geometry
114  QList<QgsGeometry*> geoms;
115  foreach ( const QgsPointLocator::Match& m, segments )
116  {
117  if ( m.hasEdge() )
118  {
119  QgsPolyline pl( 2 );
120  m.edgePoints( pl[0], pl[1] );
121  geoms << QgsGeometry::fromPolyline( pl );
122  endpoints << pl[0] << pl[1];
123  }
124  }
125 
126  QgsGeometry* g = QgsGeometry::unaryUnion( geoms );
127  qDeleteAll( geoms );
128 
129  // get intersection points
130  QList<QgsPoint> newPoints;
131  if ( g->wkbType() == QGis::WKBLineString )
132  {
133  foreach ( const QgsPoint& p, g->asPolyline() )
134  {
135  if ( !endpoints.contains( p ) )
136  newPoints << p;
137  }
138  }
139  if ( g->wkbType() == QGis::WKBMultiLineString )
140  {
141  foreach ( const QgsPolyline& pl, g->asMultiPolyline() )
142  {
143  foreach ( const QgsPoint& p, pl )
144  {
145  if ( !endpoints.contains( p ) )
146  newPoints << p;
147  }
148  }
149  }
150  delete g;
151 
152  if ( newPoints.isEmpty() )
153  return QgsPointLocator::Match();
154 
155  // find the closest points
156  QgsPoint minP;
157  double minSqrDist = 1e20; // "infinity"
158  foreach ( const QgsPoint& p, newPoints )
159  {
160  double sqrDist = pt.sqrDist( p.x(), p.y() );
161  if ( sqrDist < minSqrDist )
162  {
163  minSqrDist = sqrDist;
164  minP = p;
165  }
166  }
167 
168  return QgsPointLocator::Match( QgsPointLocator::Vertex, 0, 0, sqrt( minSqrDist ), minP );
169 }

172 static void _replaceIfBetter( QgsPointLocator::Match& mBest, const QgsPointLocator::Match& mNew, double maxDistance )
173 {
174  // is other match relevant?
175  if ( !mNew.isValid() || mNew.distance() > maxDistance )
176  return;
177 
178  // is other match actually better?
179  if ( mBest.isValid() && mBest.type() == mNew.type() && mBest.distance() - 10e-6 < mNew.distance() )
180  return;
181 
182  // prefer vertex matches to edge matches (even if they are closer)
183  if ( mBest.type() == QgsPointLocator::Vertex && mNew.type() == QgsPointLocator::Edge )
184  return;
185 
186  mBest = mNew; // the other match is better!
187 }

190 static void _updateBestMatch( QgsPointLocator::Match& bestMatch, const QgsPoint& pointMap, QgsPointLocator* loc, int type, double tolerance, QgsPointLocator::MatchFilter* filter )
191 {
192  if ( type & QgsPointLocator::Vertex )
193  {
194  _replaceIfBetter( bestMatch, loc->nearestVertex( pointMap, tolerance, filter ), tolerance );
195  }
196  if ( bestMatch.type() != QgsPointLocator::Vertex && ( type & QgsPointLocator::Edge ) )
197  {
198  _replaceIfBetter( bestMatch, loc->nearestEdge( pointMap, tolerance, filter ), tolerance );
199  }
200 }

208 QgsPointLocator::Match QgsSnappingUtils::snapToMap( const QgsPoint& pointMap, QgsPointLocator::MatchFilter* filter )
209 {
210  if ( !mMapSettings.hasValidSettings() )
211  return QgsPointLocator::Match();
212 
213  if ( mSnapToMapMode == SnapCurrentLayer )
214  {
215  if ( !mCurrentLayer )
216  return QgsPointLocator::Match();
217 
218  prepareIndex( QList<QgsVectorLayer*>() << mCurrentLayer );
219 
220  // data from project
221  double tolerance = QgsTolerance::toleranceInProjectUnits( mDefaultTolerance, mCurrentLayer, mMapSettings, mDefaultUnit );
222  int type = mDefaultType;
223 
224  // use ad-hoc locator
225  QgsPointLocator* loc = locatorForLayerUsingStrategy( mCurrentLayer, pointMap, tolerance );
226  if ( !loc )
227  return QgsPointLocator::Match();
228 
229  QgsPointLocator::Match bestMatch;
230  _updateBestMatch( bestMatch, pointMap, loc, type, tolerance, filter );
231 
232  if ( mSnapOnIntersection )
233  {
234  QgsPointLocator* locEdges = locatorForLayerUsingStrategy( mCurrentLayer, pointMap, tolerance );
235  QgsPointLocator::MatchList edges = locEdges->edgesInRect( pointMap, tolerance );
236  _replaceIfBetter( bestMatch, _findClosestSegmentIntersection( pointMap, edges ), tolerance );
237  }
238 
239  return bestMatch;
240  }
241  else if ( mSnapToMapMode == SnapAdvanced )
242  {
243  QList<QgsVectorLayer*> layers;
244  foreach ( const LayerConfig& layerConfig, mLayers )
245  layers << layerConfig.layer;
246  prepareIndex( layers );
247 
248  QgsPointLocator::Match bestMatch;
249  QgsPointLocator::MatchList edges; // for snap on intersection
250  double maxSnapIntTolerance = 0;
251 
252  foreach ( const LayerConfig& layerConfig, mLayers )
253  {
254  double tolerance = QgsTolerance::toleranceInProjectUnits( layerConfig.tolerance, layerConfig.layer, mMapSettings, layerConfig.unit );
255  if ( QgsPointLocator* loc = locatorForLayerUsingStrategy( layerConfig.layer, pointMap, tolerance ) )
256  {
257  _updateBestMatch( bestMatch, pointMap, loc, layerConfig.type, tolerance, filter );
258 
259  if ( mSnapOnIntersection )
260  {
261  edges << loc->edgesInRect( pointMap, tolerance );
262  maxSnapIntTolerance = qMax( maxSnapIntTolerance, tolerance );
263  }
264  }
265  }
266 
267  if ( mSnapOnIntersection )
268  _replaceIfBetter( bestMatch, _findClosestSegmentIntersection( pointMap, edges ), maxSnapIntTolerance );
269 
270  return bestMatch;
271  }
272  else if ( mSnapToMapMode == SnapAllLayers )
273  {
274  // data from project
275  double tolerance = QgsTolerance::toleranceInProjectUnits( mDefaultTolerance, 0, mMapSettings, mDefaultUnit );
276  int type = mDefaultType;
277 
278  QList<QgsVectorLayer*> layers;
279  foreach ( const QString& layerID, mMapSettings.layers() )
280  if ( QgsVectorLayer* vl = qobject_cast<QgsVectorLayer*>( QgsMapLayerRegistry::instance()->mapLayer( layerID ) ) )
281  layers << vl;
282  prepareIndex( layers );
283 
284  QgsPointLocator::MatchList edges; // for snap on intersection
285  QgsPointLocator::Match bestMatch;
286 
287  foreach ( QgsVectorLayer* vl, layers )
288  {
289  if ( QgsPointLocator* loc = locatorForLayerUsingStrategy( vl, pointMap, tolerance ) )
290  {
291  _updateBestMatch( bestMatch, pointMap, loc, type, tolerance, filter );
292 
293  if ( mSnapOnIntersection )
294  edges << loc->edgesInRect( pointMap, tolerance );
295  }
296  }
297 
298  if ( mSnapOnIntersection )
299  _replaceIfBetter( bestMatch, _findClosestSegmentIntersection( pointMap, edges ), tolerance );
300 
301  return bestMatch;
302  }
303 
304  return QgsPointLocator::Match();
305 }

308 void QgsSnappingUtils::prepareIndex( const QList<QgsVectorLayer*>& layers )
309 {
310  // check if we need to build any index
311  QList<QgsVectorLayer*> layersToIndex;
312  foreach ( QgsVectorLayer* vl, layers )
313  {
314  if ( willUseIndex( vl ) && !locatorForLayer( vl )->hasIndex() )
315  layersToIndex << vl;
316  }
317  if ( layersToIndex.isEmpty() )
318  return;
319 
320  // build indexes
321  QTime t; t.start();
322  int i = 0;
323  prepareIndexStarting( layersToIndex.count() );
324  foreach ( QgsVectorLayer* vl, layersToIndex )
325  {
326  QTime tt; tt.start();
327  if ( !locatorForLayer( vl )->init( mStrategy == IndexHybrid ? 1000000 : -1 ) )
328  mHybridNonindexableLayers.insert( vl->id() );
329  QgsDebugMsg( QString( "Index init: %1 ms (%2)" ).arg( tt.elapsed() ).arg( vl->id() ) );
330  prepareIndexProgress( ++i );
331  }
332  QgsDebugMsg( QString( "Prepare index total: %1 ms" ).arg( t.elapsed() ) );
333 }

387 void QgsSnappingUtils::readConfigFromProject()
388 {
389  mLayers.clear();
390 
391  QString snapMode = QgsProject::instance()->readEntry( "Digitizing", "/SnappingMode" );
392 
393  if ( mReadDefaultConfigFromProject )
394  {
395  int type = 0;
396  QString snapType = QgsProject::instance()->readEntry( "Digitizing", "/DefaultSnapType", QString( "off" ) );
397  if ( snapType == "to segment" )
398  type = QgsPointLocator::Edge;
399  else if ( snapType == "to vertex and segment" )
400  type = QgsPointLocator::Vertex | QgsPointLocator::Edge;
401  else if ( snapType == "to vertex" )
402  type = QgsPointLocator::Vertex;
403  double tolerance = QgsProject::instance()->readDoubleEntry( "Digitizing", "/DefaultSnapTolerance", 0 );
404  QgsTolerance::UnitType unit = ( QgsTolerance::UnitType ) QgsProject::instance()->readNumEntry( "Digitizing", "/DefaultSnapToleranceUnit", QgsTolerance::ProjectUnits );
405  setDefaultSettings( type, tolerance, unit );
406  }
407 
408  //snapping on intersection on?
409  setSnapOnIntersections( QgsProject::instance()->readNumEntry( "Digitizing", "/IntersectionSnapping", 0 ) );
410 
411  //read snapping settings from project
412  bool snappingDefinedInProject = false, ok;
413  QStringList layerIdList = QgsProject::instance()->readListEntry( "Digitizing", "/LayerSnappingList", QStringList(), &snappingDefinedInProject );
414  QStringList enabledList = QgsProject::instance()->readListEntry( "Digitizing", "/LayerSnappingEnabledList", QStringList(), &ok );
415  QStringList toleranceList = QgsProject::instance()->readListEntry( "Digitizing", "/LayerSnappingToleranceList", QStringList(), &ok );
416  QStringList toleranceUnitList = QgsProject::instance()->readListEntry( "Digitizing", "/LayerSnappingToleranceUnitList", QStringList(), &ok );
417  QStringList snapToList = QgsProject::instance()->readListEntry( "Digitizing", "/LayerSnapToList", QStringList(), &ok );
418 
419  // lists must have the same size, otherwise something is wrong
420  if ( layerIdList.size() != enabledList.size() ||
421  layerIdList.size() != toleranceList.size() ||
422  layerIdList.size() != toleranceUnitList.size() ||
423  layerIdList.size() != snapToList.size() )
424  return;
425 
426  if ( !snappingDefinedInProject )
427  return; // nothing defined in project
428 
429  if ( mReadDefaultConfigFromProject )
430  {
431  // Use snapping information from the project
432  if ( snapMode == "current_layer" )
433  mSnapToMapMode = SnapCurrentLayer;
434  else if ( snapMode == "all_layers" )
435  mSnapToMapMode = SnapAllLayers;
436  else // either "advanced" or empty (for background compatibility)
437  mSnapToMapMode = SnapAdvanced;
438  }
439 
440 
441  // load layers, tolerances, snap type
442  QStringList::const_iterator layerIt( layerIdList.constBegin() );
443  QStringList::const_iterator tolIt( toleranceList.constBegin() );
444  QStringList::const_iterator tolUnitIt( toleranceUnitList.constBegin() );
445  QStringList::const_iterator snapIt( snapToList.constBegin() );
446  QStringList::const_iterator enabledIt( enabledList.constBegin() );
447  for ( ; layerIt != layerIdList.constEnd(); ++layerIt, ++tolIt, ++tolUnitIt, ++snapIt, ++enabledIt )
448  {
449  // skip layer if snapping is not enabled
450  if ( *enabledIt != "enabled" )
451  continue;
452 
453  QgsVectorLayer *vlayer = qobject_cast<QgsVectorLayer *>( QgsMapLayerRegistry::instance()->mapLayer( *layerIt ) );
454  if ( !vlayer || !vlayer->hasGeometryType() )
455  continue;
456 
457  int t = ( *snapIt == "to_vertex" ? QgsPointLocator::Vertex :
458  ( *snapIt == "to_segment" ? QgsPointLocator::Edge :
459  QgsPointLocator::Vertex | QgsPointLocator::Edge ) );
460  mLayers.append( LayerConfig( vlayer, t, tolIt->toDouble(), ( QgsTolerance::UnitType ) tolUnitIt->toInt() ) );
461  }
462 
463 }
