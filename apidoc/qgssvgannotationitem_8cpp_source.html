69 void QgsSvgAnnotationItem::paint( QPainter* painter )
70 {
71  if ( !painter )
72  {
73  return;
74  }
75 
76  drawFrame( painter );
77  if ( mMapPositionFixed )
78  {
79  drawMarkerSymbol( painter );
80  }
81 
82  //keep width/height ratio of svg
83  QRect viewBox = mSvgRenderer.viewBox();
84  if ( viewBox.isValid() )
85  {
86  double widthRatio = mFrameSize.width() / viewBox.width();
87  double heightRatio = mFrameSize.height() / viewBox.height();
88  double renderWidth = 0;
89  double renderHeight = 0;
90  if ( widthRatio <= heightRatio )
91  {
92  renderWidth = mFrameSize.width();
93  renderHeight = viewBox.height() * mFrameSize.width() / viewBox.width();
94  }
95  else
96  {
97  renderHeight = mFrameSize.height();
98  renderWidth = viewBox.width() * mFrameSize.height() / viewBox.height();
99  }
100 
101  double alpha = mAngle / 180. * M_PI;
102  double rw = renderHeight * qAbs( qSin( alpha ) ) + renderWidth * qAbs( qCos( alpha ) );
103  double rh = renderHeight * qAbs( qCos( alpha ) ) + renderWidth * qAbs( qSin( alpha ) );
104  double scale = qMin( renderWidth / rw, renderHeight / rh );
105  painter->save();
106  painter->scale( scale, scale );
107  painter->rotate( mAngle );
108  mSvgRenderer.render( painter, QRectF( mOffsetFromReferencePoint.x(), mOffsetFromReferencePoint.y(),
109  renderWidth, renderHeight ) );
110  painter->restore();
111  }
112  if ( isSelected() )
113  {
114  drawSelectionBoxes( painter );
115  }
116 }

123 void QgsSvgAnnotationItem::setFilePath( const QString& filepath )
124 {
125  mFilePath = filepath;
126  svg2svgt::Logger logger;
127  svg2svgt::RuleEngine ruleEngine( logger );
128  ruleEngine.setDefaultRules();
129  svg2svgt::ProcessorEngine processor( ruleEngine, logger );
130 
131  QFile file( filepath );
132  if ( file.open( QIODevice::ReadOnly ) )
133  {
134  mSvgRenderer.load( processor.process( file.readAll() ) );
135  }
136 
137  QRect viewBox = mSvgRenderer.viewBox();
138  if ( !mIsClone )
139  {
140  setFrameSize( QSizeF( viewBox.width(), viewBox.height() ) );
141  }
142 }
