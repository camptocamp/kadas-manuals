62 int QgsMapCanvasSnapper::snapToCurrentLayer( const QPoint& p, QList<QgsSnappingResult>& results,
63  QgsSnapper::SnappingType snap_to,
64  double snappingTol,
65  const QList<QgsPoint>& excludePoints )
66 {
67  results.clear();
68 
69  if ( !mSnapper || !mMapCanvas )
70  return 1;
71 
72  //topological editing on?
73  int topologicalEditing = QgsProject::instance()->readNumEntry( "Digitizing", "/TopologicalEditing", 0 );
74  if ( topologicalEditing == 0 )
75  {
76  mSnapper->setSnapMode( QgsSnapper::SnapWithOneResult );
77  }
78  else
79  {
80  mSnapper->setSnapMode( QgsSnapper::SnapWithResultsForSamePosition );
81  }
82 
83  //current vector layer
84  QgsMapLayer* currentLayer = mMapCanvas->currentLayer();
85  if ( !currentLayer )
86  return 2;
87 
88  QgsVectorLayer* vlayer = qobject_cast<QgsVectorLayer *>( currentLayer );
89  if ( !vlayer )
90  return 3;
91 
92  QgsSnapper::SnapLayer snapLayer;
93  snapLayer.mLayer = vlayer;
94  snapLayer.mSnapTo = snap_to;
95  snapLayer.mUnitType = QgsTolerance::LayerUnits;
96 
97  if ( snappingTol < 0 )
98  {
99  //use search tolerance for vertex editing
100  snapLayer.mTolerance = QgsTolerance::vertexSearchRadius( vlayer, mMapCanvas->mapSettings() );
101  }
102  else
103  {
104  snapLayer.mTolerance = snappingTol;
105  }
106 
107  QList<QgsSnapper::SnapLayer> snapLayers;
108  snapLayers.append( snapLayer );
109  mSnapper->setSnapLayers( snapLayers );
110 
111  QgsPoint mapPoint = mMapCanvas->mapSettings().mapToPixel().toMapCoordinates( p );
112  if ( mSnapper->snapMapPoint( mapPoint, results, excludePoints ) != 0 )
113  {
114  return 4;
115  }
116 
117  return 0;
118 }

126 int QgsMapCanvasSnapper::snapToBackgroundLayers( const QgsPoint& point, QList<QgsSnappingResult>& results, const QList<QgsPoint>& excludePoints )
127 {
128  results.clear();
129 
130  if ( !mSnapper )
131  return 5;
132 
133  //topological editing on?
134  int topologicalEditing = QgsProject::instance()->readNumEntry( "Digitizing", "/TopologicalEditing", 0 );
135 
136  //snapping on intersection on?
137  int intersectionSnapping = QgsProject::instance()->readNumEntry( "Digitizing", "/IntersectionSnapping", 0 );
138 
139  if ( topologicalEditing == 0 )
140  {
141  if ( intersectionSnapping == 0 )
142  mSnapper->setSnapMode( QgsSnapper::SnapWithOneResult );
143  else
144  mSnapper->setSnapMode( QgsSnapper::SnapWithResultsWithinTolerances );
145  }
146  else if ( intersectionSnapping == 0 )
147  {
148  mSnapper->setSnapMode( QgsSnapper::SnapWithResultsForSamePosition );
149  }
150  else
151  {
152  mSnapper->setSnapMode( QgsSnapper::SnapWithResultsWithinTolerances );
153  }
154 
155  QgsVectorLayer* currentVectorLayer = dynamic_cast<QgsVectorLayer*>( mMapCanvas->currentLayer() );
156  if ( !currentVectorLayer )
157  {
158  return 1;
159  }
160 
161  //read snapping settings from project
162  QStringList layerIdList, enabledList, toleranceList, toleranceUnitList, snapToList;
163 
164  bool ok, snappingDefinedInProject;
165 
166  QSettings settings;
167  QString snappingMode = QgsProject::instance()->readEntry( "Digitizing", "/SnappingMode", "current_layer", &snappingDefinedInProject );
168  QString defaultSnapToleranceUnit = snappingDefinedInProject ? QgsProject::instance()->readEntry( "Digitizing", "/DefaultSnapToleranceUnit" ) : settings.value( "/Qgis/digitizing/default_snapping_tolerance_unit", "0" ).toString();
169  QString defaultSnapType = snappingDefinedInProject ? QgsProject::instance()->readEntry( "Digitizing", "/DefaultSnapType" ) : settings.value( "/Qgis/digitizing/default_snap_mode", "off" ).toString();
170  QString defaultSnapTolerance = snappingDefinedInProject ? QString::number( QgsProject::instance()->readDoubleEntry( "Digitizing", "/DefaultSnapTolerance" ) ) : settings.value( "/Qgis/digitizing/default_snapping_tolerance", "0" ).toString();
171 
172  if ( !snappingDefinedInProject && defaultSnapType == "off" )
173  {
174  return 0;
175  }
176 
177  if ( snappingMode == "current_layer" || !snappingDefinedInProject )
178  {
179  layerIdList.append( currentVectorLayer->id() );
180  enabledList.append( "enabled" );
181  toleranceList.append( defaultSnapTolerance );
182  toleranceUnitList.append( defaultSnapToleranceUnit );
183  snapToList.append( defaultSnapType );
184  }
185  else if ( snappingMode == "all_layers" )
186  {
187  QList<QgsMapLayer*> allLayers = mMapCanvas->layers();
188  QList<QgsMapLayer*>::const_iterator layerIt = allLayers.constBegin();
189  for ( ; layerIt != allLayers.constEnd(); ++layerIt )
190  {
191  if ( !( *layerIt ) )
192  {
193  continue;
194  }
195  layerIdList.append(( *layerIt )->id() );
196  enabledList.append( "enabled" );
197  toleranceList.append( defaultSnapTolerance );
198  toleranceUnitList.append( defaultSnapToleranceUnit );
199  snapToList.append( defaultSnapType );
200  }
201  }
202  else //advanced
203  {
204  layerIdList = QgsProject::instance()->readListEntry( "Digitizing", "/LayerSnappingList", QStringList(), &ok );
205  enabledList = QgsProject::instance()->readListEntry( "Digitizing", "/LayerSnappingEnabledList", QStringList(), &ok );
206  toleranceList = QgsProject::instance()->readListEntry( "Digitizing", "/LayerSnappingToleranceList", QStringList(), &ok );
207  toleranceUnitList = QgsProject::instance()->readListEntry( "Digitizing", "/LayerSnappingToleranceUnitList", QStringList(), &ok );
208  snapToList = QgsProject::instance()->readListEntry( "Digitizing", "/LayerSnapToList", QStringList(), &ok );
209  }
210 
211  if ( !( layerIdList.size() == enabledList.size() &&
212  layerIdList.size() == toleranceList.size() &&
213  layerIdList.size() == toleranceUnitList.size() &&
214  layerIdList.size() == snapToList.size() ) )
215  {
216  // lists must have the same size, otherwise something is wrong
217  return 1;
218  }
219 
220  QList<QgsSnapper::SnapLayer> snapLayers;
221  QgsSnapper::SnapLayer snapLayer;
222 
223 
224 
225  // set layers, tolerances, snap to segment/vertex to QgsSnapper
226  QStringList::const_iterator layerIt( layerIdList.constBegin() );
227  QStringList::const_iterator tolIt( toleranceList.constBegin() );
228  QStringList::const_iterator tolUnitIt( toleranceUnitList.constBegin() );
229  QStringList::const_iterator snapIt( snapToList.constBegin() );
230  QStringList::const_iterator enabledIt( enabledList.constBegin() );
231  for ( ; layerIt != layerIdList.constEnd(); ++layerIt, ++tolIt, ++tolUnitIt, ++snapIt, ++enabledIt )
232  {
233  if ( *enabledIt != "enabled" )
234  {
235  // skip layer if snapping is not enabled
236  continue;
237  }
238 
239  //layer
240  QgsVectorLayer *vlayer = qobject_cast<QgsVectorLayer *>( QgsMapLayerRegistry::instance()->mapLayer( *layerIt ) );
241  if ( !vlayer || !vlayer->hasGeometryType() )
242  continue;
243 
244  snapLayer.mLayer = vlayer;
245 
246  //tolerance
247  snapLayer.mTolerance = tolIt->toDouble();
248  snapLayer.mUnitType = ( QgsTolerance::UnitType ) tolUnitIt->toInt();
249 
250  // segment or vertex
251  if ( *snapIt == "to vertex" || *snapIt == "to_vertex" )
252  {
253  snapLayer.mSnapTo = QgsSnapper::SnapToVertex;
254  }
255  else if ( *snapIt == "to segment" || *snapIt == "to_segment" )
256  {
257  snapLayer.mSnapTo = QgsSnapper::SnapToSegment;
258  }
259  else if ( *snapIt == "to vertex and segment" || *snapIt == "to_vertex_and_segment" )
260  {
261  snapLayer.mSnapTo = QgsSnapper::SnapToVertexAndSegment;
262  }
263  else //off
264  {
265  continue;
266  }
267 
268  snapLayers.append( snapLayer );
269  }
270 
271  mSnapper->setSnapLayers( snapLayers );
272 
273  if ( mSnapper->snapMapPoint( point, results, excludePoints ) != 0 )
274  return 4;
275 
276  if ( intersectionSnapping != 1 )
277  return 0;
278 
279  QList<QgsSnappingResult> segments;
280  QList<QgsSnappingResult> points;
281  for ( QList<QgsSnappingResult>::const_iterator it = results.constBegin();
282  it != results.constEnd();
283  ++it )
284  {
285  if ( it->snappedVertexNr == -1 )
286  {
287  QgsDebugMsg( "segment" );
288  segments.push_back( *it );
289  }
290  else
291  {
292  QgsDebugMsg( "no segment" );
293  points.push_back( *it );
294  }
295  }
296 
297  if ( segments.length() < 2 )
298  return 0;
299 
300  QList<QgsSnappingResult> myResults;
301 
302  for ( QList<QgsSnappingResult>::const_iterator oSegIt = segments.constBegin();
303  oSegIt != segments.constEnd();
304  ++oSegIt )
305  {
306  QgsDebugMsg( QString::number( oSegIt->beforeVertexNr ) );
307 
308  QVector<QgsPoint> vertexPoints;
309  vertexPoints.append( oSegIt->beforeVertex );
310  vertexPoints.append( oSegIt->afterVertex );
311 
312  QgsGeometry* lineA = QgsGeometry::fromPolyline( vertexPoints );
313 
314  for ( QList<QgsSnappingResult>::iterator iSegIt = segments.begin();
315  iSegIt != segments.end();
316  ++iSegIt )
317  {
318  QVector<QgsPoint> vertexPoints;
319  vertexPoints.append( iSegIt->beforeVertex );
320  vertexPoints.append( iSegIt->afterVertex );
321 
322  QgsGeometry* lineB = QgsGeometry::fromPolyline( vertexPoints );
323  QgsGeometry* intersectionPoint = lineA->intersection( lineB );
324  delete lineB;
325 
326  if ( intersectionPoint && intersectionPoint->type() == QGis::Point )
327  {
328  //We have to check the intersection point is inside the tolerance distance for both layers
329  double toleranceA = 0;
330  double toleranceB = 0;
331  for ( int i = 0 ;i < snapLayers.size();++i )
332  {
333  if ( snapLayers[i].mLayer == oSegIt->layer )
334  {
335  toleranceA = QgsTolerance::toleranceInMapUnits( snapLayers[i].mTolerance, snapLayers[i].mLayer, mMapCanvas->mapSettings(), snapLayers[i].mUnitType );
336  }
337  if ( snapLayers[i].mLayer == iSegIt->layer )
338  {
339  toleranceB = QgsTolerance::toleranceInMapUnits( snapLayers[i].mTolerance, snapLayers[i].mLayer, mMapCanvas->mapSettings(), snapLayers[i].mUnitType );
340  }
341  }
342  QgsGeometry* cursorPoint = QgsGeometry::fromPoint( point );
343  double distance = intersectionPoint->distance( *cursorPoint );
344  if ( distance < toleranceA && distance < toleranceB )
345  {
346  iSegIt->snappedVertex = intersectionPoint->asPoint();
347  myResults.append( *iSegIt );
348  }
349  delete cursorPoint;
350  }
351  delete intersectionPoint;
352 
353  }
354 
355  delete lineA;
356  }
357 
358  if ( myResults.length() > 0 )
359  {
360  results.clear();
361  results = myResults;
362  }
363 
364  return 0;
365 }
