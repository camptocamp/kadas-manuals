27 void QgsBillBoardRegistry::addItem( void* parent, const QString& name, const QImage &image, const QgsPoint &worldPos , int xoffset, const QString &layerId, const QPointF& hotSpot )
28 {
29  QMap<void*, QgsBillBoardItem*>::iterator it = mItems.find( parent );
30  if ( it == mItems.end() )
31  {
32  it = mItems.insert( parent, new QgsBillBoardItem );
33  }
34  if ( xoffset == 0 )
35  {
36  it.value()->image = image;
37  }
38  else
39  {
40  QImage newimage( image.width() + 2 * qAbs( xoffset ), image.height(), image.format() );
41  newimage.fill( Qt::transparent );
42  QPainter p( &newimage );
43  p.drawImage( xoffset < 0 ? 0 : 2 * xoffset, 0, image );
44  it.value()->image = newimage;
45  }
46  it.value()->name = name;
47  it.value()->worldPos = worldPos;
48  it.value()->layerId = layerId;
49  it.value()->hotSpot = hotSpot;
50  emit itemAdded( it.value() );
51 }
