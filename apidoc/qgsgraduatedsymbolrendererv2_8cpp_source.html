342 QgsSymbolV2* QgsGraduatedSymbolRendererV2::symbolForFeature( QgsFeature& feature )
343 {
344  QgsSymbolV2* symbol = originalSymbolForFeature( feature );
345  if ( symbol == NULL )
346  return NULL;
347 
348  if ( !mRotation.data() && !mSizeScale.data() )
349  return symbol; // no data-defined rotation/scaling - just return the symbol
350 
351  // find out rotation, size scale
352  const double rotation = mRotation.data() ? mRotation->evaluate( feature ).toDouble() : 0;
353  const double sizeScale = mSizeScale.data() ? mSizeScale->evaluate( feature ).toDouble() : 1.;
354

364  markerSymbol->setScaleMethod( mScaleMethod );

367  {
368  QgsLineSymbolV2* lineSymbol = static_cast<QgsLineSymbolV2*>( tempSymbol );
369  lineSymbol->setWidth( sizeScale * static_cast<QgsLineSymbolV2*>( symbol )->width() );

591 static QList<double> _calcEqualIntervalBreaks( double minimum, double maximum, int classes )
592 {
593 
594  // Equal interval algorithm
595  //
596  // Returns breaks based on dividing the range ('minimum' to 'maximum')
597  // into 'classes' parts.
598 
599  double step = ( maximum - minimum ) / classes;
600 
601  QList<double> breaks;
602  double value = minimum;
603  for ( int i = 0; i < classes; i++ )
604  {
605  value += step;
606  breaks.append( value );
607  }
608 
609  // floating point arithmetics is not precise:
610  // set the last break to be exactly maximum so we do not miss it
611  breaks[classes-1] = maximum;
612 
613  return breaks;
614 }

616 static QList<double> _calcQuantileBreaks( QList<double> values, int classes )
617 {
618  // q-th quantile of a data set:
619  // value where q fraction of data is below and (1-q) fraction is above this value
620  // Xq = (1 - r) * X_NI1 + r * X_NI2
621  // NI1 = (int) (q * (n+1))
622  // NI2 = NI1 + 1
623  // r = q * (n+1) - (int) (q * (n+1))
624  // (indices of X: 1...n)
625 
626  // sort the values first
627  qSort( values );
628 
629  QList<double> breaks;
630 
631  // If there are no values to process: bail out
632  if ( !values.count() )
633  return breaks;
634 
635  int n = values.count();
636  double Xq = n > 0 ? values[0] : 0.0;
637 
638  for ( int i = 1; i < classes; i++ )
639  {
640  if ( n > 1 )
641  {
642  double q = i / ( double ) classes;
643  double a = q * ( n - 1 );
644  int aa = ( int )( a );
645 
646  double r = a - aa;
647  Xq = ( 1 - r ) * values[aa] + r * values[aa+1];
648  }
649  breaks.append( Xq );
650  }
651 
652  breaks.append( values[ n-1 ] );
653 
654  return breaks;
655 }

657 static QList<double> _calcPrettyBreaks( double minimum, double maximum, int classes )
658 {
659 
660  // C++ implementation of R's pretty algorithm
661  // Based on code for determining optimal tick placement for statistical graphics
662  // from the R statistical programming language.
663  // Code ported from R implementation from 'labeling' R package
664  //
665  // Computes a sequence of about 'classes' equally spaced round values
666  // which cover the range of values from 'minimum' to 'maximum'.
667  // The values are chosen so that they are 1, 2 or 5 times a power of 10.
668 
669  QList<double> breaks;
670  if ( classes < 1 )
671  {
672  breaks.append( maximum );
673  return breaks;
674  }
675 
676  int minimumCount = ( int ) classes / 3;
677  double shrink = 0.75;
678  double highBias = 1.5;
679  double adjustBias = 0.5 + 1.5 * highBias;
680  int divisions = classes;
681  double h = highBias;
682  double cell;
683  int U;
684  bool small = false;
685  double dx = maximum - minimum;
686 
687  if ( dx == 0 && maximum == 0 )
688  {
689  cell = 1.0;
690  small = true;
691  U = 1;
692  }
693  else
694  {
695  cell = qMax( qAbs( minimum ), qAbs( maximum ) );
696  if ( adjustBias >= 1.5 * h + 0.5 )
697  {
698  U = 1 + ( 1.0 / ( 1 + h ) );
699  }
700  else
701  {
702  U = 1 + ( 1.5 / ( 1 + adjustBias ) );
703  }
704  small = dx < ( cell * U * qMax( 1, divisions ) * 1e-07 * 3.0 );
705  }
706 
707  if ( small )
708  {
709  if ( cell > 10 )
710  {
711  cell = 9 + cell / 10;
712  cell = cell * shrink;
713  }
714  if ( minimumCount > 1 )
715  {
716  cell = cell / minimumCount;
717  }
718  }
719  else
720  {
721  cell = dx;
722  if ( divisions > 1 )
723  {
724  cell = cell / divisions;
725  }
726  }
727  if ( cell < 20 * 1e-07 )
728  {
729  cell = 20 * 1e-07;
730  }
731 
732  double base = pow( 10.0, floor( log10( cell ) ) );
733  double unit = base;
734  if (( 2 * base ) - cell < h *( cell - unit ) )
735  {
736  unit = 2.0 * base;
737  if (( 5 * base ) - cell < adjustBias *( cell - unit ) )
738  {
739  unit = 5.0 * base;
740  if (( 10.0 * base ) - cell < h *( cell - unit ) )
741  {
742  unit = 10.0 * base;
743  }
744  }
745  }
746  // Maybe used to correct for the epsilon here??
747  int start = floor( minimum / unit + 1e-07 );
748  int end = ceil( maximum / unit - 1e-07 );
749 
750  // Extend the range out beyond the data. Does this ever happen??
751  while ( start * unit > minimum + ( 1e-07 * unit ) )
752  {
753  start = start - 1;
754  }
755  while ( end * unit < maximum - ( 1e-07 * unit ) )
756  {
757  end = end + 1;
758  }
759  QgsDebugMsg( QString( "pretty classes: %1" ).arg( end ) );
760 
761  // If we don't have quite enough labels, extend the range out
762  // to make more (these labels are beyond the data :( )
763  int k = floor( 0.5 + end - start );
764  if ( k < minimumCount )
765  {
766  k = minimumCount - k;
767  if ( start >= 0 )
768  {
769  end = end + k / 2;
770  start = start - k / 2 + k % 2;
771  }
772  else
773  {
774  start = start - k / 2;
775  end = end + k / 2 + k % 2;
776  }
777  }
778  double minimumBreak = start * unit;
779  //double maximumBreak = end * unit;
780  int count = end - start;
781 
782  for ( int i = 1; i < count + 1; i++ )
783  {
784  breaks.append( minimumBreak + i * unit );
785  }
786 
787  if ( breaks.isEmpty() )
788  return breaks;
789 
790  if ( breaks.first() < minimum )
791  {
792  breaks[0] = minimum;
793  }
794  if ( breaks.last() > maximum )
795  {
796  breaks[breaks.count()-1] = maximum;
797  }
798 
799  return breaks;
800 } // _calcPrettyBreaks

803 static QList<double> _calcStdDevBreaks( QList<double> values, int classes, QList<double> &labels )
804 {
805 
806  // C++ implementation of the standard deviation class interval algorithm
807  // as implemented in the 'classInt' package available for the R statistical
808  // prgramming language.
809 
810  // Returns breaks based on '_calcPrettyBreaks' of the centred and scaled
811  // values of 'values', and may have a number of classes different from 'classes'.
812 
813  // If there are no values to process: bail out
814  if ( !values.count() )
815  return QList<double>();
816 
817  double mean = 0.0;
818  double stdDev = 0.0;
819  int n = values.count();
820  double minimum = values[0];
821  double maximum = values[0];
822 
823  for ( int i = 0; i < n; i++ )
824  {
825  mean += values[i];
826  minimum = qMin( values[i], minimum ); // could use precomputed max and min
827  maximum = qMax( values[i], maximum ); // but have to go through entire list anyway
828  }
829  mean = mean / ( double ) n;
830 
831  double sd = 0.0;
832  for ( int i = 0; i < n; i++ )
833  {
834  sd = values[i] - mean;
835  stdDev += sd * sd;
836  }
837  stdDev = sqrt( stdDev / n );
838 
839  QList<double> breaks = _calcPrettyBreaks(( minimum - mean ) / stdDev, ( maximum - mean ) / stdDev, classes );
840  for ( int i = 0; i < breaks.count(); i++ )
841  {
842  labels.append( breaks[i] );
843  breaks[i] = ( breaks[i] * stdDev ) + mean;
844  }
845 
846  return breaks;
847 } // _calcStdDevBreaks

849 static QList<double> _calcJenksBreaks( QList<double> values, int classes,
850  double minimum, double maximum,
851  int maximumSize = 1000 )
852 {
853  // Jenks Optimal (Natural Breaks) algorithm
854  // Based on the Jenks algorithm from the 'classInt' package available for
855  // the R statistical prgramming language, and from Python code from here:
856  // http://danieljlewis.org/2010/06/07/jenks-natural-breaks-algorithm-in-python/
857  // and is based on a JAVA and Fortran code available here:
858  // https://stat.ethz.ch/pipermail/r-sig-geo/2006-March/000811.html
859 
860  // Returns class breaks such that classes are internally homogeneous while
861  // assuring heterogeneity among classes.
862 
863  if ( !values.count() )
864  return QList<double>();
865 
866  if ( classes <= 1 )
867  {
868  return QList<double>() << maximum;
869  }
870 
871  if ( classes >= values.size() )
872  {
873  return values;
874  }
875 
876  QVector<double> sample;
877 
878  // if we have lots of values, we need to take a random sample
879  if ( values.size() > maximumSize )
880  {
881  // for now, sample at least maximumSize values or a 10% sample, whichever
882  // is larger. This will produce a more representative sample for very large
883  // layers, but could end up being computationally intensive...
884 
885  sample.resize( qMax( maximumSize, values.size() / 10 ) );
886 
887  QgsDebugMsg( QString( "natural breaks (jenks) sample size: %1" ).arg( sample.size() ) );
888  QgsDebugMsg( QString( "values:%1" ).arg( values.size() ) );
889 
890  sample[ 0 ] = minimum;
891  sample[ 1 ] = maximum;;
892  for ( int i = 2; i < sample.size(); i++ )
893  {
894  // pick a random integer from 0 to n
895  double r = qrand();
896  int j = floor( r / RAND_MAX * ( values.size() - 1 ) );
897  sample[ i ] = values[ j ];
898  }
899  }
900  else
901  {
902  sample = values.toVector();
903  }
904 
905  int n = sample.size();
906 
907  // sort the sample values
908  qSort( sample );
909 
910  QVector< QVector<int> > matrixOne( n + 1 );
911  QVector< QVector<double> > matrixTwo( n + 1 );
912 
913  for ( int i = 0; i <= n; i++ )
914  {
915  matrixOne[i].resize( classes + 1 );
916  matrixTwo[i].resize( classes + 1 );
917  }
918 
919  for ( int i = 1; i <= classes; i++ )
920  {
921  matrixOne[0][i] = 1;
922  matrixOne[1][i] = 1;
923  matrixTwo[0][i] = 0.0;
924  for ( int j = 2; j <= n; j++ )
925  {
926  matrixTwo[j][i] = std::numeric_limits<double>::max();
927  }
928  }
929 
930  for ( int l = 2; l <= n; l++ )
931  {
932  double s1 = 0.0;
933  double s2 = 0.0;
934  int w = 0;
935 
936  double v = 0.0;
937 
938  for ( int m = 1; m <= l; m++ )
939  {
940  int i3 = l - m + 1;
941 
942  double val = sample[ i3 - 1 ];
943 
944  s2 += val * val;
945  s1 += val;
946  w++;
947 
948  v = s2 - ( s1 * s1 ) / ( double ) w;
949  int i4 = i3 - 1;
950  if ( i4 != 0 )
951  {
952  for ( int j = 2; j <= classes; j++ )
953  {
954  if ( matrixTwo[l][j] >= v + matrixTwo[i4][j - 1] )
955  {
956  matrixOne[l][j] = i4;
957  matrixTwo[l][j] = v + matrixTwo[i4][j - 1];
958  }
959  }
960  }
961  }
962  matrixOne[l][1] = 1;
963  matrixTwo[l][1] = v;
964  }
965 
966  QVector<double> breaks( classes );
967  breaks[classes-1] = sample[n-1];
968 
969  for ( int j = classes, k = n; j >= 2; j-- )
970  {
971  int id = matrixOne[k][j] - 1;
972  breaks[j - 2] = sample[id];
973  k = matrixOne[k][j] - 1;
974  }
975 
976  return breaks.toList();
977 } //_calcJenksBreaks
