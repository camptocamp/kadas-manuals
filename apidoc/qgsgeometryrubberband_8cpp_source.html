70 void QgsGeometryRubberBand::paint( QPainter* painter )
71 {
72  if ( !mGeometry || !painter )
73  {
74  return;
75  }
76 
77  painter->save();
78  painter->translate( -pos() );
79 
80  if ( mGeometryType == QGis::Polygon )
81  {
82  painter->setBrush( mBrush );
83  }
84  else
85  {
86  painter->setBrush( Qt::NoBrush );
87  }
88  painter->setPen( mPen );
89 
90 
91  QgsAbstractGeometryV2* paintGeom = mGeometry->clone();
92 
93  QTransform mapToCanvas = mMapCanvas->getCoordinateTransform()->transform();
94  QTransform translationOffset = QTransform::fromTranslate( mTranslationOffset[0], mTranslationOffset[1] );
95 
96  paintGeom->transform( translationOffset * mapToCanvas );
97  paintGeom->draw( *painter );
98 
99  //draw vertices
100  QgsVertexId vertexId;
101  QgsPointV2 vertex;
102  while ( paintGeom->nextVertex( vertexId, vertex ) )
103  {
104  if ( !mHiddenNodes.contains( vertexId ) )
105  {
106  drawVertex( painter, vertex.x(), vertex.y() );
107  }
108  }
109 
110  delete paintGeom;
111  painter->restore();
112 }

114 void QgsGeometryRubberBand::drawVertex( QPainter* p, double x, double y )
115 {
116  qreal s = ( mIconSize - 1 ) / 2;
117  p->save();
118  p->setPen( mIconPen );
119  p->setBrush( mIconBrush );
120 
121  switch ( mIconType )
122  {
123  case ICON_NONE:
124  break;
125 
126  case ICON_CROSS:
127  p->drawLine( QLineF( x - s, y, x + s, y ) );
128  p->drawLine( QLineF( x, y - s, x, y + s ) );
129  break;
130 
131  case ICON_X:
132  p->drawLine( QLineF( x - s, y - s, x + s, y + s ) );
133  p->drawLine( QLineF( x - s, y + s, x + s, y - s ) );
134  break;
135 
136  case ICON_BOX:
137  p->drawLine( QLineF( x - s, y - s, x + s, y - s ) );
138  p->drawLine( QLineF( x + s, y - s, x + s, y + s ) );
139  p->drawLine( QLineF( x + s, y + s, x - s, y + s ) );
140  p->drawLine( QLineF( x - s, y + s, x - s, y - s ) );
141  break;
142 
143  case ICON_FULL_BOX:
144  p->drawRect( x - s, y - s, mIconSize, mIconSize );
145  break;
146 
147  case ICON_CIRCLE:
148  p->drawEllipse( x - s, y - s, mIconSize, mIconSize );
149  break;
150 
151  case ICON_TRIANGLE:
152  p->drawLine( QLineF( x - s, y + s, x + s, y + s ) );
153  p->drawLine( QLineF( x + s, y + s, x, y - s ) );
154  p->drawLine( QLineF( x, y - s, x - s, y + s ) );
155  break;
156 
157  case ICON_FULL_TRIANGLE:
158  p->drawPolygon( QPolygonF() <<
159  QPointF( x - s, y + s ) <<
160  QPointF( x + s, y + s ) <<
161  QPointF( x, y - s ) );
162  break;
163  }
164  p->restore();
165 }

234 bool QgsGeometryRubberBand::contains( const QgsPoint& p, double tol ) const
235 {
236  if ( !mGeometry )
237  {
238  return false;
239  }
240 
241  QgsPolygonV2 filterRect;
242  QgsLineStringV2* exterior = new QgsLineStringV2();
243  exterior->setPoints( QList<QgsPointV2>()
244  << QgsPointV2( p.x() - tol, p.y() - tol )
245  << QgsPointV2( p.x() + tol, p.y() - tol )
246  << QgsPointV2( p.x() + tol, p.y() + tol )
247  << QgsPointV2( p.x() - tol, p.y() + tol )
248  << QgsPointV2( p.x() - tol, p.y() - tol ) );
249  filterRect.setExteriorRing( exterior );
250 
251  QgsGeometryEngine* geomEngine = QgsGeometry::createGeometryEngine( mGeometry );
252  bool intersects = geomEngine->intersects( filterRect );
253  delete geomEngine;
254  return intersects;

389 {

405 void QgsGeometryRubberBand::measureGeometry( QgsAbstractGeometryV2 *geometry, int part )
406 {
407  QStringList measurements;
408  if ( mMeasurer )
409  {
410  foreach ( const Measurer::Measurement& measurement, mMeasurer->measure( mMeasurementMode, part, geometry, mPartMeasurements ) )
411  {
412  QString value = "";
413  if ( measurement.type == Measurer::Measurement::Length )
414  {
415  value = formatMeasurement( measurement.value, false );
416  }
417  else if ( measurement.type == Measurer::Measurement::Area )
418  {
419  value = formatMeasurement( measurement.value, true );
420  }
421  else if ( measurement.type == Measurer::Measurement::Angle )
422  {
423  value = formatAngle( measurement.value );
424  }
425  if ( !measurement.label.isEmpty() )
426  {
427  value = QString( "%1: %2" ).arg( measurement.label ).arg( value );
428  }
429  measurements.append( value );
430  }
431  }
432  else
433  {
434  switch ( mMeasurementMode )
435  {
436  case MEASURE_LINE:
437  if ( dynamic_cast<QgsCurveV2*>( geometry ) )
438  {
439  double length = mDa.measureLine( static_cast<QgsCurveV2*>( geometry ) );
440  mPartMeasurements.append( length );
441  measurements.append( formatMeasurement( length, false ) );
442  }
443  break;
444  case MEASURE_LINE_AND_SEGMENTS:
445  if ( dynamic_cast<QgsCurveV2*>( geometry ) )
446  {
447  QgsVertexId vid;
448  QgsPointV2 p;
449  QList<QgsPointV2> points;
450  while ( geometry->nextVertex( vid, p ) )
451  {
452  if ( !mHiddenNodes.contains( vid ) )
453  {
454  points.append( p );
455  }
456  }
457  double totLength = 0;
458  for ( int i = 0, n = points.size() - 1; i < n; ++i )
459  {
460  QgsPoint p1( points[i].x(), points[i].y() );
461  QgsPoint p2( points[i+1].x(), points[i+1].y() );
462  double segmentLength = mDa.measureLine( p1, p2 );
463  totLength += segmentLength;
464  if ( n > 1 )
465  {
466  addMeasurements( QStringList() << formatMeasurement( segmentLength, false ), QgsPointV2( 0.5 * ( p1.x() + p2.x() ), 0.5 * ( p1.y() + p2.y() ) ) );
467  }
468  }
469  if ( !points.isEmpty() )
470  {
471  mPartMeasurements.append( totLength );
472  QString totLengthStr = QApplication::translate( "QgsGeometryRubberBand", "Tot.: %1" ).arg( formatMeasurement( totLength, false ) );
473  addMeasurements( QStringList() << totLengthStr, points.last() );
474  }
475  }
476  break;
477  case MEASURE_AZIMUTH:
478  if ( dynamic_cast<QgsCurveV2*>( geometry ) )
479  {
480  QgsVertexId vid;
481  QgsPointV2 p;
482  QList<QgsPointV2> points;
483  while ( geometry->nextVertex( vid, p ) )
484  {
485  if ( !mHiddenNodes.contains( vid ) )
486  {
487  points.append( p );
488  }
489  }
490  for ( int i = 0, n = points.size() - 1; i < n; ++i )
491  {
492  QgsPoint p1( points[i].x(), points[i].y() );
493  QgsPoint p2( points[i+1].x(), points[i+1].y() );
494 
495  double angle = 0;
496  if ( mAzimuthNorth == AZIMUTH_NORTH_GEOGRAPHIC )
497  {
498  angle = mDa.bearing( p1, p2 );
499  }
500  else
501  {
502  angle = qAtan2( p2.x() - p1.x(), p2.y() - p1.y() );
503  }
504  angle = qRound( angle * 1000 ) / 1000.;
505  angle = angle < 0 ? angle + 2 * M_PI : angle;
506  angle = angle >= 2 * M_PI ? angle - 2 * M_PI : angle;
507  mPartMeasurements.append( angle );
508  QString segmentLength = formatAngle( angle );
509  addMeasurements( QStringList() << segmentLength, QgsPointV2( 0.5 * ( p1.x() + p2.x() ), 0.5 * ( p1.y() + p2.y() ) ) );
510  }
511  }
512  break;
513  case MEASURE_ANGLE:
514  // Note: only works with circular sector geometry
515  if ( dynamic_cast<QgsCurvePolygonV2*>( geometry ) && dynamic_cast<QgsCompoundCurveV2*>( static_cast<QgsCurvePolygonV2*>( geometry )->exteriorRing() ) )
516  {
517  QgsCompoundCurveV2* curve = static_cast<QgsCompoundCurveV2*>( static_cast<QgsCurvePolygonV2*>( geometry )->exteriorRing() );
518  if ( !curve->isEmpty() )
519  {
520  if ( dynamic_cast<const QgsCircularStringV2*>( curve->curveAt( 0 ) ) )
521  {
522  const QgsCircularStringV2* circularString = static_cast<const QgsCircularStringV2*>( curve->curveAt( 0 ) );
523  if ( circularString->vertexCount() == 3 )
524  {
525  QgsPointV2 p1 = circularString->pointN( 0 );
526  QgsPointV2 p2 = circularString->pointN( 1 );
527  QgsPointV2 p3 = circularString->pointN( 2 );
528  double angle;
529  if ( p1 == p3 )
530  {
531  angle = 2 * M_PI;
532  }
533  else
534  {
535  double radius, cx, cy;
536  QgsGeometryUtils::circleCenterRadius( p1, p2, p3, radius, cx, cy );
537 
538  double azimuthOne = mDa.bearing( QgsPoint( cx, cy ), QgsPoint( p1.x(), p1.y() ) );
539  double azimuthTwo = mDa.bearing( QgsPoint( cx, cy ), QgsPoint( p3.x(), p3.y() ) );
540  azimuthOne = azimuthOne < 0 ? azimuthOne + 2 * M_PI : azimuthOne;
541  azimuthTwo = azimuthTwo < 0 ? azimuthTwo + 2 * M_PI : azimuthTwo;
542  azimuthTwo = azimuthTwo < azimuthOne ? azimuthTwo + 2 * M_PI : azimuthTwo;
543  angle = azimuthTwo - azimuthOne;
544  }
545  mPartMeasurements.append( angle );
546  QString segmentLength = formatAngle( angle );
547  addMeasurements( QStringList() << segmentLength, p2 );
548  }
549  }
550  else if ( dynamic_cast<const QgsLineStringV2*>( curve->curveAt( 0 ) ) )
551  {
552  const QgsLineStringV2* lineString = static_cast<const QgsLineStringV2*>( curve->curveAt( 0 ) );
553  if ( lineString->vertexCount() == 3 )
554  {
555  mPartMeasurements.append( 0 );
556  addMeasurements( QStringList() << formatAngle( 0 ), lineString->pointN( 1 ) );
557  }
558  }
559  }
560  }
561  break;
562  case MEASURE_POLYGON:
563  if ( dynamic_cast<QgsCurvePolygonV2*>( geometry ) )
564  {
565  double area = mDa.measurePolygon( static_cast<QgsCurvePolygonV2*>( geometry )->exteriorRing() );
566  mPartMeasurements.append( area );
567  measurements.append( formatMeasurement( area, true ) );
568  }
569  break;
570  case MEASURE_RECTANGLE:
571  if ( dynamic_cast<QgsPolygonV2*>( geometry ) )
572  {
573  double area = mDa.measurePolygon( static_cast<QgsCurvePolygonV2*>( geometry )->exteriorRing() );
574  mPartMeasurements.append( area );
575  measurements.append( formatMeasurement( area, true ) );
576  QgsCurveV2* ring = static_cast<QgsPolygonV2*>( geometry )->exteriorRing();
577  if ( ring->vertexCount() >= 4 )
578  {
579  QList<QgsPointV2> points;
580  ring->points( points );
581  QgsPoint p1( points[0].x(), points[0].y() );
582  QgsPoint p2( points[2].x(), points[2].y() );
583  QString width = formatMeasurement( mDa.measureLine( p1, QgsPoint( p2.x(), p1.y() ) ), false );
584  QString height = formatMeasurement( mDa.measureLine( p1, QgsPoint( p1.x(), p2.y() ) ), false );
585  measurements.append( QString( "(%1 x %2)" ).arg( width ).arg( height ) );
586  }
587  }
588  break;
589  case MEASURE_CIRCLE:
590  if ( dynamic_cast<QgsCurvePolygonV2*>( geometry ) )
591  {
592  QgsCurveV2* ring = static_cast<QgsCurvePolygonV2*>( geometry )->exteriorRing();
593  QgsLineStringV2* polyline = ring->curveToLine();
594  double area = mDa.measurePolygon( polyline );
595  mPartMeasurements.append( area );
596  measurements.append( formatMeasurement( area, true ) );
597  delete polyline;
598  QgsPointV2 p1 = ring->vertexAt( QgsVertexId( 0, 0, 0 ) );
599  QgsPointV2 p2 = ring->vertexAt( QgsVertexId( 0, 0, 1 ) );
600  measurements.append( QApplication::translate( "QgsGeometryRubberBand", "Radius: %1" ).arg( formatMeasurement( mDa.measureLine( QgsPoint( p1.x(), p1.y() ), QgsPoint( p2.x(), p2.y() ) ), false ) ) );
601  }
602  break;
603  case MEASURE_NONE:
604  break;
605  }
606  }
607  if ( !measurements.isEmpty() )
608  {
609  addMeasurements( measurements, geometry->centroid() );
610  }
611 }

621 QString QgsGeometryRubberBand::formatAngle( double value ) const
622 {
623  int decimals = QSettings().value( "/Qgis/measure/decimalplaces", "2" ).toInt();
624  switch ( mAngleUnit )
625  {
626  case ANGLE_DEGREES:
627  return QString( "%1 deg" ).arg( QLocale::system().toString( value * 180. / M_PI, 'f', decimals ) );
628  case ANGLE_RADIANS:
629  return QString( "%1 rad" ).arg( QLocale::system().toString( value, 'f', decimals ) );
630  case ANGLE_GRADIANS:
631  return QString( "%1 gon" ).arg( QLocale::system().toString( value * 200. / M_PI, 'f', decimals ) );
632  case ANGLE_MIL:
633  return QString( "%1 mil" ).arg( QLocale::system().toString( value * 3200. / M_PI, 'f', decimals ) );
634  }
635  return "";
636 }

638 void QgsGeometryRubberBand::addMeasurements( const QStringList& measurements, const QgsPointV2& mapPos )
639 {
640  if ( measurements.isEmpty() )
641  {
642  return;
643  }
644  QGraphicsTextItem* label = new QGraphicsTextItem();
