61 void QgsClipboard::setStoredFeatures( const QgsFeatureStore &featureStore )
62 {
63  // Also store plaintext version
64  QSettings settings;
65  bool copyWKT = settings.value( "Qgis/copyGeometryAsWKT", true ).toBool();
66 
67  QStringList textLines;
68  QStringList textFields;
69 
70  // first column names
71  if ( copyWKT )
72  {
73  textFields += "wkt_geom";
74  }
75  for ( int i = 0, n = featureStore.fields().count(); i < n; ++i )
76  {
77  textFields.append( featureStore.fields().at( i ).name() );
78  }
79  textLines.append( textFields.join( "\t" ) );
80  textFields.clear();
81 
82  // then the field contents
83  foreach ( const QgsFeature& feature, featureStore.features() )
84  {
85  if ( copyWKT )
86  {
87  if ( feature.geometry() )
88  textFields.append( feature.geometry()->exportToWkt() );
89  else
90  textFields.append( settings.value( "Qgis/nullValue", "NULL" ).toString() );
91  }
92  foreach ( const QVariant& attr, feature.attributes() )
93  {
94  textFields.append( attr.toString() );
95  }
96  textLines.append( textFields.join( "\t" ) );
97  textFields.clear();
98  }
99  QMimeData* mimeData = new QMimeData();
100  mimeData->setData( "text/plain", textLines.join( "\n" ).toLocal8Bit() );
101  QApplication::clipboard()->setMimeData( mimeData );
102 
103  // Make QApplication::clipboard() emit the dataChanged signal
104  QApplication::processEvents( QEventLoop::ExcludeUserInputEvents );
105 
106  // After plaintext version, because dataChanged clears the internal feature store
107  mFeatureStore = featureStore;
108 }
