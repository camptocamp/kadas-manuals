82 bool QgsRasterResampleFilter::setInput( QgsRasterInterface* input )
83 {
84  QgsDebugMsg( "Entered" );
85 
86  // Resampler can only work with single band ARGB32_Premultiplied
87  if ( !input )
88  {
89  QgsDebugMsg( "No input" );
90  return false;
91  }
92 
93  if ( !mOn )
94  {
95  // In off mode we can connect to anything
96  QgsDebugMsg( "OK" );
97  mInput = input;
98  return true;
99  }
100 
101  if ( input->bandCount() < 1 )
102  {
103  QgsDebugMsg( "No input band" );
104  return false;
105  }
106 
107  if ( input->dataType( 1 ) != QGis::ARGB32_Premultiplied &&
108  input->dataType( 1 ) != QGis::ARGB32 )
109  {
110  QgsDebugMsg( "Unknown input data type" );
111  return false;
112  }
113 
114  mInput = input;
115  QgsDebugMsg( "OK" );
116  return true;
117 }

136 QgsRasterBlock* QgsRasterResampleFilter::block2( int bandNo, QgsRectangle const & extent, int width, int height, QgsRasterBlockFeedback *feedback )
137 {
138  Q_UNUSED( bandNo );
139  QgsDebugMsg( QString( "width = %1 height = %2 extent = %3" ).arg( width ).arg( height ).arg( extent.toString() ) );
140  QgsRasterBlock *outputBlock = new QgsRasterBlock();
141  if ( !mInput ) return outputBlock;
142 
143  double oversampling = 1.0; // approximate global oversampling factor
144 
145  if ( mZoomedInResampler || mZoomedOutResampler )
146  {
147  QgsRasterDataProvider *provider = dynamic_cast<QgsRasterDataProvider*>( mInput->srcInput() );
148  if ( provider && ( provider->capabilities() & QgsRasterDataProvider::Size ) )
149  {
150  double xRes = extent.width() / width;
151  double providerXRes = provider->extent().width() / provider->xSize();
152  double pixelRatio = xRes / providerXRes;
153  oversampling = ( pixelRatio > mMaxOversampling ) ? mMaxOversampling : pixelRatio;
154  QgsDebugMsg( QString( "xRes = %1 providerXRes = %2 pixelRatio = %3 oversampling = %4" ).arg( xRes ).arg( providerXRes ).arg( pixelRatio ).arg( oversampling ) );
155  }
156  else
157  {
158  // We don't know exact data source resolution (WMS) so we expect that
159  // server data have higher resolution (which is not always true) and use
160  // mMaxOversampling
161  oversampling = mMaxOversampling;
162  }
163  }
164 
165  QgsDebugMsg( QString( "oversampling %1" ).arg( oversampling ) );
166 
167  int bandNumber = 1;
168 
169  // Do no oversampling if no resampler for zoomed in / zoomed out (nearest neighbour)
170  // We do mZoomedInResampler if oversampling == 1 (otherwise for example reprojected
171  // zoom in rasters are never resampled because projector limits resolution.
172  if ((( oversampling < 1.0 || qgsDoubleNear( oversampling, 1.0 ) ) && !mZoomedInResampler ) || ( oversampling > 1.0 && !mZoomedOutResampler ) )
173  {
174  QgsDebugMsg( "No oversampling." );
175  delete outputBlock;
176  return mInput->block2( bandNumber, extent, width, height, feedback );
177  }
178 
179  //effective oversampling factors are different to global one because of rounding
180  double oversamplingX = (( double )width * oversampling ) / width;
181  double oversamplingY = (( double )height * oversampling ) / height;
182 
183  // TODO: we must also increase the extent to get correct result on borders of parts
184 
185  int resWidth = width * oversamplingX;
186  int resHeight = height * oversamplingY;
187 
188  QgsRasterBlock *inputBlock = mInput->block2( bandNumber, extent, resWidth, resHeight, feedback );
189  if ( !inputBlock || inputBlock->isEmpty() )
190  {
191  QgsDebugMsg( "No raster data!" );
192  delete inputBlock;
193  return outputBlock;
194  }
195 
196  if ( !outputBlock->reset( QGis::ARGB32_Premultiplied, width, height ) )
197  {
198  delete inputBlock;
199  return outputBlock;
200  }
201 
202  //resample image
203  QImage img = inputBlock->image();
204 
205  QImage dstImg = QImage( width, height, QImage::Format_ARGB32_Premultiplied );
206 
207  if ( mZoomedInResampler && ( oversamplingX < 1.0 || qgsDoubleNear( oversampling, 1.0 ) ) )
208  {
209  QgsDebugMsg( "zoomed in resampling" );
210  mZoomedInResampler->resample( img, dstImg );
211  }
212  else if ( mZoomedOutResampler && oversamplingX > 1.0 )
213  {
214  QgsDebugMsg( "zoomed out resampling" );
215  mZoomedOutResampler->resample( img, dstImg );
216  }
217  else
218  {
219  // Should not happen
220  QgsDebugMsg( "Unexpected resampling" );
221  dstImg = img.scaled( width, height );
222  }
223 
224  outputBlock->setImage( &dstImg );
225 
226  delete inputBlock;
227  return outputBlock; // No resampling
228 }

251 void QgsRasterResampleFilter::readXML( const QDomElement& filterElem )
252 {
253  if ( filterElem.isNull() )
254  {
255  return;
256  }
257 
258  mMaxOversampling = filterElem.attribute( "maxOversampling", "2.0" ).toDouble();
259 
260  QString zoomedInResamplerType = filterElem.attribute( "zoomedInResampler" );
261  if ( zoomedInResamplerType == "bilinear" )
262  {
263  mZoomedInResampler = new QgsBilinearRasterResampler();
264  }
265  else if ( zoomedInResamplerType == "cubic" )
266  {
267  mZoomedInResampler = new QgsCubicRasterResampler();
268  }
269 
270  QString zoomedOutResamplerType = filterElem.attribute( "zoomedOutResampler" );
271  if ( zoomedOutResamplerType == "bilinear" )
272  {
273  mZoomedOutResampler = new QgsBilinearRasterResampler();
274  }
