241 int QgsVectorLayerEditUtils::splitFeatures( const QList<QgsPoint>& splitLine, bool topologicalEditing )
242 {
243  if ( !L->hasGeometryType() )
244  return 4;
245 
246  QgsFeatureList newFeatures; //store all the newly created features
247  double xMin, yMin, xMax, yMax;
248  QgsRectangle bBox; //bounding box of the split line
249  int returnCode = 0;
250  int splitFunctionReturn; //return code of QgsGeometry::splitGeometry
251  int numberOfSplittedFeatures = 0;
252 
253  QgsFeatureIterator features;
254  const QgsFeatureIds selectedIds = L->selectedFeaturesIds();
255 
256  if ( selectedIds.size() > 0 ) //consider only the selected features if there is a selection
257  {
258  features = L->selectedFeaturesIterator();
259  }
260  else //else consider all the feature that intersect the bounding box of the split line
261  {
262  if ( boundingBoxFromPointList( splitLine, xMin, yMin, xMax, yMax ) == 0 )
263  {
264  bBox.setXMinimum( xMin ); bBox.setYMinimum( yMin );
265  bBox.setXMaximum( xMax ); bBox.setYMaximum( yMax );
266  }
267  else
268  {
269  return 1;
270  }
271 
272  if ( bBox.isEmpty() )
273  {
274  //if the bbox is a line, try to make a square out of it
275  if ( bBox.width() == 0.0 && bBox.height() > 0 )
276  {
277  bBox.setXMinimum( bBox.xMinimum() - bBox.height() / 2 );
278  bBox.setXMaximum( bBox.xMaximum() + bBox.height() / 2 );
279  }
280  else if ( bBox.height() == 0.0 && bBox.width() > 0 )
281  {
282  bBox.setYMinimum( bBox.yMinimum() - bBox.width() / 2 );
283  bBox.setYMaximum( bBox.yMaximum() + bBox.width() / 2 );
284  }
285  else
286  {
287  //If we have a single point, we still create a non-null box
288  double bufferDistance = 0.000001;
289  if ( L->crs().geographicFlag() )
290  bufferDistance = 0.00000001;
291  bBox.setXMinimum( bBox.xMinimum() - bufferDistance );
292  bBox.setXMaximum( bBox.xMaximum() + bufferDistance );
293  bBox.setYMinimum( bBox.yMinimum() - bufferDistance );
294  bBox.setYMaximum( bBox.yMaximum() + bufferDistance );
295  }
296  }
297 
298  features = L->getFeatures( QgsFeatureRequest().setFilterRect( bBox ).setFlags( QgsFeatureRequest::ExactIntersect ) );
299  }
300 
301  QgsFeature feat;
302  while ( features.nextFeature( feat ) )
303  {
304  if ( !feat.geometry() )
305  {
306  continue;
307  }
308  QList<QgsGeometry*> newGeometries;
309  QList<QgsPoint> topologyTestPoints;
310  QgsGeometry* newGeometry = 0;
311  splitFunctionReturn = feat.geometry()->splitGeometry( splitLine, newGeometries, topologicalEditing, topologyTestPoints );
312  if ( splitFunctionReturn == 0 )
313  {
314  //change this geometry
315  L->editBuffer()->changeGeometry( feat.id(), feat.geometry() );
316 
317  //insert new features
318  for ( int i = 0; i < newGeometries.size(); ++i )
319  {
320  newGeometry = newGeometries.at( i );
321  QgsFeature newFeature;
322  newFeature.setGeometry( newGeometry );
323 
324  //use default value where possible for primary key (e.g. autoincrement),
325  //and use the value from the original (split) feature if not primary key
326  QgsAttributes newAttributes = feat.attributes();
327  foreach ( int pkIdx, L->dataProvider()->pkAttributeIndexes() )
328  {
329  const QVariant defaultValue = L->dataProvider()->defaultValue( pkIdx );
330  if ( !defaultValue.isNull() )
331  {
332  newAttributes[ pkIdx ] = defaultValue;
333  }
334  else //try with NULL
335  {
336  newAttributes[ pkIdx ] = QVariant();
337  }
338  }
339 
340  newFeature.setAttributes( newAttributes );
341 
342  newFeatures.append( newFeature );
343  }
344 
345  if ( topologicalEditing )
346  {
347  QList<QgsPoint>::const_iterator topol_it = topologyTestPoints.constBegin();
348  for ( ; topol_it != topologyTestPoints.constEnd(); ++topol_it )
349  {
350  addTopologicalPoints( *topol_it );
351  }
352  }
353  ++numberOfSplittedFeatures;
354  }
355  else if ( splitFunctionReturn > 1 ) //1 means no split but also no error
356  {
357  returnCode = splitFunctionReturn;
358  }
359  }
360 
361  if ( numberOfSplittedFeatures == 0 && selectedIds.size() > 0 )
362  {
363  //There is a selection but no feature has been split.
364  //Maybe user forgot that only the selected features are split
365  returnCode = 4;
366  }
367 
368 
369  //now add the new features to this vectorlayer
370  L->editBuffer()->addFeatures( newFeatures );
371 
372  return returnCode;
373 }

507 int QgsVectorLayerEditUtils::addTopologicalPoints( QgsGeometry* geom )
508 {
509  if ( !L->hasGeometryType() )
510  return 1;
511 
512  if ( !geom )
513  {
514  return 1;
515  }
516 
517  int returnVal = 0;
518 
519  QGis::WkbType wkbType = geom->wkbType();
520 
521  switch ( wkbType )
522  {
523  //line
524  case QGis::WKBLineString25D:
525  case QGis::WKBLineString:
526  {
527  QgsPolyline theLine = geom->asPolyline();
528  QgsPolyline::const_iterator line_it = theLine.constBegin();
529  for ( ; line_it != theLine.constEnd(); ++line_it )
530  {
531  if ( addTopologicalPoints( *line_it ) != 0 )
532  {
533  returnVal = 2;
534  }
535  }
536  break;
537  }
538 
539  //multiline
540  case QGis::WKBMultiLineString25D:
541  case QGis::WKBMultiLineString:
542  {
543  QgsMultiPolyline theMultiLine = geom->asMultiPolyline();
544  QgsPolyline currentPolyline;
545 
546  for ( int i = 0; i < theMultiLine.size(); ++i )
547  {
548  QgsPolyline::const_iterator line_it = currentPolyline.constBegin();
549  for ( ; line_it != currentPolyline.constEnd(); ++line_it )
550  {
551  if ( addTopologicalPoints( *line_it ) != 0 )
552  {
553  returnVal = 2;
554  }
555  }
556  }
557  break;
558  }
559 
560  //polygon
561  case QGis::WKBPolygon25D:
562  case QGis::WKBPolygon:
563  {
564  QgsPolygon thePolygon = geom->asPolygon();
565  QgsPolyline currentRing;
566 
567  for ( int i = 0; i < thePolygon.size(); ++i )
568  {
569  currentRing = thePolygon.at( i );
570  QgsPolyline::const_iterator line_it = currentRing.constBegin();
571  for ( ; line_it != currentRing.constEnd(); ++line_it )
572  {
573  if ( addTopologicalPoints( *line_it ) != 0 )
574  {
575  returnVal = 2;
576  }
577  }
578  }
579  break;
580  }
581 
582  //multipolygon
583  case QGis::WKBMultiPolygon25D:
584  case QGis::WKBMultiPolygon:
585  {
586  QgsMultiPolygon theMultiPolygon = geom->asMultiPolygon();
587  QgsPolygon currentPolygon;
588  QgsPolyline currentRing;
589 
590  for ( int i = 0; i < theMultiPolygon.size(); ++i )
591  {
592  currentPolygon = theMultiPolygon.at( i );
593  for ( int j = 0; j < currentPolygon.size(); ++j )
594  {
595  currentRing = currentPolygon.at( j );
596  QgsPolyline::const_iterator line_it = currentRing.constBegin();
597  for ( ; line_it != currentRing.constEnd(); ++line_it )
598  {
599  if ( addTopologicalPoints( *line_it ) != 0 )
600  {
601  returnVal = 2;
602  }
603  }
604  }
605  }
606  break;
607  }
608  default:
609  break;
610  }
611  return returnVal;
612 }

615 int QgsVectorLayerEditUtils::addTopologicalPoints( const QgsPoint& p )
616 {
617  if ( !L->hasGeometryType() )
618  return 1;
619 
620  QMultiMap<double, QgsSnappingResult> snapResults; //results from the snapper object
621  //we also need to snap to vertex to make sure the vertex does not already exist in this geometry
622  QMultiMap<double, QgsSnappingResult> vertexSnapResults;
623 
624  QList<QgsSnappingResult> filteredSnapResults; //we filter out the results that are on existing vertices
625 
626  //work with a tolerance because coordinate projection may introduce some rounding
627  double threshold = 0.0000001;
628  if ( L->crs().mapUnits() == QGis::Meters )
629  {
630  threshold = 0.001;
631  }
632  else if ( L->crs().mapUnits() == QGis::Feet )
633  {
634  threshold = 0.0001;
635  }
636 
637 
638  if ( L->snapWithContext( p, threshold, snapResults, QgsSnapper::SnapToSegment ) != 0 )
639  {
640  return 2;
641  }
642 
643  QMultiMap<double, QgsSnappingResult>::const_iterator snap_it = snapResults.constBegin();
644  QMultiMap<double, QgsSnappingResult>::const_iterator vertex_snap_it;
645  for ( ; snap_it != snapResults.constEnd(); ++snap_it )
646  {
647  //test if p is already a vertex of this geometry. If yes, don't insert it
648  bool vertexAlreadyExists = false;
649  if ( L->snapWithContext( p, threshold, vertexSnapResults, QgsSnapper::SnapToVertex ) != 0 )
650  {
651  continue;
652  }
653 
654  vertex_snap_it = vertexSnapResults.constBegin();
655  for ( ; vertex_snap_it != vertexSnapResults.constEnd(); ++vertex_snap_it )
656  {
657  if ( snap_it.value().snappedAtGeometry == vertex_snap_it.value().snappedAtGeometry )
658  {
659  vertexAlreadyExists = true;
660  }
661  }
662 
663  if ( !vertexAlreadyExists )
664  {
665  filteredSnapResults.push_back( *snap_it );
666  }
667  }
668  insertSegmentVerticesForSnap( filteredSnapResults );
669  return 0;
670 }

699 int QgsVectorLayerEditUtils::boundingBoxFromPointList( const QList<QgsPoint>& list, double& xmin, double& ymin, double& xmax, double& ymax ) const
700 {
701  if ( list.size() < 1 )
702  {
703  return 1;
704  }
705 
706  xmin = std::numeric_limits<double>::max();
707  xmax = -std::numeric_limits<double>::max();
708  ymin = std::numeric_limits<double>::max();
709  ymax = -std::numeric_limits<double>::max();
710 
711  for ( QList<QgsPoint>::const_iterator it = list.constBegin(); it != list.constEnd(); ++it )
712  {
713  if ( it->x() < xmin )
714  {
715  xmin = it->x();
716  }
717  if ( it->x() > xmax )
718  {
719  xmax = it->x();
720  }
721  if ( it->y() < ymin )
722  {
