90 void QgsFloatingInputWidget::removeInputField( int idx )
91 {
92  if ( mFocusedInput == mInputFields.at( idx ) )
93  {
94  int n = mInputFields.size();
95  int nextIdx = ( n + mInputFields.indexOf( mFocusedInput ) - 1 ) % n;
96  for ( int i = 0; i < n && mInputFields[nextIdx]->isHidden(); ++i )
97  {
98  nextIdx = ( nextIdx - 1 ) % n;
99  }
100  setFocusedInputField( mInputFields[nextIdx] );
101  }
102  mInputFields.removeAt( idx );
103  if ( mInitiallyFocusedInput >= idx )
104  {
105  mInitiallyFocusedInput = qMax( 0, mInitiallyFocusedInput - 1 );
106  }
107  // Because re-creating the layout is actually easier than just deleting a row...
108  QGridLayout* oldLayout = static_cast<QGridLayout*>( layout() );
109  QGridLayout* newLayout = new QGridLayout();
110  newLayout->setContentsMargins( 2, 2, 2, 2 );
111  newLayout->setSpacing( 1 );
112  newLayout->setSizeConstraint( QLayout::SetFixedSize );
113  for ( int row = 0, newrow = 0, n = oldLayout->rowCount(); row < n; ++row )
114  {
115  QLayoutItem* item0 = oldLayout->itemAtPosition( row, 0 );
116  QLayoutItem* item1 = oldLayout->itemAtPosition( row, 1 );
117  if ( row != idx )
118  {
119  if ( item0 && item1 )
120  {
121  oldLayout->removeItem( item0 );
122  oldLayout->removeItem( item1 );
123  newLayout->addWidget( item0->widget(), newrow, 0 );
124  newLayout->addWidget( item1->widget(), newrow, 1 );
125  ++newrow;
126  }
127  }
128  else
129  {
130  delete item0->widget();
131  delete item1->widget();
132  }
133  }
134  delete oldLayout;
135  setLayout( newLayout );
136 }

189 void QgsFloatingInputWidget::adjustCursorAndExtent( const QgsPoint& geoPos )
190 {
191  // If position is not within visible extent, center map there
192  if ( !mCanvas->mapSettings().visibleExtent().contains( geoPos ) )
193  {
194  QgsRectangle rect = mCanvas->mapSettings().visibleExtent();
195  rect = QgsRectangle( geoPos.x() - 0.5 * rect.width(), geoPos.y() - 0.5 * rect.height(), geoPos.x() + 0.5 * rect.width(), geoPos.y() + 0.5 * rect.height() );
196  mCanvas->setExtent( rect );
197  mCanvas->refresh();
198  }
199  // Then, move cursor to corresponding screen position and simulate move event
200  double x = geoPos.x(), y = geoPos.y();
201  mCanvas->getCoordinateTransform()->transformInPlace( x, y );
202  QPoint p( qRound( x ), qRound( y ) );
203  QCursor::setPos( mCanvas->mapToGlobal( p ) );
204  move( p.x(), p.y() + 20 );
205 }

214 void QgsFloatingInputWidget::keyPressEvent( QKeyEvent *ev )
215 {
216  // Override tab handling to ensure only the input fields inside the widget receive focus
217  if ( ev->key() == Qt::Key_Tab || ev->key() == Qt::Key_Down )
218  {
219  int n = mInputFields.size();
220  int nextIdx = ( mInputFields.indexOf( mFocusedInput ) + 1 ) % n;
221  for ( int i = 0; i < n && mInputFields[nextIdx]->isHidden(); ++i )
222  {
223  nextIdx = ( nextIdx + 1 ) % n;
224  }
225  setFocusedInputField( mInputFields[nextIdx] );
226  ev->accept();
227  }
228  else if ( ev->key() == Qt::Key_Backtab || ev->key() == Qt::Key_Up )
229  {
230  int n = mInputFields.size();
231  int nextIdx = ( n + mInputFields.indexOf( mFocusedInput ) - 1 ) % n;
232  for ( int i = 0; i < n && mInputFields[nextIdx]->isHidden(); ++i )
233  {
234  nextIdx = ( nextIdx - 1 ) % n;
235  }
236  setFocusedInputField( mInputFields[nextIdx] );
237  ev->accept();
238  }
239  else
240  {
241  QWidget::keyPressEvent(( ev ) );
242  }
