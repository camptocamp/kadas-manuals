224 {
225  QRegExp regexp;
226  regexp.setMinimal( true );
227  QString safeName( aUri );
228  if ( aUri.contains( " password=" ) )
229  {
230  regexp.setPattern( " password=.* " );
231  safeName.replace( regexp, " " );
232  }
233  else if ( aUri.contains( ",password=" ) )
234  {
235  regexp.setPattern( ",password=.*," );
236  safeName.replace( regexp, "," );
237  }
238  else if ( aUri.contains( "IDB:" ) )
239  {
240  regexp.setPattern( " pass=.* " );
241  safeName.replace( regexp, " " );
242  }
243  else if (( aUri.contains( "OCI:" ) )
244  || ( aUri.contains( "ODBC:" ) ) )
245  {
246  regexp.setPattern( "/.*@" );
247  safeName.replace( regexp, "/@" );
248  }
249  else if ( aUri.contains( "SDE:" ) )
250  {
251  QStringList strlist = aUri.split( "," );
252  safeName = strlist[0] + "," + strlist[1] + "," + strlist[2] + "," + strlist[3];

380 QString QgsDataSourceURI::getValue( const QString &uri, int &i )
381 {
382  skipBlanks( uri, i );
383 
384  // Get the parameter value
385  QString pval;
386  if ( i < uri.length() && ( uri[i] == '\'' || uri[i] == '"' ) )
387  {
388  QChar delim = uri[i];
389 
390  i++;
391 
392  // value is quoted
393  for ( ;; )
394  {
395  if ( i == uri.length() )
396  {
397  QgsDebugMsg( "unterminated quoted string in connection info string" );
398  return pval;
399  }
400 
401  if ( uri[i] == '\\' )
402  {
403  i++;
404  if ( i == uri.length() )
405  continue;
406  if ( uri[i] != delim && uri[i] != '\\' )
407  i--;
408  }
409  else if ( uri[i] == delim )
410  {
411  i++;
412  break;
413  }
414 
415  pval += uri[i++];
416  }
417  }
418  else
419  {
420  // value is not quoted
421  while ( i < uri.length() )
422  {
423  if ( uri[i].isSpace() )
424  {
425  // end of value
426  break;
427  }
428 
429  if ( uri[i] == '\\' )
430  {
431  i++;
432  if ( i == uri.length() )
433  break;
434  if ( uri[i] != '\\' && uri[i] != '\'' )
435  i--;
436  }
437 
438  pval += uri[i++];
439  }
440  }
441 
442  skipBlanks( uri, i );
443 
444  return pval;
445 }

495 QString QgsDataSourceURI::uri() const
496 {
497  QString theUri = connectionInfo();
498 
499  if ( !mKeyColumn.isEmpty() )
500  {
501  theUri += QString( " key='%1'" ).arg( escape( mKeyColumn ) );
502  }
503 
504  if ( mUseEstimatedMetadata )
505  {
506  theUri += QString( " estimatedmetadata=true" );
507  }
508 
509  if ( !mSrid.isEmpty() )
510  {
511  theUri += QString( " srid=%1" ).arg( mSrid );
512  }
513 
514 
515  if ( mWkbType != QGis::WKBUnknown && mWkbType != QGis::WKBNoGeometry )
516  {
517  theUri += " type=";
518  theUri += QgsWKBTypes::displayString( QgsWKBTypes::Type( mWkbType ) );
519  }
520 
521  if ( mSelectAtIdDisabled )
522  {
523  theUri += QString( " selectatid=false" );
524  }
525 
526  for ( QMap<QString, QString>::const_iterator it = mParams.begin(); it != mParams.end(); ++it )
527  {
528  if ( it.key().contains( "=" ) || it.key().contains( " " ) )
529  {
530  QgsDebugMsg( QString( "invalid uri parameter %1 skipped" ).arg( it.key() ) );
531  continue;
532  }
533 
534  theUri += " " + it.key() + "='" + escape( it.value() ) + "'";
535  }
536 
537  QString columnName( mGeometryColumn );
538  columnName.replace( "\\", "\\\\" );
539  columnName.replace( ")", "\\)" );
540 
541  theUri += QString( " table=%1%2 sql=%3" )
542  .arg( quotedTablename() )
543  .arg( mGeometryColumn.isNull() ? QString() : QString( " (%1)" ).arg( columnName ) )
544  .arg( mSql );
545 
546  return theUri;
547 }
