151 void QgsComposerMap::adjustExtentToItemShape( double itemWidth, double itemHeight, QgsRectangle& extent ) const
152 {
153  double itemWidthHeightRatio = itemWidth / itemHeight;
154  double newWidthHeightRatio = extent.width() / extent.height();
155 
156  if ( itemWidthHeightRatio <= newWidthHeightRatio )
157  {
158  //enlarge height of new extent, ensuring the map center stays the same
159  double newHeight = extent.width() / itemWidthHeightRatio;
160  double deltaHeight = newHeight - extent.height();
161  extent.setYMinimum( extent.yMinimum() - deltaHeight / 2 );
162  extent.setYMaximum( extent.yMaximum() + deltaHeight / 2 );
163  }
164  else
165  {
166  //enlarge width of new extent, ensuring the map center stays the same
167  double newWidth = itemWidthHeightRatio * extent.height();
168  double deltaWidth = newWidth - extent.width();
169  extent.setXMinimum( extent.xMinimum() - deltaWidth / 2 );
170  extent.setXMaximum( extent.xMaximum() + deltaWidth / 2 );
171  }
172 }

263 void QgsComposerMap::cache( void )
264 {
265  if ( mPreviewMode == Rectangle )
266  {
267  return;
268  }
269 
270  if ( mPainterJob )
271  {
272  if ( mPainterCancelWait )
273  {
274  return; // Already waiting
275  }
276  QgsDebugMsg( "Aborting composer painter job" );
277  mPainterCancelWait = true;
278  mPainterJob->cancel();
279  mPainterCancelWait = false;
280  }
281 
282  Q_ASSERT( mPainterJob == nullptr );
283  Q_ASSERT( mPainter == nullptr );
284 
285  double horizontalVScaleFactor = horizontalViewScaleFactor();
286  if ( horizontalVScaleFactor < 0 )
287  {
288  //make sure scale factor is positive
289  horizontalVScaleFactor = mLastValidViewScaleFactor > 0 ? mLastValidViewScaleFactor : 1;
290  }
291 
292  const QgsRectangle &ext = *currentMapExtent();
293  double widthMM = ext.width() * mapUnitsToMM();
294  double heightMM = ext.height() * mapUnitsToMM();
295 
296  int w = widthMM * horizontalVScaleFactor;
297  int h = heightMM * horizontalVScaleFactor;
298 
299  // limit size of image for better performance
300  if ( w > 5000 || h > 5000 )
301  {
302  if ( w > h )
303  {
304  w = 5000;
305  h = w * heightMM / widthMM;
306  }
307  else
308  {
309  h = 5000;
310  w = h * widthMM / heightMM;
311  }
312  }
313 
314  mCacheImage = QImage( w, h, QImage::Format_ARGB32 );
315 
316  // set DPI of the image
317  mCacheImage.setDotsPerMeterX( 1000 * w / widthMM );
318  mCacheImage.setDotsPerMeterY( 1000 * h / heightMM );
319 
320  if ( hasBackground() )
321  {
322  //Initially fill image with specified background color. This ensures that layers with blend modes will
323  //preview correctly
324  mCacheImage.fill( backgroundColor().rgba() );
325  }
326  else
327  {
328  //no background, but start with empty fill to avoid artifacts
329  mCacheImage.fill( QColor( 255, 255, 255, 0 ).rgba() );
330  }
331 
332  mPainter = new QPainter( &mCacheImage );
333  QgsMapSettings settings( mapSettings( ext, QSizeF( w, h ), mCacheImage.logicalDpiX() ) );
334  mPainterJob = new QgsMapRendererCustomPainterJob( settings, mPainter );
335  connect( mPainterJob, SIGNAL( finished() ), this, SLOT( painterJobFinished() ) );
336  QgsDebugMsg( "Starting new composer painter job" );
337  mPainterJob->start();
338 }

352 void QgsComposerMap::paint( QPainter* painter, const QStyleOptionGraphicsItem* itemStyle, QWidget* pWidget )
353 {
354  Q_UNUSED( pWidget );
355 
356  if ( !mComposition || !painter )
357  {
358  return;
359  }
360  if ( !shouldDrawItem() )
361  {
362  return;
363  }
364 
365  QRectF thisPaintRect = QRectF( 0, 0, QGraphicsRectItem::rect().width(), QGraphicsRectItem::rect().height() );
366  painter->save();
367  painter->setClipRect( thisPaintRect );
368 
369  if ( mComposition->plotStyle() == QgsComposition::Preview && mPreviewMode == Rectangle )
370  {
371  // Fill with background color
372  drawBackground( painter );
373  QFont messageFont( "", 12 );
374  painter->setFont( messageFont );
375  painter->setPen( QColor( 0, 0, 0, 125 ) );
376  painter->drawText( thisPaintRect, tr( "Map will be printed here" ) );
377  }
378  else if ( mComposition->plotStyle() == QgsComposition::Preview )
379  {
380  //draw cached pixmap. This function does not call cache() any more because
381  //Qt 4.4.0 and 4.4.1 have problems with recursive paintings
382  //QgsComposerMap::cache() and QgsComposerMap::update() need to be called by
383  //client functions
384 
385  //Background color is already included in cached image, so no need to draw
386 
387  double imagePixelWidth = mCacheImage.width(); //how many pixels of the image are for the map extent?
388  double scale = rect().width() / imagePixelWidth;
389 
390  painter->save();
391 
392  painter->translate( mXOffset, mYOffset );
393  painter->scale( scale, scale );
394  painter->drawImage( 0, 0, mCacheImage );
395 
396  //restore rotation
397  painter->restore();
398 
399  //draw canvas items
400  drawCanvasItems( painter, itemStyle );
401  }
402  else if ( mComposition->plotStyle() == QgsComposition::Print ||
403  mComposition->plotStyle() == QgsComposition::Postscript )
404  {
405  if ( mDrawing )
406  {
407  return;
408  }
409 
410  mDrawing = true;
411  QPaintDevice* thePaintDevice = painter->device();
412  if ( !thePaintDevice )
413  {
414  return;
415  }
416 
417  // Fill with background color
418  if ( shouldDrawPart( Background ) )
419  {
420  drawBackground( painter );
421  }
422 
423  QgsRectangle cExtent = *currentMapExtent();
424 
425  QSizeF theSize( cExtent.width() * mapUnitsToMM(), cExtent.height() * mapUnitsToMM() );
426 
427  painter->save();
428  painter->translate( mXOffset, mYOffset );
429 
430  double dotsPerMM = thePaintDevice->logicalDpiX() / 25.4;
431  theSize *= dotsPerMM; // output size will be in dots (pixels)
432  painter->scale( 1 / dotsPerMM, 1 / dotsPerMM ); // scale painter from mm to dots
433  draw( painter, cExtent, theSize, thePaintDevice->logicalDpiX() );
434 
435  //restore rotation
436  painter->restore();
437 
438  //draw canvas items
439  drawCanvasItems( painter, itemStyle );
440 
441  mDrawing = false;
442  }
443 
444  painter->setClipRect( thisPaintRect, Qt::NoClip );
445  if ( shouldDrawPart( OverviewMapExtent ) &&
446  ( mComposition->plotStyle() != QgsComposition::Preview || mPreviewMode != Rectangle ) )
447  {
448  mOverviewStack->drawItems( painter );
449  }
450  if ( shouldDrawPart( Grid ) &&
451  ( mComposition->plotStyle() != QgsComposition::Preview || mPreviewMode != Rectangle ) )
452  {
453  mGridStack->drawItems( painter );
454  }
455  if ( shouldDrawPart( Frame ) )
456  {
457  drawFrame( painter );
458  }
459  if ( isSelected() && shouldDrawPart( SelectionBoxes ) )
460  {
461  drawSelectionBoxes( painter );
462  }
463 
464  painter->restore();
465 }

638 void QgsComposerMap::zoomContent( const double factor, const QPointF point, const ZoomMode mode )
639 {
640  if ( mDrawing )
641  {
642  return;
643  }
644 
645  if ( mode == QgsComposerItem::NoZoom )
646  {
647  //do nothing
648  return;
649  }
650 
651  //find out map coordinates of position
652  double mapX = currentMapExtent()->xMinimum() + ( point.x() / rect().width() ) * ( currentMapExtent()->xMaximum() - currentMapExtent()->xMinimum() );
653  double mapY = currentMapExtent()->yMinimum() + ( 1 - ( point.y() / rect().height() ) ) * ( currentMapExtent()->yMaximum() - currentMapExtent()->yMinimum() );
654 
655  //find out new center point
656  double centerX = ( currentMapExtent()->xMaximum() + currentMapExtent()->xMinimum() ) / 2;
657  double centerY = ( currentMapExtent()->yMaximum() + currentMapExtent()->yMinimum() ) / 2;
658 
659  if ( mode != QgsComposerItem::Zoom )
660  {
661  if ( mode == QgsComposerItem::ZoomRecenter )
662  {
663  centerX = mapX;
664  centerY = mapY;
665  }
666  else if ( mode == QgsComposerItem::ZoomToPoint )
667  {
668  centerX = mapX + ( centerX - mapX ) * ( 1.0 / factor );
669  centerY = mapY + ( centerY - mapY ) * ( 1.0 / factor );
670  }
671  }
672 
673  double newIntervalX, newIntervalY;
674 
675  if ( factor > 0 )
676  {
677  newIntervalX = ( currentMapExtent()->xMaximum() - currentMapExtent()->xMinimum() ) / factor;
678  newIntervalY = ( currentMapExtent()->yMaximum() - currentMapExtent()->yMinimum() ) / factor;
679  }
680  else //no need to zoom
681  {
682  return;
683  }
684 
685  currentMapExtent()->setXMaximum( centerX + newIntervalX / 2 );
686  currentMapExtent()->setXMinimum( centerX - newIntervalX / 2 );
687  currentMapExtent()->setYMaximum( centerY + newIntervalY / 2 );
688  currentMapExtent()->setYMinimum( centerY - newIntervalY / 2 );
689 
690  if ( mAtlasDriven && mAtlasScalingMode == Fixed && mComposition->atlasMode() != QgsComposition::AtlasOff )
691  {
692  //if map is atlas controlled and set to fixed scaling mode, then scale changes should be treated as permanant
693  //and also apply to the map's original extent (see #9602)

932 void QgsComposerMap::refreshMapExtents()
933 {
934  //data defined map extents set?
935  QVariant exprVal;
936 
937  QgsRectangle newExtent = *currentMapExtent();
938  bool useDdXMin = false;
939  bool useDdXMax = false;
940  bool useDdYMin = false;
941  bool useDdYMax = false;
942  double minXD = 0;
943  double minYD = 0;
944  double maxXD = 0;
945  double maxYD = 0;
946 
947  if ( dataDefinedEvaluate( QgsComposerObject::MapXMin, exprVal ) )
948  {
949  bool ok;
950  minXD = exprVal.toDouble( &ok );
951  QgsDebugMsg( QString( "exprVal Map XMin:%1" ).arg( minXD ) );
952  if ( ok && !exprVal.isNull() )
953  {
954  useDdXMin = true;
955  newExtent.setXMinimum( minXD );
956  }
957  }
958  if ( dataDefinedEvaluate( QgsComposerObject::MapYMin, exprVal ) )
959  {
960  bool ok;
961  minYD = exprVal.toDouble( &ok );
962  QgsDebugMsg( QString( "exprVal Map YMin:%1" ).arg( minYD ) );
963  if ( ok && !exprVal.isNull() )
964  {
965  useDdYMin = true;
966  newExtent.setYMinimum( minYD );
967  }
968  }
969  if ( dataDefinedEvaluate( QgsComposerObject::MapXMax, exprVal ) )
970  {
971  bool ok;
972  maxXD = exprVal.toDouble( &ok );
973  QgsDebugMsg( QString( "exprVal Map XMax:%1" ).arg( maxXD ) );
974  if ( ok && !exprVal.isNull() )
975  {
976  useDdXMax = true;
977  newExtent.setXMaximum( maxXD );
978  }
979  }
980  if ( dataDefinedEvaluate( QgsComposerObject::MapYMax, exprVal ) )
981  {
982  bool ok;
983  maxYD = exprVal.toDouble( &ok );
984  QgsDebugMsg( QString( "exprVal Map YMax:%1" ).arg( maxYD ) );
985  if ( ok && !exprVal.isNull() )
986  {
987  useDdYMax = true;
988  newExtent.setYMaximum( maxYD );
989  }
990  }
991 
992  if ( newExtent != *currentMapExtent() )
993  {
994  //calculate new extents to fit data defined extents
995 
996  //Make sure the width/height ratio is the same as in current map extent.
997  //This is to keep the map item frame and the page layout fixed
998  double currentWidthHeightRatio = currentMapExtent()->width() / currentMapExtent()->height();
999  double newWidthHeightRatio = newExtent.width() / newExtent.height();
1000 
1001  if ( currentWidthHeightRatio < newWidthHeightRatio )
1002  {
1003  //enlarge height of new extent, ensuring the map center stays the same
1004  double newHeight = newExtent.width() / currentWidthHeightRatio;
1005  double deltaHeight = newHeight - newExtent.height();
1006  newExtent.setYMinimum( newExtent.yMinimum() - deltaHeight / 2 );
1007  newExtent.setYMaximum( newExtent.yMaximum() + deltaHeight / 2 );
1008  }
1009  else
1010  {
1011  //enlarge width of new extent, ensuring the map center stays the same
1012  double newWidth = currentWidthHeightRatio * newExtent.height();
1013  double deltaWidth = newWidth - newExtent.width();
1014  newExtent.setXMinimum( newExtent.xMinimum() - deltaWidth / 2 );
1015  newExtent.setXMaximum( newExtent.xMaximum() + deltaWidth / 2 );
1016  }
1017 
1018  *currentMapExtent() = newExtent;
1019  }
1020 
1021  //now refresh scale, as this potentially overrides extents
1022 
1023  //data defined map scale set?
1024  if ( dataDefinedEvaluate( QgsComposerObject::MapScale, exprVal ) )
1025  {
1026  bool ok;
1027  double scaleD = exprVal.toDouble( &ok );
1028  QgsDebugMsg( QString( "exprVal Map Scale:%1" ).arg( scaleD ) );
1029  if ( ok && !exprVal.isNull() )
1030  {
1031  setNewScale( scaleD, false );
1032  newExtent = *currentMapExtent();
1033  }
1034  }
1035 
1036  if ( useDdXMax || useDdXMin || useDdYMax || useDdYMin )
1037  {
1038  //if only one of min/max was set for either x or y, then make sure our extent is locked on that value
1039  //as we can do this without altering the scale
1040  if ( useDdXMin && !useDdXMax )
1041  {
1042  double xMax = currentMapExtent()->xMaximum() - ( currentMapExtent()->xMinimum() - minXD );
1043  newExtent.setXMinimum( minXD );
1044  newExtent.setXMaximum( xMax );
1045  }
1046  else if ( !useDdXMin && useDdXMax )
1047  {
1048  double xMin = currentMapExtent()->xMinimum() - ( currentMapExtent()->xMaximum() - maxXD );
1049  newExtent.setXMinimum( xMin );
1050  newExtent.setXMaximum( maxXD );
1051  }
1052  if ( useDdYMin && !useDdYMax )
1053  {
1054  double yMax = currentMapExtent()->yMaximum() - ( currentMapExtent()->yMinimum() - minYD );
1055  newExtent.setYMinimum( minYD );
1056  newExtent.setYMaximum( yMax );
1057  }
1058  else if ( !useDdYMin && useDdYMax )
1059  {
1060  double yMin = currentMapExtent()->yMinimum() - ( currentMapExtent()->yMaximum() - maxYD );
1061  newExtent.setYMinimum( yMin );
1062  newExtent.setYMaximum( maxYD );
1063  }
1064 
1065  if ( newExtent != *currentMapExtent() )
1066  {
1067  *currentMapExtent() = newExtent;
1068  }
1069  }
1070 
1071  //lastly, map rotation overrides all
1072  double mapRotation = mMapRotation;
1073 
1074  //data defined map rotation set?
1075  if ( dataDefinedEvaluate( QgsComposerObject::MapRotation, exprVal ) )
1076  {
1077  bool ok;
1078  double rotationD = exprVal.toDouble( &ok );
1079  QgsDebugMsg( QString( "exprVal Map Rotation:%1" ).arg( rotationD ) );
1080  if ( ok && !exprVal.isNull() )
1081  {
1082  mapRotation = rotationD;
1083  }
1084  }
1085 
1086  if ( mEvaluatedMapRotation != mapRotation )
1087  {
1088  mEvaluatedMapRotation = mapRotation;
1089  emit mapRotationChanged( mapRotation );
1090  }
1091
