101 QTransform QgsMapToPixel::transform() const
102 {
103  // NOTE: operations are done in the reverse order in which
104  // they are configured, so translation to geographical
105  // center happens first, then scaling, then rotation
106  // and finally translation to output viewport center
107 
108  double rotation = mapRotation();
109  if ( qgsDoubleNear( rotation, 0.0 ) )
110  {
111  //no rotation, return a simplified matrix
112  return QTransform::fromScale( 1.0 / mMapUnitsPerPixel, -1.0 / mMapUnitsPerPixel )
113  .translate( -xMin, - ( yMin + mHeight * mMapUnitsPerPixel ) );
114  }
115 
116  double cy = mapHeight() / 2.0;
117  double cx = mapWidth() / 2.0;
118  return QTransform::fromTranslate( cx, cy )
119  .rotate( rotation )
120  .scale( 1 / mMapUnitsPerPixel, -1 / mMapUnitsPerPixel )
121  .translate( -xCenter, -yCenter );
122 }

169 void QgsMapToPixel::setMapRotation( double degrees, double cx, double cy )
170 {
171  double oldRotation = mRotation;
172  double oldXCenter = xCenter;
173  double oldYCenter = yCenter;
174  double oldWidth = mWidth;
175 
176  mRotation = degrees;
177  xCenter = cx;
178  yCenter = cy;
179  if ( mWidth < 0 )
180  {
181  // set width not that we can compute it
182  mWidth = (( xCenter - xMin ) * 2 ) / mMapUnitsPerPixel;
183  }
184 
185  if ( !updateMatrix() )
186  {
187  mRotation = oldRotation;
188  xCenter = oldXCenter;
189  yCenter = oldYCenter;
190  mWidth = oldWidth;
191  }
192 }
