75 QgsRenderContext QgsRenderContext::fromMapSettings( const QgsMapSettings& mapSettings )
76 {
77  QgsRenderContext ctx;
78  ctx.setMapToPixel( mapSettings.mapToPixel() );
79  ctx.setExtent( mapSettings.visibleExtent() );
80  ctx.setDrawEditingInformation( mapSettings.testFlag( QgsMapSettings::DrawEditingInfo ) );
81  ctx.setForceVectorOutput( mapSettings.testFlag( QgsMapSettings::ForceVectorOutput ) );
82  ctx.setUseAdvancedEffects( mapSettings.testFlag( QgsMapSettings::UseAdvancedEffects ) );
83  ctx.setUseRenderingOptimization( mapSettings.testFlag( QgsMapSettings::UseRenderingOptimization ) );
84  ctx.setCoordinateTransform( 0 );
85  ctx.setSelectionColor( mapSettings.selectionColor() );
86  ctx.setShowSelection( mapSettings.testFlag( QgsMapSettings::DrawSelection ) );
87  ctx.setRasterScaleFactor( 1.0 );
88  ctx.setScaleFactor( mapSettings.outputDpi() / 25.4 ); // = pixels per mm
89  ctx.setRendererScale( mapSettings.scale() );
90  ctx.setSelectionColor( mapSettings.selectionColor() );
91  ctx.setCustomRenderFlags( mapSettings.customRenderFlags() );
92  ctx.setRenderMapTile( mapSettings.testFlag( QgsMapSettings::RenderMapTile ) );
93  ctx.setRenderPartialOutput( mapSettings.testFlag( QgsMapSettings::RenderPartialOutput ) );
94 
95  //this flag is only for stopping during the current rendering progress,
96  //so must be false at every new render operation
97  ctx.setRenderingStopped( false );
98 
99  return ctx;
100 }
