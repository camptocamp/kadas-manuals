49 QSizeF QgsLegendRenderer::paintAndDetermineSize( QPainter* painter )
50 {
51  QSizeF size( 0, 0 );
52  QgsLayerTreeGroup* rootGroup = mLegendModel->rootGroup();
53  if ( !rootGroup ) return size;
54 
55  QList<Atom> atomList = createAtomList( rootGroup, mSettings.splitLayer() );
56 
57  setColumns( atomList );
58 
59  qreal maxColumnWidth = 0;
60  if ( mSettings.equalColumnWidth() )
61  {
62  foreach ( Atom atom, atomList )
63  {
64  maxColumnWidth = qMax( atom.size.width(), maxColumnWidth );
65  }
66  }
67 
68  //calculate size of title
69  QSizeF titleSize = drawTitle();
70  //add title margin to size of title text
71  titleSize.rwidth() += mSettings.boxSpace() * 2.0;
72  double columnTop = mSettings.boxSpace() + titleSize.height() + mSettings.style( QgsComposerLegendStyle::Title ).margin( QgsComposerLegendStyle::Bottom );
73 
74  QPointF point( mSettings.boxSpace(), columnTop );
75  bool firstInColumn = true;
76  double columnMaxHeight = 0;
77  qreal columnWidth = 0;
78  int column = 0;
79  foreach ( Atom atom, atomList )
80  {
81  if ( atom.column > column )
82  {
83  // Switch to next column
84  if ( mSettings.equalColumnWidth() )
85  {
86  point.rx() += mSettings.columnSpace() + maxColumnWidth;
87  }
88  else
89  {
90  point.rx() += mSettings.columnSpace() + columnWidth;
91  }
92  point.ry() = columnTop;
93  columnWidth = 0;
94  column++;
95  firstInColumn = true;
96  }
97  if ( !firstInColumn )
98  {
99  point.ry() += spaceAboveAtom( atom );
100  }
101 
102  QSizeF atomSize = drawAtom( atom, painter, point );
103  columnWidth = qMax( atomSize.width(), columnWidth );
104 
105  point.ry() += atom.size.height();
106  columnMaxHeight = qMax( point.y() - columnTop, columnMaxHeight );
107 
108  firstInColumn = false;
109  }
110  point.rx() += columnWidth + mSettings.boxSpace();
111 
112  size.rheight() = columnTop + columnMaxHeight + mSettings.boxSpace();
113  size.rwidth() = point.x();
114  if ( !mSettings.title().isEmpty() )
115  {
116  size.rwidth() = qMax( titleSize.width(), size.width() );
117  }
118 
119  // override the size if it was set by the user
120  if ( mLegendSize.isValid() )
121  {
122  qreal w = qMax( size.width(), mLegendSize.width() );
123  qreal h = qMax( size.height(), mLegendSize.height() );
124  size = QSizeF( w, h );
125  }
126 
127  // Now we have set the correct total item width and can draw the title centered
128  if ( !mSettings.title().isEmpty() )
129  {
130  if ( mSettings.titleAlignment() == Qt::AlignLeft )
131  {
132  point.rx() = mSettings.boxSpace();
133  }
134  else if ( mSettings.titleAlignment() == Qt::AlignHCenter )
135  {
136  point.rx() = size.width() / 2;
137  }
138  else
139  {
140  point.rx() = size.width() - mSettings.boxSpace();
141  }
142  point.ry() = mSettings.boxSpace();
143  drawTitle( painter, point, mSettings.titleAlignment(), size.width() );
144  }
145 
146  return size;
147 }

150 QList<QgsLegendRenderer::Atom> QgsLegendRenderer::createAtomList( QgsLayerTreeGroup* parentGroup, bool splitLayer )
151 {
152  QList<Atom> atoms;
153 
154  if ( !parentGroup ) return atoms;
155 
156  foreach ( QgsLayerTreeNode* node, parentGroup->children() )
157  {
158  if ( QgsLayerTree::isGroup( node ) )
159  {
160  QgsLayerTreeGroup* nodeGroup = QgsLayerTree::toGroup( node );
161 
162  // Group subitems
163  QList<Atom> groupAtoms = createAtomList( nodeGroup, splitLayer );
164  bool hasSubItems = groupAtoms.size() > 0;
165 
166  if ( nodeLegendStyle( nodeGroup ) != QgsComposerLegendStyle::Hidden )
167  {
168  Nucleon nucleon;
169  nucleon.item = node;
170  nucleon.size = drawGroupTitle( nodeGroup );
171 
172 
173  if ( groupAtoms.size() > 0 )
174  {
175  // Add internal space between this group title and the next nucleon
176  groupAtoms[0].size.rheight() += spaceAboveAtom( groupAtoms[0] );
177  // Prepend this group title to the first atom
178  groupAtoms[0].nucleons.prepend( nucleon );
179  groupAtoms[0].size.rheight() += nucleon.size.height();
180  groupAtoms[0].size.rwidth() = qMax( nucleon.size.width(), groupAtoms[0].size.width() );
181  }
182  else
183  {
184  // no subitems, append new atom
185  Atom atom;
186  atom.nucleons.append( nucleon );
187  atom.size.rwidth() += nucleon.size.width();
188  atom.size.rheight() += nucleon.size.height();
189  atom.size.rwidth() = qMax( nucleon.size.width(), atom.size.width() );
190  groupAtoms.append( atom );
191  }
192  }
193 
194  if ( hasSubItems ) //leave away groups without content
195  {
196  atoms.append( groupAtoms );
197  }
198 
199  }
200  else if ( QgsLayerTree::isLayer( node ) )
201  {
202  QgsLayerTreeLayer* nodeLayer = QgsLayerTree::toLayer( node );
203 
204  Atom atom;
205 
206  if ( nodeLegendStyle( nodeLayer ) != QgsComposerLegendStyle::Hidden )
207  {
208  Nucleon nucleon;
209  nucleon.item = node;
210  nucleon.size = drawLayerTitle( nodeLayer );
211  atom.nucleons.append( nucleon );
212  atom.size.rwidth() = nucleon.size.width();
213  atom.size.rheight() = nucleon.size.height();
214  }
215 
216  QList<QgsLayerTreeModelLegendNode*> legendNodes = mLegendModel->layerLegendNodes( nodeLayer );
217 
218  // workaround for the issue that "filtering by map" does not remove layer nodes that have no symbols present
219  // on the map. We explicitly skip such layers here. In future ideally that should be handled directly
220  // in the layer tree model
221  if ( legendNodes.isEmpty() && mLegendModel->legendFilterByMap() )
222  continue;
223 
224  QList<Atom> layerAtoms;
225 
226  for ( int j = 0; j < legendNodes.count(); j++ )
227  {
228  QgsLayerTreeModelLegendNode* legendNode = legendNodes.at( j );
229 
230  Nucleon symbolNucleon = drawSymbolItem( legendNode );
231 
232  if ( !mSettings.splitLayer() || j == 0 )
233  {
234  // append to layer atom
235  // the width is not correct at this moment, we must align all symbol labels
236  atom.size.rwidth() = qMax( symbolNucleon.size.width(), atom.size.width() );
237  // Add symbol space only if there is already title or another item above
238  if ( atom.nucleons.size() > 0 )
239  {
240  // TODO: for now we keep Symbol and SymbolLabel Top margin in sync
241  atom.size.rheight() += mSettings.style( QgsComposerLegendStyle::Symbol ).margin( QgsComposerLegendStyle::Top );
242  }
243  atom.size.rheight() += symbolNucleon.size.height();
244  atom.nucleons.append( symbolNucleon );
245  }
246  else
247  {
248  Atom symbolAtom;
249  symbolAtom.nucleons.append( symbolNucleon );
250  symbolAtom.size.rwidth() = symbolNucleon.size.width();
251  symbolAtom.size.rheight() = symbolNucleon.size.height();
252  layerAtoms.append( symbolAtom );
253  }
254  }
255  layerAtoms.prepend( atom );
256  atoms.append( layerAtoms );
257  }
258  }
259 
260  return atoms;
261 }

264 void QgsLegendRenderer::setColumns( QList<Atom>& atomList )
265 {
266  if ( mSettings.columnCount() == 0 ) return;
267 
268  // Divide atoms to columns
269  double totalHeight = 0;
270  // bool first = true;
271  qreal maxAtomHeight = 0;
272  foreach ( Atom atom, atomList )
273  {
274  //if ( !first )
275  //{
276  totalHeight += spaceAboveAtom( atom );
277  //}
278  totalHeight += atom.size.height();
279  maxAtomHeight = qMax( atom.size.height(), maxAtomHeight );
280  // first = false;
281  }
282 
283  // We know height of each atom and we have to split them into columns
284  // minimizing max column height. It is sort of bin packing problem, NP-hard.
285  // We are using simple heuristic, brute fore appeared to be to slow,
286  // the number of combinations is N = n!/(k!*(n-k)!) where n = atomsCount-1
287  // and k = columnsCount-1
288 
289  double avgColumnHeight = totalHeight / mSettings.columnCount();
290  int currentColumn = 0;
291  int currentColumnAtomCount = 0; // number of atoms in current column
292  double currentColumnHeight = 0;
293  double maxColumnHeight = 0;
294  double closedColumnsHeight = 0;
295  // first = true; // first in column
296  for ( int i = 0; i < atomList.size(); i++ )
297  {
298  Atom atom = atomList[i];
299  double currentHeight = currentColumnHeight;
300  //if ( !first )
301  //{
302  currentHeight += spaceAboveAtom( atom );
303  //}
304  currentHeight += atom.size.height();
305 
306  // Recalc average height for remaining columns including current
307  avgColumnHeight = ( totalHeight - closedColumnsHeight ) / ( mSettings.columnCount() - currentColumn );
308  if (( currentHeight - avgColumnHeight ) > atom.size.height() / 2 // center of current atom is over average height
309  && currentColumnAtomCount > 0 // do not leave empty column
310  && currentHeight > maxAtomHeight // no sense to make smaller columns than max atom height
311  && currentHeight > maxColumnHeight // no sense to make smaller columns than max column already created
312  && currentColumn < mSettings.columnCount() - 1 ) // must not exceed max number of columns
313  {
314  // New column
315  currentColumn++;
316  currentColumnAtomCount = 0;
317  closedColumnsHeight += currentColumnHeight;
318  currentColumnHeight = atom.size.height();
319  }
320  else
321  {
322  currentColumnHeight = currentHeight;
323  }
324  atomList[i].column = currentColumn;
325  currentColumnAtomCount++;
326  maxColumnHeight = qMax( currentColumnHeight, maxColumnHeight );
327 
328  // first = false;
329  }
330 
331  // Alling labels of symbols for each layr/column to the same labelXOffset
332  QMap<QString, qreal> maxSymbolWidth;
333  for ( int i = 0; i < atomList.size(); i++ )
334  {
335  Atom& atom = atomList[i];
336  for ( int j = 0; j < atom.nucleons.size(); j++ )
337  {
338  if ( QgsLayerTreeModelLegendNode* legendNode = qobject_cast<QgsLayerTreeModelLegendNode*>( atom.nucleons[j].item ) )
339  {
340  QString key = QString( "%1-%2" ).arg(( qulonglong )legendNode->layerNode() ).arg( atom.column );
341  maxSymbolWidth[key] = qMax( atom.nucleons[j].symbolSize.width(), maxSymbolWidth[key] );
342  }
343  }
344  }
345  for ( int i = 0; i < atomList.size(); i++ )
346  {
347  Atom& atom = atomList[i];
348  for ( int j = 0; j < atom.nucleons.size(); j++ )
349  {
350  if ( QgsLayerTreeModelLegendNode* legendNode = qobject_cast<QgsLayerTreeModelLegendNode*>( atom.nucleons[j].item ) )
351  {
352  QString key = QString( "%1-%2" ).arg(( qulonglong )legendNode->layerNode() ).arg( atom.column );
353  double space = mSettings.style( QgsComposerLegendStyle::Symbol ).margin( QgsComposerLegendStyle::Right ) +
354  mSettings.style( QgsComposerLegendStyle::SymbolLabel ).margin( QgsComposerLegendStyle::Left );
355  atom.nucleons[j].labelXOffset = maxSymbolWidth[key] + space;
356  atom.nucleons[j].size.rwidth() = maxSymbolWidth[key] + space + atom.nucleons[j].labelSize.width();
357  }
358  }
359  }
360 }

363 QSizeF QgsLegendRenderer::drawTitle( QPainter* painter, QPointF point, Qt::AlignmentFlag halignment, double legendWidth )
364 {
365  QSizeF size( 0, 0 );
366  if ( mSettings.title().isEmpty() )
367  {
368  return size;
369  }
370 
371  QStringList lines = mSettings.splitStringForWrapping( mSettings.title() );
372  double y = point.y();
373 
374  if ( painter )
375  {
376  painter->setPen( mSettings.fontColor() );
377  }
378 
379  //calculate width and left pos of rectangle to draw text into
380  double textBoxWidth;
381  double textBoxLeft;
382  switch ( halignment )
383  {
384  case Qt::AlignHCenter:
385  textBoxWidth = ( qMin(( double ) point.x(), legendWidth - point.x() ) - mSettings.boxSpace() ) * 2.0;
386  textBoxLeft = point.x() - textBoxWidth / 2.;
387  break;
388  case Qt::AlignRight:
389  textBoxLeft = mSettings.boxSpace();
390  textBoxWidth = point.x() - mSettings.boxSpace();
391  break;
392  case Qt::AlignLeft:
393  default:
394  textBoxLeft = point.x();
395  textBoxWidth = legendWidth - point.x() - mSettings.boxSpace();
396  break;
397  }
398 
399  QFont titleFont = mSettings.style( QgsComposerLegendStyle::Title ).font();
400 
401  for ( QStringList::Iterator titlePart = lines.begin(); titlePart != lines.end(); ++titlePart )
402  {
403  //last word is not drawn if rectangle width is exactly text width, so add 1
404  //TODO - correctly calculate size of italicized text, since QFontMetrics does not
405  qreal width = mSettings.textWidthMillimeters( titleFont, *titlePart ) + 1;
406  qreal height = mSettings.fontAscentMillimeters( titleFont ) + mSettings.fontDescentMillimeters( titleFont );
407 
408  QRectF r( textBoxLeft, y, textBoxWidth, height );
409 
410  if ( painter )
411  {
412  mSettings.drawText( painter, r, *titlePart, titleFont, halignment, Qt::AlignVCenter, Qt::TextDontClip );
413  }
414 
415  //update max width of title
416  size.rwidth() = qMax( width, size.rwidth() );
417 
418  y += height;
419  if ( titlePart != lines.end() )
420  {
421  y += mSettings.lineSpacing();
422  }
423  }
424  size.rheight() = y - point.y();
425 
426  return size;
427 }

430 double QgsLegendRenderer::spaceAboveAtom( Atom atom )
431 {
432  if ( atom.nucleons.size() == 0 ) return 0;
433 
434  Nucleon nucleon = atom.nucleons.first();
435 
436  if ( QgsLayerTreeGroup* nodeGroup = qobject_cast<QgsLayerTreeGroup*>( nucleon.item ) )
437  {
438  return mSettings.style( nodeLegendStyle( nodeGroup ) ).margin( QgsComposerLegendStyle::Top );
439  }
440  else if ( QgsLayerTreeLayer* nodeLayer = qobject_cast<QgsLayerTreeLayer*>( nucleon.item ) )
441  {
442  return mSettings.style( nodeLegendStyle( nodeLayer ) ).margin( QgsComposerLegendStyle::Top );
443  }
444  else if ( qobject_cast<QgsLayerTreeModelLegendNode*>( nucleon.item ) )
445  {
446  // TODO: use Symbol or SymbolLabel Top margin
447  return mSettings.style( QgsComposerLegendStyle::Symbol ).margin( QgsComposerLegendStyle::Top );
448  }
449 
450  return 0;
451 }

455 QSizeF QgsLegendRenderer::drawAtom( Atom atom, QPainter* painter, QPointF point )
456 {
457  bool first = true;
458  QSizeF size = QSizeF( atom.size );
459  foreach ( Nucleon nucleon, atom.nucleons )
460  {
461  if ( QgsLayerTreeGroup* groupItem = qobject_cast<QgsLayerTreeGroup*>( nucleon.item ) )
462  {
463  QgsComposerLegendStyle::Style s = nodeLegendStyle( groupItem );
464  if ( s != QgsComposerLegendStyle::Hidden )
465  {
466  if ( !first )
467  {
468  point.ry() += mSettings.style( s ).margin( QgsComposerLegendStyle::Top );
469  }
470  drawGroupTitle( groupItem, painter, point );
471  }
472  }
473  else if ( QgsLayerTreeLayer* layerItem = qobject_cast<QgsLayerTreeLayer*>( nucleon.item ) )
474  {
475  QgsComposerLegendStyle::Style s = nodeLegendStyle( layerItem );
476  if ( s != QgsComposerLegendStyle::Hidden )
477  {
478  if ( !first )
479  {
480  point.ry() += mSettings.style( s ).margin( QgsComposerLegendStyle::Top );
481  }
482  drawLayerTitle( layerItem, painter, point );
483  }
484  }
485  else if ( QgsLayerTreeModelLegendNode* legendNode = qobject_cast<QgsLayerTreeModelLegendNode*>( nucleon.item ) )
486  {
487  if ( !first )
488  {
489  point.ry() += mSettings.style( QgsComposerLegendStyle::Symbol ).margin( QgsComposerLegendStyle::Top );
490  }
491 
492  Nucleon symbolNucleon = drawSymbolItem( legendNode, painter, point, nucleon.labelXOffset );
493  // expand width, it may be wider because of labelXOffset
494  size.rwidth() = qMax( symbolNucleon.size.width(), size.width() );
495  }
496  point.ry() += nucleon.size.height();
497  first = false;
498  }
499  return size;
500 }
