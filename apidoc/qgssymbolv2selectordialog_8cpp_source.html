349 void QgsSymbolV2SelectorDialog::symbolChanged()
350 {
351  SymbolLayerItem *currentItem = static_cast<SymbolLayerItem*>( model->itemFromIndex( layersTree->currentIndex() ) );
352  if ( currentItem == NULL || currentItem->isLayer() )
353  return;
354  // disconnect to avoid recreating widget
355  disconnect( layersTree->selectionModel(), SIGNAL( currentChanged( const QModelIndex&, const QModelIndex& ) ), this, SLOT( layerChanged() ) );
356  if ( currentItem->parent() )
357  {
358  // it is a sub-symbol
359  QgsSymbolV2* symbol = currentItem->symbol();
360  SymbolLayerItem *parent = static_cast<SymbolLayerItem*>( currentItem->parent() );
361  parent->removeRow( 0 );
362  loadSymbol( symbol, parent );
363  layersTree->setCurrentIndex( parent->child( 0 )->index() );
364  parent->updatePreview();
365  }
366  else
367  {
368  //it is the symbol itself
369  loadSymbol();
370  QModelIndex newIndex = layersTree->model()->index( 0, 0 );
371  layersTree->setCurrentIndex( newIndex );
372  }
373  updatePreview();
374  // connect it back once things are set
375  connect( layersTree->selectionModel(), SIGNAL( currentChanged( const QModelIndex&, const QModelIndex& ) ), this, SLOT( layerChanged() ) );
376 }

399 void QgsSymbolV2SelectorDialog::addLayer()
400 {
401  QModelIndex idx = layersTree->currentIndex();
402  if ( !idx.isValid() )
403  return;
404 
405  int insertIdx = -1;
406  SymbolLayerItem *item = static_cast<SymbolLayerItem*>( model->itemFromIndex( idx ) );
407  if ( item->isLayer() )
408  {
409  insertIdx = item->row();
410  item = static_cast<SymbolLayerItem*>( item->parent() );
411  }
412 
413  QgsSymbolV2* parentSymbol = item->symbol();
414  QgsSymbolLayerV2* newLayer = QgsSymbolLayerV2Registry::instance()->defaultSymbolLayer( parentSymbol->type() );
415  if ( insertIdx == -1 )
416  parentSymbol->appendSymbolLayer( newLayer );
417  else
418  parentSymbol->insertSymbolLayer( item->rowCount() - insertIdx, newLayer );
419  SymbolLayerItem *newLayerItem = new SymbolLayerItem( newLayer );
420  item->insertRow( insertIdx == -1 ? 0 : insertIdx, newLayerItem );
421  item->updatePreview();
422 
423  layersTree->setCurrentIndex( model->indexFromItem( newLayerItem ) );
424  updateUi();
425  updatePreview();
426 }

460 void QgsSymbolV2SelectorDialog::moveLayerByOffset( int offset )
461 {

463  if ( item == NULL )
464  return;
465  int row = item->row();
466 
467  SymbolLayerItem *parent = static_cast<SymbolLayerItem*>( item->parent() );
468  QgsSymbolV2* parentSymbol = parent->symbol();
469 
470  int layerIdx = parent->rowCount() - row - 1;
471  // switch layers
472  QgsSymbolLayerV2* tmpLayer = parentSymbol->takeSymbolLayer( layerIdx );
473  parentSymbol->insertSymbolLayer( layerIdx - offset, tmpLayer );
474 
475  QList<QStandardItem*> rowItems = parent->takeRow( row );
476  parent->insertRows( row + offset, rowItems );
477  parent->updatePreview();
478 
479  QModelIndex newIdx = rowItems[ 0 ]->index();
480  layersTree->setCurrentIndex( newIdx );
481 
482  updatePreview();
483  updateUi();

522 void QgsSymbolV2SelectorDialog::changeLayer( QgsSymbolLayerV2* newLayer )
523 {
524  SymbolLayerItem *item = currentLayerItem();

526

530  }

537  layersTree->setExpanded( item->index(), true );

539

543  symbol->changeSymbolLayer( layerIdx, newLayer );
544 
545  item->updatePreview();
546  updatePreview();
547  // Important: This lets the layer to have its own layer properties widget
548  layerChanged();
