80 QgsRasterRenderer* QgsMultiBandColorRenderer::create( const QDomElement& elem, QgsRasterInterface* input )
81 {
82  if ( elem.isNull() )
83  {
84  return 0;
85  }
86 
87  //red band, green band, blue band
88  int redBand = elem.attribute( "redBand", "-1" ).toInt();
89  int greenBand = elem.attribute( "greenBand", "-1" ).toInt();

91

100  }
101 
102  QgsContrastEnhancement* greenContrastEnhancement = 0;
103  QDomElement greenContrastElem = elem.firstChildElement( "greenContrastEnhancement" );
104  if ( !greenContrastElem.isNull() )
105  {
106  greenContrastEnhancement = new QgsContrastEnhancement(( QGis::DataType )(
107  input->dataType( greenBand ) ) );
108  greenContrastEnhancement->readXML( greenContrastElem );
109  }
110 
111  QgsContrastEnhancement* blueContrastEnhancement = 0;
112  QDomElement blueContrastElem = elem.firstChildElement( "blueContrastEnhancement" );
113  if ( !blueContrastElem.isNull() )
114  {
115  blueContrastEnhancement = new QgsContrastEnhancement(( QGis::DataType )(
116  input->dataType( blueBand ) ) );
117  blueContrastEnhancement->readXML( blueContrastElem );

119 
120  QgsRasterRenderer* r = new QgsMultiBandColorRenderer( input, redBand, greenBand, blueBand, redContrastEnhancement,

131 QgsRasterBlock* QgsMultiBandColorRenderer::block2( int bandNo, QgsRectangle const & extent, int width, int height, QgsRasterBlockFeedback *feedback )
132 {
133  Q_UNUSED( bandNo );
134  QgsRasterBlock *outputBlock = new QgsRasterBlock();
135  if ( !mInput )
136  {
137  return outputBlock;
138  }
139 
140  //In some (common) cases, we can simplify the drawing loop considerably and save render time
141  bool fastDraw = ( !usesTransparency()
142  && mRedBand > 0 && mGreenBand > 0 && mBlueBand > 0
143  && mAlphaBand < 1 && !mRedContrastEnhancement && !mGreenContrastEnhancement && !mBlueContrastEnhancement );
144 
145  QSet<int> bands;
146  if ( mRedBand > 0 )
147  {
148  bands << mRedBand;
149  }
150  if ( mGreenBand > 0 )
151  {
152  bands << mGreenBand;
153  }
154  if ( mBlueBand > 0 )
155  {
156  bands << mBlueBand;
157  }
158  if ( bands.size() < 1 )
159  {
160  // no need to draw anything if no band is set
161  // TODO:: we should probably return default color block
162  return outputBlock;
163  }
164 
165  if ( mAlphaBand > 0 )
166  {
167  bands << mAlphaBand;
168  }
169 
170  QMap<int, QgsRasterBlock*> bandBlocks;
171  QgsRasterBlock* defaultPointer = 0;
172  QSet<int>::const_iterator bandIt = bands.constBegin();
173  for ( ; bandIt != bands.constEnd(); ++bandIt )
174  {
175  bandBlocks.insert( *bandIt, defaultPointer );
176  }
177 
178  QgsRasterBlock* redBlock = 0;
179  QgsRasterBlock* greenBlock = 0;
180  QgsRasterBlock* blueBlock = 0;
181  QgsRasterBlock* alphaBlock = 0;
182 
183  bandIt = bands.constBegin();
184  for ( ; bandIt != bands.constEnd(); ++bandIt )
185  {
186  bandBlocks[*bandIt] = mInput->block2( *bandIt, extent, width, height, feedback );
187  if ( !bandBlocks[*bandIt] )
188  {
189  // We should free the alloced mem from block().
190  QgsDebugMsg( "No input band" );
191  --bandIt;
192  for ( ; bandIt != bands.constBegin(); --bandIt )
193  {
194  delete bandBlocks[*bandIt];
195  }
196  return outputBlock;
197  }
198  }
199 
200  if ( mRedBand > 0 )
201  {
202  redBlock = bandBlocks[mRedBand];
203  }
204  if ( mGreenBand > 0 )
205  {
206  greenBlock = bandBlocks[mGreenBand];
207  }
208  if ( mBlueBand > 0 )
209  {
210  blueBlock = bandBlocks[mBlueBand];
211  }
212  if ( mAlphaBand > 0 )
213  {
214  alphaBlock = bandBlocks[mAlphaBand];
215  }
216 
217  if ( !outputBlock->reset( QGis::ARGB32_Premultiplied, width, height ) )
218  {
219  for ( int i = 0; i < bandBlocks.size(); i++ )
220  {
221  delete bandBlocks.value( i );
222  }
223  return outputBlock;
224  }
225 
226  QRgb myDefaultColor = NODATA_COLOR;
227 
228  for ( qgssize i = 0; i < ( qgssize )width*height; i++ )
229  {
230  if ( fastDraw ) //fast rendering if no transparency, stretching, color inversion, etc.
231  {
232  if ( redBlock->isNoData( i ) ||
233  greenBlock->isNoData( i ) ||
234  blueBlock->isNoData( i ) )
235  {
236  outputBlock->setColor( i, myDefaultColor );
237  }
238  else
239  {
240  int redVal = ( int )redBlock->value( i );
241  int greenVal = ( int )greenBlock->value( i );
242  int blueVal = ( int )blueBlock->value( i );
243  outputBlock->setColor( i, qRgba( redVal, greenVal, blueVal, 255 ) );
244  }
245  continue;
246  }
247 
248  bool isNoData = false;
249  double redVal = 0;
250  double greenVal = 0;
251  double blueVal = 0;
252  if ( mRedBand > 0 )
253  {
254  redVal = redBlock->value( i );
255  if ( redBlock->isNoData( i ) ) isNoData = true;
256  }
257  if ( !isNoData && mGreenBand > 0 )
258  {
259  greenVal = greenBlock->value( i );
260  if ( greenBlock->isNoData( i ) ) isNoData = true;
261  }
262  if ( !isNoData && mBlueBand > 0 )
263  {
264  blueVal = blueBlock->value( i );
265  if ( blueBlock->isNoData( i ) ) isNoData = true;
266  }
267  if ( isNoData )
268  {
269  outputBlock->setColor( i, myDefaultColor );
270  continue;
271  }
272 
273  //apply default color if red, green or blue not in displayable range
274  if (( mRedContrastEnhancement && !mRedContrastEnhancement->isValueInDisplayableRange( redVal ) )
275  || ( mGreenContrastEnhancement && !mGreenContrastEnhancement->isValueInDisplayableRange( redVal ) )
276  || ( mBlueContrastEnhancement && !mBlueContrastEnhancement->isValueInDisplayableRange( redVal ) ) )
277  {
278  outputBlock->setColor( i, myDefaultColor );
279  continue;
280  }
281 
282  //stretch color values
283  if ( mRedContrastEnhancement )
284  {
285  redVal = mRedContrastEnhancement->enhanceContrast( redVal );
286  }
287  if ( mGreenContrastEnhancement )
288  {
289  greenVal = mGreenContrastEnhancement->enhanceContrast( greenVal );
290  }
291  if ( mBlueContrastEnhancement )
292  {
293  blueVal = mBlueContrastEnhancement->enhanceContrast( blueVal );
294  }
295 
296  //opacity
297  double currentOpacity = mOpacity;
298  if ( mRasterTransparency )
299  {
300  currentOpacity = mRasterTransparency->alphaValue( redVal, greenVal, blueVal, mOpacity * 255 ) / 255.0;
301  }
302  if ( mAlphaBand > 0 )
303  {
304  currentOpacity *= alphaBlock->value( i ) / 255.0;
305  }
306 
307  if ( qgsDoubleNear( currentOpacity, 1.0 ) )
308  {
309  outputBlock->setColor( i, qRgba( redVal, greenVal, blueVal, 255 ) );
310  }
311  else
312  {
313  outputBlock->setColor( i, qRgba( currentOpacity * redVal, currentOpacity * greenVal, currentOpacity * blueVal, currentOpacity * 255 ) );
314  }
315  }
316 
317  //delete input blocks
318  QMap<int, QgsRasterBlock*>::const_iterator bandDelIt = bandBlocks.constBegin();
319  for ( ; bandDelIt != bandBlocks.constEnd(); ++bandDelIt )
320  {
321  delete bandDelIt.value();
322  }
323 
324  return outputBlock;
325 }

327 void QgsMultiBandColorRenderer::writeXML( QDomDocument& doc, QDomElement& parentElem ) const
328 {
329  if ( parentElem.isNull() )
330  {
331  return;
332  }
333 
334  QDomElement rasterRendererElem = doc.createElement( "rasterrenderer" );
335  _writeXML( doc, rasterRendererElem );
336  rasterRendererElem.setAttribute( "redBand", mRedBand );
337  rasterRendererElem.setAttribute( "greenBand", mGreenBand );
338  rasterRendererElem.setAttribute( "blueBand", mBlueBand );
339 
340  //contrast enhancement
341  if ( mRedContrastEnhancement )
342  {
343  QDomElement redContrastElem = doc.createElement( "redContrastEnhancement" );
344  mRedContrastEnhancement->writeXML( doc, redContrastElem );
345  rasterRendererElem.appendChild( redContrastElem );
346  }
347  if ( mGreenContrastEnhancement )
348  {
349  QDomElement greenContrastElem = doc.createElement( "greenContrastEnhancement" );
350  mGreenContrastEnhancement->writeXML( doc, greenContrastElem );
351  rasterRendererElem.appendChild( greenContrastElem );
352  }
353  if ( mBlueContrastEnhancement )
354  {
355  QDomElement blueContrastElem = doc.createElement( "blueContrastEnhancement" );
356  mBlueContrastEnhancement->writeXML( doc, blueContrastElem );
357  rasterRendererElem.appendChild( blueContrastElem );
358  }
359  parentElem.appendChild( rasterRendererElem );
360 }
