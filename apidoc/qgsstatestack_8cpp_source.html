40 void QgsStateStack::undo()

43  {

56 void QgsStateStack::redo()
57 {
58  while ( !mRedoStack.isEmpty() )
59  {
60  StateChangeCommand* command = mRedoStack.pop();
61  command->redo();
62  mUndoStack.push( command );
63  if ( mRedoStack.isEmpty() || !mRedoStack.top()->compress() )
64  {
65  break;
66  }
67  }
68  emit canUndoChanged( !mUndoStack.isEmpty() );
69  emit canRedoChanged( !mRedoStack.isEmpty() );
