79 void QgsLayerTreeMapCanvasBridge::setCustomLayerOrder( const QStringList& order )
80 {
81  if ( mCustomLayerOrder == order )
82  return;
83 
84  // verify that the new order is correct
85  QStringList defOrder( defaultLayerOrder() );
86  QStringList newOrder( order );
87  QStringList sortedNewOrder( order );
88  qSort( defOrder );
89  qSort( sortedNewOrder );
90 
91  if ( defOrder.size() < sortedNewOrder.size() )
92  {
93  // might contain bad layers, but also duplicates
94  QSet<QString> ids( defOrder.toSet() );
95 
96  for ( int i = 0; i < sortedNewOrder.size(); i++ )
97  {
98  if ( !ids.contains( sortedNewOrder[i] ) )
99  {
100  newOrder.removeAll( sortedNewOrder[i] );
101  sortedNewOrder.removeAt( i-- );
102  }
103  }
104  }
105 
106  if ( defOrder != sortedNewOrder )
107  return; // must be permutation of the default order
108 
109  mCustomLayerOrder = newOrder;
110  emit customLayerOrderChanged( mCustomLayerOrder );
111 
112  if ( mHasCustomLayerOrder )
113  deferredSetCanvasLayers();
114 }

116 void QgsLayerTreeMapCanvasBridge::setCanvasLayers()
117 {
118  QList<QgsMapCanvasLayer> layers;
119 
120  if ( mHasCustomLayerOrder )
121  {
122  foreach ( QString layerId, mCustomLayerOrder )
123  {
124  QgsLayerTreeLayer* nodeLayer = mRoot->findLayer( layerId );
125  if ( nodeLayer )
126  layers << QgsMapCanvasLayer( nodeLayer->layer(), nodeLayer->isVisible() == Qt::Checked, nodeLayer->customProperty( "overview", 0 ).toInt() );
127  }
128  }
129  else
130  setCanvasLayers( mRoot, layers );
131 
132  QList<QgsLayerTreeLayer*> layerNodes = mRoot->findLayers();
133  int redliningLayerCount = 0;
134  // Exclude redlining layers from count
135  foreach ( QgsLayerTreeLayer* node, layerNodes )
136  {
137  if ( node->layer() && node->layer()->type() == QgsMapLayer::RedliningLayer )
138  {
139  ++redliningLayerCount;
140  }
141  }
142  int currentLayerCount = layerNodes.count() - redliningLayerCount;
143  bool firstLayers = mAutoSetupOnFirstLayer && mLastLayerCount == 0 && currentLayerCount != 0;
144 
145  if ( firstLayers )
146  {
147  // also setup destination CRS and map units if the OTF projections are not yet enabled
148  if ( !mCanvas->mapSettings().hasCrsTransformEnabled() )
149  {
150  foreach ( QgsLayerTreeLayer* layerNode, layerNodes )
151  {
152  if ( layerNode->layer() &&
153  (
154  qobject_cast<QgsVectorLayer *>( layerNode->layer() ) == 0 ||
155  qobject_cast<QgsVectorLayer *>( layerNode->layer() )->geometryType() != QGis::NoGeometry
156  )
157  )
158  {
159  mCanvas->setDestinationCrs( layerNode->layer()->crs() );
160  mCanvas->setMapUnits( layerNode->layer()->crs().mapUnits() );
161  break;
162  }
163  }
164  }
165  }
166 
167  mCanvas->setLayerSet( layers );
168 
169  if ( firstLayers )
170  {
171  // if we are moving from zero to non-zero layers, let's zoom to those data
172  mCanvas->zoomToFullExtent();
173  }
174 
175  if ( !mFirstCRS.isValid() )
176  {
177  // find out what is the first used CRS in case we may need to turn on OTF projections later
178  foreach ( QgsLayerTreeLayer* layerNode, layerNodes )
179  {
180  if ( layerNode->layer() && layerNode->layer()->crs().isValid() )
181  {
182  mFirstCRS = layerNode->layer()->crs();
183  break;
184  }
185  }
186  }
187 
188  if ( mAutoEnableCrsTransform && mFirstCRS.isValid() && !mCanvas->mapSettings().hasCrsTransformEnabled() )
189  {
190  // check whether all layers still have the same CRS
191  foreach ( QgsLayerTreeLayer* layerNode, layerNodes )
192  {
193  if ( layerNode->layer() && layerNode->layer()->crs().isValid() && layerNode->layer()->crs() != mFirstCRS )
194  {
195  mCanvas->setDestinationCrs( mFirstCRS );
196  mCanvas->setCrsTransformEnabled( true );
197  break;
198  }
199  }
200  }
201 
202  mLastLayerCount = currentLayerCount;
203  if ( currentLayerCount == 0 )
204  mFirstCRS = QgsCoordinateReferenceSystem();
205 
206  mPendingCanvasUpdate = false;
207 }

209 void QgsLayerTreeMapCanvasBridge::readProject( const QDomDocument& doc )
210 {
211  mFirstCRS = QgsCoordinateReferenceSystem(); // invalidate on project load
212 
213  QDomElement elem = doc.documentElement().firstChildElement( "layer-tree-canvas" );
214  if ( elem.isNull() )
215  {
216  bool oldEnabled;
217  QStringList oldOrder;
218  if ( QgsLayerTreeUtils::readOldLegendLayerOrder( doc.documentElement().firstChildElement( "legend" ), oldEnabled, oldOrder ) )
219  {
220  setHasCustomLayerOrder( oldEnabled );
221  setCustomLayerOrder( oldOrder );
222  }
223  return;
224  }
225 
226  QDomElement customOrderElem = elem.firstChildElement( "custom-order" );
227  if ( !customOrderElem.isNull() )
228  {
229  QStringList order;
230  QDomElement itemElem = customOrderElem.firstChildElement( "item" );
231  while ( !itemElem.isNull() )
232  {
233  order.append( itemElem.text() );
234  itemElem = itemElem.nextSiblingElement( "item" );
235  }
236 
237  setHasCustomLayerOrder( customOrderElem.attribute( "enabled", 0 ).toInt() );
238  setCustomLayerOrder( order );
239  }
240 }

280 void QgsLayerTreeMapCanvasBridge::nodeAddedChildren( QgsLayerTreeNode* node, int indexFrom, int indexTo )
281 {
282  Q_ASSERT( node );
283 
284  // collect layer IDs that have been added in order to put them into custom layer order
285  QStringList layerIds;
286  QList<QgsLayerTreeNode*> children = node->children();
287  for ( int i = indexFrom; i <= indexTo; ++i )
288  {
289  QgsLayerTreeNode* child = children.at( i );
290  if ( QgsLayerTree::isLayer( child ) )
291  {
292  layerIds << QgsLayerTree::toLayer( child )->layerId();
293  }
294  else if ( QgsLayerTree::isGroup( child ) )
295  {
296  foreach ( QgsLayerTreeLayer* nodeL, QgsLayerTree::toGroup( child )->findLayers() )
297  layerIds << nodeL->layerId();
298  }
299  }
300 
301  foreach ( QString layerId, layerIds )
302  {
303  if ( !mCustomLayerOrder.contains( layerId ) )
304  mCustomLayerOrder.append( layerId );
305  }
306 
307  emit customLayerOrderChanged( mCustomLayerOrder );
308 
309  deferredSetCanvasLayers();
310 }

312 void QgsLayerTreeMapCanvasBridge::nodeRemovedChildren()
313 {
314  // no need to disconnect from removed nodes as they are deleted
315 
316  // check whether the layers are still there, if not, remove them from the layer order!
317  QList<int> toRemove;
318  for ( int i = 0; i < mCustomLayerOrder.count(); ++i )
319  {
320  QgsLayerTreeLayer* node = mRoot->findLayer( mCustomLayerOrder[i] );
321  if ( !node )
322  toRemove << i;
323  }
324  for ( int i = toRemove.count() - 1; i >= 0; --i )
325  mCustomLayerOrder.removeAt( toRemove[i] );
326  emit customLayerOrderChanged( mCustomLayerOrder );
327 
328  deferredSetCanvasLayers();
329 }
