884 template<class T> void QgsComposition::composerItems( QList<T*>& itemList )
885 {
886  itemList.clear();
887  QList<QGraphicsItem *> graphicsItemList = items();
888  QList<QGraphicsItem *>::iterator itemIt = graphicsItemList.begin();
889  for ( ; itemIt != graphicsItemList.end(); ++itemIt )
890  {
891  T* item = dynamic_cast<T*>( *itemIt );
892  if ( item )
893  {
894  itemList.push_back( item );
895  }
896  }
897 }
898 
899 template<class T> void QgsComposition::composerItemsOnPage( QList<T*>& itemList, const int pageNumber ) const
900 {
901  itemList.clear();
902  QList<QGraphicsItem *> graphicsItemList = items();
903  QList<QGraphicsItem *>::iterator itemIt = graphicsItemList.begin();
904  for ( ; itemIt != graphicsItemList.end(); ++itemIt )
905  {
906  T* item = dynamic_cast<T*>( *itemIt );
907  if ( item && itemPageNumber( item ) == pageNumber )
908  {
909  itemList.push_back( item );
910  }
911  }
