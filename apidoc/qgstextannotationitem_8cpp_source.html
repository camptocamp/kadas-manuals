68 void QgsTextAnnotationItem::paint( QPainter * painter )
69 {
70  if ( !painter || !mDocument )
71  {
72  return;
73  }
74 
75  drawFrame( painter );
76  if ( mMapPositionFixed )
77  {
78  drawMarkerSymbol( painter );
79  }
80  double frameWidth = mFrameBorderWidth;
81  mDocument->setTextWidth( mFrameSize.width() );
82 
83  painter->save();
84  painter->translate( mOffsetFromReferencePoint.x() + frameWidth / 2.0,
85  mOffsetFromReferencePoint.y() + frameWidth / 2.0 );
86 
87  QRectF clipRect = QRectF( 0, 0, mFrameSize.width() - frameWidth / 2.0, mFrameSize.height() - frameWidth / 2.0 );
88  if ( painter->hasClipping() )
89  {
90  //QTextDocument::drawContents will draw text outside of the painter's clip region
91  //when it is passed a clip rectangle. So, we need to intersect it with the
92  //painter's clip region to prevent text drawn outside clipped region (eg, outside composer maps, see #10400)
93  clipRect = clipRect.intersected( painter->clipRegion().boundingRect() );
94  }
95  //draw text document
96  mDocument->drawContents( painter, clipRect );
97  painter->restore();
98  if ( isSelected() )
99  {
100  drawSelectionBoxes( painter );
101  }
102 }
