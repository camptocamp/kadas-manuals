54 QgsSymbolV2* QgsRedliningRendererV2::originalSymbolForFeature( QgsFeature& feature )
55 {
56  switch ( QgsWKBTypes::flatType( QgsWKBTypes::singleType( feature.geometry()->geometry()->wkbType() ) ) )
57  {
58  case QgsWKBTypes::Point:

60  case QgsWKBTypes::LineString:
61  case QgsWKBTypes::CircularString:

64  case QgsWKBTypes::Polygon:
65  case QgsWKBTypes::CurvePolygon:
66  return mFillSymbol.data();

68  return 0;
69  }

80 bool QgsRedliningRendererV2::renderFeature( QgsFeature& feature, QgsRenderContext& context, int layer, bool selected, bool drawVertexMarker )
81 {
82  if ( !feature.constGeometry() || !feature.geometry()->geometry() )
83  {
84  return false;
85  }
86  // Don't draw features whose text attribute is set - they are drawn as labels only
87  if ( feature.geometry()->type() == QGis::Point && !feature.attribute( "text" ).toString().isEmpty() && !feature.attribute( "flags" ).toString().contains( "symbol=" ) )
88  {
89  return true;
90  }
91  QgsAbstractGeometryV2* geom = feature.geometry()->geometry();
92  QgsAbstractGeometryV2* segmentized = 0;
93 
94  //convert curve types to normal point/line/polygon ones
95  switch ( QgsWKBTypes::flatType( geom->wkbType() ) )
96  {
97  case QgsWKBTypes::CurvePolygon:
98  case QgsWKBTypes::CircularString:
99  case QgsWKBTypes::CompoundCurve:
100  case QgsWKBTypes::MultiSurface:
101  case QgsWKBTypes::MultiCurve:
102  {
103  segmentized = geom->segmentize();
104  geom = segmentized;
105  }
106  default:
107  break;
108  }
109  int wkbSize;
110 
111  switch ( QgsWKBTypes::flatType( geom->wkbType() ) )
112  {
113  case QgsWKBTypes::Point:
114  {
115  QPointF pt;
116  _getPoint( pt, context, geom->asWkb( wkbSize ) );
117  mMarkerSymbol->renderPoint( pt, &feature, context, layer, selected );
118  break;
119  }
120  case QgsWKBTypes::LineString:
121  {
122  QPolygonF pts;
123  _getLineString( pts, context, geom->asWkb( wkbSize ) );
124  mLineSymbol->renderPolyline( pts, &feature, context, layer, selected );
125 
126  if ( drawVertexMarker )
127  drawVertexMarkers( feature.geometry()->geometry(), context );
128  break;
129  }
130  case QgsWKBTypes::Polygon:
131  {
132  QPolygonF pts;
133  QList<QPolygonF> holes;
134  _getPolygon( pts, holes, context, geom->asWkb( wkbSize ) );
135  mFillSymbol->renderPolygon( pts, ( holes.count() ? &holes : NULL ), &feature, context, layer, selected );
136 
137  if ( drawVertexMarker )
138  drawVertexMarkers( feature.geometry()->geometry(), context );
139  break;
140  }
141  default:
142  break;
143  }
144  delete segmentized;
145  return true;
146 }

155 void QgsRedliningRendererV2::drawVertexMarkers( QgsAbstractGeometryV2 *geom, QgsRenderContext& context )
156 {
157  const QgsCoordinateTransform* ct = context.coordinateTransform();
158  const QgsMapToPixel& mtp = context.mapToPixel();
159 
160  QgsPointV2 vertexPoint;
161  QgsVertexId vertexId;
162  double x, y, z;
163  while ( geom->nextVertex( vertexId, vertexPoint ) )
164  {
165  //transform
166  x = vertexPoint.x(); y = vertexPoint.y(); z = vertexPoint.z();
167  if ( ct )
168  {
169  ct->transformInPlace( x, y, z );
170  }
171  mtp.transformInPlace( x, y );
172  renderVertexMarker( QPointF( x, y ), context );
173  }
174 }
