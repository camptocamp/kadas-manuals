75 QList<QgsMapLayer *> QgsMapLayerRegistry::addMapLayers(
76  QList<QgsMapLayer *> theMapLayers,
77  bool addToLegend,
78  bool takeOwnership )
79 {
80  QList<QgsMapLayer *> myResultList;
81  for ( int i = 0; i < theMapLayers.size(); ++i )
82  {
83  QgsMapLayer * myLayer = theMapLayers.at( i );
84  if ( !myLayer || !myLayer->isValid() )
85  {
86  QgsDebugMsg( "cannot add invalid layers" );
87  continue;
88  }
89  //check the layer is not already registered!
90  if ( !mMapLayers.contains( myLayer->id() ) )
91  {
92  mMapLayers[myLayer->id()] = myLayer;
93  myResultList << mMapLayers[myLayer->id()];
94  if ( takeOwnership )
95  mOwnedLayers << myLayer;
96  emit layerWasAdded( myLayer );
97  }
98  }
99  if ( myResultList.count() > 0 )
100  {
101  emit layersAdded( myResultList );
102 
103  if ( addToLegend )
104  emit legendLayersAdded( myResultList );
105  }
106  return myResultList;
107 } // QgsMapLayerRegistry::addMapLayers

121 //introduced in 1.8
122 void QgsMapLayerRegistry::removeMapLayers( QStringList theLayerIds )
123 {
124  if ( theLayerIds.isEmpty() )
125  {
126  return;
127  }
128  emit layersWillBeRemoved( theLayerIds );
129 
130  foreach ( const QString &myId, theLayerIds )
131  {
132  QgsMapLayer* lyr = mMapLayers[myId];
133  if ( mOwnedLayers.contains( lyr ) )
134  {
135  emit layerWillBeRemoved( myId );
136  if ( lyr == mRedliningLayer )
137  {
138  mRedliningLayer = nullptr;
139  }
140  else if ( lyr == mGpsRoutesLayer )
141  {
142  mGpsRoutesLayer = nullptr;
143  }
144  delete lyr;
145  mOwnedLayers.remove( lyr );
146  }
147  mMapLayers.remove( myId );
148  emit layerRemoved( myId );
149  }
150  emit layersRemoved( theLayerIds );
151 }
