27 QgsFeaturePicker::PickResult QgsFeaturePicker::pick( const QgsMapCanvas* canvas, const QPoint &canvasPos, const QgsPoint &mapPos, QGis::GeometryType geomType, filter_t filter )
28 {
29  PickResult pickResult;
30 
31  // First, try annotations
32  QgsAnnotationItem* annotationItem = canvas->annotationItemAtPos( canvasPos );
33  if ( annotationItem )
34  {
35  pickResult.annotation = annotationItem;
36  pickResult.boundingBox = annotationItem->screenBoundingRect();
37  return pickResult;
38  }
39 
40  // Then, try labels
41  const QgsLabelingResults* labelingResults = canvas->labelingResults();
42  QList<QgsLabelPosition> labelPositions = labelingResults ? labelingResults->labelsAtPosition( mapPos ) : QList<QgsLabelPosition>();
43  if ( !labelPositions.isEmpty() )
44  {
45  QgsMapLayer* layer = QgsMapLayerRegistry::instance()->mapLayer( labelPositions.front().layerID );
46  if ( layer )
47  {
48  pickResult.layer = layer;
49  pickResult.labelPos = labelPositions.front();
50  return pickResult;
51  }
52  }
53 
54  // Last, try layer features
55  QgsRenderContext renderContext = QgsRenderContext::fromMapSettings( canvas->mapSettings() );
56  double radiusmm = QSettings().value( "/Map/searchRadiusMM", QGis::DEFAULT_SEARCH_RADIUS_MM ).toDouble();
57  radiusmm = radiusmm > 0 ? radiusmm : QGis::DEFAULT_SEARCH_RADIUS_MM;
58  double radiusmu = radiusmm * renderContext.scaleFactor() * renderContext.mapToPixel().mapUnitsPerPixel();
59  QgsRectangle filterRect;
60  filterRect.setXMinimum( mapPos.x() - radiusmu );
61  filterRect.setXMaximum( mapPos.x() + radiusmu );
62  filterRect.setYMinimum( mapPos.y() - radiusmu );
63  filterRect.setYMaximum( mapPos.y() + radiusmu );
64 
65  QgsFeatureList features;
66  foreach ( QgsMapLayer* layer, canvas->layers() )
67  {
68  if ( layer->type() == QgsMapLayer::PluginLayer )
69  {
70  QgsPluginLayer* pluginLayer = static_cast<QgsPluginLayer*>( layer );
71  QVariant result;
72  QRect boundingBox;
73  if ( pluginLayer->testPick( mapPos, canvas->mapSettings(), result, boundingBox ) )
74  {
75  pickResult.layer = layer;
76  pickResult.otherResult = result;
77  pickResult.boundingBox = boundingBox;
78  return pickResult;
79  }
80  }
81  if (( layer->type() != QgsMapLayer::VectorLayer && layer->type() != QgsMapLayer::RedliningLayer ) )
82  {
83  continue;
84  }
85  QgsVectorLayer* vlayer = static_cast<QgsVectorLayer*>( layer );
86  if ( geomType != QGis::AnyGeometry && geomType != QGis::AnyGeometry && vlayer->geometryType() != QGis::AnyGeometry && vlayer->geometryType() != geomType )
87  {
88  continue;
89  }
90  if ( vlayer->hasScaleBasedVisibility() &&
91  ( vlayer->minimumScale() > canvas->mapSettings().scale() ||
92  vlayer->maximumScale() <= canvas->mapSettings().scale() ) )
93  {
94  continue;
95  }
96 
97  QgsFeatureRendererV2* renderer = vlayer->rendererV2();
98  bool filteredRendering = false;
99  if ( renderer && renderer->capabilities() & QgsFeatureRendererV2::ScaleDependent )
100  {
101  // setup scale for scale dependent visibility (rule based)
102  renderer->startRender( renderContext, vlayer->pendingFields() );
103  filteredRendering = renderer->capabilities() & QgsFeatureRendererV2::Filter;
104  }
105 
106  QgsRectangle layerFilterRect = canvas->mapSettings().mapToLayerCoordinates( vlayer, filterRect );
107  QgsFeatureIterator fit = vlayer->getFeatures( QgsFeatureRequest( layerFilterRect ).setFlags( QgsFeatureRequest::ExactIntersect ) );
108  QgsFeature feature;
109  while ( fit.nextFeature( feature ) )
110  {
111  if ( filteredRendering && !renderer->willRenderFeature( feature ) )
112  {
113  continue;
114  }
115  if ( filter && !filter( feature ) )
116  {
117  continue;
118  }
119  if ( geomType != QGis::AnyGeometry && feature.geometry()->type() != geomType )
120  {
121  continue;
122  }
123  features.append( feature );
124  }
125  if ( renderer && renderer->capabilities() & QgsFeatureRendererV2::ScaleDependent )
126  {
127  renderer->stopRender( renderContext );
128  }
129  if ( !features.empty() )
130  {
131  pickResult.layer = vlayer;
132  pickResult.feature = features.front();
133  return pickResult;
134  }
135  }
136  return pickResult;
137 }
