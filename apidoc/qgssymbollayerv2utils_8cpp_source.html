632 #if !defined(GEOS_VERSION_MAJOR) || !defined(GEOS_VERSION_MINOR) || \
633  ((GEOS_VERSION_MAJOR<3) || ((GEOS_VERSION_MAJOR==3) && (GEOS_VERSION_MINOR<3)))
634 // calculate line's angle and tangent
635 static bool lineInfo( QPointF p1, QPointF p2, double& angle, double& t )
636 {
637  double x1 = p1.x(), y1 = p1.y(), x2 = p2.x(), y2 = p2.y();
638 
639  if ( x1 == x2 && y1 == y2 )
640  return false;
641 
642  // tangent
643  t = ( x1 == x2 ? std::numeric_limits<double>::max() : ( y2 - y1 ) / ( x2 - x1 ) );
644 
645  // angle
646  if ( t == std::numeric_limits<double>::max() )
647  angle = ( y2 > y1 ? M_PI / 2 : M_PI * 3 / 2 ); // angle is 90 or 270
648  else if ( t == 0 )

651  angle = ( y2 > y1 ? atan( t ) : M_PI + atan( t ) );
652  else // t < 0
653  angle = ( y2 > y1 ? M_PI + atan( t ) : atan( t ) );
654

664 // calc intersection of two (infinite) lines defined by one point and tangent
665 static QPointF linesIntersection( QPointF p1, double t1, QPointF p2, double t2 )
666 {
667  // parallel lines? (or the difference between angles is less than appr. 10 degree)
668  if (( t1 == std::numeric_limits<double>::max() && t2 == std::numeric_limits<double>::max() ) || qAbs( atan( t1 ) - atan( t2 ) ) < 0.175 )
669  return QPointF();
670 
671  double x, y;
672  if ( t1 == std::numeric_limits<double>::max() || t2 == std::numeric_limits<double>::max() )
673  {
674  // in case one line is with angle 90 resp. 270 degrees (tangent undefined)
675  // swap them so that line 2 is with undefined tangent
676  if ( t1 == std::numeric_limits<double>::max() )
677  {
678  QPointF pSwp = p1; p1 = p2; p2 = pSwp;
679  double tSwp = t1; t1 = t2; t2 = tSwp;
680  }
681 
682  x = p2.x();
683  }
684  else
685  {
686  // usual case
687  x = (( p1.y() - p2.y() ) + t2 * p2.x() - t1 * p1.x() ) / ( t2 - t1 );
688  }
689 
690  y = p1.y() + t1 * ( x - p1.x() );
691  return QPointF( x, y );
692 }

722  {
723  resultLine.append( polyline );
724  return resultLine;
725  }

727  QPolygonF newLine;
728 
729  // need at least geos 3.3 for OffsetCurve tool
730 #if defined(GEOS_VERSION_MAJOR) && defined(GEOS_VERSION_MINOR) && \

732 
733  unsigned int i, pointCount = polyline.count();
734 
735  QgsPolyline tempPolyline( pointCount );
736  QPointF* tempPtr = polyline.data();
737  for ( i = 0; i < pointCount; ++i, tempPtr++ )
738  tempPolyline[i] = QgsPoint( tempPtr->rx(), tempPtr->ry() );
739 
740  QgsGeometry* tempGeometry = geometryType == QGis::Polygon ? QgsGeometry::fromPolygon( QgsPolygon() << tempPolyline ) : QgsGeometry::fromPolyline( tempPolyline );
741  if ( tempGeometry )
742  {
743  int quadSegments = 0; // we want mitre joins, not round joins
744  double mitreLimit = 2.0; // the default value in GEOS (5.0) allows for fairly sharp endings
745  QgsGeometry* offsetGeom = 0;
746  if ( geometryType == QGis::Polygon )
747  offsetGeom = tempGeometry->buffer( -dist, quadSegments, GEOSBUF_CAP_FLAT, GEOSBUF_JOIN_MITRE, mitreLimit );
748  else
749  offsetGeom = tempGeometry->offsetCurve( dist, quadSegments, GEOSBUF_JOIN_MITRE, mitreLimit );

752  {

754  tempGeometry = offsetGeom;

757  {
758  resultLine.append( makeOffsetGeometry( tempGeometry->asPolyline() ) );

760  return resultLine;
761  }
762  else if ( QGis::flatType( tempGeometry->wkbType() ) == QGis::WKBPolygon )
763  {
764  resultLine.append( makeOffsetGeometry( tempGeometry->asPolygon() ) );
765  delete tempGeometry;
766  return resultLine;
767  }
768  else if ( QGis::flatType( tempGeometry->wkbType() ) == QGis::WKBMultiLineString )
769  {
770  QgsMultiPolyline tempMPolyline = tempGeometry->asMultiPolyline();
771 
772  for ( int part = 0; part < tempMPolyline.count(); ++part )
773  {
774  resultLine.append( makeOffsetGeometry( tempMPolyline[ part ] ) );
775  }
776  delete tempGeometry;
777  return resultLine;
778  }
779  else if ( QGis::flatType( tempGeometry->wkbType() ) == QGis::WKBMultiPolygon )
780  {
781  QgsMultiPolygon tempMPolygon = tempGeometry->asMultiPolygon();
782 
783  for ( int part = 0; part < tempMPolygon.count(); ++part )
784  {
785  resultLine.append( makeOffsetGeometry( tempMPolygon[ part ] ) );
786  }
787  delete tempGeometry;
788  return resultLine;
789  }
790  }
791  delete tempGeometry;
792  }
793 
794  // returns original polyline when 'GEOSOffsetCurve' fails!
795  resultLine.append( polyline );
796  return resultLine;
797

799 
800  double angle = 0.0, t_new, t_old = 0;
801  QPointF pt_old, pt_new;
802  QPointF p1 = polyline[0], p2;
803  bool first_point = true;
804 
805  for ( int i = 1; i < polyline.count(); i++ )
806  {
807  p2 = polyline[i];
808 
809  if ( !lineInfo( p1, p2, angle, t_new ) )
810  continue; // not a line...
811 
812  pt_new = offsetPoint( p1, angle + M_PI / 2, dist );
813 
814  if ( ! first_point )
815  {
816  // if it's not the first line segment
817  // calc intersection with last line (with offset)
818  QPointF pt_tmp = linesIntersection( pt_old, t_old, pt_new, t_new );
819  if ( !pt_tmp.isNull() )
820  pt_new = pt_tmp;
821  }
822 
823  newLine.append( pt_new );
824 
825  pt_old = pt_new;
826  t_old = t_new;
827  p1 = p2;
828  first_point = false;
829  }
830 
831  // last line segment:
832  pt_new = offsetPoint( p2, angle + M_PI / 2, dist );
833  newLine.append( pt_new );
834 
835  resultLine.append( newLine );
836  return resultLine;
837 
838 #endif
839 }
