74 QgsRasterRenderer* QgsSingleBandGrayRendererWidget::renderer()
75 {
76  if ( !mRasterLayer )
77  {
78  return 0;
79  }
80  QgsRasterDataProvider* provider = mRasterLayer->dataProvider();
81  if ( !provider )
82  {
83  return 0;
84  }
85  int band = mGrayBandComboBox->itemData( mGrayBandComboBox->currentIndex() ).toInt();
86 
87  QgsContrastEnhancement* e = new QgsContrastEnhancement(( QGis::DataType )(
88  provider->dataType( band ) ) );
89  e->setMinimumValue( mMinLineEdit->text().toDouble() );
90  e->setMaximumValue( mMaxLineEdit->text().toDouble() );
91  e->setContrastEnhancementAlgorithm(( QgsContrastEnhancement::ContrastEnhancementAlgorithm )( mContrastEnhancementComboBox->itemData(
92  mContrastEnhancementComboBox->currentIndex() ).toInt() ) );
93 
94 
95  QgsSingleBandGrayRenderer* renderer = new QgsSingleBandGrayRenderer( provider, band );
96  renderer->setContrastEnhancement( e );
97 
98  renderer->setGradient(( QgsSingleBandGrayRenderer::Gradient ) mGradientComboBox->itemData( mGradientComboBox->currentIndex() ).toInt() );
99 
100  return renderer;
101 }
