41 bool QgsCredentials::get( QString realm, QString &username, QString &password, QString message, bool interactive )
42 {
43  QMap< QString, QPair<QString, QString> >::iterator it = mCredentialCache.find( realm );
44 
45  // If credentials are cached and not equal to the passed ones, try with those
46  if ( it != mCredentialCache.end() && username != it.value().first && password != it.value().second )
47  {
48  QgsDebugMsg( QString( "using cached credentials realm:%1 username:%2 password:%3" ).arg( realm ).arg( username ).arg( password ) );
49  username = it.value().first;
50  password = it.value().second;
51  return true;
52  }
53  // Else, if in interactive mode, request credentials from user
54  else if ( interactive )
55  {
56  QgsDebugMsg( "Requesting credentials from user" );
57  return request( realm, username, password, message );
58  }
59  // Else, fail
60  else
61  {
62  return false;
63  }
64 }
