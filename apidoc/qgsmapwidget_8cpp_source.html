195 void QgsMapWidget::updateLayerSelectionMenu()
196 {
197  QStringList prevDisabledLayers;
198  QStringList prevLayers;
199  foreach ( QAction* action, mLayerSelectionMenu->actions() )
200  {
201  prevLayers.append( action->data().toString() );
202  if ( !action->isChecked() )
203  {
204  prevDisabledLayers.append( action->data().toString() );
205  }
206  }
207  QStringList masterLayers;
208  foreach ( const QgsMapLayer* layer, mMasterCanvas->layers() )
209  {
210  masterLayers.append( layer->id() );
211  }
212 
213  mLayerSelectionMenu->clear();
214  // Use layerTreeRoot to get layers ordered as in the layer tree
215  foreach ( QgsLayerTreeLayer* layerTreeLayer, QgsProject::instance()->layerTreeRoot()->findLayers() )
216  {
217  QgsMapLayer* layer = layerTreeLayer->layer();
218  if ( !layer )
219  continue;
220  QAction* layerAction = new QAction( layer->name(), mLayerSelectionMenu );
221  layerAction->setData( layer->id() );
222  layerAction->setCheckable( true );
223  if ( !mInitialLayers.isEmpty() )
224  {
225  layerAction->setChecked( mInitialLayers.contains( layer->id() ) );
226  }
227  else
228  {
229  bool wasDisabled = prevDisabledLayers.contains( layer->id() );
230  bool isNewEnabledLayer = !prevLayers.contains( layer->id() ) && masterLayers.contains( layer->id() );
231  layerAction->setChecked(( prevLayers.contains( layer->id() ) && !wasDisabled ) || isNewEnabledLayer );
232  }
233  connect( layerAction, SIGNAL( toggled( bool ) ), this, SLOT( updateLayerSet() ) );
234  mLayerSelectionMenu->addAction( layerAction );
235  }
236  updateLayerSet();
237  mInitialLayers.clear();
238 }

240 void QgsMapWidget::updateLayerSet()
241 {
242  QList<QgsMapCanvasLayer> layerSet;
243  foreach ( QAction* layerAction, mLayerSelectionMenu->actions() )
244  {
245  if ( layerAction->isChecked() )
246  {
247  QgsMapLayer* layer = QgsMapLayerRegistry::instance()->mapLayer( layerAction->data().toString() );
248  connect( layer, SIGNAL( repaintRequested() ), mMapCanvas, SLOT( refresh() ) );
249  layerSet.append( QgsMapCanvasLayer( layer ) );
250  }
251  }
252  foreach ( QgsMapLayer* layer, mMapCanvas->layers() )
253  {
254  disconnect( layer, SIGNAL( repaintRequested() ), mMapCanvas, SLOT( refresh() ) );
255  }
256 
257  mMapCanvas->setLayerSet( layerSet );
258 }

279 bool QgsMapWidget::eventFilter( QObject *obj, QEvent *ev )
280 {
281  if ( obj == mTitleLabel && ev->type() == QEvent::MouseButtonPress )
282  {
283  mTitleStackedWidget->setCurrentWidget( mTitleLineEdit );
284  mTitleLineEdit->setText( mTitleLabel->text() );
285  mTitleLineEdit->setFocus();
286  mTitleLineEdit->selectAll();
287  return true;
288  }
289  else if ( obj == mTitleLineEdit && ev->type() == QEvent::FocusOut )
290  {
291  setWindowTitle( mTitleLineEdit->text() );
292  mTitleLabel->setText( mTitleLineEdit->text() );
293  mTitleStackedWidget->setCurrentWidget( mTitleLabel );
294  return true;
295  }
296  else if ( obj == mTitleLineEdit && ev->type() == QEvent::KeyPress && (
297  static_cast<QKeyEvent*>( ev )->key() == Qt::Key_Enter ||
298  static_cast<QKeyEvent*>( ev )->key() == Qt::Key_Return ||
299  static_cast<QKeyEvent*>( ev )->key() == Qt::Key_Escape ) )
300  {
301  setWindowTitle( mTitleLineEdit->text() );
302  mTitleLabel->setText( mTitleLineEdit->text() );
303  mTitleStackedWidget->setCurrentWidget( mTitleLabel );
304  return true;
305  }
306  else
307  {
308  return QObject::eventFilter( obj, ev );
309  }
310 }
